1212 {
1213  ProfilerGraph graph;
1214  RunStats stats = {};
1215  CameraOrientation cam_view_target = {};
1216  CameraOrientation cam_view = {};
1217  FpsControl draw_times;
1218  f32 dtime; // in seconds
1219 
1220  /* Clear the profiler */
1221  Profiler::GraphValues dummyvalues;
1222  g_profiler->graphGet(dummyvalues);
1223 
1224  draw_times.reset();
1225 
1226  set_light_table(g_settings->getFloat("display_gamma"));
1227 
1228 #ifdef HAVE_TOUCHSCREENGUI
1229  m_cache_hold_aux1 = g_settings->getBool("fast_move")
1230  && client->checkPrivilege("fast");
1231 #endif
1232 
1233  const irr::core::dimension2du initial_screen_size(
1234  g_settings->getU16("screen_w"),
1235  g_settings->getU16("screen_h")
1236  );
1237  const bool initial_window_maximized = g_settings->getBool("window_maximized");
1238 
1239  while (m_rendering_engine->run()
1240  && !(*kill || g_gamecallback->shutdown_requested
1241  || (server && server->isShutdownRequested()))) {
1242 
1243  // Calculate dtime =
1244  // m_rendering_engine->run() from this iteration
1245  // + Sleep time until the wanted FPS are reached
1246  draw_times.limit(device, &dtime);
1247 
1248  const auto current_dynamic_info = getCurrentDynamicInfo();
1249  if (!current_dynamic_info.equal(client_display_info)) {
1250  client_display_info = current_dynamic_info;
1251  dynamic_info_send_timer = 0.2f;
1252  }
1253 
1254  if (dynamic_info_send_timer > 0.0f) {
1255  dynamic_info_send_timer -= dtime;
1256  if (dynamic_info_send_timer <= 0.0f) {
1257  client->sendUpdateClientInfo(current_dynamic_info);
1258  }
1259  }
1260 
1261  // Prepare render data for next iteration
1262 
1263  updateStats(&stats, draw_times, dtime);
1264  updateInteractTimers(dtime);
1265 
1266  if (!checkConnection())
1267  break;
1268  if (!handleCallbacks())
1269  break;
1270 
1271  processQueues();
1272 
1273  m_game_ui->clearInfoText();
1274 
1275  updateProfilers(stats, draw_times, dtime);
1276  processUserInput(dtime);
1277  // Update camera before player movement to avoid camera lag of one frame
1278  updateCameraDirection(&cam_view_target, dtime);
1279  cam_view.camera_yaw += (cam_view_target.camera_yaw -
1280  cam_view.camera_yaw) * m_cache_cam_smoothing;
1281  cam_view.camera_pitch += (cam_view_target.camera_pitch -
1282  cam_view.camera_pitch) * m_cache_cam_smoothing;
1283  updatePlayerControl(cam_view);
1284 
1285  {
1286  bool was_paused = m_is_paused;
1287  m_is_paused = simple_singleplayer_mode && g_menumgr.pausesGame();
1288  if (m_is_paused)
1289  dtime = 0.0f;
1290 
1291  if (!was_paused && m_is_paused)
1292  pauseAnimation();
1293  else if (was_paused && !m_is_paused)
1294  resumeAnimation();
1295  }
1296 
1297  if (!m_is_paused)
1298  step(dtime);
