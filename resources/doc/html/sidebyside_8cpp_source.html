49 void populateSideBySidePipeline(RenderPipeline *pipeline, Client *client, bool horizontal, bool flipped, v2f &virtual_size_scale)
50 {
51  static const u8 TEXTURE_LEFT = 0;
52  static const u8 TEXTURE_RIGHT = 1;
53 
54  v2f offset;
55  if (horizontal) {
56  virtual_size_scale = v2f(1.0f, 0.5f);
57  offset = v2f(0.0f, 0.5f);
58  }
59  else {
60  virtual_size_scale = v2f(0.5f, 1.0f);
61  offset = v2f(0.5f, 0.0f);
62  }
63 
64  TextureBuffer *buffer = pipeline->createOwned<TextureBuffer>();
65  buffer->setTexture(TEXTURE_LEFT, virtual_size_scale, "3d_render_left", video::ECF_A8R8G8B8);
66  buffer->setTexture(TEXTURE_RIGHT, virtual_size_scale, "3d_render_right", video::ECF_A8R8G8B8);
67 
68  auto step3D = pipeline->own(create3DStage(client, virtual_size_scale));
69 
70  // eyes
71  for (bool right : { false, true }) {
72  pipeline->addStep<OffsetCameraStep>(flipped ? !right : right);
73  auto output = pipeline->createOwned<TextureBufferOutput>(buffer, right ? TEXTURE_RIGHT : TEXTURE_LEFT);
74  pipeline->addStep<SetRenderTargetStep>(step3D, output);
75  pipeline->addStep(step3D);
76  pipeline->addStep<DrawWield>();
77  pipeline->addStep<MapPostFxStep>();
78  pipeline->addStep<DrawHUD>();
79  }
80 
81  pipeline->addStep<OffsetCameraStep>(0.0f);
82 
83  auto screen = pipeline->createOwned<ScreenTarget>();
84 
85  for (bool right : { false, true }) {
86  auto step = pipeline->addStep<DrawImageStep>(
87  right ? TEXTURE_RIGHT : TEXTURE_LEFT,
88  right ? offset : v2f());
89  step->setRenderSource(buffer);
90  step->setRenderTarget(screen);
91  }
92 }
