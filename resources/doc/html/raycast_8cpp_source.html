26 bool RaycastSort::operator() (const PointedThing &pt1,
27  const PointedThing &pt2) const
28 {
29  // "nothing" cannot be sorted
30  assert(pt1.type != POINTEDTHING_NOTHING);
31  assert(pt2.type != POINTEDTHING_NOTHING);
32  f32 pt1_distSq = pt1.distanceSq;
33 
34  // Add some bonus when one of them is an object
35  if (pt1.type != pt2.type) {
36  if (pt1.type == POINTEDTHING_OBJECT)
37  pt1_distSq -= BS * BS;
38  else if (pt2.type == POINTEDTHING_OBJECT)
39  pt1_distSq += BS * BS;
40  }
41 
42  // returns false if pt1 is nearer than pt2
43  if (pt1_distSq < pt2.distanceSq) {
44  return false;
45  }
46 
47  if (pt1_distSq == pt2.distanceSq) {
48  // Sort them to allow only one order
49  if (pt1.type == POINTEDTHING_OBJECT)
50  return (pt2.type == POINTEDTHING_OBJECT
51  && pt1.object_id < pt2.object_id);
52 
53  return (pt2.type == POINTEDTHING_OBJECT
54  || pt1.node_undersurface < pt2.node_undersurface);
55  }
56  return true;
57 }

71 bool boxLineCollision(const aabb3f &box, const v3f &start,
72  const v3f &dir, v3f *collision_point, v3f *collision_normal)
73 {
74  if (box.isPointInside(start)) {
75  *collision_point = start;
76  collision_normal->set(0, 0, 0);
77  return true;
78  }
79  float m = 0;
80 
81  // Test X collision
82  if (dir.X != 0) {
83  if (dir.X > 0)
84  m = (box.MinEdge.X - start.X) / dir.X;
85  else
86  m = (box.MaxEdge.X - start.X) / dir.X;
87 
88  if (m >= 0 && m <= 1) {
89  *collision_point = start + dir * m;
90  if ((collision_point->Y >= box.MinEdge.Y)
91  && (collision_point->Y <= box.MaxEdge.Y)
92  && (collision_point->Z >= box.MinEdge.Z)
93  && (collision_point->Z <= box.MaxEdge.Z)) {
94  collision_normal->set((dir.X > 0) ? -1 : 1, 0, 0);
95  return true;
96  }
97  }
98  }
99 
100  // Test Y collision
101  if (dir.Y != 0) {
102  if (dir.Y > 0)
103  m = (box.MinEdge.Y - start.Y) / dir.Y;
104  else
105  m = (box.MaxEdge.Y - start.Y) / dir.Y;
106 
107  if (m >= 0 && m <= 1) {
108  *collision_point = start + dir * m;
109  if ((collision_point->X >= box.MinEdge.X)
110  && (collision_point->X <= box.MaxEdge.X)
111  && (collision_point->Z >= box.MinEdge.Z)
112  && (collision_point->Z <= box.MaxEdge.Z)) {
113  collision_normal->set(0, (dir.Y > 0) ? -1 : 1, 0);
114  return true;
115  }
116  }
117  }
118 
119  // Test Z collision
120  if (dir.Z != 0) {
121  if (dir.Z > 0)
122  m = (box.MinEdge.Z - start.Z) / dir.Z;
123  else
124  m = (box.MaxEdge.Z - start.Z) / dir.Z;
125 
126  if (m >= 0 && m <= 1) {
127  *collision_point = start + dir * m;
128  if ((collision_point->X >= box.MinEdge.X)
129  && (collision_point->X <= box.MaxEdge.X)
130  && (collision_point->Y >= box.MinEdge.Y)
131  && (collision_point->Y <= box.MaxEdge.Y)) {
132  collision_normal->set(0, 0, (dir.Z > 0) ? -1 : 1);
133  return true;
134  }
135  }
136  }
137  return false;
138 }

140 bool boxLineCollision(const aabb3f &box, const v3f &rotation,
141  const v3f &start, const v3f &dir,
142  v3f *collision_point, v3f *collision_normal, v3f *raw_collision_normal)
143 {
144  // Inversely transform the ray rather than rotating the box faces;
145  // this allows us to continue using a simple ray - AABB intersection
146  core::quaternion rot(rotation * core::DEGTORAD);
147  rot.makeInverse();
148 
149  bool collision = boxLineCollision(box, rot * start, rot * dir, collision_point, collision_normal);
150  if (!collision) return collision;
151 
152  // Transform the results back
153  rot.makeInverse();
154  *collision_point = rot * *collision_point;
155  *raw_collision_normal = *collision_normal;
156  *collision_normal = rot * *collision_normal;
157  return collision;
158 }
