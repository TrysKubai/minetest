203 inline float modulo360f(float f)
204 {
205  int sign;
206  int whole;
207  float fraction;
208 
209  if (f < 0) {
210  f = -f;
211  sign = -1;
212  } else {
213  sign = 1;
214  }
215 
216  whole = f;
217 
218  fraction = f - whole;
219  whole %= 360;
220 
221  return sign * (whole + fraction);
222 }

393 /*
394  Splits a list into "pages". For example, the list [1,2,3,4,5] split
395  into two pages would be [1,2,3],[4,5]. This function computes the
396  minimum and maximum indices of a single page.
397 
398  length: Length of the list that should be split
399  page: Page number, 1 <= page <= pagecount
400  pagecount: The number of pages, >= 1
401  minindex: Receives the minimum index (inclusive).
402  maxindex: Receives the maximum index (exclusive).
403 
404  Ensures 0 <= minindex <= maxindex <= length.
405 */
406 inline void paging(u32 length, u32 page, u32 pagecount, u32 &minindex, u32 &maxindex)
407 {
408  if (length < 1 || pagecount < 1 || page < 1 || page > pagecount) {
409  // Special cases or invalid parameters
410  minindex = maxindex = 0;
411  } else if(pagecount <= length) {
412  // Less pages than entries in the list:
413  // Each page contains at least one entry
414  minindex = (length * (page-1) + (pagecount-1)) / pagecount;
415  maxindex = (length * page + (pagecount-1)) / pagecount;
416  } else {
417  // More pages than entries in the list:
418  // Make sure the empty pages are at the end
419  if (page < length) {
420  minindex = page-1;
421  maxindex = page;
422  } else {
423  minindex = 0;
424  maxindex = 0;
425  }
426  }
427 }

453 // Gradual steps towards the target value in a wrapped (circular) system
454 // using the shorter of both ways
455 template<typename T>
456 inline void wrappedApproachShortest(T &current, const T target, const T stepsize,
457  const T maximum)
458 {
459  T delta = target - current;
460  if (delta < 0)
461  delta += maximum;
462 
463  if (delta > stepsize && maximum - delta > stepsize) {
464  current += (delta < maximum / 2) ? stepsize : -stepsize;
465  if (current >= maximum)
466  current -= maximum;
467  } else {
468  current = target;
469  }
470 }
