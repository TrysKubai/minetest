112 bool GUIVolumeChange::OnEvent(const SEvent& event)
113 {
114  if (event.EventType == EET_KEY_INPUT_EVENT) {
115  if (event.KeyInput.Key == KEY_ESCAPE && event.KeyInput.PressedDown) {
116  quitMenu();
117  return true;
118  }
119 
120  if (event.KeyInput.Key == KEY_RETURN && event.KeyInput.PressedDown) {
121  quitMenu();
122  return true;
123  }
124  } else if (event.EventType == EET_GUI_EVENT) {
125  if (event.GUIEvent.EventType == gui::EGET_CHECKBOX_CHANGED) {
126  gui::IGUIElement *e = getElementFromId(ID_soundMuteButton);
127  if (e != NULL && e->getType() == gui::EGUIET_CHECK_BOX) {
128  g_settings->setBool("mute_sound", ((gui::IGUICheckBox*)e)->isChecked());
129  }
130 
131  Environment->setFocus(this);
132  return true;
133  }
134 
135  if (event.GUIEvent.EventType == gui::EGET_BUTTON_CLICKED) {
136  if (event.GUIEvent.Caller->getID() == ID_soundExitButton) {
137  quitMenu();
138  return true;
139  }
140  Environment->setFocus(this);
141  }
142 
143  if (event.GUIEvent.EventType == gui::EGET_ELEMENT_FOCUS_LOST
144  && isVisible()) {
145  if (!canTakeFocus(event.GUIEvent.Element)) {
146  infostream << "GUIVolumeChange: Not allowing focus change."
147  << std::endl;
148  // Returning true disables focus change
149  return true;
150  }
151  }
152  if (event.GUIEvent.EventType == gui::EGET_SCROLL_BAR_CHANGED) {
153  if (event.GUIEvent.Caller->getID() == ID_soundSlider) {
154  s32 pos = ((gui::IGUIScrollBar*)event.GUIEvent.Caller)->getPos();
155  g_settings->setFloat("sound_volume", (float) pos / 100);
156 
157  gui::IGUIElement *e = getElementFromId(ID_soundText);
158  e->setText(fwgettext("Sound Volume: %d%%", pos).c_str());
159  return true;
160  }
161  }
162 
163  }
164 
165  return Parent ? Parent->OnEvent(event) : false;
166 }
