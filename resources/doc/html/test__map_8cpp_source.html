53 void TestMap::testMaxMapgenLimit()
54 {
55  // limit must end on a mapblock boundary
56  UASSERTEQ(int, MAX_MAP_GENERATION_LIMIT % MAP_BLOCKSIZE, MAP_BLOCKSIZE - 1);
57 
58  // objectpos_over_limit should do exactly this except the last node
59  // actually spans from LIMIT-0.5 to LIMIT+0.5
60  float limit_times_bs = MAX_MAP_GENERATION_LIMIT * BS;
61  UASSERT(objectpos_over_limit(v3f(limit_times_bs-BS/2)) == false);
62  UASSERT(objectpos_over_limit(v3f(limit_times_bs)) == false);
63  UASSERT(objectpos_over_limit(v3f(limit_times_bs+BS/2)) == false);
64  UASSERT(objectpos_over_limit(v3f(limit_times_bs+BS)) == true);
65 
66  UASSERT(objectpos_over_limit(v3f(-limit_times_bs+BS/2)) == false);
67  UASSERT(objectpos_over_limit(v3f(-limit_times_bs)) == false);
68  UASSERT(objectpos_over_limit(v3f(-limit_times_bs-BS/2)) == false);
69  UASSERT(objectpos_over_limit(v3f(-limit_times_bs-BS)) == true);
70 
71  // blockpos_over_max_limit
72  s16 limit_block = MAX_MAP_GENERATION_LIMIT / MAP_BLOCKSIZE;
73  UASSERT(blockpos_over_max_limit(v3s16(limit_block)) == false);
74  UASSERT(blockpos_over_max_limit(v3s16(limit_block+1)) == true);
75  UASSERT(blockpos_over_max_limit(v3s16(-limit_block)) == false);
76  UASSERT(blockpos_over_max_limit(v3s16(-limit_block-1)) == true);
77 }

79 void TestMap::testForEachNodeInArea(IGameDef *gamedef)
80 {
81  v3s16 minp_visit(-10, -10, -10);
82  v3s16 maxp_visit(20, 20, 10);
83  v3s16 dims_visit = maxp_visit - minp_visit + v3s16(1, 1, 1);
84  s32 volume_visit = (s32)dims_visit.X * (s32)dims_visit.Y * (s32)dims_visit.Z;
85 
86  v3s16 minp = minp_visit - v3s16(1, 1, 1);
87  v3s16 maxp = maxp_visit + v3s16(1, 1, 1);
88  DummyMap map(gamedef, getNodeBlockPos(minp), getNodeBlockPos(maxp));
89 
90  v3s16 p1(0, 10, 5);
91  MapNode n1(t_CONTENT_STONE);
92  map.setNode(p1, n1);
93 
94  v3s16 p2(-1, 15, 5);
95  MapNode n2(t_CONTENT_TORCH);
96  map.setNode(p2, n2);
97 
98  v3s16 p3 = minp_visit;
99  MapNode n3(CONTENT_AIR);
100  map.setNode(p3, n3);
101 
102  v3s16 p4 = maxp_visit;
103  MapNode n4(t_CONTENT_LAVA);
104  map.setNode(p4, n4);
105 
106  // These positions should not be visited.
107  map.setNode(minp, MapNode(t_CONTENT_WATER));
108  map.setNode(maxp, MapNode(t_CONTENT_WATER));
109 
110  s32 n_visited = 0;
111  std::unordered_set<v3s16> visited;
112  v3s16 minp_visited(0, 0, 0);
113  v3s16 maxp_visited(0, 0, 0);
114  std::unordered_map<v3s16, MapNode> found;
115  map.forEachNodeInArea(minp_visit, maxp_visit, [&](v3s16 p, MapNode n) -> bool {
116  n_visited++;
117  visited.insert(p);
118  minp_visited.X = std::min(minp_visited.X, p.X);
119  minp_visited.Y = std::min(minp_visited.Y, p.Y);
120  minp_visited.Z = std::min(minp_visited.Z, p.Z);
121  maxp_visited.X = std::max(maxp_visited.X, p.X);
122  maxp_visited.Y = std::max(maxp_visited.Y, p.Y);
123  maxp_visited.Z = std::max(maxp_visited.Z, p.Z);
124 
125  if (n.getContent() != CONTENT_IGNORE)
126  found[p] = n;
127 
128  return true;
129  });
130 
131  UASSERTEQ(s32, n_visited, volume_visit);
132  UASSERTEQ(s32, (s32)visited.size(), volume_visit);
133  UASSERT(minp_visited == minp_visit);
134  UASSERT(maxp_visited == maxp_visit);
135 
136  UASSERTEQ(size_t, found.size(), 4);
137  UASSERT(found.find(p1) != found.end());
138  UASSERTEQ(content_t, found[p1].getContent(), n1.getContent());
139  UASSERT(found.find(p2) != found.end());
140  UASSERTEQ(content_t, found[p2].getContent(), n2.getContent());
141  UASSERT(found.find(p3) != found.end());
142  UASSERTEQ(content_t, found[p3].getContent(), n3.getContent());
143  UASSERT(found.find(p4) != found.end());
144  UASSERTEQ(content_t, found[p4].getContent(), n4.getContent());
145 }

147 void TestMap::testForEachNodeInAreaBlank(IGameDef *gamedef)
148 {
149  DummyMap map(gamedef, v3s16(0, 0, 0), v3s16(-1, -1, -1));
150 
151  v3s16 invalid_p(0, 0, 0);
152  bool visited = false;
153  map.forEachNodeInArea(invalid_p, invalid_p, [&](v3s16 p, MapNode n) -> bool {
154  bool is_valid_position = true;
155  UASSERT(n == map.getNode(p, &is_valid_position));
156  UASSERT(!is_valid_position);
157  UASSERT(!visited);
158  visited = true;
159  return true;
160  });
161  UASSERT(visited);
162 }
