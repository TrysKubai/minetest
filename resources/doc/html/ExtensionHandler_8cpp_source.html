18  void COpenGL3ExtensionHandler::initExtensionsOld()
19  {
20  auto extensions_string = reinterpret_cast<const char *>(glGetString(GL_EXTENSIONS));
21  const char *pos = extensions_string;
22  while (const char *next = strchr(pos, ' ')) {
23  addExtension(std::string{pos, next});
