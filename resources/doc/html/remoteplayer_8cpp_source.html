73 RemotePlayerChatResult RemotePlayer::canSendChatMessage()
74 {
75  // Rate limit messages
76  u32 now = time(NULL);
77  float time_passed = now - m_last_chat_message_sent;
78  m_last_chat_message_sent = now;
79 
80  // If this feature is disabled
81  if (m_setting_chat_message_limit_per_10sec <= 0.0) {
82  return RPLAYER_CHATRESULT_OK;
83  }
84 
85  m_chat_message_allowance += time_passed * (m_setting_chat_message_limit_per_10sec / 8.0f);
86  if (m_chat_message_allowance > m_setting_chat_message_limit_per_10sec) {
87  m_chat_message_allowance = m_setting_chat_message_limit_per_10sec;
88  }
89 
90  if (m_chat_message_allowance < 1.0f) {
91  infostream << "Player " << m_name
92  << " chat limited due to excessive message amount." << std::endl;
93 
94  // Kick player if flooding is too intensive
95  m_message_rate_overhead++;
96  if (m_message_rate_overhead > RemotePlayer::m_setting_chat_message_limit_trigger_kick) {
97  return RPLAYER_CHATRESULT_KICK;
98  }
99 
100  return RPLAYER_CHATRESULT_FLOODING;
101  }
102 
103  // Reinit message overhead
104  if (m_message_rate_overhead > 0) {
105  m_message_rate_overhead = 0;
106  }
107 
108  m_chat_message_allowance -= 1.0f;
109  return RPLAYER_CHATRESULT_OK;
110 }
