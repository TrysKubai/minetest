267 bool CGUIEditBox::processKey(const SEvent& event)
268 {
269  if (!event.KeyInput.PressedDown)
270  return false;
271 
272  bool textChanged = false;
273  s32 newMarkBegin = MarkBegin;
274  s32 newMarkEnd = MarkEnd;
275 
276  // control shortcut handling
277 
278  if (event.KeyInput.Control)
279  {
280  // german backlash '\' entered with control + '?'
281  if ( event.KeyInput.Char == '\\' )
282  {
283  inputChar(event.KeyInput.Char);
284  return true;
285  }
286 
287  switch(event.KeyInput.Key)
288  {
289  case KEY_KEY_A:
290  // select all
291  newMarkBegin = 0;
292  newMarkEnd = Text.size();
293  break;
294  case KEY_KEY_C:
295  // copy to clipboard
296  if (!PasswordBox && Operator && MarkBegin != MarkEnd)
297  {
298  const s32 realmbgn = MarkBegin < MarkEnd ? MarkBegin : MarkEnd;
299  const s32 realmend = MarkBegin < MarkEnd ? MarkEnd : MarkBegin;
300 
301  core::stringc s;
302  wStringToMultibyte(s, Text.subString(realmbgn, realmend - realmbgn));
303  Operator->copyToClipboard(s.c_str());
304  }
305  break;
306  case KEY_KEY_X:
307  // cut to the clipboard
308  if (!PasswordBox && Operator && MarkBegin != MarkEnd)
309  {
310  const s32 realmbgn = MarkBegin < MarkEnd ? MarkBegin : MarkEnd;
311  const s32 realmend = MarkBegin < MarkEnd ? MarkEnd : MarkBegin;
312 
313  // copy
314  core::stringc sc;
315  wStringToMultibyte(sc, Text.subString(realmbgn, realmend - realmbgn));
316  Operator->copyToClipboard(sc.c_str());
317 
318  if (isEnabled())
319  {
320  // delete
321  core::stringw s;
322  s = Text.subString(0, realmbgn);
323  s.append( Text.subString(realmend, Text.size()-realmend) );
324  Text = s;
325 
326  CursorPos = realmbgn;
327  newMarkBegin = 0;
328  newMarkEnd = 0;
329  textChanged = true;
330  }
331  }
332  break;
333  case KEY_KEY_V:
334  if ( !isEnabled() )
335  break;
336 
337  // paste from the clipboard
338  if (Operator)
339  {
340  const s32 realmbgn = MarkBegin < MarkEnd ? MarkBegin : MarkEnd;
341  const s32 realmend = MarkBegin < MarkEnd ? MarkEnd : MarkBegin;
342 
343  // add the string
344  const c8 *p = Operator->getTextFromClipboard();
345  if (p)
346  {
347  irr::core::stringw widep;
348  core::multibyteToWString(widep, p);
349 
350  if (MarkBegin == MarkEnd)
351  {
352  // insert text
353  core::stringw s = Text.subString(0, CursorPos);
354  s.append(widep);
355  s.append( Text.subString(CursorPos, Text.size()-CursorPos) );
356 
357  if (!Max || s.size()<=Max) // thx to Fish FH for fix
358  {
359  Text = s;
360  s = widep;
361  CursorPos += s.size();
362  }
363  }
364  else
365  {
366  // replace text
367 
368  core::stringw s = Text.subString(0, realmbgn);
369  s.append(widep);
370  s.append( Text.subString(realmend, Text.size()-realmend) );
371 
372  if (!Max || s.size()<=Max) // thx to Fish FH for fix
373  {
374  Text = s;
375  s = widep;
376  CursorPos = realmbgn + s.size();
377  }
378  }
379  }
380 
381  newMarkBegin = 0;
382  newMarkEnd = 0;
383  textChanged = true;
384  }
385  break;
386  case KEY_HOME:
387  // move/highlight to start of text
388  if (event.KeyInput.Shift)
389  {
390  newMarkEnd = CursorPos;
391  newMarkBegin = 0;
392  CursorPos = 0;
393  }
394  else
395  {
396  CursorPos = 0;
397  newMarkBegin = 0;
398  newMarkEnd = 0;
399  }
400  break;
401  case KEY_END:
402  // move/highlight to end of text
403  if (event.KeyInput.Shift)
404  {
405  newMarkBegin = CursorPos;
406  newMarkEnd = Text.size();
407  CursorPos = 0;
408  }
409  else
410  {
411  CursorPos = Text.size();
412  newMarkBegin = 0;
413  newMarkEnd = 0;
414  }
415  break;
416  default:
417  return false;
418  }
419  }
420  // Some special keys - but only handle them if KeyInput.Char is null as on some systems (X11) they might have same key-code as ansi-keys otherwise
421  else if (event.KeyInput.Char == 0)
422  {
423  switch(event.KeyInput.Key)
424  {
425  case KEY_END:
426  {
427  s32 p = Text.size();
428  if (WordWrap || MultiLine)
429  {
430  p = getLineFromPos(CursorPos);
431  p = BrokenTextPositions[p] + (s32)BrokenText[p].size();
432  if (p > 0 && (Text[p-1] == L'\r' || Text[p-1] == L'\n' ))
433  p-=1;
434  }
435 
436  if (event.KeyInput.Shift)
437  {
438  if (MarkBegin == MarkEnd)
439  newMarkBegin = CursorPos;
440 
441  newMarkEnd = p;
442  }
443  else
444  {
445  newMarkBegin = 0;
446  newMarkEnd = 0;
447  }
448  CursorPos = p;
449  BlinkStartTime = os::Timer::getTime();
450  }
451  break;
452  case KEY_HOME:
453  {
454 
455  s32 p = 0;
456  if (WordWrap || MultiLine)
457  {
458  p = getLineFromPos(CursorPos);
459  p = BrokenTextPositions[p];
460  }
461 
462  if (event.KeyInput.Shift)
463  {
464  if (MarkBegin == MarkEnd)
465  newMarkBegin = CursorPos;
466  newMarkEnd = p;
467  }
468  else
469  {
470  newMarkBegin = 0;
471  newMarkEnd = 0;
472  }
473  CursorPos = p;
474  BlinkStartTime = os::Timer::getTime();
475  }
476  break;
477  case KEY_LEFT:
478 
479  if (event.KeyInput.Shift)
480  {
481  if (CursorPos > 0)
482  {
483  if (MarkBegin == MarkEnd)
484  newMarkBegin = CursorPos;
485 
486  newMarkEnd = CursorPos-1;
487  }
488  }
489  else
490  {
491  newMarkBegin = 0;
492  newMarkEnd = 0;
493  }
494 
495  if (CursorPos > 0) CursorPos--;
496  BlinkStartTime = os::Timer::getTime();
497  break;
498 
499  case KEY_RIGHT:
500  if (event.KeyInput.Shift)
501  {
502  if (Text.size() > (u32)CursorPos)
503  {
504  if (MarkBegin == MarkEnd)
505  newMarkBegin = CursorPos;
506 
507  newMarkEnd = CursorPos+1;
508  }
509  }
510  else
511  {
512  newMarkBegin = 0;
513  newMarkEnd = 0;
514  }
515 
516  if (Text.size() > (u32)CursorPos) CursorPos++;
517  BlinkStartTime = os::Timer::getTime();
518  break;
519  case KEY_UP:
520  if (MultiLine || (WordWrap && BrokenText.size() > 1) )
521  {
522  s32 lineNo = getLineFromPos(CursorPos);
523  s32 mb = (MarkBegin == MarkEnd) ? CursorPos : (MarkBegin > MarkEnd ? MarkBegin : MarkEnd);
524  if (lineNo > 0)
525  {
526  s32 cp = CursorPos - BrokenTextPositions[lineNo];
527  if ((s32)BrokenText[lineNo-1].size() < cp)
528  CursorPos = BrokenTextPositions[lineNo-1] + core::max_((u32)1, BrokenText[lineNo-1].size())-1;
529  else
530  CursorPos = BrokenTextPositions[lineNo-1] + cp;
531  }
532 
533  if (event.KeyInput.Shift)
534  {
535  newMarkBegin = mb;
536  newMarkEnd = CursorPos;
537  }
538  else
539  {
540  newMarkBegin = 0;
541  newMarkEnd = 0;
542  }
543 
544  }
545  else
546  {
547  return false;
548  }
549  break;
550  case KEY_DOWN:
551  if (MultiLine || (WordWrap && BrokenText.size() > 1) )
552  {
553  s32 lineNo = getLineFromPos(CursorPos);
554  s32 mb = (MarkBegin == MarkEnd) ? CursorPos : (MarkBegin < MarkEnd ? MarkBegin : MarkEnd);
555  if (lineNo < (s32)BrokenText.size()-1)
556  {
557  s32 cp = CursorPos - BrokenTextPositions[lineNo];
558  if ((s32)BrokenText[lineNo+1].size() < cp)
559  CursorPos = BrokenTextPositions[lineNo+1] + core::max_((u32)1, BrokenText[lineNo+1].size())-1;
560  else
561  CursorPos = BrokenTextPositions[lineNo+1] + cp;
562  }
563 
564  if (event.KeyInput.Shift)
565  {
566  newMarkBegin = mb;
567  newMarkEnd = CursorPos;
568  }
569  else
570  {
571  newMarkBegin = 0;
572  newMarkEnd = 0;
573  }
574 
575  }
576  else
577  {
578  return false;
579  }
580  break;
581  case KEY_INSERT:
582  if ( !isEnabled() )
583  break;
584 
585  OverwriteMode = !OverwriteMode;
586  break;
587  case KEY_DELETE:
588  if ( !isEnabled() )
589  break;
590 
591  if (keyDelete())
592  {
593  BlinkStartTime = os::Timer::getTime();
594  newMarkBegin = 0;
595  newMarkEnd = 0;
596  textChanged = true;
597  }
598  break;
599  default:
600  return false;
601  }
602  }
603  else
604  {
605  // default keyboard handling
606  switch(event.KeyInput.Key)
607  {
608  case KEY_RETURN:
609  if (MultiLine)
610  {
611  inputChar(L'\n');
612  }
613  else
614  {
615  calculateScrollPos();
616  sendGuiEvent( EGET_EDITBOX_ENTER );
617  }
618  return true;
619 
620  case KEY_BACK:
621  if ( !isEnabled() )
622  break;
623 
624  if (Text.size())
625  {
626  core::stringw s;
627 
628  if (MarkBegin != MarkEnd)
629  {
630  // delete marked text
631  const s32 realmbgn = MarkBegin < MarkEnd ? MarkBegin : MarkEnd;
632  const s32 realmend = MarkBegin < MarkEnd ? MarkEnd : MarkBegin;
633 
634  s = Text.subString(0, realmbgn);
635  s.append( Text.subString(realmend, Text.size()-realmend) );
636  Text = s;
637 
638  CursorPos = realmbgn;
639  }
640  else
641  {
642  // delete text behind cursor
643  if (CursorPos>0)
644  s = Text.subString(0, CursorPos-1);
645  else
646  s = L"";
647  s.append( Text.subString(CursorPos, Text.size()-CursorPos) );
648  Text = s;
649  --CursorPos;
650  }
651 
652  if (CursorPos < 0)
653  CursorPos = 0;
654  BlinkStartTime = os::Timer::getTime();
655  newMarkBegin = 0;
656  newMarkEnd = 0;
657  textChanged = true;
658  }
659  break;
660 
661  case KEY_DELETE:
662 
663  // At least on X11 we get a char with 127 when the delete key is pressed.
664  // We get no char when the delete key on numkeys is pressed with numlock off (handled in the other case calling keyDelete as Char is then 0).
665  // We get a keykode != 127 when delete key on numlock is pressed with numlock on.
666  if (event.KeyInput.Char == 127)
667  {
668  if ( !isEnabled() )
669  break;
670 
671  if (keyDelete())
672  {
673  BlinkStartTime = os::Timer::getTime();
674  newMarkBegin = 0;
675  newMarkEnd = 0;
676  textChanged = true;
677  }
678  break;
679  }
680  else
681  {
682  inputChar(event.KeyInput.Char);
683  return true;
684  }
685 
686  case KEY_ESCAPE:
687  case KEY_TAB:
688  case KEY_SHIFT:
689  case KEY_F1:
690  case KEY_F2:
691  case KEY_F3:
692  case KEY_F4:
693  case KEY_F5:
694  case KEY_F6:
695  case KEY_F7:
696  case KEY_F8:
697  case KEY_F9:
698  case KEY_F10:
699  case KEY_F11:
700  case KEY_F12:
701  case KEY_F13:
702  case KEY_F14:
703  case KEY_F15:
704  case KEY_F16:
705  case KEY_F17:
706  case KEY_F18:
707  case KEY_F19:
708  case KEY_F20:
709  case KEY_F21:
710  case KEY_F22:
711  case KEY_F23:
712  case KEY_F24:
713  // ignore these keys
714  return false;
715 
716  default:
717  inputChar(event.KeyInput.Char);
718  return true;
719  }
720  }
721 
722  // Set new text markers
723  setTextMarkers( newMarkBegin, newMarkEnd );
724 
725  // break the text if it has changed
726  if (textChanged)
727  {
728  breakText();
729  calculateScrollPos();
730  sendGuiEvent(EGET_EDITBOX_CHANGED);
731  }
732  else
733  {
734  calculateScrollPos();
735  }
736 
737  return true;
738 }

740 bool CGUIEditBox::keyDelete()
741 {
742  if (Text.size() != 0)
743  {
744  core::stringw s;
745 
746  if (MarkBegin != MarkEnd)
747  {
748  // delete marked text
749  const s32 realmbgn = MarkBegin < MarkEnd ? MarkBegin : MarkEnd;
750  const s32 realmend = MarkBegin < MarkEnd ? MarkEnd : MarkBegin;
751 
752  s = Text.subString(0, realmbgn);
753  s.append( Text.subString(realmend, Text.size()-realmend) );
754  Text = s;
755 
756  CursorPos = realmbgn;
757  }
758  else
759  {
760  // delete text before cursor
761  s = Text.subString(0, CursorPos);
762  s.append( Text.subString(CursorPos+1, Text.size()-CursorPos-1) );
763  Text = s;
764  }
765 
766  if (CursorPos > (s32)Text.size())
767  CursorPos = (s32)Text.size();
768 
769  return true;
770  }
771 
772  return false;

776 void CGUIEditBox::draw()
777 {
778  if (!IsVisible)
779  return;
780 
781  const bool focus = Environment->hasFocus(this);
782 
783  IGUISkin* skin = Environment->getSkin();
784  if (!skin)
785  return;
786 
787  EGUI_DEFAULT_COLOR bgCol = EGDC_GRAY_EDITABLE;
788  if ( isEnabled() )
789  bgCol = focus ? EGDC_FOCUSED_EDITABLE : EGDC_EDITABLE;
790 
791  if (!Border && Background)
792  {
793  skin->draw2DRectangle(this, skin->getColor(bgCol), AbsoluteRect, &AbsoluteClippingRect);
794  }
795 
796  if (Border)
797  {
798  // draw the border
799  skin->draw3DSunkenPane(this, skin->getColor(bgCol), false, Background, AbsoluteRect, &AbsoluteClippingRect);
800 
801  calculateFrameRect();
802  }
803 
804  core::rect<s32> localClipRect = FrameRect;
805  localClipRect.clipAgainst(AbsoluteClippingRect);
806 
807  // draw the text
808 
809  IGUIFont* font = getActiveFont();
810 
811  s32 cursorLine = 0;
812  s32 charcursorpos = 0;
813 
814  if (font)
815  {
816  if (LastBreakFont != font)
817  {
818  breakText();
819  }
820 
821  // calculate cursor pos
822 
823  core::stringw *txtLine = &Text;
824  s32 startPos = 0;
825 
826  core::stringw s, s2;
827 
828  // get mark position
829  const bool ml = (!PasswordBox && (WordWrap || MultiLine));
830  const s32 realmbgn = MarkBegin < MarkEnd ? MarkBegin : MarkEnd;
831  const s32 realmend = MarkBegin < MarkEnd ? MarkEnd : MarkBegin;
832  const s32 hlineStart = ml ? getLineFromPos(realmbgn) : 0;
833  const s32 hlineCount = ml ? getLineFromPos(realmend) - hlineStart + 1 : 1;
834  const s32 lineCount = ml ? BrokenText.size() : 1;
835 
836  // Save the override color information.
837  // Then, alter it if the edit box is disabled.
838  const bool prevOver = OverrideColorEnabled;
839  const video::SColor prevColor = OverrideColor;
840 
841  if (Text.size())
842  {
843  if (!isEnabled() && !OverrideColorEnabled)
844  {
845  OverrideColorEnabled = true;
846  OverrideColor = skin->getColor(EGDC_GRAY_TEXT);
847  }
848 
849  for (s32 i=0; i < lineCount; ++i)
850  {
851  setTextRect(i);
852 
853  // clipping test - don't draw anything outside the visible area
854  core::rect<s32> c = localClipRect;
855  c.clipAgainst(CurrentTextRect);
856  if (!c.isValid())
857  continue;
858 
859  // get current line
860  if (PasswordBox)
861  {
862  if (BrokenText.size() != 1)
863  {
864  BrokenText.clear();
865  BrokenText.push_back(core::stringw());
866  }
867  if (BrokenText[0].size() != Text.size())
868  {
869  BrokenText[0] = Text;
870  for (u32 q = 0; q < Text.size(); ++q)
871  {
872  BrokenText[0] [q] = PasswordChar;
873  }
874  }
