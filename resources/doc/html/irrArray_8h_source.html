312  s32 binary_search(const T& element, s32 left, s32 right) const
313  {
314  if (left > right)
315  return -1;
316  auto lpos = std::next(m_data.begin(), left);
317  auto rpos = std::next(m_data.begin(), right);
318  auto it = std::lower_bound(lpos, rpos, element);
319  // *it = first element in [first, last) that is >= element, or last if not found.
320  if (*it < element || element < *it)
321  return -1;
322  return it - m_data.begin();
323  }

334  s32 binary_search_multi(const T& element, s32 &last)
335  {
336  sort();
337  auto iters = std::equal_range(m_data.begin(), m_data.end(), element);
338  if (iters.first == iters.second)
339  return -1;
340  last = (iters.second - m_data.begin()) - 1;
341  return iters.first - m_data.begin();
342  }

367  {
368  auto it = std::find(m_data.rbegin(), m_data.rend(), element);
369  if (it == m_data.rend())

393  void erase(u32 index, s32 count)
394  {
395  if (index >= m_data.size() || count < 1)
396  return;
397  count = core::min_(count, (s32)m_data.size() - (s32)index);
398  auto first = std::next(m_data.begin(), index);
399  auto last = std::next(first, count);
