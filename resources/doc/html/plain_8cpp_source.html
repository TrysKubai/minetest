106 std::unique_ptr<RenderStep> create3DStage(Client *client, v2f scale)

126 RenderStep* addUpscaling(RenderPipeline *pipeline, RenderStep *previousStep, v2f downscale_factor)
127 {
128  const int TEXTURE_UPSCALE = 0;
129 
130  if (downscale_factor.X == 1.0f && downscale_factor.Y == 1.0f)
131  return previousStep;
132 
133  // Initialize buffer
134  TextureBuffer *buffer = pipeline->createOwned<TextureBuffer>();
135  buffer->setTexture(TEXTURE_UPSCALE, downscale_factor, "upscale", video::ECF_A8R8G8B8);
136 
137  // Attach previous step to the buffer
138  TextureBufferOutput *buffer_output = pipeline->createOwned<TextureBufferOutput>(buffer, TEXTURE_UPSCALE);
139  previousStep->setRenderTarget(buffer_output);
140 
141  // Add upscaling step
142  RenderStep *upscale = pipeline->createOwned<UpscaleStep>();
143  upscale->setRenderSource(buffer);
144  pipeline->addStep(upscale);
145 
146  return upscale;
147 }
