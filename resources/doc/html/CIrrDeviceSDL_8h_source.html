214  void updateCursorPos()
215  {
216 #ifdef _IRR_EMSCRIPTEN_PLATFORM_
217  EmscriptenPointerlockChangeEvent pointerlockStatus; // let's hope that test is not expensive ...
218  if ( emscripten_get_pointerlock_status(&pointerlockStatus) == EMSCRIPTEN_RESULT_SUCCESS )
219  {
220  if ( pointerlockStatus.isActive )
221  {
222  CursorPos.X += Device->MouseXRel;
223  CursorPos.Y += Device->MouseYRel;
224  Device->MouseXRel = 0;
225  Device->MouseYRel = 0;
226  }
227  else
228  {
229  CursorPos.X = Device->MouseX;
230  CursorPos.Y = Device->MouseY;
231  }
232  }
233 #else
234  CursorPos.X = Device->MouseX;
235  CursorPos.Y = Device->MouseY;
236 
237  if (CursorPos.X < 0)
238  CursorPos.X = 0;
239  if (CursorPos.X > (s32)Device->Width)
240  CursorPos.X = Device->Width;
241  if (CursorPos.Y < 0)
242  CursorPos.Y = 0;
243  if (CursorPos.Y > (s32)Device->Height)
244  CursorPos.Y = Device->Height;
245 #endif
246  }
