78 void TextureBuffer::reset(PipelineContext &context)
79 {
80  if (!m_driver)
81  m_driver = context.device->getVideoDriver();
82 
83  // remove extra textures
84  if (m_textures.size() > m_definitions.size()) {
85  for (unsigned i = m_definitions.size(); i < m_textures.size(); i++)
86  if (m_textures[i])
87  m_driver->removeTexture(m_textures[i]);
88 
89  m_textures.set_used(m_definitions.size());
90  }
91 
92  // add placeholders for new definitions
93  while (m_textures.size() < m_definitions.size())
94  m_textures.push_back(nullptr);
95 
96  // change textures to match definitions
97  for (u32 i = 0; i < m_definitions.size(); i++) {
98  video::ITexture **ptr = &m_textures[i];
99 
100  ensureTexture(ptr, m_definitions[i], context);
101  m_definitions[i].dirty = false;
102  }
103 
104  RenderSource::reset(context);
105 }

117 bool TextureBuffer::ensureTexture(video::ITexture **texture, const TextureDefinition& definition, PipelineContext &context)
118 {
119  bool modify;
120  core::dimension2du size;
121  if (definition.valid) {
122  if (definition.fixed_size)
123  size = definition.size;
124  else
125  size = core::dimension2du(
126  (u32)(context.target_size.X * definition.scale_factor.X),
127  (u32)(context.target_size.Y * definition.scale_factor.Y));
128 
129  modify = definition.dirty || (*texture == nullptr) || (*texture)->getSize() != size;
130  }
131  else {
132  modify = (*texture != nullptr);
133  }
134 
135  if (!modify)
136  return false;
137 
138  if (*texture)
139  m_driver->removeTexture(*texture);
140 
141  if (definition.valid) {
142  if (definition.clear) {
143  video::IImage *image = m_driver->createImage(definition.format, size);
144  // Cannot use image->fill because it's not implemented for all formats.
145  std::memset(image->getData(), 0, image->getDataSizeFromFormat(definition.format, size.Width, size.Height));
146  *texture = m_driver->addTexture(definition.name.c_str(), image);
147  image->drop();
148  }
149  else {
150  *texture = m_driver->addRenderTargetTexture(size, definition.name.c_str(), definition.format);
151  }
152  }
153  else {
154  *texture = nullptr;
155  }
156 
157  return true;
158 }

178 void TextureBufferOutput::activate(PipelineContext &context)
179 {
180  if (!driver)
181  driver = context.device->getVideoDriver();
182 
183  if (!render_target)
184  render_target = driver->addRenderTarget();
185 
186  core::array<video::ITexture *> textures;
187  core::dimension2du size(0, 0);
188  for (size_t i = 0; i < texture_map.size(); i++) {
189  video::ITexture *texture = buffer->getTexture(texture_map[i]);
190  textures.push_back(texture);
191  if (texture && size.Width == 0)
192  size = texture->getSize();
193  }
194 
195  // Use legacy call when there's single texture without depth texture
196  // This binds default depth buffer to the FBO
197  if (textures.size() == 1 && depth_stencil == NO_DEPTH_TEXTURE) {
198  driver->setRenderTarget(textures[0], m_clear, m_clear, context.clear_color);
199  return;
200  }
201 
202  video::ITexture *depth_texture = nullptr;
203  if (depth_stencil != NO_DEPTH_TEXTURE)
204  depth_texture = buffer->getTexture(depth_stencil);
205 
206  render_target->setTexture(textures, depth_texture);
207 
208  driver->setRenderTargetEx(render_target, m_clear ? video::ECBF_ALL : video::ECBF_NONE, context.clear_color);
209  driver->OnResize(size);
210 
211  RenderTarget::activate(context);
212 }

277 void RenderPipeline::run(PipelineContext &context)
278 {
279  v2u32 original_size = context.target_size;
280  context.target_size = v2u32(original_size.X * scale.X, original_size.Y * scale.Y);
281 
282  for (auto &object : m_objects)
283  object->reset(context);
284 
285  for (auto &step: m_pipeline)
286  step->run(context);
287 
288  context.target_size = original_size;
289 }
