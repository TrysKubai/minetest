115 bool GUIModalMenu::DoubleClickDetection(const SEvent &event)
116 {
117  /* The following code is for capturing double-clicks of the mouse button
118  * and translating the double-click into an EET_KEY_INPUT_EVENT event
119  * -- which closes the form -- under some circumstances.
120  *
121  * There have been many github issues reporting this as a bug even though it
122  * was an intended feature. For this reason, remapping the double-click as
123  * an ESC must be explicitly set when creating this class via the
124  * /p remap_dbl_click parameter of the constructor.
125  */
126 
127  if (!m_remap_dbl_click)
128  return false;
129 
130  if (event.MouseInput.Event == EMIE_LMOUSE_PRESSED_DOWN) {
131  m_doubleclickdetect[0].pos = m_doubleclickdetect[1].pos;
132  m_doubleclickdetect[0].time = m_doubleclickdetect[1].time;
133 
134  m_doubleclickdetect[1].pos = m_pointer;
135  m_doubleclickdetect[1].time = porting::getTimeMs();
136  } else if (event.MouseInput.Event == EMIE_LMOUSE_LEFT_UP) {
137  u64 delta = porting::getDeltaMs(
138  m_doubleclickdetect[0].time, porting::getTimeMs());
139  if (delta > 400)
140  return false;
141 
142  double squaredistance = m_doubleclickdetect[0].pos.
143  getDistanceFromSQ(m_doubleclickdetect[1].pos);
144 
145  if (squaredistance > (30 * 30)) {
146  return false;
147  }
148 
149  SEvent translated{};
150  // translate doubleclick to escape
151  translated.EventType = EET_KEY_INPUT_EVENT;
152  translated.KeyInput.Key = KEY_ESCAPE;
153  translated.KeyInput.Control = false;
154  translated.KeyInput.Shift = false;
155  translated.KeyInput.PressedDown = true;
156  translated.KeyInput.Char = 0;
157  OnEvent(translated);
158 
159  return true;
160  }
161 
162  return false;
163 }

179 bool GUIModalMenu::simulateMouseEvent(
180  gui::IGUIElement *target, ETOUCH_INPUT_EVENT touch_event)
181 {
182  SEvent mouse_event{}; // value-initialized, not unitialized
183  mouse_event.EventType = EET_MOUSE_INPUT_EVENT;
184  mouse_event.MouseInput.X = m_pointer.X;
185  mouse_event.MouseInput.Y = m_pointer.Y;
186  switch (touch_event) {
187  case ETIE_PRESSED_DOWN:
188  mouse_event.MouseInput.Event = EMIE_LMOUSE_PRESSED_DOWN;
189  mouse_event.MouseInput.ButtonStates = EMBSM_LEFT;
190  break;
191  case ETIE_MOVED:
192  mouse_event.MouseInput.Event = EMIE_MOUSE_MOVED;
193  mouse_event.MouseInput.ButtonStates = EMBSM_LEFT;
194  break;
195  case ETIE_LEFT_UP:
196  mouse_event.MouseInput.Event = EMIE_LMOUSE_LEFT_UP;
197  mouse_event.MouseInput.ButtonStates = 0;
198  break;
199  default:
200  return false;
201  }
202  if (preprocessEvent(mouse_event))
203  return true;
204  if (!target)
205  return false;
206  return target->OnEvent(mouse_event);
207 }

209 void GUIModalMenu::enter(gui::IGUIElement *hovered)
210 {
211  if (!hovered)

214  m_hovered.grab(hovered);
215  SEvent gui_event{};
216  gui_event.EventType = EET_GUI_EVENT;
217  gui_event.GUIEvent.Caller = m_hovered.get();
218  gui_event.GUIEvent.EventType = EGET_ELEMENT_HOVERED;
219  gui_event.GUIEvent.Element = gui_event.GUIEvent.Caller;
220  m_hovered->OnEvent(gui_event);
221 }

237 bool GUIModalMenu::preprocessEvent(const SEvent &event)
238 {
239 #ifdef __ANDROID__
240  // clang-format off
241  // display software keyboard when clicking edit boxes
242  if (event.EventType == EET_MOUSE_INPUT_EVENT &&
243  event.MouseInput.Event == EMIE_LMOUSE_PRESSED_DOWN) {
244  gui::IGUIElement *hovered =
245  Environment->getRootGUIElement()->getElementFromPoint(
246  core::position2d<s32>(event.MouseInput.X, event.MouseInput.Y));
247  if ((hovered) && (hovered->getType() == irr::gui::EGUIET_EDIT_BOX)) {
248  bool retval = hovered->OnEvent(event);
249  if (retval)
250  Environment->setFocus(hovered);
251 
252  std::string field_name = getNameByID(hovered->getID());
253  // read-only field
254  if (field_name.empty())
255  return retval;
256 
257  m_jni_field_name = field_name;
258 
259  // single line text input
260  int type = 2;
261 
262  // multi line text input
263  if (((gui::IGUIEditBox *)hovered)->isMultiLineEnabled())
264  type = 1;
265 
266  // passwords are always single line
267  if (((gui::IGUIEditBox *)hovered)->isPasswordBox())
268  type = 3;
269 
270  porting::showInputDialog(gettext("OK"), "",
271  wide_to_utf8(((gui::IGUIEditBox *)hovered)->getText()), type);
272  return retval;
273  }
274  }
275 #endif
276 
277 #ifdef HAVE_TOUCHSCREENGUI
278  if (event.EventType == EET_TOUCH_INPUT_EVENT) {
279  irr_ptr<GUIModalMenu> holder;
280  holder.grab(this); // keep this alive until return (it might be dropped downstream [?])
281 
282  switch ((int)event.TouchInput.touchedCount) {
283  case 1: {
284  if (event.TouchInput.Event == ETIE_PRESSED_DOWN || event.TouchInput.Event == ETIE_MOVED)
285  m_pointer = v2s32(event.TouchInput.X, event.TouchInput.Y);
286  if (event.TouchInput.Event == ETIE_PRESSED_DOWN)
287  m_down_pos = m_pointer;
288  gui::IGUIElement *hovered = Environment->getRootGUIElement()->getElementFromPoint(core::position2d<s32>(m_pointer));
289  if (event.TouchInput.Event == ETIE_PRESSED_DOWN)
290  Environment->setFocus(hovered);
291  if (m_hovered != hovered) {
292  leave();
293  enter(hovered);
294  }
295  gui::IGUIElement *focused = Environment->getFocus();
296  bool ret = simulateMouseEvent(focused, event.TouchInput.Event);
297  if (!ret && m_hovered != focused)
298  ret = simulateMouseEvent(m_hovered.get(), event.TouchInput.Event);
299  if (event.TouchInput.Event == ETIE_LEFT_UP)
300  leave();
301  return ret;
302  }
303  case 2: {
304  if (event.TouchInput.Event != ETIE_PRESSED_DOWN)
305  return true; // ignore
306  auto focused = Environment->getFocus();
307  if (!focused)
308  return true;
309  SEvent rclick_event{};
310  rclick_event.EventType = EET_MOUSE_INPUT_EVENT;
311  rclick_event.MouseInput.Event = EMIE_RMOUSE_PRESSED_DOWN;
312  rclick_event.MouseInput.ButtonStates = EMBSM_LEFT | EMBSM_RIGHT;
313  rclick_event.MouseInput.X = m_pointer.X;
314  rclick_event.MouseInput.Y = m_pointer.Y;
315  focused->OnEvent(rclick_event);
316  rclick_event.MouseInput.Event = EMIE_RMOUSE_LEFT_UP;
317  rclick_event.MouseInput.ButtonStates = EMBSM_LEFT;
318  focused->OnEvent(rclick_event);
319  return true;
320  }
321  default: // ignored
322  return true;
323  }
324  }
325 #endif
326 
327  if (event.EventType == EET_MOUSE_INPUT_EVENT) {
328  s32 x = event.MouseInput.X;
329  s32 y = event.MouseInput.Y;
330  gui::IGUIElement *hovered =
331  Environment->getRootGUIElement()->getElementFromPoint(
332  core::position2d<s32>(x, y));
333  if (!isChild(hovered, this)) {
334  if (DoubleClickDetection(event)) {
335  return true;
336  }
337  }
338  }
339  return false;
340 }
