43 T RangedParameter<T>::pickWithin() const
44 {
45  typename T::pickFactors values;
46  auto p = numericAbsolute(bias) + 1;

57 template<typename T>
58 T TweenedParameter<T>::blend(float fac) const
59 {
60  // warp time coordinates in accordance w/ settings
61  if (fac > beginning) {
62  // remap for beginning offset
63  auto len = 1 - beginning;
64  fac -= beginning;
65  fac /= len;
66 
67  // remap for repetitions
68  fac *= reps;
69  if (fac > 1) // poor man's modulo
70  fac -= (decltype(reps))fac;
71 
72  // remap for style
73  switch (style) {
74  case TweenStyle::fwd: /* do nothing */ break;
75  case TweenStyle::rev: fac = 1.0f - fac; break;
76  case TweenStyle::pulse:
77  case TweenStyle::flicker: {
78  if (fac > 0.5f) {
79  fac = 1.f - (fac*2.f - 1.f);
80  } else {
81  fac = fac * 2;
82  }
83  if (style == TweenStyle::flicker) {
84  fac *= myrand_range(0.7f, 1.0f);
85  }
86  }
87  }
88  if (fac>1.f)
89  fac = 1.f;
90  else if (fac<0.f)
91  fac = 0.f;
92  } else {
93  fac = (style == TweenStyle::rev) ? 1.f : 0.f;
94  }
95 
96  return start.interpolate(fac, end);
97 }

177 enum class ParticleTextureFlags : u8 {
178  /* each value specifies a bit in a bitmask; if the maximum value
179  * goes above 1<<7 the type of the flags field must be changed
180  * from u8, which will necessitate a protocol change! */
181 
182  // the first bit indicates whether the texture is animated
183  animated = 1,
184 
185  /* the next three bits indicate the blending mode of the texture
186  * blendmode is encoded by (flags |= (u8)blend << 1); retrieve with
187  * (flags & ParticleTextureFlags::blend) >> 1. note that the third
188  * bit is currently reserved for adding more blend modes in the future */
189  blend = 0x7 << 1,
190 };

196 void ServerParticleTexture::serialize(std::ostream &os, u16 protocol_ver, bool newPropertiesOnly) const
197 {
198  /* newPropertiesOnly is used to de/serialize parameters of the legacy texture
199  * field, which are encoded separately from the texspec string */
200  FlagT flags = 0;
201  if (animated)
202  flags |= FlagT(ParticleTextureFlags::animated);
203  if (blendmode != BlendMode::alpha)
204  flags |= FlagT(blendmode) << 1;
205  serializeParameterValue(os, flags);
206 
207  alpha.serialize(os);
208  scale.serialize(os);
209  if (!newPropertiesOnly)
210  os << serializeString32(string);
211 
212  if (animated)
213  animation.serialize(os, protocol_ver);
214 }
