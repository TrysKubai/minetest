82 void COSOperator::copyToClipboard(const c8 *text) const
83 {
84  if (strlen(text)==0)
85  return;
86 
87 #if defined(_IRR_COMPILE_WITH_SDL_DEVICE_)
88  SDL_SetClipboardText(text);
89 
90 #elif defined(_IRR_WINDOWS_API_)
91  if (!OpenClipboard(NULL) || text == 0)
92  return;
93 
94  EmptyClipboard();
95 
96  core::stringw tempbuffer;
97  core::multibyteToWString(tempbuffer, text);
98  const u32 size = (tempbuffer.size() + 1) * sizeof(wchar_t);
99 
100  HGLOBAL clipbuffer;
101  void * buffer;
102 
103  clipbuffer = GlobalAlloc(GMEM_MOVEABLE, size);
104  buffer = GlobalLock(clipbuffer);
105 
106  memcpy(buffer, tempbuffer.c_str(), size);
107 
108  GlobalUnlock(clipbuffer);
109  SetClipboardData(CF_UNICODETEXT, clipbuffer);
110  CloseClipboard();
111 
112 #elif defined(_IRR_COMPILE_WITH_OSX_DEVICE_)
113  NSString *str = nil;
114  NSPasteboard *board = nil;
115 
116  if ((text != NULL) && (strlen(text) > 0))
117  {
118  str = [NSString stringWithCString:text encoding:NSUTF8StringEncoding];
119  board = [NSPasteboard generalPasteboard];
120  [board declareTypes:[NSArray arrayWithObject:NSStringPboardType] owner:NSApp];
121  [board setString:str forType:NSStringPboardType];
122  }
123 
124 #elif defined(_IRR_COMPILE_WITH_X11_DEVICE_)
125  if ( IrrDeviceLinux )
126  IrrDeviceLinux->copyToClipboard(text);
127 #endif
128 }

224 bool COSOperator::getSystemMemory(u32* Total, u32* Avail) const
225 {
226 #if defined(_IRR_WINDOWS_API_)
227 
228  MEMORYSTATUSEX MemoryStatusEx;
229  MemoryStatusEx.dwLength = sizeof(MEMORYSTATUSEX);
230 
231  // cannot fail
232  GlobalMemoryStatusEx(&MemoryStatusEx);
233 
234  if (Total)
235  *Total = (u32)(MemoryStatusEx.ullTotalPhys>>10);
236  if (Avail)
237  *Avail = (u32)(MemoryStatusEx.ullAvailPhys>>10);
238  return true;
239 
240 #elif defined(_IRR_POSIX_API_) && defined(_SC_PHYS_PAGES) && defined(_SC_AVPHYS_PAGES)
241  long ps = sysconf(_SC_PAGESIZE);
242  long pp = sysconf(_SC_PHYS_PAGES);
243  long ap = sysconf(_SC_AVPHYS_PAGES);
244 
245  if (ps == -1 || (Total && pp == -1) || (Avail && ap == -1))
246  return false;
247 
248  if (Total)
249  *Total = (u32)((pp>>10)*ps);
250  if (Avail)
251  *Avail = (u32)((ap>>10)*ps);
252  return true;
253 #elif defined(_IRR_OSX_PLATFORM_)
254  int mib[2];
255  int64_t physical_memory;
256  size_t length;
257 
258  // Get the Physical memory size
259  mib[0] = CTL_HW;
260  mib[1] = HW_MEMSIZE;
261  length = sizeof(int64_t);
262  sysctl(mib, 2, &physical_memory, &length, NULL, 0);
263 
264  if (Total)
265  *Total = (u32)(physical_memory>>10);
266  if (Avail)
267  *Avail = (u32)(physical_memory>>10); // we don't know better
268  return true;
269 #else
270  // TODO: implement for others
271  return false;
272 #endif
273 }
