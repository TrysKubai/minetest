103 void ModApiBase::registerClass(lua_State *L, const char *name,
104  const luaL_Reg *methods,
105  const luaL_Reg *metamethods)
106 {
107  luaL_newmetatable(L, name);
108  luaL_register(L, NULL, metamethods);
109  int metatable = lua_gettop(L);
110 
111  lua_newtable(L);
112  luaL_register(L, NULL, methods);
113  int methodtable = lua_gettop(L);
114 
115  lua_pushvalue(L, methodtable);
116  lua_setfield(L, metatable, "__index");
117 
118  // Protect the real metatable.
119  lua_pushvalue(L, methodtable);
120  lua_setfield(L, metatable, "__metatable");
121 
122  // Pop methodtable and metatable.
123  lua_pop(L, 2);
124 }

126 int ModApiBase::l_deprecated_function(lua_State *L, const char *good, const char *bad, lua_CFunction func)
127 {
128  thread_local std::vector<u64> deprecated_logged;
129 
130  DeprecatedHandlingMode dep_mode = get_deprecated_handling_mode();
131  if (dep_mode == DeprecatedHandlingMode::Ignore)
132  return func(L);
133 
134  u64 start_time = porting::getTimeUs();
135  lua_Debug ar;
136 
137  // Get caller name with line and script backtrace
138  FATAL_ERROR_IF(!lua_getstack(L, 1, &ar), "lua_getstack() failed");
139  FATAL_ERROR_IF(!lua_getinfo(L, "Sl", &ar), "lua_getinfo() failed");
140 
141  // Get backtrace and hash it to reduce the warning flood
142  std::string backtrace = ar.short_src;
143  backtrace.append(":").append(std::to_string(ar.currentline));
144  u64 hash = murmur_hash_64_ua(backtrace.data(), backtrace.length(), 0xBADBABE);
145 
146  if (std::find(deprecated_logged.begin(), deprecated_logged.end(), hash)
147  == deprecated_logged.end()) {
148 
149  deprecated_logged.emplace_back(hash);
150  warningstream << "Call to deprecated function '" << bad << "', please use '"
151  << good << "' at " << backtrace << std::endl;
152 
153  if (dep_mode == DeprecatedHandlingMode::Error)
154  script_error(L, LUA_ERRRUN, NULL, NULL);
155  }
156 
157  u64 end_time = porting::getTimeUs();
158  g_profiler->avg("l_deprecated_function", end_time - start_time);
159 
160  return func(L);
161 }
