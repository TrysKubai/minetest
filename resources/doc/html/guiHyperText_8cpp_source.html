176 void ParsedText::parse(const wchar_t *text)
177 {
178  wchar_t c;
179  u32 cursor = 0;
180  bool escape = false;
181 
182  while ((c = text[cursor]) != L'\0') {
183  cursor++;
184 
185  if (c == L'\r') { // Mac or Windows breaks
186  if (text[cursor] == L'\n')
187  cursor++;
188  // If text has begun, don't skip empty line
189  if (m_paragraph) {
190  endParagraph(ER_NEWLINE);
191  enterElement(ELEMENT_SEPARATOR);
192  }
193  escape = false;
194  continue;
195  }
196 
197  if (c == L'\n') { // Unix breaks
198  // If text has begun, don't skip empty line
199  if (m_paragraph) {
200  endParagraph(ER_NEWLINE);
201  enterElement(ELEMENT_SEPARATOR);
202  }
203  escape = false;
204  continue;
205  }
206 
207  if (escape) {
208  escape = false;
209  pushChar(c);
210  continue;
211  }
212 
213  if (c == L'\\') {
214  escape = true;
215  continue;
216  }
217 
218  // Tag check
219  if (c == L'<') {
220  u32 newcursor = parseTag(text, cursor);
221  if (newcursor > 0) {
222  cursor = newcursor;
223  continue;
224  }
225  }
226 
227  // Default behavior
228  pushChar(c);
229  }
230 
231  endParagraph(ER_NONE);
232 }

239 void ParsedText::endParagraph(EndReason reason)
240 {
241  if (!m_paragraph)
242  return;
243 
244  EndReason previous = m_end_paragraph_reason;
245  m_end_paragraph_reason = reason;
246  if (m_empty_paragraph && (reason == ER_TAG ||
247  (reason == ER_NEWLINE && previous == ER_TAG))) {
248  // Ignore last empty paragraph
249  m_paragraph = nullptr;
250  m_paragraphs.pop_back();
251  return;
252  }
253  endElement();
254  m_paragraph = NULL;
255 }

325  const std::string &name, const std::string &value, StyleList &style)
326 {

328  if (name == "color" || name == "hovercolor") {

352 void ParsedText::globalTag(const AttrsList &attrs)
353 {
354  for (const auto &attr : attrs) {
355  // Only page level style
356  if (attr.first == "margin") {
357  if (check_integer(attr.second))
358  margin = stoi(attr.second.c_str());
359 
360  } else if (attr.first == "valign") {
361  if (attr.second == "top")
362  valign = ParsedText::VALIGN_TOP;
363  else if (attr.second == "bottom")
364  valign = ParsedText::VALIGN_BOTTOM;
365  else if (attr.second == "middle")
366  valign = ParsedText::VALIGN_MIDDLE;
367  } else if (attr.first == "background") {
368  irr::video::SColor color;
369  if (attr.second == "none") {
370  background_type = BACKGROUND_NONE;
371  } else if (parseColorString(attr.second, color, false)) {
372  background_type = BACKGROUND_COLOR;
373  background_color = color;
374  }
375 
376  // Inheriting styles
377 
378  } else if (attr.first == "halign") {
379  if (attr.second == "left" || attr.second == "center" ||
380  attr.second == "right" ||
381  attr.second == "justify")
382  m_root_tag.style["halign"] = attr.second;
383 
384  // Generic default styles
385 
386  } else {
387  parseGenericStyleAttr(attr.first, attr.second, m_root_tag.style);
388  }
389  }

392 u32 ParsedText::parseTag(const wchar_t *text, u32 cursor)
393 {
394  // Tag name
395  bool end = false;
396  std::string name = "";
397  wchar_t c = text[cursor];
398 
399  if (c == L'/') {
400  end = true;
401  c = text[++cursor];
402  if (c == L'\0')
403  return 0;
404  }
405 
406  while (c != ' ' && c != '>') {
407  name += c;
408  c = text[++cursor];
409  if (c == L'\0')
410  return 0;
411  }
412 
413  // Tag attributes
414  AttrsList attrs;
415  while (c != L'>') {
416  std::string attr_name = "";
417  core::stringw attr_val = L"";
418 
419  while (c == ' ') {
420  c = text[++cursor];
421  if (c == L'\0' || c == L'=')
422  return 0;
423  }
424 
425  while (c != L' ' && c != L'=') {
426  attr_name += (char)c;
427  c = text[++cursor];
428  if (c == L'\0' || c == L'>')
429  return 0;
430  }
431 
432  while (c == L' ') {
433  c = text[++cursor];
434  if (c == L'\0' || c == L'>')
435  return 0;
436  }
437 
438  if (c != L'=')
439  return 0;
440 
441  c = text[++cursor];
442 
443  if (c == L'\0')
444  return 0;
445 
446  while (c != L'>' && c != L' ') {
447  attr_val += c;
448  c = text[++cursor];
449  if (c == L'\0')
450  return 0;
451  }
452 
453  attrs[attr_name] = stringw_to_utf8(attr_val);
454  }
455 
456  ++cursor; // Last ">"
457 
458  // Tag specific processing
459  StyleList style;
460 
461  if (name == "global") {
462  if (end)
463  return 0;
464  globalTag(attrs);
465 
466  } else if (name == "style") {
467  if (end) {
468  closeTag(name);
469  } else {
470  parseStyles(attrs, style);
471  openTag(name, attrs)->style = style;
472  }
473  endElement();
474  } else if (name == "img" || name == "item") {
475  if (end)
476  return 0;
477 
478  // Name is a required attribute
479  if (!attrs.count("name"))
480  return 0;
481 
482  // Rotate attribute is only for <item>
483  if (attrs.count("rotate") && name != "item")
484  return 0;
485 
486  // Angle attribute is only for <item>
487  if (attrs.count("angle") && name != "item")
488  return 0;
489 
490  // Ok, element can be created
491  newTag(name, attrs);
492 
493  if (name == "img")
494  enterElement(ELEMENT_IMAGE);
495  else
496  enterElement(ELEMENT_ITEM);
497 
498  m_element->text = utf8_to_stringw(attrs["name"]);
499 
500  if (attrs.count("float")) {
501  if (attrs["float"] == "left")
502  m_element->floating = FLOAT_LEFT;
503  if (attrs["float"] == "right")
504  m_element->floating = FLOAT_RIGHT;
505  }
506 
507  if (attrs.count("width")) {
508  int width = stoi(attrs["width"]);
509  if (width > 0)
510  m_element->dim.Width = width;
511  }
512 
513  if (attrs.count("height")) {
514  int height = stoi(attrs["height"]);
515  if (height > 0)
516  m_element->dim.Height = height;
517  }
518 
519  if (attrs.count("angle")) {
520  std::string str = attrs["angle"];
521  std::vector<std::string> parts = split(str, ',');
522  if (parts.size() == 3) {
523  m_element->angle = v3s16(
524  rangelim(stoi(parts[0]), -180, 180),
525  rangelim(stoi(parts[1]), -180, 180),
526  rangelim(stoi(parts[2]), -180, 180));
527  m_element->rotation = v3s16(0, 0, 0);
528  }
529  }
530 
531  if (attrs.count("rotate")) {
532  if (attrs["rotate"] == "yes") {
533  m_element->rotation = v3s16(0, 100, 0);
534  } else {
535  std::string str = attrs["rotate"];
536  std::vector<std::string> parts = split(str, ',');
537  if (parts.size() == 3) {
538  m_element->rotation = v3s16 (
539  rangelim(stoi(parts[0]), -1000, 1000),
540  rangelim(stoi(parts[1]), -1000, 1000),
541  rangelim(stoi(parts[2]), -1000, 1000));
542  }
543  }
544  }
545 
546  endElement();
547 
548  } else if (name == "tag") {
549  // Required attributes
550  if (!attrs.count("name"))
551  return 0;
552 
553  StyleList tagstyle;
554  parseStyles(attrs, tagstyle);
555 
556  if (is_yes(attrs["paragraph"]))
557  m_paragraphtags[attrs["name"]] = tagstyle;
558  else
559  m_elementtags[attrs["name"]] = tagstyle;
560 
561  } else if (name == "action") {
562  if (end) {
563  closeTag(name);
564  } else {
565  if (!attrs.count("name"))
566  return 0;
567  openTag(name, attrs)->style = m_elementtags["action"];
568  }
569 
570  } else if (m_elementtags.count(name)) {
571  if (end) {
572  closeTag(name);
573  } else {
574  openTag(name, attrs)->style = m_elementtags[name];
575  }
576  endElement();
577 
578  } else if (m_paragraphtags.count(name)) {
579  if (end) {
580  closeTag(name);
581  } else {
582  openTag(name, attrs)->style = m_paragraphtags[name];
583  }
584  endParagraph(ER_TAG);
585 
586  } else
587  return 0; // Unknown tag
588 
589  // Update styles accordingly
590  m_style.clear();
591  for (auto tag = m_active_tags.crbegin(); tag != m_active_tags.crend(); ++tag)
592  for (const auto &prop : (*tag)->style)
593  m_style[prop.first] = prop.second;
594 
595  return cursor;
596 }

601 TextDrawer::TextDrawer(const wchar_t *text, Client *client,
602  gui::IGUIEnvironment *environment, ISimpleTextureSource *tsrc) :
603  m_text(text),
604  m_client(client), m_environment(environment)
605 {
606  // Size all elements
607  for (auto &p : m_text.m_paragraphs) {
608  for (auto &e : p.elements) {
609  switch (e.type) {
610  case ParsedText::ELEMENT_SEPARATOR:
611  case ParsedText::ELEMENT_TEXT:
612  if (e.font) {
613  e.dim.Width = e.font->getDimension(e.text.c_str()).Width;
614  e.dim.Height = e.font->getDimension(L"Yy").Height;
615  if (e.font->getType() == irr::gui::EGFT_CUSTOM) {
616  CGUITTFont *tmp = static_cast<CGUITTFont*>(e.font);
617  e.baseline = e.dim.Height - 1 - tmp->getAscender() / 64;
618  }
619  } else {
620  e.dim = {0, 0};
621  }
622  break;
623 
624  case ParsedText::ELEMENT_IMAGE:
625  case ParsedText::ELEMENT_ITEM:
626  // Resize only non sized items
627  if (e.dim.Height != 0 && e.dim.Width != 0)
628  break;
629 
630  // Default image and item size
631  core::dimension2d<u32> dim(80, 80);
632 
633  if (e.type == ParsedText::ELEMENT_IMAGE) {
634  video::ITexture *texture =
635  m_client->getTextureSource()->
636  getTexture(stringw_to_utf8(e.text));
637  if (texture)
638  dim = texture->getOriginalSize();
639  }
640 
641  if (e.dim.Height == 0)
642  if (e.dim.Width == 0)
643  e.dim = dim;
644  else
645  e.dim.Height = dim.Height * e.dim.Width /
646  dim.Width;
647  else
648  e.dim.Width = dim.Width * e.dim.Height /
649  dim.Height;
650  break;
651  }
652  }
653  }

671 /*
672  This function places all elements according to given width. Elements have
673  been previously sized by constructor and will be later drawed by draw.
674  It may be called each time width changes and resulting height can be
675  retrieved using getHeight. See GUIHyperText constructor, it uses it once to
676  test if text fits in window and eventually another time if width is reduced
677  m_floating because of scrollbar added.
678 */
679 void TextDrawer::place(const core::rect<s32> &dest_rect)
680 {
681  m_floating.clear();
682  s32 y = 0;
683  s32 ymargin = m_text.margin;
684 
685  // Iterator used :
686  // p - Current paragraph, walked only once
687  // el - Current element, walked only once
688  // e and f - local element and floating operators
689 
690  for (auto &p : m_text.m_paragraphs) {
691  // Find and place floating stuff in paragraph
692  for (auto e = p.elements.begin(); e != p.elements.end(); ++e) {
693  if (e->floating != ParsedText::FLOAT_NONE) {
694  if (y)
695  e->pos.Y = y + std::max(ymargin, e->margin);
696  else
697  e->pos.Y = ymargin;
698 
699  if (e->floating == ParsedText::FLOAT_LEFT)
700  e->pos.X = m_text.margin;
701  if (e->floating == ParsedText::FLOAT_RIGHT)
702  e->pos.X = dest_rect.getWidth() - e->dim.Width -
703  m_text.margin;
704 
705  RectWithMargin floating;
706  floating.rect = core::rect<s32>(e->pos, e->dim);
707  floating.margin = e->margin;
708 
709  m_floating.push_back(floating);
710  }
711  }
712 
713  if (y)
714  y = y + std::max(ymargin, p.margin);
715 
716  ymargin = p.margin;
717 
718  // Place non floating stuff
719  std::vector<ParsedText::Element>::iterator el = p.elements.begin();
720 
721  while (el != p.elements.end()) {
722  // Determine line width and y pos
723  s32 left, right;
724  s32 nexty = y;
725  do {
726  y = nexty;
727  nexty = 0;
728 
729  // Inner left & right
730  left = m_text.margin;
731  right = dest_rect.getWidth() - m_text.margin;
732 
733  for (const auto &f : m_floating) {
734  // Does floating rect intersect paragraph y line?
735  if (f.rect.UpperLeftCorner.Y - f.margin <= y &&
736  f.rect.LowerRightCorner.Y + f.margin >= y) {
737 
738  // Next Y to try if no room left
739  if (!nexty || f.rect.LowerRightCorner.Y +
740  std::max(f.margin, p.margin) < nexty) {
741  nexty = f.rect.LowerRightCorner.Y +
742  std::max(f.margin, p.margin) + 1;
743  }
744 
745  if (f.rect.UpperLeftCorner.X - f.margin <= left &&
746  f.rect.LowerRightCorner.X + f.margin < right) {
747  // float on left
748  if (f.rect.LowerRightCorner.X +
749  std::max(f.margin, p.margin) > left) {
750  left = f.rect.LowerRightCorner.X +
751  std::max(f.margin, p.margin);
752  }
753  } else if (f.rect.LowerRightCorner.X + f.margin >= right &&
754  f.rect.UpperLeftCorner.X - f.margin > left) {
755  // float on right
756  if (f.rect.UpperLeftCorner.X -
757  std::max(f.margin, p.margin) < right)
758  right = f.rect.UpperLeftCorner.X -
759  std::max(f.margin, p.margin);
760 
761  } else if (f.rect.UpperLeftCorner.X - f.margin <= left &&
762  f.rect.LowerRightCorner.X + f.margin >= right) {
763  // float taking all space
764  left = right;
765  }
766  else
767  { // float in the middle -- should not occure yet, see that later
768  }
769  }
770  }
771  } while (nexty && right <= left);
772 
773  u32 linewidth = right - left;
774  float x = left;
775 
776  u32 charsheight = 0;
777  u32 charswidth = 0;
778  u32 wordcount = 0;
779 
780  // Skip beginning of line separators but include them in height
781  // computation.
782  while (el != p.elements.end() &&
783  el->type == ParsedText::ELEMENT_SEPARATOR) {
784  if (el->floating == ParsedText::FLOAT_NONE) {
785  el->drawwidth = 0;
786  if (charsheight < el->dim.Height)
787  charsheight = el->dim.Height;
788  }
789  el++;
790  }
791 
792  std::vector<ParsedText::Element>::iterator linestart = el;
793  std::vector<ParsedText::Element>::iterator lineend = p.elements.end();
794 
795  // First pass, find elements fitting into line
796  // (or at least one element)
797  while (el != p.elements.end() && (charswidth == 0 ||
798  charswidth + el->dim.Width <= linewidth)) {
799  if (el->floating == ParsedText::FLOAT_NONE) {
800  if (el->type != ParsedText::ELEMENT_SEPARATOR) {
801  lineend = el;
802  wordcount++;
803  }
804  charswidth += el->dim.Width;
805  if (charsheight < el->dim.Height)
806  charsheight = el->dim.Height;
807  }
808  el++;
809  }
810 
811  // Empty line, nothing to place only go down line height
812  if (lineend == p.elements.end()) {
813  y += charsheight;
814  continue;
815  }
816 
817  // Point to the first position outside line (may be end())
818  lineend++;
819 
820  // Second pass, compute printable line width and adjustments
821  charswidth = 0;
822  s32 top = 0;
823  s32 bottom = 0;
824  for (auto e = linestart; e != lineend; ++e) {
825  if (e->floating == ParsedText::FLOAT_NONE) {
826  charswidth += e->dim.Width;
827  if (top < (s32)e->dim.Height - e->baseline)
828  top = e->dim.Height - e->baseline;
829  if (bottom < e->baseline)
830  bottom = e->baseline;
831  }
832  }
833 
834  float extraspace = 0.f;
835 
836  switch (p.halign) {
837  case ParsedText::HALIGN_CENTER:
838  x += (linewidth - charswidth) / 2.f;
839  break;
840  case ParsedText::HALIGN_JUSTIFY:
841  if (wordcount > 1 && // Justification only if at least two words
842  !(lineend == p.elements.end())) // Don't justify last line
843  extraspace = ((float)(linewidth - charswidth)) / (wordcount - 1);
844  break;
845  case ParsedText::HALIGN_RIGHT:
846  x += linewidth - charswidth;
847  break;
848  case ParsedText::HALIGN_LEFT:
849  break;
850  }
851 
852  // Third pass, actually place everything
853  for (auto e = linestart; e != lineend; ++e) {
854  if (e->floating != ParsedText::FLOAT_NONE)
855  continue;
856 
857  e->pos.X = x;
858  e->pos.Y = y;
859 
860  switch (e->type) {
861  case ParsedText::ELEMENT_TEXT:
862  case ParsedText::ELEMENT_SEPARATOR:
863  e->pos.X = x;
864 
865  // Align char baselines
866  e->pos.Y = y + top + e->baseline - e->dim.Height;
867 
868  x += e->dim.Width;
869  if (e->type == ParsedText::ELEMENT_SEPARATOR)
870  x += extraspace;
871  break;
872 
873  case ParsedText::ELEMENT_IMAGE:
874  case ParsedText::ELEMENT_ITEM:
875  x += e->dim.Width;
876  break;
877  }
878 
879  // Draw width for separator can be different than element
880  // width. This will be important for char effects like
881  // underline.
882  e->drawwidth = x - e->pos.X;
883  }
884  y += charsheight;
885  } // Elements (actually lines)
886  } // Paragraph
887 
888  // Check if float goes under paragraph
889  for (const auto &f : m_floating) {
890  if (f.rect.LowerRightCorner.Y >= y)
891  y = f.rect.LowerRightCorner.Y;
892  }
893 
894  m_height = y + m_text.margin;
895  // Compute vertical offset according to vertical alignment
896  if (m_height < dest_rect.getHeight())
897  switch (m_text.valign) {
898  case ParsedText::VALIGN_BOTTOM:
899  m_voffset = dest_rect.getHeight() - m_height;
900  break;
901  case ParsedText::VALIGN_MIDDLE:
902  m_voffset = (dest_rect.getHeight() - m_height) / 2;
903  break;
904  case ParsedText::VALIGN_TOP:
905  default:
906  m_voffset = 0;
907  }
908  else
909  m_voffset = 0;
910 }

914 void TextDrawer::draw(const core::rect<s32> &clip_rect,
