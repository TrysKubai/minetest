50 
51  if (Driver)
52  Driver->grab();
53 
54  if (CursorControl)
55  CursorControl->grab();
56 
57  // create mesh cache if not there already
58  if (!MeshCache)
59  MeshCache = new CMeshCache();
60  else
61  MeshCache->grab();
62 
63  // set scene parameters
64  Parameters = new io::CAttributes();
65 
66  // create collision manager
67  CollisionManager = new CSceneCollisionManager(this, Driver);
68 
69  // add file format loaders. add the least commonly used ones first,
70  // as these are checked last
71 
72  // TODO: now that we have multiple scene managers, these should be
73  // shallow copies from the previous manager if there is one.
74 
75  MeshLoaderList.push_back(new CXMeshFileLoader(this));
76  MeshLoaderList.push_back(new COBJMeshFileLoader(this));
77  MeshLoaderList.push_back(new CB3DMeshFileLoader(this));

138 // load and create a mesh which we know already isn't in the cache and put it in there
139 IAnimatedMesh* CSceneManager::getUncachedMesh(io::IReadFile* file, const io::path& filename, const io::path& cachename)
140 {
141  IAnimatedMesh* msh = 0;
142 
143  // iterate the list in reverse order so user-added loaders can override the built-in ones
144  s32 count = MeshLoaderList.size();
145  for (s32 i=count-1; i>=0; --i)
146  {
147  if (MeshLoaderList[i]->isALoadableFileExtension(filename))
148  {
149  // reset file to avoid side effects of previous calls to createMesh
150  file->seek(0);
151  msh = MeshLoaderList[i]->createMesh(file);
152  if (msh)
153  {
154  MeshCache->addMesh(cachename, msh);
155  msh->drop();
156  break;
157  }
158  }
159  }
160 
161  if (!msh)
162  os::Printer::log("Could not load mesh, file format seems to be unsupported", filename, ELL_ERROR);
163  else
164  os::Printer::log("Loaded mesh", filename, ELL_DEBUG);
165 
166  return msh;
167 }

334 bool CSceneManager::isCulled(const ISceneNode* node) const
335 {
336  const ICameraSceneNode* cam = getActiveCamera();
337  if (!cam)
338  {
339  return false;
340  }
341  bool result = false;
342 
343  // has occlusion query information
344  if (node->getAutomaticCulling() & scene::EAC_OCC_QUERY)
345  {
346  result = (Driver->getOcclusionQueryResult(const_cast<ISceneNode*>(node))==0);
347  }
348 
349  // can be seen by a bounding box ?
350  if (!result && (node->getAutomaticCulling() & scene::EAC_BOX))
351  {
352  core::aabbox3d<f32> tbox = node->getBoundingBox();
353  node->getAbsoluteTransformation().transformBoxEx(tbox);
354  result = !(tbox.intersectsWithBox(cam->getViewFrustum()->getBoundingBox() ));
355  }
356 
357  // can be seen by a bounding sphere
358  if (!result && (node->getAutomaticCulling() & scene::EAC_FRUSTUM_SPHERE))
359  {
360  const core::aabbox3df nbox = node->getTransformedBoundingBox();
361  const float rad = nbox.getRadius();
362  const core::vector3df center = nbox.getCenter();
363 
364  const float camrad = cam->getViewFrustum()->getBoundingRadius();
365  const core::vector3df camcenter = cam->getViewFrustum()->getBoundingCenter();
366 
367  const float dist = (center - camcenter).getLengthSQ();
368  const float maxdist = (rad + camrad) * (rad + camrad);
369 
370  result = dist > maxdist;
371  }
372 
373  // can be seen by cam pyramid planes ?
374  if (!result && (node->getAutomaticCulling() & scene::EAC_FRUSTUM_BOX))
375  {
376  SViewFrustum frust = *cam->getViewFrustum();
377 
378  //transform the frustum to the node's current absolute transformation
379  core::matrix4 invTrans(node->getAbsoluteTransformation(), core::matrix4::EM4CONST_INVERSE);
380  //invTrans.makeInverse();
381  frust.transform(invTrans);
382 
383  core::vector3df edges[8];
384  node->getBoundingBox().getEdges(edges);
385 
386  for (s32 i=0; i<scene::SViewFrustum::VF_PLANE_COUNT; ++i)
387  {
388  bool boxInFrustum=false;
389  for (u32 j=0; j<8; ++j)
390  {
391  if (frust.planes[i].classifyPointRelation(edges[j]) != core::ISREL3D_FRONT)
392  {
393  boxInFrustum=true;
394  break;
395  }
396  }
397 
398  if (!boxInFrustum)
399  {
400  result = true;
401  break;
402  }
403  }
404  }
405 
406  return result;
407 }

411 u32 CSceneManager::registerNodeForRendering(ISceneNode* node, E_SCENE_NODE_RENDER_PASS pass)
412 {
413  u32 taken = 0;
414 
415  switch(pass)
416  {
417  // take camera if it is not already registered
418  case ESNRP_CAMERA:
419  {
420  taken = 1;
421  for (u32 i = 0; i != CameraList.size(); ++i)
422  {
423  if (CameraList[i] == node)
424  {
425  taken = 0;
426  break;
427  }
428  }
429  if (taken)
430  {
431  CameraList.push_back(node);
432  }
433  }
434  break;
435  case ESNRP_SKY_BOX:
436  SkyBoxList.push_back(node);
437  taken = 1;
438  break;
439  case ESNRP_SOLID:
440  if (!isCulled(node))
441  {
442  SolidNodeList.push_back(node);
443  taken = 1;
444  }
445  break;
446  case ESNRP_TRANSPARENT:
447  if (!isCulled(node))
448  {
449  TransparentNodeList.push_back(TransparentNodeEntry(node, camWorldPos));
450  taken = 1;
451  }
452  break;
453  case ESNRP_TRANSPARENT_EFFECT:
454  if (!isCulled(node))
455  {
456  TransparentEffectNodeList.push_back(TransparentNodeEntry(node, camWorldPos));
457  taken = 1;
458  }
459  break;
460  case ESNRP_AUTOMATIC:
461  if (!isCulled(node))
462  {
463  const u32 count = node->getMaterialCount();
464 
465  taken = 0;
466  for (u32 i=0; i<count; ++i)
467  {
468  if (Driver->needsTransparentRenderPass(node->getMaterial(i)))
469  {
470  // register as transparent node
471  TransparentNodeEntry e(node, camWorldPos);
472  TransparentNodeList.push_back(e);
473  taken = 1;
474  break;
475  }
476  }
477 
478  // not transparent, register as solid
479  if (!taken)
480  {
481  SolidNodeList.push_back(node);
482  taken = 1;
483  }
484  }
485  break;
486  case ESNRP_GUI:
487  if (!isCulled(node))
488  {
489  GuiNodeList.push_back(node);
490  taken = 1;
491  }
492 
493  // as of yet unused
494  case ESNRP_LIGHT:
495  case ESNRP_SHADOW:
496  case ESNRP_NONE: // ignore this one
497  break;
498  }
499 
500  return taken;
501 }

515 void CSceneManager::drawAll()
516 {
517  if (!Driver)
518  return;
519 
520  u32 i; // new ISO for scoping problem in some compilers
521 
522  // reset all transforms
523  Driver->setMaterial(video::SMaterial());
524  Driver->setTransform ( video::ETS_PROJECTION, core::IdentityMatrix );
525  Driver->setTransform ( video::ETS_VIEW, core::IdentityMatrix );
526  Driver->setTransform ( video::ETS_WORLD, core::IdentityMatrix );
527  for (i=video::ETS_COUNT-1; i>=video::ETS_TEXTURE_0; --i)
528  Driver->setTransform ( (video::E_TRANSFORMATION_STATE)i, core::IdentityMatrix );
529  // TODO: This should not use an attribute here but a real parameter when necessary (too slow!)
530  Driver->setAllowZWriteOnTransparent(Parameters->getAttributeAsBool(ALLOW_ZWRITE_ON_TRANSPARENT));
531 
532  // do animations and other stuff.
533  OnAnimate(os::Timer::getTime());
534 
539  camWorldPos.set(0,0,0);
540  if (ActiveCamera)
541  {
542  ActiveCamera->render();
543  camWorldPos = ActiveCamera->getAbsolutePosition();
544  }
545 
546  // let all nodes register themselves
547  OnRegisterSceneNode();
548 
549  //render camera scenes
550  {
551  CurrentRenderPass = ESNRP_CAMERA;
552  Driver->getOverrideMaterial().Enabled = ((Driver->getOverrideMaterial().EnablePasses & CurrentRenderPass) != 0);
553 
554  for (i=0; i<CameraList.size(); ++i)
555  CameraList[i]->render();
556 
557  CameraList.set_used(0);
558  }
559 
560  // render skyboxes
561  {
562  CurrentRenderPass = ESNRP_SKY_BOX;
563  Driver->getOverrideMaterial().Enabled = ((Driver->getOverrideMaterial().EnablePasses & CurrentRenderPass) != 0);
564 
565  for (i=0; i<SkyBoxList.size(); ++i)
566  SkyBoxList[i]->render();
567 
568  SkyBoxList.set_used(0);
569  }
570 
571  // render default objects
572  {
573  CurrentRenderPass = ESNRP_SOLID;
574  Driver->getOverrideMaterial().Enabled = ((Driver->getOverrideMaterial().EnablePasses & CurrentRenderPass) != 0);
575 
576  SolidNodeList.sort(); // sort by textures
577 
578  for (i=0; i<SolidNodeList.size(); ++i)
579  SolidNodeList[i].Node->render();
580 
581  SolidNodeList.set_used(0);
582  }
583 
584  // render transparent objects.
585  {
586  CurrentRenderPass = ESNRP_TRANSPARENT;
587  Driver->getOverrideMaterial().Enabled = ((Driver->getOverrideMaterial().EnablePasses & CurrentRenderPass) != 0);
588 
589  TransparentNodeList.sort(); // sort by distance from camera
590  for (i=0; i<TransparentNodeList.size(); ++i)
591  TransparentNodeList[i].Node->render();
592 
593  TransparentNodeList.set_used(0);
594  }
595 
596  // render transparent effect objects.
597  {
598  CurrentRenderPass = ESNRP_TRANSPARENT_EFFECT;
599  Driver->getOverrideMaterial().Enabled = ((Driver->getOverrideMaterial().EnablePasses & CurrentRenderPass) != 0);
600 
601  TransparentEffectNodeList.sort(); // sort by distance from camera
602 
603  for (i=0; i<TransparentEffectNodeList.size(); ++i)
604  TransparentEffectNodeList[i].Node->render();
605 
606  TransparentEffectNodeList.set_used(0);
607  }
608 
609  // render custom gui nodes
610  {
611  CurrentRenderPass = ESNRP_GUI;
612  Driver->getOverrideMaterial().Enabled = ((Driver->getOverrideMaterial().EnablePasses & CurrentRenderPass) != 0);
613 
614  for (i=0; i<GuiNodeList.size(); ++i)
615  GuiNodeList[i]->render();
616 
617  GuiNodeList.set_used(0);
618  }
619  clearDeletionList();
620 
621  CurrentRenderPass = ESNRP_NONE;
622 }
