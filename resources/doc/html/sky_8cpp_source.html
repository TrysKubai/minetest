38 static video::SMaterial baseMaterial()
39 {
40  video::SMaterial mat;
41  mat.Lighting = false;
42  mat.ZBuffer = video::ECFN_DISABLED;
43  mat.ZWriteEnable = video::EZW_OFF;
44  mat.AntiAliasing = 0;
45  mat.TextureLayer[0].TextureWrapU = video::ETC_CLAMP_TO_EDGE;
46  mat.TextureLayer[0].TextureWrapV = video::ETC_CLAMP_TO_EDGE;
47  mat.BackfaceCulling = false;
48  return mat;
49 }

67 
68  m_enable_shaders = g_settings->getBool("enable_shaders");
69

71  m_sun_params = SkyboxDefaults::getSunDefaults();
72  m_moon_params = SkyboxDefaults::getMoonDefaults();
73  m_star_params = SkyboxDefaults::getStarDefaults();
74 
75  // Create materials
76 
77  m_materials[0] = baseMaterial();

113 void Sky::render()
114 {
115  video::IVideoDriver *driver = SceneManager->getVideoDriver();
116  scene::ICameraSceneNode *camera = SceneManager->getActiveCamera();
117 
118  if (!camera || !driver)
119  return;
120 
121  ScopeProfiler sp(g_profiler, "Sky::render()", SPT_AVG);
122 
123  // Draw perspective skybox
124 
125  core::matrix4 translate(AbsoluteTransformation);
126  translate.setTranslation(camera->getAbsolutePosition());
127 
128  // Draw the sky box between the near and far clip plane
129  const f32 viewDistance = (camera->getNearValue() + camera->getFarValue()) * 0.5f;
130  core::matrix4 scale;
131  scale.setScale(core::vector3df(viewDistance, viewDistance, viewDistance));
132 
133  driver->setTransform(video::ETS_WORLD, translate * scale);
134 
135  if (m_sunlight_seen) {
136  video::SColorf suncolor_f(1, 1, 0, 1);
137  //suncolor_f.r = 1;
138  //suncolor_f.g = MYMAX(0.3, MYMIN(1.0, 0.7 + m_time_brightness * 0.5));
139  //suncolor_f.b = MYMAX(0.0, m_brightness * 0.95);
140  video::SColorf suncolor2_f(1, 1, 1, 1);
141  // The values below were probably meant to be suncolor2_f instead of a
142  // reassignment of suncolor_f. However, the resulting colour was chosen
143  // and is our long-running classic colour. So preserve, but comment-out
144  // the unnecessary first assignments above.
145  suncolor_f.r = 1;
146  suncolor_f.g = MYMAX(0.3, MYMIN(1.0, 0.85 + m_time_brightness * 0.5));
147  suncolor_f.b = MYMAX(0.0, m_brightness);
148 
149  video::SColorf mooncolor_f(0.50, 0.57, 0.65, 1);
150  video::SColorf mooncolor2_f(0.85, 0.875, 0.9, 1);
151 
152  float wicked_time_of_day = getWickedTimeOfDay(m_time_of_day);
153 
154  video::SColor suncolor = suncolor_f.toSColor();
155  video::SColor suncolor2 = suncolor2_f.toSColor();
156  video::SColor mooncolor = mooncolor_f.toSColor();
157  video::SColor mooncolor2 = mooncolor2_f.toSColor();
158 
159  // Calculate offset normalized to the X dimension of a 512x1 px tonemap
160  float offset = (1.0 - fabs(sin((m_time_of_day - 0.5) * irr::core::PI))) * 511;
161 
162  if (m_sun_tonemap) {
163  u8 * texels = (u8 *)m_sun_tonemap->lock();
164  video::SColor* texel = (video::SColor *)(texels + (u32)offset * 4);
165  video::SColor texel_color (255, texel->getRed(),
166  texel->getGreen(), texel->getBlue());
167  m_sun_tonemap->unlock();
168  m_materials[3].EmissiveColor = texel_color;
169  }
170 
171  if (m_moon_tonemap) {
172  u8 * texels = (u8 *)m_moon_tonemap->lock();
173  video::SColor* texel = (video::SColor *)(texels + (u32)offset * 4);
174  video::SColor texel_color (255, texel->getRed(),
175  texel->getGreen(), texel->getBlue());
176  m_moon_tonemap->unlock();
177  m_materials[4].EmissiveColor = texel_color;
178  }
179 
180  const f32 t = 1.0f;
181  const f32 o = 0.0f;
182  static const u16 indices[6] = {0, 1, 2, 0, 2, 3};
183  video::S3DVertex vertices[4];
184 
185  driver->setMaterial(m_materials[1]);
186 
187  video::SColor cloudyfogcolor = m_bgcolor;
188 
189  // Abort rendering if we're in the clouds.
190  // Stops rendering a pure white hole in the bottom of the skybox.
191  if (m_in_clouds)
192  return;
193 
194  // Draw the six sided skybox,
195  if (m_sky_params.textures.size() == 6) {
196  for (u32 j = 5; j < 11; j++) {
197  video::SColor c(255, 255, 255, 255);
198  driver->setMaterial(m_materials[j]);
199  // Use 1.05 rather than 1.0 to avoid colliding with the
200  // sun, moon and stars, as this is a background skybox.
201  vertices[0] = video::S3DVertex(-1.05, -1.05, -1.05, 0, 0, 1, c, t, t);
202  vertices[1] = video::S3DVertex( 1.05, -1.05, -1.05, 0, 0, 1, c, o, t);
203  vertices[2] = video::S3DVertex( 1.05, 1.05, -1.05, 0, 0, 1, c, o, o);
204  vertices[3] = video::S3DVertex(-1.05, 1.05, -1.05, 0, 0, 1, c, t, o);
205  for (video::S3DVertex &vertex : vertices) {
206  if (j == 5) { // Top texture
207  vertex.Pos.rotateYZBy(90);
208  vertex.Pos.rotateXZBy(90);
209  } else if (j == 6) { // Bottom texture
210  vertex.Pos.rotateYZBy(-90);
211  vertex.Pos.rotateXZBy(90);
212  } else if (j == 7) { // Left texture
213  vertex.Pos.rotateXZBy(90);
214  } else if (j == 8) { // Right texture
215  vertex.Pos.rotateXZBy(-90);
216  } else if (j == 9) { // Front texture, do nothing
217  // Irrlicht doesn't like it when vertexes are left
218  // alone and not rotated for some reason.
219  vertex.Pos.rotateXZBy(0);
220  } else {// Back texture
221  vertex.Pos.rotateXZBy(180);
222  }
223  }
224  driver->drawIndexedTriangleList(&vertices[0], 4, indices, 2);
225  }
226  }
227 
228  // Draw far cloudy fog thing blended with skycolor
229  if (m_visible) {
230  driver->setMaterial(m_materials[1]);
231  for (u32 j = 0; j < 4; j++) {
232  vertices[0] = video::S3DVertex(-1, -0.02, -1, 0, 0, 1, m_bgcolor, t, t);
233  vertices[1] = video::S3DVertex( 1, -0.02, -1, 0, 0, 1, m_bgcolor, o, t);
234  vertices[2] = video::S3DVertex( 1, 0.45, -1, 0, 0, 1, m_skycolor, o, o);
235  vertices[3] = video::S3DVertex(-1, 0.45, -1, 0, 0, 1, m_skycolor, t, o);
236  for (video::S3DVertex &vertex : vertices) {
237  if (j == 0)
238  // Don't switch
239  {}
240  else if (j == 1)
241  // Switch from -Z (south) to +X (east)
242  vertex.Pos.rotateXZBy(90);
243  else if (j == 2)
244  // Switch from -Z (south) to -X (west)
245  vertex.Pos.rotateXZBy(-90);
246  else
247  // Switch from -Z (south) to +Z (north)
248  vertex.Pos.rotateXZBy(-180);
249  }
250  driver->drawIndexedTriangleList(&vertices[0], 4, indices, 2);
251  }
252  }
253 
254  // Draw stars before sun and moon to be behind them
255  if (m_star_params.visible)
256  draw_stars(driver, wicked_time_of_day);
257 
258  // Draw sunrise/sunset horizon glow texture
259  // (textures/base/pack/sunrisebg.png)
260  if (m_sun_params.sunrise_visible) {
261  driver->setMaterial(m_materials[2]);
262  float mid1 = 0.25;
263  float mid = wicked_time_of_day < 0.5 ? mid1 : (1.0 - mid1);
264  float a_ = 1.0f - std::fabs(wicked_time_of_day - mid) * 35.0f;
265  float a = easeCurve(MYMAX(0, MYMIN(1, a_)));
266  //std::cerr<<"a_="<<a_<<" a="<<a<<std::endl;
267  video::SColor c(255, 255, 255, 255);
268  float y = -(1.0 - a) * 0.22;
269  vertices[0] = video::S3DVertex(-1, -0.05 + y, -1, 0, 0, 1, c, t, t);
270  vertices[1] = video::S3DVertex( 1, -0.05 + y, -1, 0, 0, 1, c, o, t);
271  vertices[2] = video::S3DVertex( 1, 0.2 + y, -1, 0, 0, 1, c, o, o);
272  vertices[3] = video::S3DVertex(-1, 0.2 + y, -1, 0, 0, 1, c, t, o);
273  for (video::S3DVertex &vertex : vertices) {
274  if (wicked_time_of_day < 0.5)
275  // Switch from -Z (south) to +X (east)
276  vertex.Pos.rotateXZBy(90);
277  else
278  // Switch from -Z (south) to -X (west)
279  vertex.Pos.rotateXZBy(-90);
280  }
281  driver->drawIndexedTriangleList(&vertices[0], 4, indices, 2);
282  }
283 
284  // Draw sun
285  if (m_sun_params.visible)
286  draw_sun(driver, suncolor, suncolor2, wicked_time_of_day);
287 
288  // Draw moon
289  if (m_moon_params.visible)
290  draw_moon(driver, mooncolor, mooncolor2, wicked_time_of_day);
291 
292  // Draw far cloudy fog thing below all horizons in front of sun, moon
293  // and stars.
294  if (m_visible) {
295  driver->setMaterial(m_materials[1]);
296 
297  for (u32 j = 0; j < 4; j++) {
298  video::SColor c = cloudyfogcolor;
299  vertices[0] = video::S3DVertex(-1, -1.0, -1, 0, 0, 1, c, t, t);
300  vertices[1] = video::S3DVertex( 1, -1.0, -1, 0, 0, 1, c, o, t);
301  vertices[2] = video::S3DVertex( 1, -0.02, -1, 0, 0, 1, c, o, o);
302  vertices[3] = video::S3DVertex(-1, -0.02, -1, 0, 0, 1, c, t, o);
303  for (video::S3DVertex &vertex : vertices) {
304  if (j == 0)
305  // Don't switch
306  {}
307  else if (j == 1)
308  // Switch from -Z (south) to +X (east)
309  vertex.Pos.rotateXZBy(90);
310  else if (j == 2)
311  // Switch from -Z (south) to -X (west)
312  vertex.Pos.rotateXZBy(-90);
313  else
314  // Switch from -Z (south) to +Z (north)
315  vertex.Pos.rotateXZBy(-180);
316  }
317  driver->drawIndexedTriangleList(&vertices[0], 4, indices, 2);
318  }
319 
320  // Draw bottom far cloudy fog thing in front of sun, moon and stars
321  video::SColor c = cloudyfogcolor;
322  vertices[0] = video::S3DVertex(-1, -1.0, -1, 0, 1, 0, c, t, t);
323  vertices[1] = video::S3DVertex( 1, -1.0, -1, 0, 1, 0, c, o, t);
324  vertices[2] = video::S3DVertex( 1, -1.0, 1, 0, 1, 0, c, o, o);
325  vertices[3] = video::S3DVertex(-1, -1.0, 1, 0, 1, 0, c, t, o);
326  driver->drawIndexedTriangleList(&vertices[0], 4, indices, 2);
327  }
328  }
329 }

331 void Sky::update(float time_of_day, float time_brightness,
332  float direct_brightness, bool sunlight_seen,
333  CameraMode cam_mode, float yaw, float pitch)
334 {
335  // Stabilize initial brightness and color values by flooding updates
336  if (m_first_update) {
337  /*dstream<<"First update with time_of_day="<<time_of_day
338  <<" time_brightness="<<time_brightness
339  <<" direct_brightness="<<direct_brightness
340  <<" sunlight_seen="<<sunlight_seen<<std::endl;*/
341  m_first_update = false;
342  for (u32 i = 0; i < 100; i++) {
343  update(time_of_day, time_brightness, direct_brightness,
344  sunlight_seen, cam_mode, yaw, pitch);
345  }
346  return;
347  }
348 
349  m_time_of_day = time_of_day;
350  m_time_brightness = time_brightness;
351  m_sunlight_seen = sunlight_seen;
352  m_in_clouds = false;
353 
354  bool is_dawn = (time_brightness >= 0.20 && time_brightness < 0.35);
355 
356  video::SColorf bgcolor_bright_normal_f = m_sky_params.sky_color.day_horizon;
357  video::SColorf bgcolor_bright_indoor_f = m_sky_params.sky_color.indoors;
358  video::SColorf bgcolor_bright_dawn_f = m_sky_params.sky_color.dawn_horizon;
359  video::SColorf bgcolor_bright_night_f = m_sky_params.sky_color.night_horizon;
360 
361  video::SColorf skycolor_bright_normal_f = m_sky_params.sky_color.day_sky;
362  video::SColorf skycolor_bright_dawn_f = m_sky_params.sky_color.dawn_sky;
363  video::SColorf skycolor_bright_night_f = m_sky_params.sky_color.night_sky;
364 
365  video::SColorf cloudcolor_bright_normal_f = m_cloudcolor_day_f;
366  video::SColorf cloudcolor_bright_dawn_f = m_cloudcolor_dawn_f;
367 
368  float cloud_color_change_fraction = 0.95;
369  if (sunlight_seen) {
370  if (std::fabs(time_brightness - m_brightness) < 0.2f) {
371  m_brightness = m_brightness * 0.95 + time_brightness * 0.05;
372  } else {
373  m_brightness = m_brightness * 0.80 + time_brightness * 0.20;
374  cloud_color_change_fraction = 0.0;
375  }
376  } else {
377  if (direct_brightness < m_brightness)
378  m_brightness = m_brightness * 0.95 + direct_brightness * 0.05;
379  else
380  m_brightness = m_brightness * 0.98 + direct_brightness * 0.02;
381  }
382 
383  m_clouds_visible = true;
384  float color_change_fraction = 0.98f;
385  if (sunlight_seen) {
386  if (is_dawn) { // Dawn
387  m_bgcolor_bright_f = m_bgcolor_bright_f.getInterpolated(
388  bgcolor_bright_dawn_f, color_change_fraction);
389  m_skycolor_bright_f = m_skycolor_bright_f.getInterpolated(
390  skycolor_bright_dawn_f, color_change_fraction);
391  m_cloudcolor_bright_f = m_cloudcolor_bright_f.getInterpolated(
392  cloudcolor_bright_dawn_f, color_change_fraction);
393  } else {
394  if (time_brightness < 0.13f) { // Night
395  m_bgcolor_bright_f = m_bgcolor_bright_f.getInterpolated(
396  bgcolor_bright_night_f, color_change_fraction);
397  m_skycolor_bright_f = m_skycolor_bright_f.getInterpolated(
398  skycolor_bright_night_f, color_change_fraction);
399  } else { // Day
400  m_bgcolor_bright_f = m_bgcolor_bright_f.getInterpolated(
401  bgcolor_bright_normal_f, color_change_fraction);
402  m_skycolor_bright_f = m_skycolor_bright_f.getInterpolated(
403  skycolor_bright_normal_f, color_change_fraction);
404  }
405 
406  m_cloudcolor_bright_f = m_cloudcolor_bright_f.getInterpolated(
407  cloudcolor_bright_normal_f, color_change_fraction);
408  }
409  } else {
410  m_bgcolor_bright_f = m_bgcolor_bright_f.getInterpolated(
411  bgcolor_bright_indoor_f, color_change_fraction);
412  m_skycolor_bright_f = m_skycolor_bright_f.getInterpolated(
413  bgcolor_bright_indoor_f, color_change_fraction);
414  m_cloudcolor_bright_f = m_cloudcolor_bright_f.getInterpolated(
415  cloudcolor_bright_normal_f, color_change_fraction);
416  m_clouds_visible = false;
417  }
418 
419  video::SColor bgcolor_bright = m_bgcolor_bright_f.toSColor();
420  m_bgcolor = video::SColor(
421  255,
422  bgcolor_bright.getRed() * m_brightness,
423  bgcolor_bright.getGreen() * m_brightness,
424  bgcolor_bright.getBlue() * m_brightness
425  );
426 
427  video::SColor skycolor_bright = m_skycolor_bright_f.toSColor();
428  m_skycolor = video::SColor(
429  255,
430  skycolor_bright.getRed() * m_brightness,
431  skycolor_bright.getGreen() * m_brightness,
432  skycolor_bright.getBlue() * m_brightness
433  );
434 
435  // Horizon coloring based on sun and moon direction during sunset and sunrise
436  video::SColor pointcolor = video::SColor(m_bgcolor.getAlpha(), 255, 255, 255);
437  if (m_directional_colored_fog) {
438  if (m_horizon_blend() != 0) {
439  // Calculate hemisphere value from yaw, (inverted in third person front view)
440  s8 dir_factor = 1;
441  if (cam_mode > CAMERA_MODE_THIRD)
442  dir_factor = -1;
443  f32 pointcolor_blend = wrapDegrees_0_360(yaw * dir_factor + 90);
444  if (pointcolor_blend > 180)
445  pointcolor_blend = 360 - pointcolor_blend;
446  pointcolor_blend /= 180;
447  // Bound view angle to determine where transition starts and ends
448  pointcolor_blend = rangelim(1 - pointcolor_blend * 1.375, 0, 1 / 1.375) *
449  1.375;
450  // Combine the colors when looking up or down, otherwise turning looks weird
451  pointcolor_blend += (0.5 - pointcolor_blend) *
452  (1 - MYMIN((90 - std::fabs(pitch)) / 90 * 1.5, 1));
453  // Invert direction to match where the sun and moon are rising
454  if (m_time_of_day > 0.5)
455  pointcolor_blend = 1 - pointcolor_blend;
456  // Horizon colors of sun and moon
457  f32 pointcolor_light = rangelim(m_time_brightness * 3, 0.2, 1);
458 
459  video::SColorf pointcolor_sun_f(1, 1, 1, 1);
460  // Use tonemap only if default sun/moon tinting is used
461  // which keeps previous behavior.
462  if (m_sun_tonemap && m_default_tint) {
463  pointcolor_sun_f.r = pointcolor_light *
464  (float)m_materials[3].EmissiveColor.getRed() / 255;
465  pointcolor_sun_f.b = pointcolor_light *
466  (float)m_materials[3].EmissiveColor.getBlue() / 255;
467  pointcolor_sun_f.g = pointcolor_light *
468  (float)m_materials[3].EmissiveColor.getGreen() / 255;
469  } else if (!m_default_tint) {
470  pointcolor_sun_f = m_sky_params.fog_sun_tint;
471  } else {
472  pointcolor_sun_f.r = pointcolor_light * 1;
473  pointcolor_sun_f.b = pointcolor_light *
474  (0.25 + (rangelim(m_time_brightness, 0.25, 0.75) - 0.25) * 2 * 0.75);
475  pointcolor_sun_f.g = pointcolor_light * (pointcolor_sun_f.b * 0.375 +
476  (rangelim(m_time_brightness, 0.05, 0.15) - 0.05) * 10 * 0.625);
477  }
478 
479  video::SColorf pointcolor_moon_f;
480  if (m_default_tint) {
481  pointcolor_moon_f = video::SColorf(
482  0.5 * pointcolor_light,
483  0.6 * pointcolor_light,
484  0.8 * pointcolor_light,
485  1
486  );
487  } else {
488  pointcolor_moon_f = video::SColorf(
489  (m_sky_params.fog_moon_tint.getRed() / 255) * pointcolor_light,
490  (m_sky_params.fog_moon_tint.getGreen() / 255) * pointcolor_light,
491  (m_sky_params.fog_moon_tint.getBlue() / 255) * pointcolor_light,
492  1
493  );
494  }
495  if (m_moon_tonemap && m_default_tint) {
496  pointcolor_moon_f.r = pointcolor_light *
497  (float)m_materials[4].EmissiveColor.getRed() / 255;
498  pointcolor_moon_f.b = pointcolor_light *
499  (float)m_materials[4].EmissiveColor.getBlue() / 255;
500  pointcolor_moon_f.g = pointcolor_light *
501  (float)m_materials[4].EmissiveColor.getGreen() / 255;

670 void Sky::draw_stars(video::IVideoDriver * driver, float wicked_time_of_day)
671 {
672  // Tune values so that stars first appear just after the sun
673  // disappears over the horizon, and disappear just before the sun
674  // appears over the horizon.
675  // Also tune so that stars are at full brightness from time 20000
676  // to time 4000.
677 
678  float tod = wicked_time_of_day < 0.5f ? wicked_time_of_day : (1.0f - wicked_time_of_day);
679  float day_opacity = clamp(m_star_params.day_opacity, 0.0f, 1.0f);
680  float starbrightness = (0.25f - fabs(tod)) * 20.0f;
681  float alpha = clamp(starbrightness, day_opacity, 1.0f);
682 
683  m_star_color = m_star_params.starcolor;
684  m_star_color.a *= alpha;
685  if (m_star_color.a <= 0.0f) // Stars are only drawn when not fully transparent
686  return;
687  m_materials[0].DiffuseColor = m_materials[0].EmissiveColor = m_star_color.toSColor();
688  auto sky_rotation = core::matrix4().setRotationAxisRadians(2.0f * M_PI * (wicked_time_of_day - 0.25f), v3f(0.0f, 0.0f, 1.0f));
689  auto world_matrix = driver->getTransform(video::ETS_WORLD);
690  driver->setTransform(video::ETS_WORLD, world_matrix * sky_rotation);
691  driver->setMaterial(m_materials[0]);
692  driver->drawMeshBuffer(m_stars.get());
693  driver->setTransform(video::ETS_WORLD, world_matrix);

817 void Sky::updateStars()
818 {
819  m_stars.reset(new scene::SMeshBuffer());
820  // Stupid IrrLicht doesnt allow non-indexed rendering, and indexed quad
821  // rendering is slow due to lack of hardware support. So as indices are
822  // 16-bit and there are 4 vertices per star... the limit is 2^16/4 = 0x4000.
823  // That should be well enough actually.
824  if (m_star_params.count > 0x4000) {
825  warningstream << "Requested " << m_star_params.count << " stars but " << 0x4000 << " is the max\n";
826  m_star_params.count = 0x4000;
827  }
828  m_stars->Vertices.reallocate(4 * m_star_params.count);
829  m_stars->Indices.reallocate(6 * m_star_params.count);
830 
831  video::SColor fallback_color = m_star_params.starcolor; // used on GLES 2 without shaders
832  PcgRandom rgen(m_seed);
833  float d = (0.006 / 2) * m_star_params.scale;
834  for (u16 i = 0; i < m_star_params.count; i++) {
835  v3f r = v3f(
836  rgen.range(-10000, 10000),
837  rgen.range(-10000, 10000),
838  rgen.range(-10000, 10000)
839  );
840  core::CMatrix4<f32> a;
841  a.buildRotateFromTo(v3f(0, 1, 0), r);
842  v3f p = v3f(-d, 1, -d);
843  v3f p1 = v3f(d, 1, -d);
844  v3f p2 = v3f(d, 1, d);
845  v3f p3 = v3f(-d, 1, d);
846  a.rotateVect(p);
847  a.rotateVect(p1);
848  a.rotateVect(p2);
849  a.rotateVect(p3);
850  m_stars->Vertices.push_back(video::S3DVertex(p, {}, fallback_color, {}));
851  m_stars->Vertices.push_back(video::S3DVertex(p1, {}, fallback_color, {}));
852  m_stars->Vertices.push_back(video::S3DVertex(p2, {}, fallback_color, {}));
853  m_stars->Vertices.push_back(video::S3DVertex(p3, {}, fallback_color, {}));
854  }
855  for (u16 i = 0; i < m_star_params.count; i++) {
856  m_stars->Indices.push_back(i * 4 + 0);
857  m_stars->Indices.push_back(i * 4 + 1);
858  m_stars->Indices.push_back(i * 4 + 2);
859  m_stars->Indices.push_back(i * 4 + 2);
860  m_stars->Indices.push_back(i * 4 + 3);
861  m_stars->Indices.push_back(i * 4 + 0);
862  }
863  m_stars->setHardwareMappingHint(scene::EHM_STATIC);
864 }
