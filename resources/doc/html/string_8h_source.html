574 {
575  std::basic_string<T> output;
576  size_t i = 0;
577  while (i < s.length()) {
578  if (s[i] == '\x1b') {
579  ++i;

581  if (s[i] == '(') {
582  ++i;

601 template <typename T>
602 std::vector<std::basic_string<T> > split(const std::basic_string<T> &s, T delim)
603 {
604  std::vector<std::basic_string<T> > tokens;
605 
606  std::basic_string<T> current;
607  bool last_was_escape = false;
608  for (size_t i = 0; i < s.length(); i++) {
609  T si = s[i];
610  if (last_was_escape) {
611  current += '\\';
612  current += si;
613  last_was_escape = false;
614  } else {
615  if (si == delim) {
616  tokens.push_back(current);
617  current.clear();
618  last_was_escape = false;
619  } else if (si == '\\') {
620  last_was_escape = true;
621  } else {
622  current += si;
623  last_was_escape = false;
624  }
625  }
626  }
627  //push last element
628  tokens.push_back(current);

676 {
677  std::ostringstream ss;
678  const char *neg = "";
679  if (sec < 0) {
680  sec = -sec;
681  neg = "-";
682  }
683  int total_sec = sec;
684  int min = sec / 60;
685  sec %= 60;
686  int hour = min / 60;
687  min %= 60;
688  int day = hour / 24;
689  hour %= 24;
690 
691  if (day > 0) {
692  ss << neg << day << "d";
693  if (hour > 0 || min > 0 || sec > 0)
694  ss << " ";
695  }
696 
697  if (hour > 0) {
698  ss << neg << hour << "h";
699  if (min > 0 || sec > 0)
700  ss << " ";
701  }
702 
703  if (min > 0) {
704  ss << neg << min << "min";
705  if (sec > 0)
706  ss << " ";
707  }
708 
709  if (sec > 0 || total_sec == 0) {
710  ss << neg << sec << "s";
