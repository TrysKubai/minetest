30 LuaEntitySAO::LuaEntitySAO(ServerEnvironment *env, v3f pos, const std::string &data)
31  : UnitSAO(env, pos)
32 {
33  std::string name;
34  std::string state;
35  u16 hp = 1;
36  v3f velocity;
37  v3f rotation;
38 
39  while (!data.empty()) { // breakable, run for one iteration
40  std::istringstream is(data, std::ios::binary);
41  // 'version' does not allow to incrementally extend the parameter list thus
42  // we need another variable to build on top of 'version=1'. Ugly hack but worksâ„¢
43  u8 version2 = 0;
44  u8 version = readU8(is);
45 
46  name = deSerializeString16(is);
47  state = deSerializeString32(is);
48 
49  if (version < 1)
50  break;
51 
52  hp = readU16(is);
53  velocity = readV3F1000(is);
54  // yaw must be yaw to be backwards-compatible
55  rotation.Y = readF1000(is);
56 
57  if (is.good()) // EOF for old formats
58  version2 = readU8(is);
59 
60  if (version2 < 1) // PROTOCOL_VERSION < 37
61  break;
62 
63  // version2 >= 1
64  rotation.X = readF1000(is);
65  rotation.Z = readF1000(is);
66 
67  // if (version2 < 2)
68  // break;
69  // <read new values>
70  break;
71  }
72  // create object
73  infostream << "LuaEntitySAO::create(name=\"" << name << "\" state=\""
74  << state << "\")" << std::endl;
75 
76  m_init_name = name;
77  m_init_state = state;
78  m_hp = hp;
79  m_velocity = velocity;
80  m_rotation = rotation;
81 }

130 void LuaEntitySAO::step(float dtime, bool send_recommended)
131 {
132  if(!m_properties_sent)
133  {
134  m_properties_sent = true;
135  std::string str = getPropertyPacket();
136  // create message and add to list
137  m_messages_out.emplace(getId(), true, str);
138  }
139 
140  // If attached, check that our parent is still there. If it isn't, detach.
141  if (m_attachment_parent_id && !isAttached()) {
142  // This is handled when objects are removed from the map
143  warningstream << "LuaEntitySAO::step() id=" << m_id <<
144  " is attached to nonexistent parent. This is a bug." << std::endl;
145  clearParentAttachment();
146  sendPosition(false, true);
147  }
148 
149  m_last_sent_position_timer += dtime;
150 
151  collisionMoveResult moveresult, *moveresult_p = nullptr;
152 
153  // Each frame, parent position is copied if the object is attached, otherwise it's calculated normally
154  // If the object gets detached this comes into effect automatically from the last known origin
155  if (auto *parent = getParent()) {
156  m_base_position = parent->getBasePosition();
157  m_velocity = v3f(0,0,0);
158  m_acceleration = v3f(0,0,0);
159  } else {
160  if(m_prop.physical){
161  aabb3f box = m_prop.collisionbox;
162  box.MinEdge *= BS;
163  box.MaxEdge *= BS;
164  f32 pos_max_d = BS*0.25; // Distance per iteration
165  v3f p_pos = m_base_position;
166  v3f p_velocity = m_velocity;
167  v3f p_acceleration = m_acceleration;
168  moveresult = collisionMoveSimple(m_env, m_env->getGameDef(),
169  pos_max_d, box, m_prop.stepheight, dtime,
170  &p_pos, &p_velocity, p_acceleration,
171  this, m_prop.collideWithObjects);
172  moveresult_p = &moveresult;
173 
174  // Apply results
175  m_base_position = p_pos;
176  m_velocity = p_velocity;
177  m_acceleration = p_acceleration;
178  } else {
179  m_base_position += (m_velocity + m_acceleration * 0.5f * dtime) * dtime;
180  m_velocity += dtime * m_acceleration;
181  }
182 
183  if (m_prop.automatic_face_movement_dir &&
184  (fabs(m_velocity.Z) > 0.001 || fabs(m_velocity.X) > 0.001)) {
185  float target_yaw = atan2(m_velocity.Z, m_velocity.X) * 180 / M_PI
186  + m_prop.automatic_face_movement_dir_offset;
187  float max_rotation_per_sec =
188  m_prop.automatic_face_movement_max_rotation_per_sec;
189 
190  if (max_rotation_per_sec > 0) {
191  m_rotation.Y = wrapDegrees_0_360(m_rotation.Y);
192  wrappedApproachShortest(m_rotation.Y, target_yaw,
193  dtime * max_rotation_per_sec, 360.f);
194  } else {
195  // Negative values of max_rotation_per_sec mean disabled.
196  m_rotation.Y = target_yaw;
197  }
198  }
199  }
200 
201  if (fabs(m_prop.automatic_rotate) > 0.001f) {
202  m_rotation_add_yaw = modulo360f(m_rotation_add_yaw + dtime * core::RADTODEG *
203  m_prop.automatic_rotate);
204  }
205 
206  if(m_registered) {
207  m_env->getScriptIface()->luaentity_Step(m_id, dtime, moveresult_p);
208  }
209 
210  if (!send_recommended)
211  return;
212 
213  if(!isAttached())
214  {
215  // TODO: force send when acceleration changes enough?
216  float minchange = 0.2*BS;
217  if(m_last_sent_position_timer > 1.0){
218  minchange = 0.01*BS;
219  } else if(m_last_sent_position_timer > 0.2){
220  minchange = 0.05*BS;
221  }
222  float move_d = m_base_position.getDistanceFrom(m_last_sent_position);
223  move_d += m_last_sent_move_precision;
224  float vel_d = m_velocity.getDistanceFrom(m_last_sent_velocity);
225  if (move_d > minchange || vel_d > minchange ||
226  std::fabs(m_rotation.X - m_last_sent_rotation.X) > 1.0f ||
227  std::fabs(m_rotation.Y - m_last_sent_rotation.Y) > 1.0f ||
228  std::fabs(m_rotation.Z - m_last_sent_rotation.Z) > 1.0f) {
229 
230  sendPosition(true, false);
231  }
232  }
233 
234  sendOutdatedData();
235 }

237 std::string LuaEntitySAO::getClientInitializationData(u16 protocol_version)
238 {
239  std::ostringstream os(std::ios::binary);
240 
241  // PROTOCOL_VERSION >= 37
242  writeU8(os, 1); // version
243  os << serializeString16(""); // name
244  writeU8(os, 0); // is_player
245  writeU16(os, getId()); //id
246  writeV3F32(os, m_base_position);
247  writeV3F32(os, m_rotation);
248  writeU16(os, m_hp);
249 
250  std::ostringstream msg_os(std::ios::binary);
251  msg_os << serializeString32(getPropertyPacket()); // message 1
252  msg_os << serializeString32(generateUpdateArmorGroupsCommand()); // 2
253  msg_os << serializeString32(generateUpdateAnimationCommand()); // 3
254  for (const auto &bone_pos : m_bone_position) {
255  msg_os << serializeString32(generateUpdateBonePositionCommand(
256  bone_pos.first, bone_pos.second.X, bone_pos.second.Y)); // 3 + N
257  }
258  msg_os << serializeString32(generateUpdateAttachmentCommand()); // 4 + m_bone_position.size
259 
260  int message_count = 4 + m_bone_position.size();
261 
262  for (const auto &id : getAttachmentChildIds()) {
263  if (ServerActiveObject *obj = m_env->getActiveObject(id)) {
264  message_count++;
265  msg_os << serializeString32(obj->generateUpdateInfantCommand(
266  id, protocol_version));
267  }
268  }
269 
270  msg_os << serializeString32(generateSetTextureModCommand());
271  message_count++;
272 
273  writeU8(os, message_count);
274  std::string serialized = msg_os.str();
275  os.write(serialized.c_str(), serialized.size());
276 
277  // return result
278  return os.str();
279 }

313 u32 LuaEntitySAO::punch(v3f dir,
314  const ToolCapabilities *toolcap,
315  ServerActiveObject *puncher,
316  float time_from_last_punch,
317  u16 initial_wear)
318 {
319  if (!m_registered) {
320  // Delete unknown LuaEntities when punched
321  markForRemoval();
322  return 0;
323  }
324 
325  FATAL_ERROR_IF(!puncher, "Punch action called without SAO");
326 
327  s32 old_hp = getHP();
328  ItemStack selected_item, hand_item;
329  ItemStack tool_item = puncher->getWieldedItem(&selected_item, &hand_item);
330 
331  PunchDamageResult result = getPunchDamage(
332  m_armor_groups,
333  toolcap,
334  &tool_item,
335  time_from_last_punch,
336  initial_wear);
337 
338  bool damage_handled = m_env->getScriptIface()->luaentity_Punch(m_id, puncher,
339  time_from_last_punch, toolcap, dir, result.did_punch ? result.damage : 0);
340 
341  if (!damage_handled) {
342  if (result.did_punch) {
343  setHP((s32)getHP() - result.damage,
344  PlayerHPChangeReason(PlayerHPChangeReason::PLAYER_PUNCH, puncher));
345  }
346  }
347 
348  actionstream << puncher->getDescription() << " (id=" << puncher->getId() <<
349  ", hp=" << puncher->getHP() << ") punched " <<
350  getDescription() << " (id=" << m_id << ", hp=" << m_hp <<
351  "), damage=" << (old_hp - (s32)getHP()) <<
352  (damage_handled ? " (handled by Lua)" : "") << std::endl;
353 
354  // TODO: give Lua control over wear
355  return result.wear;
356 }
