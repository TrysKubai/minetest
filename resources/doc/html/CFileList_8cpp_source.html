62 u32 CFileList::addItem(const io::path& fullPath, u32 offset, u32 size, bool isDirectory, u32 id)
63 {
64  SFileListEntry entry;
65  entry.ID = id ? id : Files.size();
66  entry.Offset = offset;
67  entry.Size = size;
68  entry.Name = fullPath;
69  entry.Name.replace('\\', '/');
70  entry.IsDirectory = isDirectory;
71 
72  // remove trailing slash
73  if (entry.Name.lastChar() == '/')
74  {
75  entry.IsDirectory = true;
76  entry.Name[entry.Name.size()-1] = 0;
77  entry.Name.validate();
78  }
79 
80  if (IgnoreCase)
81  entry.Name.make_lower();
82 
83  entry.FullName = entry.Name;
84 
85  core::deletePathFromFilename(entry.Name);
86 
87  if (IgnorePaths)
88  entry.FullName = entry.Name;
89 
90  //os::Printer::log(Path.c_str(), entry.FullName);
91 
92  Files.push_back(entry);
93 
94  return Files.size() - 1;
