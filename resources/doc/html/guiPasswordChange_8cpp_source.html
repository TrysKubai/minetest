199 bool GUIPasswordChange::OnEvent(const SEvent &event)
200 {
201  if (event.EventType == EET_KEY_INPUT_EVENT) {
202  // clang-format off
203  if ((event.KeyInput.Key == KEY_ESCAPE ||
204  event.KeyInput.Key == KEY_CANCEL) &&
205  event.KeyInput.PressedDown) {
206  quitMenu();
207  return true;
208  }
209  // clang-format on
210  if (event.KeyInput.Key == KEY_RETURN && event.KeyInput.PressedDown) {
211  acceptInput();
212  if (processInput())
213  quitMenu();
214  return true;
215  }
216  }
217  if (event.EventType == EET_GUI_EVENT) {
218  if (event.GUIEvent.EventType == gui::EGET_ELEMENT_FOCUS_LOST &&
219  isVisible()) {
220  if (!canTakeFocus(event.GUIEvent.Element)) {
221  infostream << "GUIPasswordChange: Not allowing focus change."
222  << std::endl;
223  // Returning true disables focus change
224  return true;
225  }
226  }
227  if (event.GUIEvent.EventType == gui::EGET_BUTTON_CLICKED) {
228  switch (event.GUIEvent.Caller->getID()) {
229  case ID_change:
230  acceptInput();
231  if (processInput())
232  quitMenu();
233  return true;
234  case ID_cancel:
235  quitMenu();
236  return true;
237  }
238  }
239  if (event.GUIEvent.EventType == gui::EGET_EDITBOX_ENTER) {
240  switch (event.GUIEvent.Caller->getID()) {
241  case ID_oldPassword:
242  case ID_newPassword1:
243  case ID_newPassword2:
244  acceptInput();
245  if (processInput())
246  quitMenu();
247  return true;
248  }
249  }
250  }
251 
252  return Parent ? Parent->OnEvent(event) : false;
253 }
