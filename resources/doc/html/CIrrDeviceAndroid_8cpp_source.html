35 CIrrDeviceAndroid::CIrrDeviceAndroid(const SIrrlichtCreationParameters& param)
36  : CIrrDeviceStub(param), Accelerometer(0), Gyroscope(0), Focused(false), Initialized(false), Paused(true), JNIEnvAttachedToVM(0)
37 {
38 #ifdef _DEBUG
39  setDebugName("CIrrDeviceAndroid");
40 #endif
41 
42  // Get the interface to the native Android activity.
43  Android = (android_app*)(param.PrivateData);
44 
45  // Set the private data so we can use it in any static callbacks.
46  Android->userData = this;
47 
48  // Set the default command handler. This is a callback function that the Android
49  // OS invokes to send the native activity messages.
50  Android->onAppCmd = handleAndroidCommand;
51 
52  createKeyMap();
53 
54  // Create a sensor manager to receive touch screen events from the java activity.
55  SensorManager = ASensorManager_getInstance();
56  SensorEventQueue = ASensorManager_createEventQueue(SensorManager, Android->looper, LOOPER_ID_USER, 0, 0);
57  Android->onInputEvent = handleInput;
58 
59  // Create EGL manager.
60  ContextManager = new video::CEGLManager();
61 
62  os::Printer::log("Waiting for Android activity window to be created.", ELL_DEBUG);
63 
64  do
65  {
66  s32 Events = 0;
67  android_poll_source* Source = 0;
68 
69  while ((ALooper_pollAll(((Focused && !Paused) || !Initialized) ? 0 : -1, 0, &Events, (void**)&Source)) >= 0)
70  {
71  if(Source)
72  Source->process(Android, Source);
73  }
74  }
75  while(!Initialized);
76 }

100 bool CIrrDeviceAndroid::run()
101 {
102  if (!Initialized)
103  return false;
104 
105  os::Timer::tick();
106 
107  s32 id;
108  s32 Events = 0;
109  android_poll_source* Source = 0;
110 
111  while ((id = ALooper_pollAll(((Focused && !Paused) || !Initialized) ? 0 : -1, 0, &Events, (void**)&Source)) >= 0)
112  {
113  if(Source)
114  Source->process(Android, Source);
115 
116  // if a sensor has data, we'll process it now.
117  if (id == LOOPER_ID_USER)
118  {
119  ASensorEvent sensorEvent;
120  while (ASensorEventQueue_getEvents(SensorEventQueue, &sensorEvent, 1) > 0)
121  {
122  switch (sensorEvent.type)
123  {
124  case ASENSOR_TYPE_ACCELEROMETER:
125  SEvent accEvent;
126  accEvent.EventType = EET_ACCELEROMETER_EVENT;
127  accEvent.AccelerometerEvent.X = sensorEvent.acceleration.x;
128  accEvent.AccelerometerEvent.Y = sensorEvent.acceleration.y;
129  accEvent.AccelerometerEvent.Z = sensorEvent.acceleration.z;
130 
131  postEventFromUser(accEvent);
132  break;
133 
134  case ASENSOR_TYPE_GYROSCOPE:
135  SEvent gyroEvent;
136  gyroEvent.EventType = EET_GYROSCOPE_EVENT;
137  gyroEvent.GyroscopeEvent.X = sensorEvent.vector.x;
138  gyroEvent.GyroscopeEvent.Y = sensorEvent.vector.y;
139  gyroEvent.GyroscopeEvent.Z = sensorEvent.vector.z;
140 
141  postEventFromUser(gyroEvent);
142  break;
143  default:
144  break;
145  }
146  }
147  }
148 
149  if(!Initialized)
150  break;
151  }
152 
153  return Initialized;
154 }

162 void CIrrDeviceAndroid::sleep(u32 timeMs, bool pauseTimer)
163 {
164  const bool wasStopped = Timer ? Timer->isStopped() : true;
165 
166  struct timespec ts;
167  ts.tv_sec = (time_t) (timeMs / 1000);
168  ts.tv_nsec = (long) (timeMs % 1000) * 1000000;
169 
170  if (pauseTimer && !wasStopped)
171  Timer->stop();
172 
173  nanosleep(&ts, NULL);
174 
175  if (pauseTimer && !wasStopped)
176  Timer->start();
177 }

229 void CIrrDeviceAndroid::handleAndroidCommand(android_app* app, int32_t cmd)
230 {
231  CIrrDeviceAndroid* device = (CIrrDeviceAndroid*)app->userData;
232 
233  SEvent event;
234  event.EventType = EET_SYSTEM_EVENT;
235  event.SystemEvent.EventType = ESET_ANDROID_CMD;
236  event.SystemEvent.AndroidCmd.Cmd = cmd;
237  if ( device->postEventFromUser(event) )
238  return;
239 
240  switch (cmd)
241  {
242  case APP_CMD_INPUT_CHANGED:
243  os::Printer::log("Android command APP_CMD_INPUT_CHANGED", ELL_DEBUG);
244  break;
245  case APP_CMD_WINDOW_RESIZED:
246  os::Printer::log("Android command APP_CMD_WINDOW_RESIZED", ELL_DEBUG);
247  break;
248  case APP_CMD_WINDOW_REDRAW_NEEDED:
249  os::Printer::log("Android command APP_CMD_WINDOW_REDRAW_NEEDED", ELL_DEBUG);
250  break;
251  case APP_CMD_SAVE_STATE:
252  os::Printer::log("Android command APP_CMD_SAVE_STATE", ELL_DEBUG);
253  break;
254  case APP_CMD_CONTENT_RECT_CHANGED:
255  os::Printer::log("Android command APP_CMD_CONTENT_RECT_CHANGED", ELL_DEBUG);
256  break;
257  case APP_CMD_CONFIG_CHANGED:
258  os::Printer::log("Android command APP_CMD_CONFIG_CHANGED", ELL_DEBUG);
259  break;
260  case APP_CMD_LOW_MEMORY:
261  os::Printer::log("Android command APP_CMD_LOW_MEMORY", ELL_DEBUG);
262  break;
263  case APP_CMD_START:
264  os::Printer::log("Android command APP_CMD_START", ELL_DEBUG);
265  break;
266  case APP_CMD_INIT_WINDOW:
267  os::Printer::log("Android command APP_CMD_INIT_WINDOW", ELL_DEBUG);
268  device->getExposedVideoData().OGLESAndroid.Window = app->window;
269 
270  if (device->CreationParams.WindowSize.Width == 0 || device->CreationParams.WindowSize.Height == 0)
271  {
272  device->CreationParams.WindowSize.Width = ANativeWindow_getWidth(app->window);
273  device->CreationParams.WindowSize.Height = ANativeWindow_getHeight(app->window);
274  }
275 
276  device->getContextManager()->initialize(device->CreationParams, device->ExposedVideoData);
277  device->getContextManager()->generateSurface();
278  device->getContextManager()->generateContext();
279  device->getContextManager()->activateContext(device->getContextManager()->getContext());
280 
281  if (!device->Initialized)
282  {
283  io::CAndroidAssetFileArchive* assets = new io::CAndroidAssetFileArchive( device->Android->activity->assetManager, false, false);
284  assets->addDirectoryToFileList("");
285  device->FileSystem->addFileArchive(assets);
286  assets->drop();
287 
288  device->createDriver();
289 
290  if (device->VideoDriver)
291  device->createGUIAndScene();
292  }
293  device->Initialized = true;
294  break;
295  case APP_CMD_TERM_WINDOW:
296  os::Printer::log("Android command APP_CMD_TERM_WINDOW", ELL_DEBUG);
297  device->getContextManager()->destroySurface();
298  break;
299  case APP_CMD_GAINED_FOCUS:
300  os::Printer::log("Android command APP_CMD_GAINED_FOCUS", ELL_DEBUG);
301  device->Focused = true;
302  break;
303  case APP_CMD_LOST_FOCUS:
304  os::Printer::log("Android command APP_CMD_LOST_FOCUS", ELL_DEBUG);
305  device->Focused = false;
306  break;
307  case APP_CMD_DESTROY:
308  os::Printer::log("Android command APP_CMD_DESTROY", ELL_DEBUG);
309  if ( device->JNIEnvAttachedToVM )
310  {
311  device->JNIEnvAttachedToVM = 0;
312  device->Android->activity->vm->DetachCurrentThread();
313  }
314  device->Initialized = false;
315  break;
316  case APP_CMD_PAUSE:
317  os::Printer::log("Android command APP_CMD_PAUSE", ELL_DEBUG);
318  device->Paused = true;
319  break;
320  case APP_CMD_STOP:
321  os::Printer::log("Android command APP_CMD_STOP", ELL_DEBUG);
322  break;
323  case APP_CMD_RESUME:
324  os::Printer::log("Android command APP_CMD_RESUME", ELL_DEBUG);
325  device->Paused = false;
326  break;
327  default:
328  break;
329  }
330 }

332 s32 CIrrDeviceAndroid::handleInput(android_app* app, AInputEvent* androidEvent)
333 {
334  CIrrDeviceAndroid* device = (CIrrDeviceAndroid*)app->userData;
335  s32 status = 0;
336 
337  switch ( AInputEvent_getType(androidEvent) )
338  {
339  case AINPUT_EVENT_TYPE_MOTION:
340  {
341  SEvent event;
342  event.EventType = EET_TOUCH_INPUT_EVENT;
343 
344  s32 eventAction = AMotionEvent_getAction(androidEvent);
345  s32 eventType = eventAction & AMOTION_EVENT_ACTION_MASK;
346 
347 #if 0
348  // Useful for debugging. We might have to pass some of those infos on at some point.
349  // but preferably device independent (so iphone can use same irrlicht flags).
350  int32_t flags = AMotionEvent_getFlags(androidEvent);
351  os::Printer::log("flags: ", core::stringc(flags).c_str(), ELL_DEBUG);
352  int32_t metaState = AMotionEvent_getMetaState(androidEvent);
353  os::Printer::log("metaState: ", core::stringc(metaState).c_str(), ELL_DEBUG);
354  int32_t edgeFlags = AMotionEvent_getEdgeFlags(androidEvent);
355  os::Printer::log("edgeFlags: ", core::stringc(flags).c_str(), ELL_DEBUG);
356 #endif
357 
358  bool touchReceived = true;
359 
360  switch (eventType)
361  {
362  case AMOTION_EVENT_ACTION_DOWN:
363  case AMOTION_EVENT_ACTION_POINTER_DOWN:
364  event.TouchInput.Event = ETIE_PRESSED_DOWN;
365  break;
366  case AMOTION_EVENT_ACTION_MOVE:
367  event.TouchInput.Event = ETIE_MOVED;
368  break;
369  case AMOTION_EVENT_ACTION_UP:
370  case AMOTION_EVENT_ACTION_POINTER_UP:
371  case AMOTION_EVENT_ACTION_CANCEL:
372  event.TouchInput.Event = ETIE_LEFT_UP;
373  break;
374  default:
375  touchReceived = false;
376  break;
377  }
378 
379  if (touchReceived)
380  {
381  // Process all touches for move action.
382  if (event.TouchInput.Event == ETIE_MOVED)
383  {
384  s32 pointerCount = AMotionEvent_getPointerCount(androidEvent);
385 
386  for (s32 i = 0; i < pointerCount; ++i)
387  {
388  event.TouchInput.ID = AMotionEvent_getPointerId(androidEvent, i);
389  event.TouchInput.X = AMotionEvent_getX(androidEvent, i);
390  event.TouchInput.Y = AMotionEvent_getY(androidEvent, i);
391  event.TouchInput.touchedCount = AMotionEvent_getPointerCount(androidEvent);
392 
393  device->postEventFromUser(event);
394  }
395  }
396  else // Process one touch for other actions.
397  {
398  s32 pointerIndex = (eventAction & AMOTION_EVENT_ACTION_POINTER_INDEX_MASK) >> AMOTION_EVENT_ACTION_POINTER_INDEX_SHIFT;
399 
400  event.TouchInput.ID = AMotionEvent_getPointerId(androidEvent, pointerIndex);
401  event.TouchInput.X = AMotionEvent_getX(androidEvent, pointerIndex);
402  event.TouchInput.Y = AMotionEvent_getY(androidEvent, pointerIndex);
403  event.TouchInput.touchedCount = AMotionEvent_getPointerCount(androidEvent);
404 
405  device->postEventFromUser(event);
406  }
407 
408  status = 1;
409  }
410  }
411  break;
412  case AINPUT_EVENT_TYPE_KEY:
413  {
414  SEvent event;
415  event.EventType = EET_KEY_INPUT_EVENT;
416 
417  int32_t keyCode = AKeyEvent_getKeyCode(androidEvent);
418  // os::Printer::log("keyCode: ", core::stringc(keyCode).c_str(), ELL_DEBUG);
419 
420  int32_t keyAction = AKeyEvent_getAction(androidEvent);
421  int32_t keyMetaState = AKeyEvent_getMetaState(androidEvent);
422 
423  if ( keyCode >= 0 && (u32)keyCode < device->KeyMap.size() )
424  event.KeyInput.Key = device->KeyMap[keyCode];
425  else
426  event.KeyInput.Key = KEY_UNKNOWN;
427  event.KeyInput.SystemKeyCode = (u32)keyCode;
428  if ( keyAction == AKEY_EVENT_ACTION_DOWN )
429  event.KeyInput.PressedDown = true;
430  else if ( keyAction == AKEY_EVENT_ACTION_UP )
431  event.KeyInput.PressedDown = false;
432  else if ( keyAction == AKEY_EVENT_ACTION_MULTIPLE )
433  {
434  // TODO: Multiple duplicate key events have occurred in a row,
435  // or a complex string is being delivered. The repeat_count
436  // property of the key event contains the number of times the
437  // given key code should be executed.
438  // I guess this might necessary for more complicated i18n key input,
439  // but don't see yet how to handle this correctly.
440  }
441 
442  /* no use for meta keys so far.
443  if ( keyMetaState & AMETA_ALT_ON
444  || keyMetaState & AMETA_ALT_LEFT_ON
445  || keyMetaState & AMETA_ALT_RIGHT_ON )
446  ;
447  // what is a sym?
448  if ( keyMetaState & AMETA_SYM_ON )
449  ;
450  */
451  if ( keyMetaState & AMETA_SHIFT_ON
452  || keyMetaState & AMETA_SHIFT_LEFT_ON
453  || keyMetaState & AMETA_SHIFT_RIGHT_ON )
454  event.KeyInput.Shift = true;
455  else
456  event.KeyInput.Shift = false;
457  event.KeyInput.Control = false;
458 
459  // Having memory allocations + going through JNI for each key-press is pretty bad (slow).
460  // So we do it only for those keys which are likely text-characters and avoid it for all other keys.
461  // So it's fast for keys like game controller input and special keys. And text keys are typically
462  // only used or entering text and not for gaming on Android, so speed likely doesn't matter there too much.
463  if ( event.KeyInput.Key > 0 )
464  {
465  // TODO:
466  // Not sure why we have to attach a JNIEnv here, but it won't work when doing that in the constructor or
467  // trying to use the activity->env. My best guess is that the event-handling happens in an own thread.
468  // It means JNIEnvAttachedToVM will never get detached as I don't know a safe way where to do that
469  // (we could attach & detach each time, but that would probably be slow)
470  // Also - it has to be each time as it get's invalid when the application mode changes.
471  if ( device->Initialized && device->Android && device->Android->activity && device->Android->activity->vm )
472  {
473  JavaVMAttachArgs attachArgs;
474  attachArgs.version = JNI_VERSION_1_6;
475  attachArgs.name = 0;
476  attachArgs.group = NULL;
477 
478  // Not a big problem calling it each time - it's a no-op when the thread already is attached.
479  // And we have to do that as someone else can have detached the thread in the meantime.
480  jint result = device->Android->activity->vm->AttachCurrentThread(&device->JNIEnvAttachedToVM, &attachArgs);
481  if(result == JNI_ERR)
482  {
483  os::Printer::log("AttachCurrentThread for the JNI environment failed.", ELL_WARNING);
484  device->JNIEnvAttachedToVM = 0;
485  }
486 
487  if ( device->JNIEnvAttachedToVM )
488  {
489  jni::CKeyEventWrapper * keyEventWrapper = new jni::CKeyEventWrapper(device->JNIEnvAttachedToVM, keyAction, keyCode);
490  event.KeyInput.Char = keyEventWrapper->getUnicodeChar(keyMetaState);
491  delete keyEventWrapper;
492  }
493  }
494  if ( event.KeyInput.Key == KEY_BACK )
495  {
496  event.KeyInput.Char = 0x08; // same key-code as on other operating systems. Otherwise we have to handle too much system specific stuff in the editbox.
497  }
498  //os::Printer::log("char-code: ", core::stringc((int)event.KeyInput.Char).c_str(), ELL_DEBUG);
499 
500  }
501  else
502  {
503  // os::Printer::log("keyCode: ", core::stringc(keyCode).c_str(), ELL_DEBUG);
504  event.KeyInput.Char = 0;
505  }
506 
507  status = device->postEventFromUser(event);
508  }
509  break;
510  default:
511  break;
512  }
513 
514  return status;
515 }

517 void CIrrDeviceAndroid::createDriver()
518 {
519  switch(CreationParams.DriverType)
520  {
521  case video::EDT_OGLES1:
522 #ifdef _IRR_COMPILE_WITH_OGLES1_
523  VideoDriver = video::createOGLES1Driver(CreationParams, FileSystem, ContextManager);
524 #else
525  os::Printer::log("No OpenGL ES 1.0 support compiled in.", ELL_ERROR);
526 #endif
527  break;
528  case video::EDT_OGLES2:
529 #ifdef _IRR_COMPILE_WITH_OGLES2_
530  VideoDriver = video::createOGLES2Driver(CreationParams, FileSystem, ContextManager);
531 #else
532  os::Printer::log("No OpenGL ES 2.0 support compiled in.", ELL_ERROR);
533 #endif
534  break;
535  case video::EDT_NULL:
536  VideoDriver = video::createNullDriver(FileSystem, CreationParams.WindowSize);
537  break;
538  case video::EDT_SOFTWARE:
539  case video::EDT_BURNINGSVIDEO:
540  case video::EDT_OPENGL:
541  case video::DEPRECATED_EDT_DIRECT3D8_NO_LONGER_EXISTS:
542  case video::EDT_DIRECT3D9:
543  os::Printer::log("This driver is not available in Android. Try OpenGL ES 1.0 or ES 2.0.", ELL_ERROR);
544  break;
545  default:
546  os::Printer::log("Unable to create video driver of unknown type.", ELL_ERROR);
547  break;
548  }
