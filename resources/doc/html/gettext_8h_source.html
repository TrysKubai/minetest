95 template <typename ...Args>
96 inline std::string fmtgettext(const char *format, Args&&... args)
97 {
98  std::string buf;
99  std::size_t buf_size = 256;
100  buf.resize(buf_size);
101 
102  format = gettext(format);
103 
104  int len = porting::mt_snprintf(&buf[0], buf_size, format, std::forward<Args>(args)...);
105  if (len <= 0) throw std::runtime_error("gettext format error: " + std::string(format));
106  if ((size_t)len >= buf.size()) {
107  buf.resize(len+1); // extra null byte
108  porting::mt_snprintf(&buf[0], buf.size(), format, std::forward<Args>(args)...);
109  }
110  buf.resize(len); // remove null bytes
111 
112  return buf;
113 }
