344 void COpenGLExtensionHandler::initExtensions(bool stencilBuffer)
345 {
346  const f32 ogl_ver = core::fast_atof(reinterpret_cast<const c8*>(glGetString(GL_VERSION)));
347  Version = static_cast<u16>(core::floor32(ogl_ver)*100+core::round32(core::fract(ogl_ver)*10.0f));
348  if ( Version >= 102)
349  os::Printer::log("OpenGL driver version is 1.2 or better.", ELL_INFORMATION);
350  else
351  os::Printer::log("OpenGL driver version is not 1.2 or better.", ELL_WARNING);
352 
353  {
354  const char* t = reinterpret_cast<const char*>(glGetString(GL_EXTENSIONS));
355  size_t len = 0;
356  c8 *str = 0;
357  if (t)
358  {
359  len = strlen(t);
360  str = new c8[len+1];
361  }
362  c8* p = str;
363 
364  for (size_t i=0; i<len; ++i)
365  {
366  str[i] = static_cast<char>(t[i]);
367 
368  if (str[i] == ' ')
369  {
370  str[i] = 0;
371  for (u32 j=0; j<IRR_OpenGL_Feature_Count; ++j)
372  {
373  if (!strcmp(OpenGLFeatureStrings[j], p))
374  {
375  FeatureAvailable[j] = true;
376  break;
377  }
378  }
379 
380  p = p + strlen(p) + 1;
381  }
382  }
383 
384  delete [] str;
385  }
386 
387  TextureCompressionExtension = FeatureAvailable[IRR_ARB_texture_compression];
388  StencilBuffer=stencilBuffer;
389 
390  const char* renderer = (const char*)glGetString(GL_RENDERER);
391  if ( renderer )
392  {

394  }

402  // Accessing the correct function is quite complex
403  // All libraries should support the ARB version, however
404  // since GLX 1.4 the non-ARB version is the official one
405  // So we have to check the runtime environment and
406  // choose the proper symbol
407  // In case you still have problems please enable the
408  // next line by uncommenting it
409  // #define _IRR_GETPROCADDRESS_WORKAROUND_
410

413  #ifdef GLX_VERSION_1_4

418  IRR_OGL_LOAD_EXTENSION_FUNCP=glXGetProcAddress;
419  else
420  #endif

422  #define IRR_OGL_LOAD_EXTENSION(X) IRR_OGL_LOAD_EXTENSION_FUNCP(reinterpret_cast<const GLubyte*>(X))
423  #else

426 #endif // Windows, SDL, or Linux
