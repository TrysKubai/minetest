82 RenderingEngine::RenderingEngine(IEventReceiver *receiver)
83 {
84  sanity_check(!s_singleton);
85 
86  // Resolution selection
87  bool fullscreen = g_settings->getBool("fullscreen");
88 #ifdef __ANDROID__
89  u16 screen_w = 0, screen_h = 0;
90  bool window_maximized = false;
91 #else
92  u16 screen_w = std::max<u16>(g_settings->getU16("screen_w"), 1);
93  u16 screen_h = std::max<u16>(g_settings->getU16("screen_h"), 1);
94  bool window_maximized = g_settings->getBool("window_maximized");
95 #endif
96 
97  // bpp, fsaa, vsync
98  bool vsync = g_settings->getBool("vsync");
99  u16 fsaa = g_settings->getU16("fsaa");
100 
101  // Determine driver
102  video::E_DRIVER_TYPE driverType;
103  const std::string &driverstring = g_settings->get("video_driver");
104  std::vector<video::E_DRIVER_TYPE> drivers =
105  RenderingEngine::getSupportedVideoDrivers();
106  u32 i;
107  for (i = 0; i != drivers.size(); i++) {
108  auto &driverinfo = RenderingEngine::getVideoDriverInfo(drivers[i]);
109  if (!strcasecmp(driverstring.c_str(), driverinfo.name.c_str())) {
110  driverType = drivers[i];
111  break;
112  }
113  }
114  if (i == drivers.size()) {
115  driverType = drivers.at(0);
116  auto &name = RenderingEngine::getVideoDriverInfo(driverType).name;
117  if (driverstring.empty()) {
118  infostream << "Defaulting to video_driver = " << name << std::endl;
119  } else {
120  errorstream << "Invalid video_driver specified; defaulting to "
121  << name << std::endl;
122  }
123  }
124 
125  SIrrlichtCreationParameters params = SIrrlichtCreationParameters();
126  if (tracestream)
127  params.LoggingLevel = irr::ELL_DEBUG;
128  params.DriverType = driverType;
129  params.WindowSize = core::dimension2d<u32>(screen_w, screen_h);
130  params.AntiAlias = fsaa;
131  params.Fullscreen = fullscreen;
132  params.WindowMaximized = window_maximized;
133  params.WindowResizable = 1; // 1 means always (required for window_maximized)
134  params.Stencilbuffer = false;
135  params.Vsync = vsync;
136  params.EventReceiver = receiver;
137  params.HighPrecisionFPU = true;
138 #ifdef __ANDROID__
139  params.PrivateData = porting::app_global;
140 #endif
141  // there is no standardized path for these on desktop
142  std::string rel_path = std::string("client") + DIR_DELIM
143  + "shaders" + DIR_DELIM + "Irrlicht";
144  params.OGLES2ShaderPath = (porting::path_share + DIR_DELIM + rel_path + DIR_DELIM).c_str();
145 
146  m_device = createDeviceEx(params);
147  driver = m_device->getVideoDriver();
148 
149  s_singleton = this;
150 
151  auto skin = createSkin(m_device->getGUIEnvironment(),
152  gui::EGST_WINDOWS_METALLIC, driver);
153  m_device->getGUIEnvironment()->setSkin(skin);
154  skin->drop();
155 }

208 void RenderingEngine::setupTopLevelXorgWindow(const std::string &name)
209 {
210 #ifdef XORG_USED
211  const video::SExposedVideoData exposedData = driver->getExposedVideoData();
212 
213  Display *x11_dpl = reinterpret_cast<Display *>(exposedData.OpenGLLinux.X11Display);
214  if (x11_dpl == NULL) {
215  warningstream << "Client: Could not find X11 Display in ExposedVideoData"
216  << std::endl;
217  return;
218  }
219 
220  verbosestream << "Client: Configuring X11-specific top level"
221  << " window properties"
222  << std::endl;
223 
224 
225  Window x11_win = reinterpret_cast<Window>(exposedData.OpenGLLinux.X11Window);
226 
227  // Set application name and class hints. For now name and class are the same.
228  XClassHint *classhint = XAllocClassHint();
229  classhint->res_name = const_cast<char *>(name.c_str());
230  classhint->res_class = const_cast<char *>(name.c_str());
231 
232  XSetClassHint(x11_dpl, x11_win, classhint);
233  XFree(classhint);
234 
235  // FIXME: In the future WMNormalHints should be set ... e.g see the
236  // gtk/gdk code (gdk/x11/gdksurface-x11.c) for the setup_top_level
237  // method. But for now (as it would require some significant changes)
238  // leave the code as is.
239 
240  // The following is borrowed from the above gdk source for setting top
241  // level windows. The source indicates and the Xlib docs suggest that
242  // this will set the WM_CLIENT_MACHINE and WM_LOCAL_NAME. This will not
243  // set the WM_CLIENT_MACHINE to a Fully Qualified Domain Name (FQDN) which is
244  // required by the Extended Window Manager Hints (EWMH) spec when setting
245  // the _NET_WM_PID (see further down) but running Minetest in an env
246  // where the window manager is on another machine from Minetest (therefore
247  // making the PID useless) is not expected to be a problem. Further
248  // more, using gtk/gdk as the model it would seem that not using a FQDN is
249  // not an issue for modern Xorg window managers.
250 
251  verbosestream << "Client: Setting Xorg window manager Properties"
252  << std::endl;
253 
254  XSetWMProperties (x11_dpl, x11_win, NULL, NULL, NULL, 0, NULL, NULL, NULL);
255 
256  // Set the _NET_WM_PID window property according to the EWMH spec. _NET_WM_PID
257  // (in conjunction with WM_CLIENT_MACHINE) can be used by window managers to
258  // force a shutdown of an application if it doesn't respond to the destroy
259  // window message.
260 
261  verbosestream << "Client: Setting Xorg _NET_WM_PID extended window manager property"
262  << std::endl;
263 
264  Atom NET_WM_PID = XInternAtom(x11_dpl, "_NET_WM_PID", false);
265 
266  pid_t pid = getpid();
267 
268  XChangeProperty(x11_dpl, x11_win, NET_WM_PID,
269  XA_CARDINAL, 32, PropModeReplace,
270  reinterpret_cast<unsigned char *>(&pid),1);
271 
272  // Set the WM_CLIENT_LEADER window property here. Minetest has only one
273  // window and that window will always be the leader.
274 
275  verbosestream << "Client: Setting Xorg WM_CLIENT_LEADER property"
276  << std::endl;
277 
278  Atom WM_CLIENT_LEADER = XInternAtom(x11_dpl, "WM_CLIENT_LEADER", false);
279 
280  XChangeProperty (x11_dpl, x11_win, WM_CLIENT_LEADER,
281  XA_WINDOW, 32, PropModeReplace,
282  reinterpret_cast<unsigned char *>(&x11_win), 1);
283 #endif
284 }

305 bool RenderingEngine::setWindowIcon()
306 {

308 #if RUN_IN_PLACE
309  return setXorgWindowIconFromPath(
310  porting::path_share + "/misc/" PROJECT_NAME "-xorg-icon-128.png");
311 #else
312  // We have semi-support for reading in-place data if we are

319 #endif
320 #elif defined(_WIN32)
321  HWND hWnd; // Window handle
322  if (!getWindowHandle(driver, hWnd))
323  return false;
324 
325  // Load the ICON from resource file
326  const HICON hicon = LoadIcon(GetModuleHandle(NULL),
327  MAKEINTRESOURCE(130) // The ID of the ICON defined in
328  // winresource.rc
329  );
330 
331  if (hicon) {
332  SendMessage(hWnd, WM_SETICON, ICON_BIG, reinterpret_cast<LPARAM>(hicon));
333  SendMessage(hWnd, WM_SETICON, ICON_SMALL,
334  reinterpret_cast<LPARAM>(hicon));
335  return true;
336  }
337  return false;
338 #else
339  return false;

343 bool RenderingEngine::setXorgWindowIconFromPath(const std::string &icon_file)
344 {
345 #ifdef XORG_USED
346 
347  video::IImageLoader *image_loader = NULL;
348  u32 cnt = driver->getImageLoaderCount();
349  for (u32 i = 0; i < cnt; i++) {
350  if (driver->getImageLoader(i)->isALoadableFileExtension(
351  icon_file.c_str())) {
352  image_loader = driver->getImageLoader(i);
353  break;
354  }
355  }
356 
357  if (!image_loader) {
358  warningstream << "Could not find image loader for file '" << icon_file
359  << "'" << std::endl;
360  return false;
361  }
362 
363  io::IReadFile *icon_f =
364  m_device->getFileSystem()->createAndOpenFile(icon_file.c_str());
365 
366  if (!icon_f) {
367  warningstream << "Could not load icon file '" << icon_file << "'"
368  << std::endl;
369  return false;
370  }
371 
372  video::IImage *img = image_loader->loadImage(icon_f);
373 
374  if (!img) {
375  warningstream << "Could not load icon file '" << icon_file << "'"
376  << std::endl;
377  icon_f->drop();
378  return false;
379  }
380 
381  u32 height = img->getDimension().Height;
382  u32 width = img->getDimension().Width;
383 
384  size_t icon_buffer_len = 2 + height * width;
385  long *icon_buffer = new long[icon_buffer_len];
386 
387  icon_buffer[0] = width;
388  icon_buffer[1] = height;
389 
390  for (u32 x = 0; x < width; x++) {
391  for (u32 y = 0; y < height; y++) {
392  video::SColor col = img->getPixel(x, y);
393  long pixel_val = 0;
394  pixel_val |= (u8)col.getAlpha() << 24;
395  pixel_val |= (u8)col.getRed() << 16;
396  pixel_val |= (u8)col.getGreen() << 8;
397  pixel_val |= (u8)col.getBlue();
398  icon_buffer[2 + x + y * width] = pixel_val;
399  }
400  }
401 
402  img->drop();
403  icon_f->drop();
404 
405  const video::SExposedVideoData &video_data = driver->getExposedVideoData();
406 
407  Display *x11_dpl = (Display *)video_data.OpenGLLinux.X11Display;
408 
409  if (x11_dpl == NULL) {
410  warningstream << "Could not find x11 display for setting its icon."
411  << std::endl;
412  delete[] icon_buffer;
413  return false;
414  }
415 
416  Window x11_win = (Window)video_data.OpenGLLinux.X11Window;
417 
418  Atom net_wm_icon = XInternAtom(x11_dpl, "_NET_WM_ICON", False);
419  Atom cardinal = XInternAtom(x11_dpl, "CARDINAL", False);
420  XChangeProperty(x11_dpl, x11_win, net_wm_icon, cardinal, 32, PropModeReplace,
421  (const unsigned char *)icon_buffer, icon_buffer_len);
422 
423  delete[] icon_buffer;
424 
425 #endif
426  return true;
427 }

429 /*
430  Draws a screen with a single text on it.
431  Text will be removed when the screen is drawn the next time.
432  Additionally, a progressbar can be drawn when percent is set between 0 and 100.
433 */
434 void RenderingEngine::draw_load_screen(const std::wstring &text,
435  gui::IGUIEnvironment *guienv, ITextureSource *tsrc, float dtime,
436  int percent, bool sky)
437 {
438  v2u32 screensize = getWindowSize();
439 
440  v2s32 textsize(g_fontengine->getTextWidth(text), g_fontengine->getLineHeight());
441  v2s32 center(screensize.X / 2, screensize.Y / 2);
442  core::rect<s32> textrect(center - textsize / 2, center + textsize / 2);
443 
444  gui::IGUIStaticText *guitext =
445  guienv->addStaticText(text.c_str(), textrect, false, false);
446  guitext->setTextAlignment(gui::EGUIA_CENTER, gui::EGUIA_UPPERLEFT);
447 
448  if (sky && g_settings->getBool("menu_clouds")) {
449  g_menuclouds->step(dtime * 3);
450  g_menuclouds->render();
451  get_video_driver()->beginScene(true, true, RenderingEngine::MENU_SKY_COLOR);
452  g_menucloudsmgr->drawAll();
453  } else if (sky)
454  get_video_driver()->beginScene(true, true, RenderingEngine::MENU_SKY_COLOR);
455  else
456  get_video_driver()->beginScene(true, true, video::SColor(255, 0, 0, 0));
457 
458  // draw progress bar
459  if ((percent >= 0) && (percent <= 100)) {
460  video::ITexture *progress_img = tsrc->getTexture("progress_bar.png");
461  video::ITexture *progress_img_bg =
462  tsrc->getTexture("progress_bar_bg.png");
463 
464  if (progress_img && progress_img_bg) {
465 #ifndef __ANDROID__
466  const core::dimension2d<u32> &img_size =
467  progress_img_bg->getSize();
468  u32 imgW = rangelim(img_size.Width, 200, 600) * getDisplayDensity();
469  u32 imgH = rangelim(img_size.Height, 24, 72) * getDisplayDensity();
470 #else
471  const core::dimension2d<u32> img_size(256, 48);
472  float imgRatio = (float)img_size.Height / img_size.Width;
473  u32 imgW = screensize.X / 2.2f;
474  u32 imgH = floor(imgW * imgRatio);
475 #endif
476  v2s32 img_pos((screensize.X - imgW) / 2,
477  (screensize.Y - imgH) / 2);
478 
479  draw2DImageFilterScaled(get_video_driver(), progress_img_bg,
480  core::rect<s32>(img_pos.X, img_pos.Y,
481  img_pos.X + imgW,
482  img_pos.Y + imgH),
483  core::rect<s32>(0, 0, img_size.Width,
484  img_size.Height),
485  0, 0, true);
486 
487  draw2DImageFilterScaled(get_video_driver(), progress_img,
488  core::rect<s32>(img_pos.X, img_pos.Y,
489  img_pos.X + (percent * imgW) / 100,
490  img_pos.Y + imgH),
491  core::rect<s32>(0, 0,
492  (percent * img_size.Width) / 100,
493  img_size.Height),
494  0, 0, true);
495  }
496  }
497 
498  guienv->drawAll();
499  get_video_driver()->endScene();
500  guitext->remove();
501 }

503 std::vector<video::E_DRIVER_TYPE> RenderingEngine::getSupportedVideoDrivers()
504 {
505  // Only check these drivers. We do not support software and D3D in any capacity.
506  // Order by preference (best first)
507  static const video::E_DRIVER_TYPE glDrivers[] = {
508  video::EDT_OPENGL,
509  video::EDT_OGLES2,
510  video::EDT_OGLES1,
511  video::EDT_NULL,
512  };
513  std::vector<video::E_DRIVER_TYPE> drivers;
514 
515  for (u32 i = 0; i < ARRLEN(glDrivers); i++) {
516  if (IrrlichtDevice::isDriverSupported(glDrivers[i]))
517  drivers.push_back(glDrivers[i]);
518  }
519 
520  return drivers;
521 }

554 static float calcDisplayDensity()
555 {
556  const char *current_display = getenv("DISPLAY");
557 
558  if (current_display != NULL) {
559  Display *x11display = XOpenDisplay(current_display);
560 
561  if (x11display != NULL) {
562  /* try x direct */
563  int dh = DisplayHeight(x11display, 0);
564  int dw = DisplayWidth(x11display, 0);
565  int dh_mm = DisplayHeightMM(x11display, 0);
566  int dw_mm = DisplayWidthMM(x11display, 0);
567  XCloseDisplay(x11display);
568 
569  if (dh_mm != 0 && dw_mm != 0) {
570  float dpi_height = floor(dh / (dh_mm * 0.039370) + 0.5);
571  float dpi_width = floor(dw / (dw_mm * 0.039370) + 0.5);
572  return std::max(dpi_height, dpi_width) / 96.0;
573  }
574  }
575  }
576 
577  /* return manually specified dpi */
578  return g_settings->getFloat("screen_dpi") / 96.0;
579 }
