31 // To disable copy constructors and assignment operations for some class
32 // 'Foobar', add the macro DISABLE_CLASS_COPY(Foobar) in the class definition.
33 // Note this also disables copying for any classes derived from 'Foobar' as well
34 // as classes having a 'Foobar' member.
35 #define DISABLE_CLASS_COPY(C) \
36  C(const C &) = delete; \
37  C &operator=(const C &) = delete;
38 
39 // If you have used DISABLE_CLASS_COPY with a class but still want to permit moving
40 // use this macro to add the default move constructors back.
41 #define ALLOW_CLASS_MOVE(C) \
42  C(C &&other) = default; \
43  C &operator=(C &&) = default;
