45 /*
46  A cache from shader name to shader path
47 */
48 MutexedMap<std::string, std::string> g_shadername_to_path_cache;
49 
50 /*
51  Gets the path to a shader by first checking if the file
52  name_of_shader/filename
53  exists in shader_path and if not, using the data path.
54 
55  If not found, returns "".
56 
57  Utilizes a thread-safe cache.
58 */
59 std::string getShaderPath(const std::string &name_of_shader,
60  const std::string &filename)
61 {
62  std::string combined = name_of_shader + DIR_DELIM + filename;
63  std::string fullpath;
64  /*
65  Check from cache
66  */
67  bool incache = g_shadername_to_path_cache.get(combined, &fullpath);
68  if(incache)
69  return fullpath;
70 
71  /*
72  Check from shader_path
73  */
74  std::string shader_path = g_settings->get("shader_path");
75  if (!shader_path.empty()) {
76  std::string testpath = shader_path + DIR_DELIM + combined;
77  if(fs::PathExists(testpath))
78  fullpath = testpath;
79  }
80 
81  /*
82  Check from default data directory
83  */
84  if (fullpath.empty()) {
85  std::string rel_path = std::string("client") + DIR_DELIM
86  + "shaders" + DIR_DELIM
87  + name_of_shader + DIR_DELIM
88  + filename;
89  std::string testpath = porting::path_share + DIR_DELIM + rel_path;
90  if(fs::PathExists(testpath))
91  fullpath = testpath;
92  }
93 
94  // Add to cache (also an empty result is cached)
95  g_shadername_to_path_cache.set(combined, fullpath);
96 
97  // Finally return it
98  return fullpath;
99 }

101 /*
102  SourceShaderCache: A cache used for storing source shaders.
103 */
104 
105 class SourceShaderCache
106 {
107 public:
108  void insert(const std::string &name_of_shader, const std::string &filename,
109  const std::string &program, bool prefer_local)
110  {
111  std::string combined = name_of_shader + DIR_DELIM + filename;
112  // Try to use local shader instead if asked to
113  if(prefer_local){
114  std::string path = getShaderPath(name_of_shader, filename);
115  if(!path.empty()){
116  std::string p = readFile(path);
117  if (!p.empty()) {
118  m_programs[combined] = p;
119  return;
120  }
121  }
122  }
123  m_programs[combined] = program;
124  }
125 
126  std::string get(const std::string &name_of_shader,
127  const std::string &filename)
128  {
129  std::string combined = name_of_shader + DIR_DELIM + filename;
130  StringMap::iterator n = m_programs.find(combined);
131  if (n != m_programs.end())
132  return n->second;
133  return "";
134  }
135 
136  // Primarily fetches from cache, secondarily tries to read from filesystem
137  std::string getOrLoad(const std::string &name_of_shader,
138  const std::string &filename)
139  {
140  std::string combined = name_of_shader + DIR_DELIM + filename;
141  StringMap::iterator n = m_programs.find(combined);
142  if (n != m_programs.end())
143  return n->second;
144  std::string path = getShaderPath(name_of_shader, filename);
145  if (path.empty()) {
146  infostream << "SourceShaderCache::getOrLoad(): No path found for \""
147  << combined << "\"" << std::endl;
148  return "";
149  }
150  infostream << "SourceShaderCache::getOrLoad(): Loading path \""
151  << path << "\"" << std::endl;
152  std::string p = readFile(path);
153  if (!p.empty()) {
154  m_programs[combined] = p;
155  return p;
156  }
157  return "";
158  }
159 private:
160  StringMap m_programs;
161 
162  std::string readFile(const std::string &path)
163  {
164  std::ifstream is(path.c_str(), std::ios::binary);
165  if(!is.is_open())
166  return "";
167  std::ostringstream tmp_os;
168  tmp_os << is.rdbuf();
169  return tmp_os.str();
170  }

267 
268  // Set world matrix
269  core::matrix4 world = driver->getTransform(video::ETS_WORLD);
270  m_world.set(*reinterpret_cast<float(*)[16]>(world.pointer()), services);
271 
272  // Set clip matrix
273  core::matrix4 worldView;
274  worldView = driver->getTransform(video::ETS_VIEW);
275  worldView *= world;
276 
277  core::matrix4 worldViewProj;
278  worldViewProj = driver->getTransform(video::ETS_PROJECTION);
279  worldViewProj *= worldView;
280  m_world_view_proj.set(*reinterpret_cast<float(*)[16]>(worldViewProj.pointer()), services);
281 
282  if (driver->getDriverType() == video::EDT_OGLES2) {
283  core::matrix4 texture = driver->getTransform(video::ETS_TEXTURE_0);
284  m_world_view.set(*reinterpret_cast<float(*)[16]>(worldView.pointer()), services);
285  m_texture.set(*reinterpret_cast<float(*)[16]>(texture.pointer()), services);
286 
287  core::matrix4 normal;
288  worldView.getTransposed(normal);
289  sanity_check(normal.makeInverse());
290  float m[9] = {
291  normal[0], normal[1], normal[2],
292  normal[4], normal[5], normal[6],
293  normal[8], normal[9], normal[10],
294  };
295  m_normal.set(m, services);
296  }
297 
298  // Set uniforms for Shadow shader
299  if (ShadowRenderer *shadow = RenderingEngine::get_shadow_renderer()) {
300  const auto &light = shadow->getDirectionalLight();
301 
302  core::matrix4 shadowViewProj = light.getProjectionMatrix();
303  shadowViewProj *= light.getViewMatrix();
304  m_shadow_view_proj.set(shadowViewProj.pointer(), services);
305 
306  f32 v_LightDirection[3];
307  light.getDirection().getAs3Values(v_LightDirection);
308  m_light_direction.set(v_LightDirection, services);
309 
310  f32 TextureResolution = light.getMapResolution();
311  m_texture_res.set(&TextureResolution, services);
312 
313  f32 ShadowStrength = shadow->getShadowStrength();
314  m_shadow_strength.set(&ShadowStrength, services);
315 
316  f32 timeOfDay = shadow->getTimeOfDay();
317  m_time_of_day.set(&timeOfDay, services);
318 
319  f32 shadowFar = shadow->getMaxShadowFar();
320  m_shadowfar.set(&shadowFar, services);
321 
322  f32 cam_pos[4];
323  shadowViewProj.transformVect(cam_pos, light.getPlayerPos());
324  m_camera_pos.set(cam_pos, services);
325 
326  // I don't like using this hardcoded value. maybe something like
327  // MAX_TEXTURE - 1 or somthing like that??
328  s32 TextureLayerID = 3;
329  m_shadow_texture.set(&TextureLayerID, services);

333  m_perspective_bias0_pixel.set(&bias0, services);

382  MaterialType material_type, NodeDrawType drawtype) override;

385 
386  // Processes queued shader requests from other threads.
387  // Shall be called from the main thread.
388  void processQueue() override;
389 
390  // Insert a shader program into the cache without touching the
391  // filesystem. Shall be called from the main thread.
392  void insertSourceShader(const std::string &name_of_shader,
393  const std::string &filename, const std::string &program) override;
394 
395  // Rebuild shaders from the current set of source shaders
396  // Shall be called from the main thread.
397  void rebuildShaders() override;
398 
399  void addShaderConstantSetterFactory(IShaderConstantSetterFactory *setter) override
400  {
401  m_setter_factories.emplace_back(setter);
402  }
403 
404 private:
405 
406  // The id of the thread that is allowed to use irrlicht directly

446 u32 ShaderSource::getShader(const std::string &name,
447  MaterialType material_type, NodeDrawType drawtype)
448 {
449  /*
450  Get shader
451  */
452 
453  if (std::this_thread::get_id() == m_main_thread) {
454  return getShaderIdDirect(name, material_type, drawtype);
455  }
456 
457  /*errorstream<<"getShader(): Queued: name=\""<<name<<"\""<<std::endl;*/
458 
459  // We're gonna ask the result to be put into here
460 
461  static ResultQueue<std::string, u32, u8, u8> result_queue;
462 
463  // Throw a request in
464  m_get_shader_queue.add(name, 0, 0, &result_queue);
465 
466  /* infostream<<"Waiting for shader from main thread, name=\""
467  <<name<<"\""<<std::endl;*/
468 
469  while(true) {
470  GetResult<std::string, u32, u8, u8>
471  result = result_queue.pop_frontNoEx();
472 
473  if (result.key == name) {
474  return result.item;
475  }
476 
477  errorstream << "Got shader with invalid name: " << result.key << std::endl;
478  }
479 
480  infostream << "getShader(): Failed" << std::endl;
481 
482  return 0;
483 }

485 /*
486  This method generates all the shaders
487 */
488 u32 ShaderSource::getShaderIdDirect(const std::string &name,
489  MaterialType material_type, NodeDrawType drawtype)
490 {
491  //infostream<<"getShaderIdDirect(): name=\""<<name<<"\""<<std::endl;
492 
493  // Empty name means shader 0
494  if (name.empty()) {
495  infostream<<"getShaderIdDirect(): name is empty"<<std::endl;
496  return 0;
497  }
498 
499  // Check if already have such instance
500  for(u32 i=0; i<m_shaderinfo_cache.size(); i++){
501  ShaderInfo *info = &m_shaderinfo_cache[i];
502  if(info->name == name && info->material_type == material_type &&
503  info->drawtype == drawtype)
504  return i;
505  }
506 
507  /*
508  Calling only allowed from main thread
509  */
510  if (std::this_thread::get_id() != m_main_thread) {
511  errorstream<<"ShaderSource::getShaderIdDirect() "
512  "called not from main thread"<<std::endl;
513  return 0;
514  }
515 
516  ShaderInfo info = generateShader(name, material_type, drawtype);
517 
518  /*
519  Add shader to caches (add dummy shaders too)
520  */
521 
522  MutexAutoLock lock(m_shaderinfo_cache_mutex);
523 
524  u32 id = m_shaderinfo_cache.size();
525  m_shaderinfo_cache.push_back(info);
526 
527  infostream<<"getShaderIdDirect(): "
528  <<"Returning id="<<id<<" for name \""<<name<<"\""<<std::endl;
529 
530  return id;
531 }

586 ShaderInfo ShaderSource::generateShader(const std::string &name,
587  MaterialType material_type, NodeDrawType drawtype)
588 {
589  ShaderInfo shaderinfo;
590  shaderinfo.name = name;
591  shaderinfo.material_type = material_type;
592  shaderinfo.drawtype = drawtype;
593  switch (material_type) {
594  case TILE_MATERIAL_OPAQUE:
595  case TILE_MATERIAL_LIQUID_OPAQUE:
596  case TILE_MATERIAL_WAVING_LIQUID_OPAQUE:
597  shaderinfo.base_material = video::EMT_SOLID;
598  break;
599  case TILE_MATERIAL_ALPHA:
600  case TILE_MATERIAL_PLAIN_ALPHA:
601  case TILE_MATERIAL_LIQUID_TRANSPARENT:
602  case TILE_MATERIAL_WAVING_LIQUID_TRANSPARENT:
603  shaderinfo.base_material = video::EMT_TRANSPARENT_ALPHA_CHANNEL;
604  break;
605  case TILE_MATERIAL_BASIC:
606  case TILE_MATERIAL_PLAIN:
607  case TILE_MATERIAL_WAVING_LEAVES:
608  case TILE_MATERIAL_WAVING_PLANTS:
609  case TILE_MATERIAL_WAVING_LIQUID_BASIC:
610  shaderinfo.base_material = video::EMT_TRANSPARENT_ALPHA_CHANNEL_REF;
611  break;
612  }
613  shaderinfo.material = shaderinfo.base_material;
614 
615  bool enable_shaders = g_settings->getBool("enable_shaders");
616  if (!enable_shaders)
617  return shaderinfo;
618 
619  video::IVideoDriver *driver = RenderingEngine::get_video_driver();
620  if (!driver->queryFeature(video::EVDF_ARB_GLSL)) {
621  errorstream << "Shaders are enabled but GLSL is not supported by the driver\n";
622  return shaderinfo;
623  }
624  video::IGPUProgrammingServices *gpu = driver->getGPUProgrammingServices();
625 
626  // Create shaders header
627  bool use_gles = driver->getDriverType() == video::EDT_OGLES2;
628  std::stringstream shaders_header;
629  shaders_header
630  << std::noboolalpha
631  << std::showpoint // for GLSL ES
632  ;
633  std::string vertex_header, fragment_header, geometry_header;
634  if (use_gles) {
635  shaders_header << R"(
636  #version 100
637  )";
638  vertex_header = R"(
639  precision mediump float;
640 
641  uniform highp mat4 mWorldView;
642  uniform highp mat4 mWorldViewProj;
643  uniform mediump mat4 mTexture;
644  uniform mediump mat3 mNormal;
645 
646  attribute highp vec4 inVertexPosition;
647  attribute lowp vec4 inVertexColor;
648  attribute mediump vec4 inTexCoord0;
649  attribute mediump vec3 inVertexNormal;
650  attribute mediump vec4 inVertexTangent;
651  attribute mediump vec4 inVertexBinormal;
652  )";
653  fragment_header = R"(
654  precision mediump float;
655  )";
656  } else {
657  shaders_header << R"(
658  #version 120
659  #define lowp
660  #define mediump
661  #define highp
662  )";
663  vertex_header = R"(
664  #define mWorldView gl_ModelViewMatrix
665  #define mWorldViewProj gl_ModelViewProjectionMatrix
666  #define mTexture (gl_TextureMatrix[0])
667  #define mNormal gl_NormalMatrix
668 
669  #define inVertexPosition gl_Vertex
670  #define inVertexColor gl_Color
671  #define inTexCoord0 gl_MultiTexCoord0
672  #define inVertexNormal gl_Normal
673  #define inVertexTangent gl_MultiTexCoord1
674  #define inVertexBinormal gl_MultiTexCoord2
675  )";
676  }
677 
678  // map legacy semantic texture names to texture identifiers
679  fragment_header += R"(
680  #define baseTexture texture0
681  #define normalTexture texture1
682  #define textureFlags texture2
683  )";
684 
685  // Since this is the first time we're using the GL bindings be extra careful.
686  // This should be removed before 5.6.0 or similar.
687  if (!GL.GetString) {
688  errorstream << "OpenGL procedures were not loaded correctly, "
689  "please open a bug report with details about your platform/OS." << std::endl;
690  abort();
