313 void CGUISkin::draw3DButtonPaneStandard(IGUIElement* element,
314  const core::rect<s32>& r,
315  const core::rect<s32>* clip)
316 {
317  if (!Driver)
318  return;
319 
320  core::rect<s32> rect = r;
321 
322  if ( Type == EGST_BURNING_SKIN )
323  {
324  rect.UpperLeftCorner.X -= 1;
325  rect.UpperLeftCorner.Y -= 1;
326  rect.LowerRightCorner.X += 1;
327  rect.LowerRightCorner.Y += 1;
328  draw3DSunkenPane(element,
329  getColor( EGDC_WINDOW ).getInterpolated( 0xFFFFFFFF, 0.9f )
330  ,false, true, rect, clip);
331  return;
332  }
333 
334  Driver->draw2DRectangle(getColor(EGDC_3D_DARK_SHADOW), rect, clip);
335 
336  rect.LowerRightCorner.X -= 1;
337  rect.LowerRightCorner.Y -= 1;
338  Driver->draw2DRectangle(getColor(EGDC_3D_HIGH_LIGHT), rect, clip);
339 
340  rect.UpperLeftCorner.X += 1;
341  rect.UpperLeftCorner.Y += 1;
342  Driver->draw2DRectangle(getColor(EGDC_3D_SHADOW), rect, clip);
343 
344  rect.LowerRightCorner.X -= 1;
345  rect.LowerRightCorner.Y -= 1;
346 
347  if (!UseGradient)
348  {
349  Driver->draw2DRectangle(getColor(EGDC_3D_FACE), rect, clip);
350  }
351  else
352  {
353  const video::SColor c1 = getColor(EGDC_3D_FACE);
354  const video::SColor c2 = c1.getInterpolated(getColor(EGDC_3D_DARK_SHADOW), 0.4f);
355  Driver->draw2DRectangle(rect, c1, c1, c2, c2, clip);
356  }
357 }

413 void CGUISkin::draw3DSunkenPane(IGUIElement* element, video::SColor bgcolor,
414  bool flat, bool fillBackGround,
415  const core::rect<s32>& r,
416  const core::rect<s32>* clip)
417 {
418  if (!Driver)
419  return;
420 
421  core::rect<s32> rect = r;
422 
423  if (fillBackGround)
424  Driver->draw2DRectangle(bgcolor, rect, clip);
425 
426  if (flat)
427  {
428  // draw flat sunken pane
429 
430  rect.LowerRightCorner.Y = rect.UpperLeftCorner.Y + 1;
431  Driver->draw2DRectangle(getColor(EGDC_3D_SHADOW), rect, clip); // top
432 
433  ++rect.UpperLeftCorner.Y;
434  rect.LowerRightCorner.Y = r.LowerRightCorner.Y;
435  rect.LowerRightCorner.X = rect.UpperLeftCorner.X + 1;
436  Driver->draw2DRectangle(getColor(EGDC_3D_SHADOW), rect, clip); // left
437 
438  rect = r;
439  ++rect.UpperLeftCorner.Y;
440  rect.UpperLeftCorner.X = rect.LowerRightCorner.X - 1;
441  Driver->draw2DRectangle(getColor(EGDC_3D_HIGH_LIGHT), rect, clip); // right
442 
443  rect = r;
444  ++rect.UpperLeftCorner.X;
445  rect.UpperLeftCorner.Y = r.LowerRightCorner.Y - 1;
446  --rect.LowerRightCorner.X;
447  Driver->draw2DRectangle(getColor(EGDC_3D_HIGH_LIGHT), rect, clip); // bottom
448  }
449  else
450  {
451  // draw deep sunken pane
452  rect.LowerRightCorner.Y = rect.UpperLeftCorner.Y + 1;
453  Driver->draw2DRectangle(getColor(EGDC_3D_SHADOW), rect, clip); // top
454  ++rect.UpperLeftCorner.X;
455  ++rect.UpperLeftCorner.Y;
456  --rect.LowerRightCorner.X;
457  ++rect.LowerRightCorner.Y;
458  Driver->draw2DRectangle(getColor(EGDC_3D_DARK_SHADOW), rect, clip);
459 
460  rect.UpperLeftCorner.X = r.UpperLeftCorner.X;
461  rect.UpperLeftCorner.Y = r.UpperLeftCorner.Y+1;
462  rect.LowerRightCorner.X = rect.UpperLeftCorner.X + 1;
463  rect.LowerRightCorner.Y = r.LowerRightCorner.Y;
464  Driver->draw2DRectangle(getColor(EGDC_3D_SHADOW), rect, clip); // left
465  ++rect.UpperLeftCorner.X;
466  ++rect.UpperLeftCorner.Y;
467  ++rect.LowerRightCorner.X;
468  --rect.LowerRightCorner.Y;
469  Driver->draw2DRectangle(getColor(EGDC_3D_DARK_SHADOW), rect, clip);
470 
471  rect = r;
472  rect.UpperLeftCorner.X = rect.LowerRightCorner.X - 1;
473  ++rect.UpperLeftCorner.Y;
474  Driver->draw2DRectangle(getColor(EGDC_3D_HIGH_LIGHT), rect, clip); // right
475  --rect.UpperLeftCorner.X;
476  ++rect.UpperLeftCorner.Y;
477  --rect.LowerRightCorner.X;
478  --rect.LowerRightCorner.Y;
479  Driver->draw2DRectangle(getColor(EGDC_3D_LIGHT), rect, clip);
480 
481  rect = r;
482  ++rect.UpperLeftCorner.X;
483  rect.UpperLeftCorner.Y = r.LowerRightCorner.Y - 1;
484  --rect.LowerRightCorner.X;
485  Driver->draw2DRectangle(getColor(EGDC_3D_HIGH_LIGHT), rect, clip); // bottom
486  ++rect.UpperLeftCorner.X;
487  --rect.UpperLeftCorner.Y;
488  --rect.LowerRightCorner.X;
489  --rect.LowerRightCorner.Y;
490  Driver->draw2DRectangle(getColor(EGDC_3D_LIGHT), rect, clip);
491  }
492 }

496 // return where to draw title bar text.
497 core::rect<s32> CGUISkin::draw3DWindowBackground(IGUIElement* element,
498  bool drawTitleBar, video::SColor titleBarColor,
499  const core::rect<s32>& r,
500  const core::rect<s32>* clip,
501  core::rect<s32>* checkClientArea)
502 {
503  if (!Driver)
504  {
505  if ( checkClientArea )
506  {
507  *checkClientArea = r;
508  }
509  return r;
510  }
511 
512  core::rect<s32> rect = r;
513 
514  // top border
515  rect.LowerRightCorner.Y = rect.UpperLeftCorner.Y + 1;
516  if ( !checkClientArea )
517  {
518  Driver->draw2DRectangle(getColor(EGDC_3D_HIGH_LIGHT), rect, clip);
519  }
520 
521  // left border
522  rect.LowerRightCorner.Y = r.LowerRightCorner.Y;
523  rect.LowerRightCorner.X = rect.UpperLeftCorner.X + 1;
524  if ( !checkClientArea )
525  {
526  Driver->draw2DRectangle(getColor(EGDC_3D_HIGH_LIGHT), rect, clip);
527  }
528 
529  // right border dark outer line
530  rect.UpperLeftCorner.X = r.LowerRightCorner.X - 1;
531  rect.LowerRightCorner.X = r.LowerRightCorner.X;
532  rect.UpperLeftCorner.Y = r.UpperLeftCorner.Y;
533  rect.LowerRightCorner.Y = r.LowerRightCorner.Y;
534  if ( !checkClientArea )
535  {
536  Driver->draw2DRectangle(getColor(EGDC_3D_DARK_SHADOW), rect, clip);
537  }
538 
539  // right border bright innner line
540  rect.UpperLeftCorner.X -= 1;
541  rect.LowerRightCorner.X -= 1;
542  rect.UpperLeftCorner.Y += 1;
543  rect.LowerRightCorner.Y -= 1;
544  if ( !checkClientArea )
545  {
546  Driver->draw2DRectangle(getColor(EGDC_3D_SHADOW), rect, clip);
547  }
548 
549  // bottom border dark outer line
550  rect.UpperLeftCorner.X = r.UpperLeftCorner.X;
551  rect.UpperLeftCorner.Y = r.LowerRightCorner.Y - 1;
552  rect.LowerRightCorner.Y = r.LowerRightCorner.Y;
553  rect.LowerRightCorner.X = r.LowerRightCorner.X;
554  if ( !checkClientArea )
555  {
556  Driver->draw2DRectangle(getColor(EGDC_3D_DARK_SHADOW), rect, clip);
557  }
558 
559  // bottom border bright inner line
560  rect.UpperLeftCorner.X += 1;
561  rect.LowerRightCorner.X -= 1;
562  rect.UpperLeftCorner.Y -= 1;
563  rect.LowerRightCorner.Y -= 1;
564  if ( !checkClientArea )
565  {
566  Driver->draw2DRectangle(getColor(EGDC_3D_SHADOW), rect, clip);
567  }
568 
569  // client area for background
570  rect = r;
571  rect.UpperLeftCorner.X +=1;
572  rect.UpperLeftCorner.Y +=1;
573  rect.LowerRightCorner.X -= 2;
574  rect.LowerRightCorner.Y -= 2;
575  if (checkClientArea)
576  {
577  *checkClientArea = rect;
578  }
579 
580  if ( !checkClientArea )
581  {
582  if (!UseGradient)
583  {
584  Driver->draw2DRectangle(getColor(EGDC_3D_FACE), rect, clip);
585  }
586  else if ( Type == EGST_BURNING_SKIN )
587  {
588  const video::SColor c1 = getColor(EGDC_WINDOW).getInterpolated ( 0xFFFFFFFF, 0.9f );
589  const video::SColor c2 = getColor(EGDC_WINDOW).getInterpolated ( 0xFFFFFFFF, 0.8f );
590 
591  Driver->draw2DRectangle(rect, c1, c1, c2, c2, clip);
592  }
593  else
594  {
595  const video::SColor c2 = getColor(EGDC_3D_SHADOW);
596  const video::SColor c1 = getColor(EGDC_3D_FACE);
597  Driver->draw2DRectangle(rect, c1, c1, c1, c2, clip);
598  }
599  }
600 
601  // title bar
602  rect = r;
603  rect.UpperLeftCorner.X += 2;
604  rect.UpperLeftCorner.Y += 2;
605  rect.LowerRightCorner.X -= 2;
606  rect.LowerRightCorner.Y = rect.UpperLeftCorner.Y + getSize(EGDS_WINDOW_BUTTON_WIDTH) + 2;
607 
608  if (drawTitleBar )
609  {
610  if (checkClientArea)
611  {
612  (*checkClientArea).UpperLeftCorner.Y = rect.LowerRightCorner.Y;
613  }
614  else
615  {
