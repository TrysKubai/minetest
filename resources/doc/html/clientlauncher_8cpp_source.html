92  * - Singleplayer (address but map provided)

94  * - Local server (for main menu only)
95  */

97  init_args(start_data, cmd_args);
98

189  /*
190  GUI stuff

193  ChatBackend chat_backend;

195  // If an error occurs, this is set to something by menu().
196  // It is then displayed before the menu shows on the next call to menu()

198  bool reconnect_requested = false;
199 
200  bool first_loop = true;
201 
202  /*
203  Menu-game loop
204  */
205  bool retval = true;
206  bool *kill = porting::signal_handler_killstatus();
207 
208  while (m_rendering_engine->run() && !*kill &&
209  !g_gamecallback->shutdown_requested) {

216  try { // This is used for catching disconnects
217 
218  m_rendering_engine->get_gui_env()->clear();
219

222  custom gui elements directly on the screen.
223  Otherwise they won't be automatically drawn.
224  */

226  core::rect<s32>(0, 0, 10000, 10000));
227

229  start_data, cmd_args);
230 
231  // Reset the reconnect_requested flag
232  reconnect_requested = false;
233 
234  // If skip_main_menu, we only want to startup once
235  if (skip_main_menu && !first_loop)
236  break;
237 
238  first_loop = false;
239 
240  if (!game_has_run) {
241  if (skip_main_menu)
242  break;
243 
244  continue;
245  }
246 
247  // Break out of menu-game loop to shut down cleanly
248  if (!m_rendering_engine->run() || *kill) {
249  if (!g_settings_path.empty())
250  g_settings->updateConfigFile(g_settings_path.c_str());
251  break;
252  }
253 
254  m_rendering_engine->get_video_driver()->setTextureCreationFlag(
255  video::ETCF_CREATE_MIP_MAPS, g_settings->getBool("mip_map"));
256 
257 #ifdef HAVE_TOUCHSCREENGUI
258  receiver->m_touchscreengui = new TouchScreenGUI(m_rendering_engine->get_raw_device(), receiver);
259  g_touchscreengui = receiver->m_touchscreengui;
260 #endif
261 
262  the_game(
263  kill,
264  input,
265  m_rendering_engine,
266  start_data,
267  error_message,
268  chat_backend,
269  &reconnect_requested
270  );
271  } //try
272  catch (con::PeerNotFoundException &e) {
273  error_message = gettext("Connection error (timed out?)");
274  errorstream << error_message << std::endl;
275  }
276 
277 #ifdef NDEBUG
278  catch (std::exception &e) {
279  std::string error_message = "Some exception: \"";
280  error_message += e.what();
281  error_message += "\"";
282  errorstream << error_message << std::endl;
283  }
284 #endif
285 
286  m_rendering_engine->get_scene_manager()->clear();
287 
288 #ifdef HAVE_TOUCHSCREENGUI
289  delete g_touchscreengui;
290  g_touchscreengui = NULL;
291  receiver->m_touchscreengui = NULL;
292 #endif
293 
294  // If no main menu, show error and exit
295  if (skip_main_menu) {
296  if (!error_message.empty()) {
297  verbosestream << "error_message = "
298  << error_message << std::endl;
299  retval = false;
300  }
301  break;
302  }
303  } // Menu-game loop
304 
305  g_menuclouds->drop();
306  g_menucloudsmgr->drop();
307 
308  return retval;
309 }

368 bool ClientLauncher::launch_game(std::string &error_message,
369  bool reconnect_requested, GameStartData &start_data,
370  const Settings &cmd_args)
371 {
372  // Prepare and check the start data to launch a game
373  std::string error_message_lua = error_message;
374  error_message.clear();
375 
376  if (cmd_args.exists("password"))
377  start_data.password = cmd_args.get("password");
378 
379  if (cmd_args.exists("password-file")) {
380  std::ifstream passfile(cmd_args.get("password-file"));
381  if (passfile.good()) {
382  getline(passfile, start_data.password);
383  } else {
384  error_message = gettext("Provided password file "
385  "failed to open: ")
386  + cmd_args.get("password-file");
387  errorstream << error_message << std::endl;
388  return false;
389  }
390  }
391 
392  // If a world was commanded, append and select it
393  // This is provieded by "get_world_from_cmdline()", main.cpp
394  if (!start_data.world_path.empty()) {
395  auto &spec = start_data.world_spec;
396 
397  spec.path = start_data.world_path;
398  spec.gameid = getWorldGameId(spec.path, true);
399  spec.name = _("[--world parameter]");
400 
401  if (spec.gameid.empty()) { // Create new
402  spec.gameid = g_settings->get("default_game");
403  spec.name += " [new]";
404  }
405  }
406 
407  /* Show the GUI menu
408  */
409  std::string server_name, server_description;
410  if (!skip_main_menu) {
411  // Initialize menu data
412  // TODO: Re-use existing structs (GameStartData)
413  MainMenuData menudata;
414  menudata.address = start_data.address;
415  menudata.name = start_data.name;
416  menudata.password = start_data.password;
417  menudata.port = itos(start_data.socket_port);
418  menudata.script_data.errormessage = std::move(error_message_lua);
419  menudata.script_data.reconnect_requested = reconnect_requested;
420 
421  main_menu(&menudata);
422 
423  // Skip further loading if there was an exit signal.
424  if (*porting::signal_handler_killstatus())
425  return false;
426 
427  if (!menudata.script_data.errormessage.empty()) {
428  /* The calling function will pass this back into this function upon the
429  * next iteration (if any) causing it to be displayed by the GUI
430  */
431  error_message = menudata.script_data.errormessage;
432  return false;
433  }
434 
435  int newport = stoi(menudata.port);
436  if (newport != 0)
437  start_data.socket_port = newport;
438 
439  // Update world information using main menu data
440  std::vector<WorldSpec> worldspecs = getAvailableWorlds();
441 
442  int world_index = menudata.selected_world;
443  if (world_index >= 0 && world_index < (int)worldspecs.size()) {
444  g_settings->set("selected_world_path",
445  worldspecs[world_index].path);
446  start_data.world_spec = worldspecs[world_index];
447  }
448 
449  start_data.name = menudata.name;
450  start_data.password = menudata.password;
451  start_data.address = std::move(menudata.address);
452  start_data.allow_login_or_register = menudata.allow_login_or_register;
453  server_name = menudata.servername;
454  server_description = menudata.serverdescription;
455 
456  start_data.local_server = !menudata.simple_singleplayer_mode &&
457  start_data.address.empty();
458  } else {
459  start_data.local_server = !start_data.world_path.empty() &&
460  start_data.address.empty() && !start_data.name.empty();
461  }
462 
463  if (!m_rendering_engine->run())
464  return false;
465 
466  if (!start_data.isSinglePlayer() && start_data.name.empty()) {
467  error_message = gettext("Please choose a name!");
468  errorstream << error_message << std::endl;
469  return false;
470  }
471 
472  // If using simple singleplayer mode, override
473  if (start_data.isSinglePlayer()) {
474  start_data.name = "singleplayer";
475  start_data.password = "";
476  start_data.socket_port = myrand_range(49152, 65535);
477  } else {
478  g_settings->set("name", start_data.name);
479  }
480 
481  if (start_data.name.length() > PLAYERNAME_SIZE - 1) {
482  error_message = gettext("Player name too long.");
483  start_data.name.resize(PLAYERNAME_SIZE);
484  g_settings->set("name", start_data.name);
485  return false;
486  }
487 
488  auto &worldspec = start_data.world_spec;
489  infostream << "Selected world: " << worldspec.name
490  << " [" << worldspec.path << "]" << std::endl;
491 
492  if (start_data.address.empty()) {
493  // For singleplayer and local server
494  if (worldspec.path.empty()) {
495  error_message = gettext("No world selected and no address "
496  "provided. Nothing to do.");
497  errorstream << error_message << std::endl;
498  return false;
499  }
500 
501  if (!fs::PathExists(worldspec.path)) {
502  error_message = gettext("Provided world path doesn't exist: ")
503  + worldspec.path;
504  errorstream << error_message << std::endl;
505  return false;
506  }
507 
508  // Load gamespec for required game
509  start_data.game_spec = findWorldSubgame(worldspec.path);
510  if (!start_data.game_spec.isValid()) {
511  error_message = gettext("Could not find or load game: ")
512  + worldspec.gameid;
513  errorstream << error_message << std::endl;
514  return false;
515  }
516 
517  if (porting::signal_handler_killstatus())
518  return true;
519 
520  if (!start_data.game_spec.isValid()) {
521  error_message = gettext("Invalid gamespec.");
522  error_message += " (world.gameid=" + worldspec.gameid + ")";
523  errorstream << error_message << std::endl;
524  return false;
525  }
526  }
527 
528  start_data.world_path = start_data.world_spec.path;
529  return true;
530 }

567  volatile static s16 temp16;
568  volatile static f32 tempf;

581  {

583  TimeTaker timer("Testing std::string speed");
584  const u32 jj = 10000;
585  for (u32 j = 0; j < jj; j++) {
586  tempstring.clear();
587  tempstring2.clear();
588  const u32 ii = 10;
589  for (u32 i = 0; i < ii; i++) {
590  tempstring2 += "asd";
591  }

593  tempstring += "asd";
594  if (tempstring == tempstring2)
595  break;
596  }
597  }
598  }
599 
600  infostream << "All of the following tests should take around 100ms each."
601  << std::endl;
602 
603  {
604  TimeTaker timer("Testing floating-point conversion speed");
605  tempf = 0.001;
606  for (u32 i = 0; i < 4000000; i++) {
607  temp16 += tempf;
608  tempf += 0.001;
609  }
610  }
611 
612  {
613  TimeTaker timer("Testing floating-point vector speed");
614 
615  tempv3f1 = v3f(1, 2, 3);
616  tempv3f2 = v3f(4, 5, 6);
617  for (u32 i = 0; i < 10000000; i++) {
618  tempf += tempv3f1.dotProduct(tempv3f2);
619  tempv3f2 += v3f(7, 8, 9);
620  }
621  }
622 
623  {
624  TimeTaker timer("Testing std::map speed");
625 
626  std::map<v2s16, f32> map1;
627  tempf = -324;
628  const s16 ii = 300;
629  for (s16 y = 0; y < ii; y++) {
630  for (s16 x = 0; x < ii; x++) {
631  map1[v2s16(x, y)] = tempf;
632  tempf += 1;
633  }
634  }
635  for (s16 y = ii - 1; y >= 0; y--) {
636  for (s16 x = 0; x < ii; x++) {
637  tempf = map1[v2s16(x, y)];
638  }
639  }
640  }
641 
642  {
643  infostream << "Around 5000/ms should do well here." << std::endl;
644  TimeTaker timer("Testing mutex speed");
645 
646  std::mutex m;
647  u32 n = 0;
648  u32 i = 0;
649  do {
650  n += 10000;
651  for (; i < n; i++) {
652  m.lock();
653  m.unlock();
654  }
655  }
656  // Do at least 10ms
657  while(timer.getTimerTime() < 10);
658 
659  u32 dtime = timer.stop();
660  u32 per_ms = n / dtime;
661  infostream << "Done. " << dtime << "ms, " << per_ms << "/ms" << std::endl;
662  }
