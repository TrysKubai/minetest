94 
95  SEvent irrevent;
96

98  irrevent.MouseInput.Event = irr::EMIE_MOUSE_ENTER_CANVAS;

196 int CIrrDeviceSDL::findCharToPassToIrrlicht(int assumedChar, EKEY_CODE key) {
197  // SDL in-place ORs values with no character representation with 1<<30
198  // https://wiki.libsdl.org/SDL2/SDLKeycodeLookup
199  if (assumedChar & (1<<30))
200  return 0;
201 
202  switch (key) {
203  case KEY_PRIOR:
204  case KEY_NEXT:
205  case KEY_HOME:
206  case KEY_END:
207  case KEY_LEFT:
208  case KEY_UP:
209  case KEY_RIGHT:
210  case KEY_DOWN:
211  case KEY_NUMLOCK:
212  return 0;
213  default:
214  return assumedChar;
215  }

227 CIrrDeviceSDL::CIrrDeviceSDL(const SIrrlichtCreationParameters& param)
228  : CIrrDeviceStub(param),
229  Window((SDL_Window*)param.WindowId), SDL_Flags(0),
230  MouseX(0), MouseY(0), MouseXRel(0), MouseYRel(0), MouseButtonStates(0),
231  Width(param.WindowSize.Width), Height(param.WindowSize.Height),
232  Resizable(param.WindowResizable == 1 ? true : false)
233 {
234  #ifdef _DEBUG
235  setDebugName("CIrrDeviceSDL");
236  #endif
237 
238  if ( ++SDLDeviceInstances == 1 )
239  {
240  u32 flags = SDL_INIT_TIMER | SDL_INIT_EVENTS;
241  if (CreationParams.DriverType != video::EDT_NULL)
242  flags |= SDL_INIT_VIDEO;
243 #if defined(_IRR_COMPILE_WITH_JOYSTICK_EVENTS_)
244  flags |= SDL_INIT_JOYSTICK;
245 #endif
246  if (SDL_Init(flags) < 0)
247  {
248  os::Printer::log("Unable to initialize SDL", SDL_GetError(), ELL_ERROR);
249  Close = true;
250  }
251  else
252  {
253  os::Printer::log("SDL initialized", ELL_INFORMATION);
254  }
255  }
256 
257  // create keymap
258  createKeyMap();
259 
260  // create window
261  if (CreationParams.DriverType != video::EDT_NULL)
262  {
263  // create the window, only if we do not use the null device
264  createWindow();
265  }
266 
267 
268  SDL_VERSION(&Info.version);
269 
270 #ifndef _IRR_EMSCRIPTEN_PLATFORM_
271  SDL_GetWindowWMInfo(Window,&Info);
272 #endif //_IRR_EMSCRIPTEN_PLATFORM_
273  core::stringc sdlversion = "SDL Version ";
274  sdlversion += Info.version.major;
275  sdlversion += ".";
276  sdlversion += Info.version.minor;
277  sdlversion += ".";
278  sdlversion += Info.version.patch;
279 
280  Operator = new COSOperator(sdlversion);

282  os::Printer::log(sdlversion.c_str(), ELL_INFORMATION);
283  }
284 
285  // create cursor control
286  CursorControl = new CCursorControl(this);
287 
288  // create driver
289  createDriver();
290 
291  if (VideoDriver) {
292  createGUIAndScene();
293  VideoDriver->OnResize(core::dimension2d<u32>(Width, Height));

347 bool CIrrDeviceSDL::createWindow()
348 {
349  if (CreationParams.Fullscreen) {
350 #ifdef _IRR_EMSCRIPTEN_PLATFORM_
351  SDL_Flags |= SDL_WINDOW_FULLSCREEN;
352 #else
353  SDL_Flags |= SDL_WINDOW_FULLSCREEN_DESKTOP;
354 #endif
355  }
356  if (Resizable)
357  SDL_Flags |= SDL_WINDOW_RESIZABLE;
358  if (CreationParams.WindowMaximized)
359  SDL_Flags |= SDL_WINDOW_MAXIMIZED;
360  SDL_Flags |= SDL_WINDOW_OPENGL;
361 
362 #ifdef _IRR_EMSCRIPTEN_PLATFORM_
363  if ( Width != 0 || Height != 0 )
364  emscripten_set_canvas_size( Width, Height);
365  else
366  {
367  int w, h, fs;
368  emscripten_get_canvas_size(&w, &h, &fs);
369  Width = w;
370  Height = h;
371  }
372 
373  SDL_GL_SetAttribute( SDL_GL_RED_SIZE, 8 );
374  SDL_GL_SetAttribute( SDL_GL_GREEN_SIZE, 8 );
375  SDL_GL_SetAttribute( SDL_GL_BLUE_SIZE, 8 );
376  SDL_GL_SetAttribute( SDL_GL_ALPHA_SIZE, CreationParams.WithAlphaChannel?8:0 );
377 
378  SDL_GL_SetAttribute( SDL_GL_DEPTH_SIZE, CreationParams.ZBufferBits);
379  SDL_GL_SetAttribute( SDL_GL_STENCIL_SIZE, CreationParams.Stencilbuffer ? 8 : 0);
380  SDL_GL_SetAttribute( SDL_GL_DOUBLEBUFFER, CreationParams.Doublebuffer ? 1 : 0);
381 
382  if (CreationParams.AntiAlias>1)
383  {
384  SDL_GL_SetAttribute( SDL_GL_MULTISAMPLEBUFFERS, 1 );
385  SDL_GL_SetAttribute( SDL_GL_MULTISAMPLESAMPLES, CreationParams.AntiAlias );
386  }
387  else
388  {
389  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, 0);
390  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, 0);
391  }
392 
393  SDL_CreateWindowAndRenderer(0, 0, SDL_Flags, &Window, &Renderer); // 0,0 will use the canvas size
394 
395  logAttributes();
396 
397  // "#canvas" is for the opengl context
398  emscripten_set_mousedown_callback("#canvas", (void*)this, true, MouseUpDownCallback);
399  emscripten_set_mouseup_callback("#canvas", (void*)this, true, MouseUpDownCallback);
400  emscripten_set_mouseenter_callback("#canvas", (void*)this, false, MouseEnterCallback);
401  emscripten_set_mouseleave_callback("#canvas", (void*)this, false, MouseLeaveCallback);
402 
403  return true;
404 #else // !_IRR_EMSCRIPTEN_PLATFORM_
405  if ( Close )
406  return false;
407 
408  switch (CreationParams.DriverType) {
409  case video::EDT_OPENGL:
410  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 2);
411  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 1);
412  break;
413  case video::EDT_OPENGL3:
414  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 3);
415  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 2);
416  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_COMPATIBILITY);
417  break;
418  case video::EDT_OGLES1:
419  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 1);
420  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 1);
421  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_ES);
422  break;
423  case video::EDT_OGLES2:
424  case video::EDT_WEBGL1:
425  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MAJOR_VERSION, 2);
426  SDL_GL_SetAttribute(SDL_GL_CONTEXT_MINOR_VERSION, 0);
427  SDL_GL_SetAttribute(SDL_GL_CONTEXT_PROFILE_MASK, SDL_GL_CONTEXT_PROFILE_ES);
428  break;
429  default:;
430  }
431 
432 #ifdef _DEBUG
433  SDL_GL_SetAttribute(SDL_GL_CONTEXT_FLAGS, SDL_GL_CONTEXT_DEBUG_FLAG | SDL_GL_CONTEXT_ROBUST_ACCESS_FLAG);
434 #endif
435 
436  if (CreationParams.Bits == 16) {
437  SDL_GL_SetAttribute(SDL_GL_RED_SIZE, 5);
438  SDL_GL_SetAttribute(SDL_GL_GREEN_SIZE, 5);
439  SDL_GL_SetAttribute(SDL_GL_BLUE_SIZE, 5);
440  SDL_GL_SetAttribute(SDL_GL_ALPHA_SIZE, CreationParams.WithAlphaChannel ? 1 : 0);
441  } else {
442  SDL_GL_SetAttribute(SDL_GL_RED_SIZE, 8);
443  SDL_GL_SetAttribute(SDL_GL_GREEN_SIZE, 8);
444  SDL_GL_SetAttribute(SDL_GL_BLUE_SIZE, 8);
445  SDL_GL_SetAttribute(SDL_GL_ALPHA_SIZE, CreationParams.WithAlphaChannel ? 8 : 0);
446  }
447  SDL_GL_SetAttribute(SDL_GL_DEPTH_SIZE, CreationParams.ZBufferBits);
448  SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, CreationParams.Doublebuffer);
449  SDL_GL_SetAttribute(SDL_GL_STENCIL_SIZE, CreationParams.Stencilbuffer ? 8 : 0);
450  SDL_GL_SetAttribute(SDL_GL_STEREO, CreationParams.Stereobuffer);
451  if (CreationParams.AntiAlias > 1) {
452  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, 1);
453  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, CreationParams.AntiAlias);
454  }
455  if (!Window)
456  Window = SDL_CreateWindow("", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, Width, Height, SDL_Flags);
457  if (!Window) {
458  os::Printer::log("Could not create window...", SDL_GetError(), ELL_WARNING);
459  }
460  if (!Window && CreationParams.AntiAlias > 1) {
461  while (--CreationParams.AntiAlias > 1) {
462  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, CreationParams.AntiAlias);
463  Window = SDL_CreateWindow("", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, Width, Height, SDL_Flags);
464  if (Window)
465  break;
466  }
467  if (!Window) {
468  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLEBUFFERS, 0);
469  SDL_GL_SetAttribute(SDL_GL_MULTISAMPLESAMPLES, 0);
470  Window = SDL_CreateWindow("", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, Width, Height, SDL_Flags);
471  if (Window)
472  os::Printer::log("AntiAliasing disabled due to lack of support!", ELL_WARNING);
473  }
474  }
475 
476  if ( !Window && CreationParams.Doublebuffer)
477  {
478  // Try single buffer
479  if (CreationParams.DriverType == video::EDT_OPENGL)
480  SDL_GL_SetAttribute(SDL_GL_DOUBLEBUFFER, 1);
481  Window = SDL_CreateWindow("", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, Width, Height, SDL_Flags);
482  }
483  if ( !Window )
484  {
485  os::Printer::log("Could not initialize display", SDL_GetError(), ELL_ERROR);
486  return false;
487  }
488 
489  Context = SDL_GL_CreateContext(Window);
490  if (!Context) {
491  os::Printer::log("Could not initialize context", SDL_GetError(), ELL_ERROR);
492  SDL_DestroyWindow(Window);
493  return false;
494  }
495 
496  return true;
497 #endif // !_IRR_EMSCRIPTEN_PLATFORM_
498 }

524 bool CIrrDeviceSDL::run()
525 {
526  os::Timer::tick();
527 
528  SEvent irrevent;
529  SDL_Event SDL_event;
530 
531  while ( !Close && SDL_PollEvent( &SDL_event ) )
532  {
533  // os::Printer::log("event: ", core::stringc((int)SDL_event.type).c_str(), ELL_INFORMATION); // just for debugging
534 
535  switch ( SDL_event.type )
536  {
537  case SDL_MOUSEMOTION: {
538  SDL_Keymod keymod = SDL_GetModState();
539 
540  irrevent.EventType = irr::EET_MOUSE_INPUT_EVENT;
541  irrevent.MouseInput.Event = irr::EMIE_MOUSE_MOVED;
542  MouseX = irrevent.MouseInput.X = SDL_event.motion.x;
543  MouseY = irrevent.MouseInput.Y = SDL_event.motion.y;
544  MouseXRel = SDL_event.motion.xrel;
545  MouseYRel = SDL_event.motion.yrel;
546  irrevent.MouseInput.ButtonStates = MouseButtonStates;
547  irrevent.MouseInput.Shift = (keymod & KMOD_SHIFT) != 0;
548  irrevent.MouseInput.Control = (keymod & KMOD_CTRL) != 0;
549 
550  postEventFromUser(irrevent);
551  break;
552  }
553  case SDL_MOUSEWHEEL: {
554  SDL_Keymod keymod = SDL_GetModState();
555 
556  irrevent.EventType = irr::EET_MOUSE_INPUT_EVENT;
557  irrevent.MouseInput.Event = irr::EMIE_MOUSE_WHEEL;
558  irrevent.MouseInput.Wheel = static_cast<float>(SDL_event.wheel.y);
559  irrevent.MouseInput.Shift = (keymod & KMOD_SHIFT) != 0;
560  irrevent.MouseInput.Control = (keymod & KMOD_CTRL) != 0;
561  irrevent.MouseInput.X = MouseX;
562  irrevent.MouseInput.Y = MouseY;
563 
564  postEventFromUser(irrevent);
565  break;
566  }
567  case SDL_MOUSEBUTTONDOWN:
568  case SDL_MOUSEBUTTONUP: {
569  SDL_Keymod keymod = SDL_GetModState();
570 
571  irrevent.EventType = irr::EET_MOUSE_INPUT_EVENT;
572  irrevent.MouseInput.X = SDL_event.button.x;
573  irrevent.MouseInput.Y = SDL_event.button.y;
574  irrevent.MouseInput.Shift = (keymod & KMOD_SHIFT) != 0;
575  irrevent.MouseInput.Control = (keymod & KMOD_CTRL) != 0;
576 
577  irrevent.MouseInput.Event = irr::EMIE_MOUSE_MOVED;
578 
579 
580 #ifdef _IRR_EMSCRIPTEN_PLATFORM_
581  // Handle mouselocking in emscripten in Windowed mode.
582  // In fullscreen SDL will handle it.
583  // The behavior we want windowed is - when the canvas was clicked then
584  // we will lock the mouse-pointer if it should be invisible.
585  // For security reasons this will be delayed until the next mouse-up event.
586  // We do not pass on this event as we don't want the activation click to do anything.
587  if ( SDL_event.type == SDL_MOUSEBUTTONDOWN && !isFullscreen() )
588  {
589  EmscriptenPointerlockChangeEvent pointerlockStatus; // let's hope that test is not expensive ...
590  if ( emscripten_get_pointerlock_status(&pointerlockStatus) == EMSCRIPTEN_RESULT_SUCCESS )
591  {
592  if ( CursorControl->isVisible() && pointerlockStatus.isActive )
593  {
594  emscripten_exit_pointerlock();
595  return !Close;
596  }
597  else if ( !CursorControl->isVisible() && !pointerlockStatus.isActive )
598  {
599  emscripten_request_pointerlock(0, true);
600  return !Close;
601  }
602  }
603  }
604 #endif
605 
606  switch(SDL_event.button.button)
607  {
608  case SDL_BUTTON_LEFT:
609  if (SDL_event.type == SDL_MOUSEBUTTONDOWN)
610  {
611  irrevent.MouseInput.Event = irr::EMIE_LMOUSE_PRESSED_DOWN;
612  MouseButtonStates |= irr::EMBSM_LEFT;
613  }
614  else
615  {
616  irrevent.MouseInput.Event = irr::EMIE_LMOUSE_LEFT_UP;
617  MouseButtonStates &= ~irr::EMBSM_LEFT;
618  }
619  break;
620 
621  case SDL_BUTTON_RIGHT:
622  if (SDL_event.type == SDL_MOUSEBUTTONDOWN)
623  {
624  irrevent.MouseInput.Event = irr::EMIE_RMOUSE_PRESSED_DOWN;
625  MouseButtonStates |= irr::EMBSM_RIGHT;
626  }
627  else
628  {
629  irrevent.MouseInput.Event = irr::EMIE_RMOUSE_LEFT_UP;
630  MouseButtonStates &= ~irr::EMBSM_RIGHT;
631  }
632  break;
633 
634  case SDL_BUTTON_MIDDLE:
635  if (SDL_event.type == SDL_MOUSEBUTTONDOWN)
636  {
637  irrevent.MouseInput.Event = irr::EMIE_MMOUSE_PRESSED_DOWN;
638  MouseButtonStates |= irr::EMBSM_MIDDLE;
639  }
640  else
641  {
642  irrevent.MouseInput.Event = irr::EMIE_MMOUSE_LEFT_UP;
643  MouseButtonStates &= ~irr::EMBSM_MIDDLE;
644  }
645  break;
646  }
647 
648  irrevent.MouseInput.ButtonStates = MouseButtonStates;
649 
650  if (irrevent.MouseInput.Event != irr::EMIE_MOUSE_MOVED)
651  {
652  postEventFromUser(irrevent);
653 
654  if ( irrevent.MouseInput.Event >= EMIE_LMOUSE_PRESSED_DOWN && irrevent.MouseInput.Event <= EMIE_MMOUSE_PRESSED_DOWN )
655  {
656  u32 clicks = checkSuccessiveClicks(irrevent.MouseInput.X, irrevent.MouseInput.Y, irrevent.MouseInput.Event);
657  if ( clicks == 2 )
658  {
659  irrevent.MouseInput.Event = (EMOUSE_INPUT_EVENT)(EMIE_LMOUSE_DOUBLE_CLICK + irrevent.MouseInput.Event-EMIE_LMOUSE_PRESSED_DOWN);
660  postEventFromUser(irrevent);
661  }
662  else if ( clicks == 3 )
663  {
664  irrevent.MouseInput.Event = (EMOUSE_INPUT_EVENT)(EMIE_LMOUSE_TRIPLE_CLICK + irrevent.MouseInput.Event-EMIE_LMOUSE_PRESSED_DOWN);
665  postEventFromUser(irrevent);
666  }
667  }
668  }
669  break;
670  }
671 
672  case SDL_TEXTINPUT:
673  {
674  irrevent.EventType = irr::EET_STRING_INPUT_EVENT;
675  irrevent.StringInput.Str = new core::stringw();
676  irr::core::multibyteToWString(*irrevent.StringInput.Str, SDL_event.text.text);
677  postEventFromUser(irrevent);
678  delete irrevent.StringInput.Str;
679  irrevent.StringInput.Str = NULL;
680  }
681  break;
682 
683  case SDL_KEYDOWN:
684  case SDL_KEYUP:
685  {
686  SKeyMap mp;
687  mp.SDLKey = SDL_event.key.keysym.sym;
688  s32 idx = KeyMap.binary_search(mp);
689 
690  EKEY_CODE key;
691  if (idx == -1)
692  key = (EKEY_CODE)0;
693  else
694  key = (EKEY_CODE)KeyMap[idx].Win32Key;
695 
696  // Make sure to only input special characters if something is in focus, as SDL_TEXTINPUT handles normal unicode already
697  if (SDL_IsTextInputActive() && !keyIsKnownSpecial(key) && (SDL_event.key.keysym.mod & KMOD_CTRL) == 0)
698  break;
699 
700 #ifdef _IRR_WINDOWS_API_
701  // handle alt+f4 in Windows, because SDL seems not to
702  if ( (SDL_event.key.keysym.mod & KMOD_LALT) && key == KEY_F4)
703  {
704  Close = true;
705  break;
706  }
707 #endif
708  irrevent.EventType = irr::EET_KEY_INPUT_EVENT;
709  irrevent.KeyInput.Key = key;
710  irrevent.KeyInput.PressedDown = (SDL_event.type == SDL_KEYDOWN);
711  irrevent.KeyInput.Shift = (SDL_event.key.keysym.mod & KMOD_SHIFT) != 0;
712  irrevent.KeyInput.Control = (SDL_event.key.keysym.mod & KMOD_CTRL ) != 0;
713  irrevent.KeyInput.Char = findCharToPassToIrrlicht(mp.SDLKey, key);
714  postEventFromUser(irrevent);
715  }
716  break;
717 
718  case SDL_QUIT:
719  Close = true;
720  break;
721 
722  case SDL_WINDOWEVENT:
723  switch (SDL_event.window.event)
724  {
725  case SDL_WINDOWEVENT_RESIZED:
726  if ((SDL_event.window.data1 != (int)Width) || (SDL_event.window.data2 != (int)Height))
727  {
728  Width = SDL_event.window.data1;
729  Height = SDL_event.window.data2;
730  if (VideoDriver)
731  VideoDriver->OnResize(core::dimension2d<u32>(Width, Height));
732  }
733  break;
734  }
735 
736  case SDL_USEREVENT:
737  irrevent.EventType = irr::EET_USER_EVENT;
738  irrevent.UserEvent.UserData1 = reinterpret_cast<uintptr_t>(SDL_event.user.data1);
739  irrevent.UserEvent.UserData2 = reinterpret_cast<uintptr_t>(SDL_event.user.data2);
740 
741  postEventFromUser(irrevent);
742  break;
743 
744  default:
745  break;
746  } // end switch
747  resetReceiveTextInputEvents();
748  } // end while
749 
750 #if defined(_IRR_COMPILE_WITH_JOYSTICK_EVENTS_)
751  // TODO: Check if the multiple open/close calls are too expensive, then
752  // open/close in the constructor/destructor instead
753 
754  // update joystick states manually
755  SDL_JoystickUpdate();
756  // we'll always send joystick input events...
757  SEvent joyevent;
758  joyevent.EventType = EET_JOYSTICK_INPUT_EVENT;
759  for (u32 i=0; i<Joysticks.size(); ++i)
760  {
761  SDL_Joystick* joystick = Joysticks[i];
762  if (joystick)
763  {
764  int j;
765  // query all buttons
766  const int numButtons = core::min_(SDL_JoystickNumButtons(joystick), 32);
767  joyevent.JoystickEvent.ButtonStates=0;
768  for (j=0; j<numButtons; ++j)
769  joyevent.JoystickEvent.ButtonStates |= (SDL_JoystickGetButton(joystick, j)<<j);
770 
771  // query all axes, already in correct range
772  const int numAxes = core::min_(SDL_JoystickNumAxes(joystick), (int)SEvent::SJoystickEvent::NUMBER_OF_AXES);
773  joyevent.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_X]=0;
774  joyevent.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_Y]=0;
775  joyevent.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_Z]=0;
776  joyevent.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_R]=0;
777  joyevent.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_U]=0;
778  joyevent.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_V]=0;
779  for (j=0; j<numAxes; ++j)
780  joyevent.JoystickEvent.Axis[j] = SDL_JoystickGetAxis(joystick, j);
781 
782  // we can only query one hat, SDL only supports 8 directions
783  if (SDL_JoystickNumHats(joystick)>0)
784  {
785  switch (SDL_JoystickGetHat(joystick, 0))
786  {
787  case SDL_HAT_UP:
788  joyevent.JoystickEvent.POV=0;
789  break;
790  case SDL_HAT_RIGHTUP:
791  joyevent.JoystickEvent.POV=4500;
792  break;
793  case SDL_HAT_RIGHT:
794  joyevent.JoystickEvent.POV=9000;
795  break;
796  case SDL_HAT_RIGHTDOWN:
797  joyevent.JoystickEvent.POV=13500;
798  break;
799  case SDL_HAT_DOWN:
800  joyevent.JoystickEvent.POV=18000;
801  break;
802  case SDL_HAT_LEFTDOWN:
803  joyevent.JoystickEvent.POV=22500;
804  break;
805  case SDL_HAT_LEFT:
806  joyevent.JoystickEvent.POV=27000;
807  break;
808  case SDL_HAT_LEFTUP:
809  joyevent.JoystickEvent.POV=31500;
810  break;
811  case SDL_HAT_CENTERED:
812  default:
813  joyevent.JoystickEvent.POV=65535;
814  break;
815  }
816  }
817  else
818  {
819  joyevent.JoystickEvent.POV=65535;
820  }
821 
822  // we map the number directly
823  joyevent.JoystickEvent.Joystick=static_cast<u8>(i);
824  // now post the event
825  postEventFromUser(joyevent);
826  // and close the joystick
827  }
828  }
829 #endif
830  return !Close;
831 }
