126 BiomeGenOriginal::BiomeGenOriginal(BiomeManager *biomemgr,
127  const BiomeParamsOriginal *params, v3s16 chunksize)
128 {
129  m_bmgr = biomemgr;
130  m_params = params;
131  m_csize = chunksize;
132 
133  noise_heat = new Noise(&params->np_heat,
134  params->seed, m_csize.X, m_csize.Z);
135  noise_humidity = new Noise(&params->np_humidity,
136  params->seed, m_csize.X, m_csize.Z);
137  noise_heat_blend = new Noise(&params->np_heat_blend,
138  params->seed, m_csize.X, m_csize.Z);
139  noise_humidity_blend = new Noise(&params->np_humidity_blend,
140  params->seed, m_csize.X, m_csize.Z);
141 
142  heatmap = noise_heat->result;
143  humidmap = noise_humidity->result;
144 
145  biomemap = new biome_t[m_csize.X * m_csize.Z];
146  // Initialise with the ID of 'BIOME_NONE' so that cavegen can get the
147  // fallback biome when biome generation (which calculates the biomemap IDs)
148  // is disabled.
149  memset(biomemap, 0, sizeof(biome_t) * m_csize.X * m_csize.Z);
150 
151  // Calculating the bounding position of each biome so we know when we might switch
152  // First gathering all heights where we might switch
153  std::vector<s16> temp_transition_heights;
154  temp_transition_heights.reserve(m_bmgr->getNumObjects() * 2);
155  for (size_t i = 0; i < m_bmgr->getNumObjects(); i++) {
156  Biome *b = (Biome *)m_bmgr->getRaw(i);
157  temp_transition_heights.push_back(b->max_pos.Y);
158  temp_transition_heights.push_back(b->min_pos.Y);
159  }
160 
161  // Sorting the biome transition points
162  std::sort(temp_transition_heights.begin(), temp_transition_heights.end(), std::greater<int>());
163 
164  // Getting rid of duplicate biome transition points
165  s16 last = temp_transition_heights[0];
166  size_t out_pos = 1;
167  for (size_t i = 1; i < temp_transition_heights.size(); i++){
168  if (temp_transition_heights[i] != last) {
169  last = temp_transition_heights[i];
170  temp_transition_heights[out_pos++] = last;
171  }
172  }
173 
174  biome_transitions = new s16[out_pos];
175  memcpy(biome_transitions, temp_transition_heights.data(), sizeof(s16) * out_pos);
176 }

234 {

236  for (s16 xr = 0; xr < m_csize.X; xr++) {

267 Biome *BiomeGenOriginal::calcBiomeFromNoise(float heat, float humidity, v3s16 pos) const
268 {
269  Biome *biome_closest = nullptr;
270  Biome *biome_closest_blend = nullptr;
271  float dist_min = FLT_MAX;
272  float dist_min_blend = FLT_MAX;
273 
274  for (size_t i = 1; i < m_bmgr->getNumObjects(); i++) {
275  Biome *b = (Biome *)m_bmgr->getRaw(i);
276  if (!b ||
277  pos.Y < b->min_pos.Y || pos.Y > b->max_pos.Y + b->vertical_blend ||
278  pos.X < b->min_pos.X || pos.X > b->max_pos.X ||
279  pos.Z < b->min_pos.Z || pos.Z > b->max_pos.Z)
280  continue;
281 
282  float d_heat = heat - b->heat_point;
283  float d_humidity = humidity - b->humidity_point;
284  float dist = (d_heat * d_heat) + (d_humidity * d_humidity);
285 
286  if (pos.Y <= b->max_pos.Y) { // Within y limits of biome b
287  if (dist < dist_min) {
288  dist_min = dist;
289  biome_closest = b;
290  }
291  } else if (dist < dist_min_blend) { // Blend area above biome b
292  dist_min_blend = dist;
293  biome_closest_blend = b;
294  }
295  }
296 
297  // Carefully tune pseudorandom seed variation to avoid single node dither
298  // and create larger scale blending patterns similar to horizontal biome
299  // blend.
300  const u64 seed = pos.Y + (heat + humidity) * 0.9f;
301  PcgRandom rng(seed);
302 
303  if (biome_closest_blend && dist_min_blend <= dist_min &&
304  rng.range(0, biome_closest_blend->vertical_blend) >=
305  pos.Y - biome_closest_blend->max_pos.Y)
306  return biome_closest_blend;
307 
308  return (biome_closest) ? biome_closest : (Biome *)m_bmgr->getRaw(BIOME_NONE);
309 }
