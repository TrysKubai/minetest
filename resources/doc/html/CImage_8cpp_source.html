45 void CImage::setPixel(u32 x, u32 y, const SColor &color, bool blend)
46 {
47  if (x >= Size.Width || y >= Size.Height)
48  return;
49 
50  switch(Format)
51  {
52  case ECF_A1R5G5B5:
53  {
54  u16 * dest = (u16*) (Data + ( y * Pitch ) + ( x << 1 ));
55  *dest = video::A8R8G8B8toA1R5G5B5( color.color );
56  } break;
57 
58  case ECF_R5G6B5:
59  {
60  u16 * dest = (u16*) (Data + ( y * Pitch ) + ( x << 1 ));
61  *dest = video::A8R8G8B8toR5G6B5( color.color );
62  } break;
63 
64  case ECF_R8G8B8:
65  {
66  u8* dest = Data + ( y * Pitch ) + ( x * 3 );
67  dest[0] = (u8)color.getRed();
68  dest[1] = (u8)color.getGreen();
69  dest[2] = (u8)color.getBlue();
70  } break;
71 
72  case ECF_A8R8G8B8:
73  {
74  u32 * dest = (u32*) (Data + ( y * Pitch ) + ( x << 2 ));
75  *dest = blend ? PixelBlend32 ( *dest, color.color ) : color.color;
76  } break;
77 
78  IRR_CASE_IIMAGE_COMPRESSED_FORMAT
79  os::Printer::log("IImage::setPixel method doesn't work with compressed images.", ELL_WARNING);
80  return;
81 
82  case ECF_UNKNOWN:
83  os::Printer::log("IImage::setPixel unknown format.", ELL_WARNING);
84  return;
85 
86  default:
87  break;
88  }

93 SColor CImage::getPixel(u32 x, u32 y) const
94 {
95  if (x >= Size.Width || y >= Size.Height)
96  return SColor(0);
97 
98  switch(Format)
99  {
100  case ECF_A1R5G5B5:
101  return A1R5G5B5toA8R8G8B8(((u16*)Data)[y*Size.Width + x]);
102  case ECF_R5G6B5:
103  return R5G6B5toA8R8G8B8(((u16*)Data)[y*Size.Width + x]);
104  case ECF_A8R8G8B8:
105  return ((u32*)Data)[y*Size.Width + x];
106  case ECF_R8G8B8:
107  {
108  u8* p = Data+(y*3)*Size.Width + (x*3);
109  return SColor(255,p[0],p[1],p[2]);
110  }
111 
112  IRR_CASE_IIMAGE_COMPRESSED_FORMAT
113  os::Printer::log("IImage::getPixel method doesn't work with compressed images.", ELL_WARNING);
114  break;
115 
116  case ECF_UNKNOWN:
117  os::Printer::log("IImage::getPixel unknown format.", ELL_WARNING);
118  break;
119 
120  default:
121  break;
122  }
123 
124  return SColor(0);

177 bool CImage::copyToNoScaling(void *target, u32 width, u32 height, ECOLOR_FORMAT format, u32 pitch) const
178 {
179  if (IImage::isCompressedFormat(Format))
180  {
181  os::Printer::log("IImage::copyToNoScaling method doesn't work with compressed images.", ELL_WARNING);
182  return false;
183  }
184 
185  if (!target || !width || !height || !Size.Width || !Size.Height)
186  return false;
187 
188  const u32 bpp=getBitsPerPixelFromFormat(format)/8;
189  if (0==pitch)
190  pitch = width*bpp;
191 
192  if (!(Format==format && Size.Width==width && Size.Height==height))
193  return false;
194 
195  if (pitch==Pitch)
196  {
197  memcpy(target, Data, (size_t)height*pitch);
198  }
199  else
200  {
201  u8* tgtpos = (u8*) target;
202  u8* srcpos = Data;
203  const u32 bwidth = width*bpp;
204  const u32 rest = pitch-bwidth;
205  for (u32 y=0; y<height; ++y)
206  {
207  // copy scanline
208  memcpy(tgtpos, srcpos, bwidth);
209  // clear pitch
210  memset(tgtpos+bwidth, 0, rest);
211  tgtpos += pitch;
212  srcpos += Pitch;
213  }
214  }
215 
216  return true;
217 }

222 void CImage::copyToScaling(void* target, u32 width, u32 height, ECOLOR_FORMAT format, u32 pitch)
223 {
224  if (IImage::isCompressedFormat(Format))
225  {
226  os::Printer::log("IImage::copyToScaling method doesn't work with compressed images.", ELL_WARNING);
227  return;
228  }
229 
230  if (!target || !width || !height || !Size.Width || !Size.Height)
231  return;
232 
233  const u32 bpp=getBitsPerPixelFromFormat(format)/8;
234  if (0==pitch)
235  pitch = width*bpp;
236 
237  if (copyToNoScaling(target, width, height, format, pitch))
238  return;
239 
240  // NOTE: Scaling is coded to keep the border pixels intact.
241  // Alternatively we could for example work with first pixel being taken at half step-size.
242  // Then we have one more step here and it would be:
243  // sourceXStep = (f32)(Size.Width-1) / (f32)(width);
244  // And sx would start at 0.5f + sourceXStep / 2.f;
245  // Similar for y.
246  // As scaling is done without any antialiasing it doesn't matter too much which outermost pixels we use and keeping
247  // border pixels intact is probably mostly better (with AA the other solution would be more correct).
248  // This is however unnecessary (and unexpected) for scaling to integer multiples, so don't do it there.
249  f32 sourceXStep, sourceYStep;
250  f32 sourceXStart = 0.f, sourceYStart = 0.f;
251  if (width % Size.Width == 0)
252  sourceXStep = (f32)(Size.Width) / (f32)(width);
253  else
254  {
255  sourceXStep = width > 1 ? (f32)(Size.Width-1) / (f32)(width-1) : 0.f;
256  sourceXStart = 0.5f; // for rounding to nearest pixel
257  }
258  if (height % Size.Height == 0)
259  sourceYStep = (f32)(Size.Height) / (f32)(height);
260  else
261  {
262  sourceYStep = height > 1 ? (f32)(Size.Height-1) / (f32)(height-1) : 0.f;
263  sourceYStart = 0.5f; // for rounding to nearest pixel
264  }
265 
266  s32 yval=0, syval=0;
267  f32 sy = sourceYStart;
268  for (u32 y=0; y<height; ++y)
269  {
270  f32 sx = sourceXStart;
271  for (u32 x=0; x<width; ++x)
272  {
273  CColorConverter::convert_viaFormat(Data+ syval + ((s32)sx)*BytesPerPixel, Format, 1, ((u8*)target)+ yval + (x*bpp), format);
274  sx+=sourceXStep;
275  }
276  sy+=sourceYStep;
277  syval=(s32)(sy)*Pitch;
278  yval+=pitch;
279  }
280 }

309 void CImage::copyToScalingBoxFilter(IImage* target, s32 bias, bool blend)
310 {
311  if (IImage::isCompressedFormat(Format))
312  {
313  os::Printer::log("IImage::copyToScalingBoxFilter method doesn't work with compressed images.", ELL_WARNING);
314  return;
315  }
316 
317  const core::dimension2d<u32> destSize = target->getDimension();
318 
319  const f32 sourceXStep = (f32) Size.Width / (f32) destSize.Width;
320  const f32 sourceYStep = (f32) Size.Height / (f32) destSize.Height;
321 
322  s32 fx = core::ceil32( sourceXStep );
323  s32 fy = core::ceil32( sourceYStep );
324  f32 sx;
325  f32 sy;
326 
327  sy = 0.f;
328  for ( u32 y = 0; y != destSize.Height; ++y )
329  {
330  sx = 0.f;
331  for ( u32 x = 0; x != destSize.Width; ++x )
332  {
333  target->setPixel( x, y,
334  getPixelBox( core::floor32(sx), core::floor32(sy), fx, fy, bias ), blend );
335  sx += sourceXStep;
336  }
337  sy += sourceYStep;
338  }

343 void CImage::fill(const SColor &color)
344 {
345  if (IImage::isCompressedFormat(Format))
346  {
347  os::Printer::log("IImage::fill method doesn't work with compressed images.", ELL_WARNING);
348  return;
349  }
350 
351  u32 c;
352 
353  switch ( Format )
354  {
355  case ECF_A1R5G5B5:
356  c = color.toA1R5G5B5();
357  c |= c << 16;
358  break;
359  case ECF_R5G6B5:
360  c = video::A8R8G8B8toR5G6B5( color.color );
361  c |= c << 16;
362  break;
363  case ECF_A8R8G8B8:
364  c = color.color;
365  break;
366  case ECF_R8G8B8:
367  {
368  u8 rgb[3];
369  CColorConverter::convert_A8R8G8B8toR8G8B8(&color, 1, rgb);
370  const u32 size = getImageDataSizeInBytes();
371  for (u32 i=0; i<size; i+=3)
372  {
373  memcpy(Data+i, rgb, 3);
374  }
375  return;
376  }
377  break;
378  default:
379  // TODO: Handle other formats
380  return;
381  }
382  memset32( Data, c, getImageDataSizeInBytes() );
383 }

387 inline SColor CImage::getPixelBox( s32 x, s32 y, s32 fx, s32 fy, s32 bias ) const
388 {
389  if (IImage::isCompressedFormat(Format))
390  {
391  os::Printer::log("IImage::getPixelBox method doesn't work with compressed images.", ELL_WARNING);
392  return SColor(0);
393  }
394 
395  SColor c;
396  s32 a = 0, r = 0, g = 0, b = 0;
397 
398  for ( s32 dx = 0; dx != fx; ++dx )
399  {
400  for ( s32 dy = 0; dy != fy; ++dy )
401  {
402  c = getPixel( core::s32_min ( x + dx, Size.Width - 1 ) ,
403  core::s32_min ( y + dy, Size.Height - 1 )
404  );
405 
406  a += c.getAlpha();
407  r += c.getRed();
408  g += c.getGreen();
409  b += c.getBlue();
410  }
411 
412  }
413 
414  s32 sdiv = s32_log2_s32(fx * fy);
415 
416  a = core::s32_clamp( ( a >> sdiv ) + bias, 0, 255 );
417  r = core::s32_clamp( ( r >> sdiv ) + bias, 0, 255 );
418  g = core::s32_clamp( ( g >> sdiv ) + bias, 0, 255 );
419  b = core::s32_clamp( ( b >> sdiv ) + bias, 0, 255 );
420 
421  c.set( a, r, g, b );
422  return c;
423 }
