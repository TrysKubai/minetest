225 bool GUIKeyChangeMenu::acceptInput()
226 {
227  for (key_setting *k : key_settings) {
228  std::string default_key;
229  Settings::getLayer(SL_DEFAULTS)->getNoEx(k->setting_name, default_key);
230 
231  if (k->key.sym() != default_key)
232  g_settings->set(k->setting_name, k->key.sym());
233  else
234  g_settings->remove(k->setting_name);
235  }
236 
237  {
238  gui::IGUIElement *e = getElementFromId(GUI_ID_CB_AUX1_DESCENDS);
239  if(e && e->getType() == gui::EGUIET_CHECK_BOX)
240  g_settings->setBool("aux1_descends", ((gui::IGUICheckBox*)e)->isChecked());
241  }
242  {
243  gui::IGUIElement *e = getElementFromId(GUI_ID_CB_DOUBLETAP_JUMP);
244  if(e && e->getType() == gui::EGUIET_CHECK_BOX)
245  g_settings->setBool("doubletap_jump", ((gui::IGUICheckBox*)e)->isChecked());
246  }
247  {
248  gui::IGUIElement *e = getElementFromId(GUI_ID_CB_AUTOJUMP);
249  if(e && e->getType() == gui::EGUIET_CHECK_BOX)
250  g_settings->setBool("autojump", ((gui::IGUICheckBox*)e)->isChecked());
251  }
252 
253  clearKeyCache();
254 
255  g_gamecallback->signalKeyConfigChange();
256 
257  return true;
258 }

269 bool GUIKeyChangeMenu::OnEvent(const SEvent& event)
270 {
271  if (event.EventType == EET_KEY_INPUT_EVENT && active_key
272  && event.KeyInput.PressedDown) {
273 
274  bool prefer_character = shift_down;
275  KeyPress kp(event.KeyInput, prefer_character);
276 
277  if (event.KeyInput.Key == irr::KEY_DELETE)
278  kp = KeyPress(""); // To erase key settings
279  else if (event.KeyInput.Key == irr::KEY_ESCAPE)
280  kp = active_key->key; // Cancel
281 
282  bool shift_went_down = false;
283  if(!shift_down &&
284  (event.KeyInput.Key == irr::KEY_SHIFT ||
285  event.KeyInput.Key == irr::KEY_LSHIFT ||
286  event.KeyInput.Key == irr::KEY_RSHIFT))
287  shift_went_down = true;
288 
289  // Display Key already in use message
290  bool key_in_use = false;
291  if (strcmp(kp.sym(), "") != 0) {
292  for (key_setting *ks : key_settings) {
293  if (ks != active_key && ks->key == kp) {
294  key_in_use = true;
295  break;
296  }
297  }
298  }
299 
300  if (key_in_use && !this->key_used_text) {
301  core::rect<s32> rect(0, 0, 600, 40);
302  rect += v2s32(0, 0) + v2s32(25, 30);
303  this->key_used_text = Environment->addStaticText(
304  wstrgettext("Key already in use").c_str(),
305  rect, false, true, this, -1);
306  } else if (!key_in_use && this->key_used_text) {
307  this->key_used_text->remove();
308  this->key_used_text = nullptr;
309  }
310 
311  // But go on
312  {
313  active_key->key = kp;
314  active_key->button->setText(wstrgettext(kp.name()).c_str());
315 
316  // Allow characters made with shift
317  if (shift_went_down){
318  shift_down = true;
319  return false;
320  }
321 
322  active_key = nullptr;
323  return true;
324  }
325  } else if (event.EventType == EET_KEY_INPUT_EVENT && !active_key
326  && event.KeyInput.PressedDown
327  && event.KeyInput.Key == irr::KEY_ESCAPE) {
328  quitMenu();
329  return true;
330  } else if (event.EventType == EET_GUI_EVENT) {
331  if (event.GUIEvent.EventType == gui::EGET_ELEMENT_FOCUS_LOST
332  && isVisible())
333  {
334  if (!canTakeFocus(event.GUIEvent.Element))
335  {
336  infostream << "GUIKeyChangeMenu: Not allowing focus change."
337  << std::endl;
338  // Returning true disables focus change
339  return true;
340  }
341  }
342  if (event.GUIEvent.EventType == gui::EGET_BUTTON_CLICKED)
343  {
344  switch (event.GUIEvent.Caller->getID())
345  {
346  case GUI_ID_BACK_BUTTON: //back
347  acceptInput();
348  quitMenu();
349  return true;
350  case GUI_ID_ABORT_BUTTON: //abort
351  quitMenu();
352  return true;
353  default:
354  resetMenu();
355  for (key_setting *ks : key_settings) {
356  if (ks->id == event.GUIEvent.Caller->getID()) {
357  active_key = ks;
358  break;
359  }
360  }
361  FATAL_ERROR_IF(!active_key, "Key setting not found");
362 
363  shift_down = false;
364  active_key->button->setText(wstrgettext("press key").c_str());
365  break;
366  }
367  Environment->setFocus(this);
368  }
369  }
370  return Parent ? Parent->OnEvent(event) : false;
371 }
