47 IAnimatedMesh* CB3DMeshFileLoader::createMesh(io::IReadFile* file)
48 {
49  if (!file)
50  return 0;
51 
52  B3DFile = file;
53  AnimatedMesh = new scene::CSkinnedMesh();
54  ShowWarning = true; // If true a warning is issued if too many textures are used
55  VerticesStart=0;
56 
57  if ( load() )
58  {
59  AnimatedMesh->finalize();
60  }
61  else
62  {
63  AnimatedMesh->drop();
64  AnimatedMesh = 0;
65  }
66 
67  return AnimatedMesh;
68 }

71 bool CB3DMeshFileLoader::load()
72 {
73  B3dStack.clear();
74 
75  NormalsInFile=false;
76  HasVertexColors=false;
77 
78  //------ Get header ------
79 
80  SB3dChunkHeader header;
81  B3DFile->read(&header, sizeof(header));
82 #ifdef __BIG_ENDIAN__
83  header.size = os::Byteswap::byteswap(header.size);
84 #endif
85 
86  if ( strncmp( header.name, "BB3D", 4 ) != 0 )
87  {
88  os::Printer::log("File is not a b3d file. Loading failed (No header found)", B3DFile->getFileName(), ELL_ERROR);
89  return false;
90  }
91 
92  // Add main chunk...
93  B3dStack.push_back(SB3dChunk(header, B3DFile->getPos()-8));
94 
95  // Get file version, but ignore it, as it's not important with b3d files...
96  s32 fileVersion;
97  B3DFile->read(&fileVersion, sizeof(fileVersion));
98 #ifdef __BIG_ENDIAN__
99  fileVersion = os::Byteswap::byteswap(fileVersion);
100 #endif
101 
102  //------ Read main chunk ------
103 
104  while ( (B3dStack.getLast().startposition + B3dStack.getLast().length) > B3DFile->getPos() )
105  {
106  B3DFile->read(&header, sizeof(header));
107 #ifdef __BIG_ENDIAN__
108  header.size = os::Byteswap::byteswap(header.size);
109 #endif
110  B3dStack.push_back(SB3dChunk(header, B3DFile->getPos()-8));
111 
112  if ( strncmp( B3dStack.getLast().name, "TEXS", 4 ) == 0 )
113  {
114  if (!readChunkTEXS())
115  return false;
116  }
117  else if ( strncmp( B3dStack.getLast().name, "BRUS", 4 ) == 0 )
118  {
119  if (!readChunkBRUS())
120  return false;
121  }
122  else if ( strncmp( B3dStack.getLast().name, "NODE", 4 ) == 0 )
123  {
124  if (!readChunkNODE((CSkinnedMesh::SJoint*)0) )
125  return false;
126  }
127  else
128  {
129  os::Printer::log("Unknown chunk found in mesh base - skipping");
130  if (!B3DFile->seek(B3dStack.getLast().startposition + B3dStack.getLast().length))
131  return false;
132  B3dStack.erase(B3dStack.size()-1);
133  }
134  }
135 
136  B3dStack.clear();
137 
138  BaseVertices.clear();
139  AnimatedVertices_VertexID.clear();
140  AnimatedVertices_BufferID.clear();
141 
142  Materials.clear();
143  Textures.clear();
144 
145  return true;
146 }

149 bool CB3DMeshFileLoader::readChunkNODE(CSkinnedMesh::SJoint *inJoint)
150 {
151  CSkinnedMesh::SJoint *joint = AnimatedMesh->addJoint(inJoint);
152  readString(joint->Name);
153 
154 #ifdef _B3D_READER_DEBUG
155  core::stringc logStr;
156  for ( u32 i=1; i < B3dStack.size(); ++i )
157  logStr += "-";
158  logStr += "read ChunkNODE";
159  os::Printer::log(logStr.c_str(), joint->Name.c_str(), ELL_DEBUG);
160 #endif
161 
162  f32 position[3], scale[3], rotation[4];
163 
164  readFloats(position, 3);
165  readFloats(scale, 3);
166  readFloats(rotation, 4);
167 
168  joint->Animatedposition = core::vector3df(position[0],position[1],position[2]) ;
169  joint->Animatedscale = core::vector3df(scale[0],scale[1],scale[2]);
170  joint->Animatedrotation = core::quaternion(rotation[1], rotation[2], rotation[3], rotation[0]);
171 
172  //Build LocalMatrix:
173 
174  core::matrix4 positionMatrix;
175  positionMatrix.setTranslation( joint->Animatedposition );
176  core::matrix4 scaleMatrix;
177  scaleMatrix.setScale( joint->Animatedscale );
178  core::matrix4 rotationMatrix;
179  joint->Animatedrotation.getMatrix_transposed(rotationMatrix);
180 
181  joint->LocalMatrix = positionMatrix * rotationMatrix * scaleMatrix;
182 
183  if (inJoint)
184  joint->GlobalMatrix = inJoint->GlobalMatrix * joint->LocalMatrix;
185  else
186  joint->GlobalMatrix = joint->LocalMatrix;
187 
188  while(B3dStack.getLast().startposition + B3dStack.getLast().length > B3DFile->getPos()) // this chunk repeats
189  {
190  SB3dChunkHeader header;
191  B3DFile->read(&header, sizeof(header));
192 #ifdef __BIG_ENDIAN__
193  header.size = os::Byteswap::byteswap(header.size);
194 #endif
195 
196  B3dStack.push_back(SB3dChunk(header, B3DFile->getPos()-8));
197 
198  if ( strncmp( B3dStack.getLast().name, "NODE", 4 ) == 0 )
199  {
200  if (!readChunkNODE(joint))
201  return false;
202  }
203  else if ( strncmp( B3dStack.getLast().name, "MESH", 4 ) == 0 )
204  {
205  VerticesStart=BaseVertices.size();
206  if (!readChunkMESH(joint))
207  return false;
208  }
209  else if ( strncmp( B3dStack.getLast().name, "BONE", 4 ) == 0 )
210  {
211  if (!readChunkBONE(joint))
212  return false;
213  }
214  else if ( strncmp( B3dStack.getLast().name, "KEYS", 4 ) == 0 )
215  {
216  if(!readChunkKEYS(joint))
217  return false;
218  }
219  else if ( strncmp( B3dStack.getLast().name, "ANIM", 4 ) == 0 )
220  {
221  if (!readChunkANIM())
222  return false;
223  }
224  else
225  {
226  os::Printer::log("Unknown chunk found in node chunk - skipping");
227  if (!B3DFile->seek(B3dStack.getLast().startposition + B3dStack.getLast().length))
228  return false;
229  B3dStack.erase(B3dStack.size()-1);
230  }
231  }
232 
233  B3dStack.erase(B3dStack.size()-1);
234 
235  return true;
236 }

239 bool CB3DMeshFileLoader::readChunkMESH(CSkinnedMesh::SJoint *inJoint)
240 {
241 #ifdef _B3D_READER_DEBUG
242  core::stringc logStr;
243  for ( u32 i=1; i < B3dStack.size(); ++i )
244  logStr += "-";
245  logStr += "read ChunkMESH";
246  os::Printer::log(logStr.c_str(), ELL_DEBUG);
247 #endif
248 
249  s32 brushID;
250  B3DFile->read(&brushID, sizeof(brushID));
251 #ifdef __BIG_ENDIAN__
252  brushID = os::Byteswap::byteswap(brushID);
253 #endif
254 
255  NormalsInFile=false;
256  HasVertexColors=false;
257 
258  while((B3dStack.getLast().startposition + B3dStack.getLast().length) > B3DFile->getPos()) //this chunk repeats
259  {
260  SB3dChunkHeader header;
261  B3DFile->read(&header, sizeof(header));
262 #ifdef __BIG_ENDIAN__
263  header.size = os::Byteswap::byteswap(header.size);
264 #endif
265 
266  B3dStack.push_back(SB3dChunk(header, B3DFile->getPos()-8));
267 
268  if ( strncmp( B3dStack.getLast().name, "VRTS", 4 ) == 0 )
269  {
270  if (!readChunkVRTS(inJoint))
271  return false;
272  }
273  else if ( strncmp( B3dStack.getLast().name, "TRIS", 4 ) == 0 )
274  {
275  scene::SSkinMeshBuffer *meshBuffer = AnimatedMesh->addMeshBuffer();
276 
277  if (brushID!=-1)
278  {
279  meshBuffer->Material=Materials[brushID].Material;
280  }
281 
282  if(readChunkTRIS(meshBuffer,AnimatedMesh->getMeshBuffers().size()-1, VerticesStart)==false)
283  return false;
284 
285  if (!NormalsInFile)
286  {
287  s32 i;
288 
289  for ( i=0; i<(s32)meshBuffer->Indices.size(); i+=3)
290  {
291  core::plane3df p(meshBuffer->getVertex(meshBuffer->Indices[i+0])->Pos,
292  meshBuffer->getVertex(meshBuffer->Indices[i+1])->Pos,
293  meshBuffer->getVertex(meshBuffer->Indices[i+2])->Pos);
294 
295  meshBuffer->getVertex(meshBuffer->Indices[i+0])->Normal += p.Normal;
296  meshBuffer->getVertex(meshBuffer->Indices[i+1])->Normal += p.Normal;
297  meshBuffer->getVertex(meshBuffer->Indices[i+2])->Normal += p.Normal;
298  }
299 
300  for ( i = 0; i<(s32)meshBuffer->getVertexCount(); ++i )
301  {
302  meshBuffer->getVertex(i)->Normal.normalize();
303  BaseVertices[VerticesStart+i].Normal=meshBuffer->getVertex(i)->Normal;
304  }
305  }
306  }
307  else
308  {
309  os::Printer::log("Unknown chunk found in mesh - skipping");
310  if (!B3DFile->seek(B3dStack.getLast().startposition + B3dStack.getLast().length))
311  return false;
312  B3dStack.erase(B3dStack.size()-1);
313  }
314  }
315 
316  B3dStack.erase(B3dStack.size()-1);
317 
318  return true;
319 }

322 /*
323 VRTS:
324  int flags ;1=normal values present, 2=rgba values present
325  int tex_coord_sets ;texture coords per vertex (eg: 1 for simple U/V) max=8
326  but we only support 3
327  int tex_coord_set_size ;components per set (eg: 2 for simple U/V) max=4
328  {
329  float x,y,z ;always present
330  float nx,ny,nz ;vertex normal: present if (flags&1)
331  float red,green,blue,alpha ;vertex color: present if (flags&2)
332  float tex_coords[tex_coord_sets][tex_coord_set_size] ;tex coords
333  }
334 */
335 bool CB3DMeshFileLoader::readChunkVRTS(CSkinnedMesh::SJoint *inJoint)
336 {
337 #ifdef _B3D_READER_DEBUG
338  core::stringc logStr;
339  for ( u32 i=1; i < B3dStack.size(); ++i )
340  logStr += "-";
341  logStr += "ChunkVRTS";
342  os::Printer::log(logStr.c_str(), ELL_DEBUG);
343 #endif
344 
345  const s32 max_tex_coords = 3;
346  s32 flags, tex_coord_sets, tex_coord_set_size;
347 
348  B3DFile->read(&flags, sizeof(flags));
349  B3DFile->read(&tex_coord_sets, sizeof(tex_coord_sets));
350  B3DFile->read(&tex_coord_set_size, sizeof(tex_coord_set_size));
351 #ifdef __BIG_ENDIAN__
352  flags = os::Byteswap::byteswap(flags);
353  tex_coord_sets = os::Byteswap::byteswap(tex_coord_sets);
354  tex_coord_set_size = os::Byteswap::byteswap(tex_coord_set_size);
355 #endif
356 
357  if (tex_coord_sets >= max_tex_coords || tex_coord_set_size >= 4) // Something is wrong
358  {
359  os::Printer::log("tex_coord_sets or tex_coord_set_size too big", B3DFile->getFileName(), ELL_ERROR);
360  return false;
361  }
362 
363  //------ Allocate Memory, for speed -----------//
364 
365  s32 numberOfReads = 3;
366 
367  if (flags & 1)
368  {
369  NormalsInFile = true;
370  numberOfReads += 3;
371  }
372  if (flags & 2)
373  {
374  numberOfReads += 4;
375  HasVertexColors=true;
376  }
377 
378  numberOfReads += tex_coord_sets*tex_coord_set_size;
379 
380  const s32 memoryNeeded = (B3dStack.getLast().length / sizeof(f32)) / numberOfReads;
381 
382  BaseVertices.reallocate(memoryNeeded + BaseVertices.size() + 1);
383  AnimatedVertices_VertexID.reallocate(memoryNeeded + AnimatedVertices_VertexID.size() + 1);
384 
385  //--------------------------------------------//
386 
387  while( (B3dStack.getLast().startposition + B3dStack.getLast().length) > B3DFile->getPos()) // this chunk repeats
388  {
389  f32 position[3];
390  f32 normal[3]={0.f, 0.f, 0.f};
391  f32 color[4]={1.0f, 1.0f, 1.0f, 1.0f};
392  f32 tex_coords[max_tex_coords][4];
393 
394  readFloats(position, 3);
395 
396  if (flags & 1)
397  readFloats(normal, 3);
398  if (flags & 2)
399  readFloats(color, 4);
400 
401  for (s32 i=0; i<tex_coord_sets; ++i)
402  readFloats(tex_coords[i], tex_coord_set_size);
403 
404  f32 tu=0.0f, tv=0.0f;
405  if (tex_coord_sets >= 1 && tex_coord_set_size >= 2)
406  {
407  tu=tex_coords[0][0];
408  tv=tex_coords[0][1];
409  }
410 
411  f32 tu2=0.0f, tv2=0.0f;
412  if (tex_coord_sets>1 && tex_coord_set_size>1)
413  {
414  tu2=tex_coords[1][0];
415  tv2=tex_coords[1][1];
416  }
417 
418  // Create Vertex...
419  video::S3DVertex2TCoords Vertex(position[0], position[1], position[2],
420  normal[0], normal[1], normal[2],
421  video::SColorf(color[0], color[1], color[2], color[3]).toSColor(),
422  tu, tv, tu2, tv2);
423 
424  // Transform the Vertex position by nested node...
425  inJoint->GlobalMatrix.transformVect(Vertex.Pos);
426  inJoint->GlobalMatrix.rotateVect(Vertex.Normal);
427 
428  //Add it...
429  BaseVertices.push_back(Vertex);
430 
431  AnimatedVertices_VertexID.push_back(-1);
432  AnimatedVertices_BufferID.push_back(-1);
433  }
434 
435  B3dStack.erase(B3dStack.size()-1);
436 
437  return true;
438 }

441 bool CB3DMeshFileLoader::readChunkTRIS(scene::SSkinMeshBuffer *meshBuffer, u32 meshBufferID, s32 vertices_Start)
442 {
443 #ifdef _B3D_READER_DEBUG
444  core::stringc logStr;
445  for ( u32 i=1; i < B3dStack.size(); ++i )
446  logStr += "-";
447  logStr += "ChunkTRIS";
448  os::Printer::log(logStr.c_str(), ELL_DEBUG);
449 #endif
450 
451  bool showVertexWarning=false;
452 
453  s32 triangle_brush_id; // Note: Irrlicht can't have different brushes for each triangle (using a workaround)
454  B3DFile->read(&triangle_brush_id, sizeof(triangle_brush_id));
455 #ifdef __BIG_ENDIAN__
456  triangle_brush_id = os::Byteswap::byteswap(triangle_brush_id);
457 #endif
458 
459  SB3dMaterial *B3dMaterial;
460 
461  if (triangle_brush_id != -1)
462  {
463  B3dMaterial = &Materials[triangle_brush_id];
464  meshBuffer->Material = B3dMaterial->Material;
465  }
466  else
467  B3dMaterial = 0;
468 
469  const s32 memoryNeeded = B3dStack.getLast().length / sizeof(s32);
470  meshBuffer->Indices.reallocate(memoryNeeded + meshBuffer->Indices.size() + 1);
471 
472  while((B3dStack.getLast().startposition + B3dStack.getLast().length) > B3DFile->getPos()) // this chunk repeats
473  {
474  s32 vertex_id[3];
475 
476  B3DFile->read(vertex_id, 3*sizeof(s32));
477 #ifdef __BIG_ENDIAN__
478  vertex_id[0] = os::Byteswap::byteswap(vertex_id[0]);
479  vertex_id[1] = os::Byteswap::byteswap(vertex_id[1]);
480  vertex_id[2] = os::Byteswap::byteswap(vertex_id[2]);
481 #endif
482 
483  //Make Ids global:
484  vertex_id[0] += vertices_Start;
485  vertex_id[1] += vertices_Start;
486  vertex_id[2] += vertices_Start;
487 
488  for(s32 i=0; i<3; ++i)
489  {
490  if ((u32)vertex_id[i] >= AnimatedVertices_VertexID.size())
491  {
492  os::Printer::log("Illegal vertex index found", B3DFile->getFileName(), ELL_ERROR);
493  return false;
494  }
495 
496  if (AnimatedVertices_VertexID[ vertex_id[i] ] != -1)
497  {
498  if ( AnimatedVertices_BufferID[ vertex_id[i] ] != (s32)meshBufferID ) //If this vertex is linked in a different meshbuffer
499  {
500  AnimatedVertices_VertexID[ vertex_id[i] ] = -1;
501  AnimatedVertices_BufferID[ vertex_id[i] ] = -1;
502  showVertexWarning=true;
503  }
504  }
505  if (AnimatedVertices_VertexID[ vertex_id[i] ] == -1) //If this vertex is not in the meshbuffer
506  {
507  //Check for lightmapping:
508  if (BaseVertices[ vertex_id[i] ].TCoords2 != core::vector2df(0.f,0.f))
509  meshBuffer->convertTo2TCoords(); //Will only affect the meshbuffer the first time this is called
510 
511  //Add the vertex to the meshbuffer:
512  if (meshBuffer->VertexType == video::EVT_STANDARD)
513  meshBuffer->Vertices_Standard.push_back( BaseVertices[ vertex_id[i] ] );
514  else
515  meshBuffer->Vertices_2TCoords.push_back(BaseVertices[ vertex_id[i] ] );
516 
517  //create vertex id to meshbuffer index link:
518  AnimatedVertices_VertexID[ vertex_id[i] ] = meshBuffer->getVertexCount()-1;
519  AnimatedVertices_BufferID[ vertex_id[i] ] = meshBufferID;
520 
521  if (B3dMaterial)
522  {
523  // Apply Material/Color/etc...
524  video::S3DVertex *Vertex=meshBuffer->getVertex(meshBuffer->getVertexCount()-1);
525 
526  if (!HasVertexColors)
527  Vertex->Color=B3dMaterial->Material.DiffuseColor;
528  else if (Vertex->Color.getAlpha() == 255)
529  Vertex->Color.setAlpha( (s32)(B3dMaterial->alpha * 255.0f) );
530 
531  // Use texture's scale
532  if (B3dMaterial->Textures[0])
533  {
534  Vertex->TCoords.X *= B3dMaterial->Textures[0]->Xscale;
535  Vertex->TCoords.Y *= B3dMaterial->Textures[0]->Yscale;
536  }
537  /*
538  if (B3dMaterial->Textures[1])
539  {
540  Vertex->TCoords2.X *=B3dMaterial->Textures[1]->Xscale;
541  Vertex->TCoords2.Y *=B3dMaterial->Textures[1]->Yscale;
542  }
543  */
544  }
545  }
546  }
547 
548  meshBuffer->Indices.push_back( AnimatedVertices_VertexID[ vertex_id[0] ] );
549  meshBuffer->Indices.push_back( AnimatedVertices_VertexID[ vertex_id[1] ] );
550  meshBuffer->Indices.push_back( AnimatedVertices_VertexID[ vertex_id[2] ] );
551  }
552 
553  B3dStack.erase(B3dStack.size()-1);
554 
555  if (showVertexWarning)
556  os::Printer::log("B3dMeshLoader: Warning, different meshbuffers linking to the same vertex, this will cause problems with animated meshes");
557 
558  return true;
559 }

562 bool CB3DMeshFileLoader::readChunkBONE(CSkinnedMesh::SJoint *inJoint)
563 {
564 #ifdef _B3D_READER_DEBUG
565  core::stringc logStr;
566  for ( u32 i=1; i < B3dStack.size(); ++i )
567  logStr += "-";
568  logStr += "read ChunkBONE";
569  os::Printer::log(logStr.c_str(), ELL_DEBUG);
570 #endif
571 
572  if (B3dStack.getLast().length > 8)
573  {
574  while((B3dStack.getLast().startposition + B3dStack.getLast().length) > B3DFile->getPos()) // this chunk repeats
575  {
576  u32 globalVertexID;
577  f32 strength;
578  B3DFile->read(&globalVertexID, sizeof(globalVertexID));
579  B3DFile->read(&strength, sizeof(strength));
580 #ifdef __BIG_ENDIAN__
581  globalVertexID = os::Byteswap::byteswap(globalVertexID);
582  strength = os::Byteswap::byteswap(strength);
583 #endif
584  globalVertexID += VerticesStart;
585 
586  if (AnimatedVertices_VertexID[globalVertexID]==-1)
587  {
588  os::Printer::log("B3dMeshLoader: Weight has bad vertex id (no link to meshbuffer index found)");
589  }
590  else if (strength >0)
591  {
592  CSkinnedMesh::SWeight *weight=AnimatedMesh->addWeight(inJoint);
593  weight->strength=strength;
594  //Find the meshbuffer and Vertex index from the Global Vertex ID:
595  weight->vertex_id = AnimatedVertices_VertexID[globalVertexID];
596  weight->buffer_id = AnimatedVertices_BufferID[globalVertexID];
597  }
598  }
599  }
600 
601  B3dStack.erase(B3dStack.size()-1);
602  return true;
603 }

606 bool CB3DMeshFileLoader::readChunkKEYS(CSkinnedMesh::SJoint *inJoint)
607 {
608 #ifdef _B3D_READER_DEBUG
609  // Only print first, that's just too much output otherwise
610  if ( !inJoint || (inJoint->PositionKeys.empty() && inJoint->ScaleKeys.empty() && inJoint->RotationKeys.empty()) )
611  {
612  core::stringc logStr;
613  for ( u32 i=1; i < B3dStack.size(); ++i )
614  logStr += "-";
615  logStr += "read ChunkKEYS";
616  os::Printer::log(logStr.c_str(), ELL_DEBUG);
617  }
618 #endif
619 
620  s32 flags;
621  B3DFile->read(&flags, sizeof(flags));
622 #ifdef __BIG_ENDIAN__
623  flags = os::Byteswap::byteswap(flags);
624 #endif
625 
626  CSkinnedMesh::SPositionKey *oldPosKey=0;
627  core::vector3df oldPos[2];
628  CSkinnedMesh::SScaleKey *oldScaleKey=0;
629  core::vector3df oldScale[2];
630  CSkinnedMesh::SRotationKey *oldRotKey=0;
631  core::quaternion oldRot[2];
632  bool isFirst[3]={true,true,true};
633  while((B3dStack.getLast().startposition + B3dStack.getLast().length) > B3DFile->getPos()) //this chunk repeats
634  {
635  s32 frame;
636 
637  B3DFile->read(&frame, sizeof(frame));
638  #ifdef __BIG_ENDIAN__
639  frame = os::Byteswap::byteswap(frame);
640  #endif
641 
642  // Add key frames, frames in Irrlicht are zero-based
643  f32 data[4];
644  if (flags & 1)
645  {
646  readFloats(data, 3);
647  if ((oldPosKey!=0) && (oldPos[0]==oldPos[1]))
648  {
649  const core::vector3df pos(data[0], data[1], data[2]);

831 bool CB3DMeshFileLoader::readChunkBRUS()
832 {
833 #ifdef _B3D_READER_DEBUG
834  core::stringc logStr;
835  for ( u32 i=1; i < B3dStack.size(); ++i )
836  logStr += "-";
837  logStr += "read ChunkBRUS";
838  os::Printer::log(logStr.c_str(), ELL_DEBUG);
839 #endif
840 
841  u32 n_texs;
842  B3DFile->read(&n_texs, sizeof(u32));
843 #ifdef __BIG_ENDIAN__
844  n_texs = os::Byteswap::byteswap(n_texs);
845 #endif
846 
847  // number of texture ids read for Irrlicht
848  const u32 num_textures = core::min_(n_texs, video::MATERIAL_MAX_TEXTURES);
849  // number of bytes to skip (for ignored texture ids)
850  const u32 n_texs_offset = (num_textures<n_texs)?(n_texs-num_textures):0;
851 
852  while((B3dStack.getLast().startposition + B3dStack.getLast().length) > B3DFile->getPos()) //this chunk repeats
853  {
854  // This is what blitz basic calls a brush, like a Irrlicht Material
855 
856  core::stringc name;
857  readString(name);
858 #ifdef _B3D_READER_DEBUG
859  os::Printer::log("read Material", name, ELL_DEBUG);
860 #endif
861  Materials.push_back(SB3dMaterial());
862  SB3dMaterial& B3dMaterial=Materials.getLast();
863 
864  readFloats(&B3dMaterial.red, 1);
865  readFloats(&B3dMaterial.green, 1);
866  readFloats(&B3dMaterial.blue, 1);
867  readFloats(&B3dMaterial.alpha, 1);
868  readFloats(&B3dMaterial.shininess, 1);
869 
870  B3DFile->read(&B3dMaterial.blend, sizeof(B3dMaterial.blend));
871  B3DFile->read(&B3dMaterial.fx, sizeof(B3dMaterial.fx));
872 #ifdef __BIG_ENDIAN__
873  B3dMaterial.blend = os::Byteswap::byteswap(B3dMaterial.blend);
874  B3dMaterial.fx = os::Byteswap::byteswap(B3dMaterial.fx);
875 #endif
876 #ifdef _B3D_READER_DEBUG
877  os::Printer::log("Blend", core::stringc(B3dMaterial.blend).c_str(), ELL_DEBUG);
878  os::Printer::log("FX", core::stringc(B3dMaterial.fx).c_str(), ELL_DEBUG);
879 #endif
880 
881  u32 i;
882  for (i=0; i<num_textures; ++i)
883  {
884  s32 texture_id=-1;
885  B3DFile->read(&texture_id, sizeof(s32));
886 #ifdef __BIG_ENDIAN__
887  texture_id = os::Byteswap::byteswap(texture_id);
888 #endif

890  if ((u32)texture_id < Textures.size())
891  {
892  B3dMaterial.Textures[i]=&Textures[texture_id];
893 #ifdef _B3D_READER_DEBUG
894  os::Printer::log("Layer", core::stringc(i).c_str(), ELL_DEBUG);
895  os::Printer::log("using texture", Textures[texture_id].TextureName.c_str(), ELL_DEBUG);
896 #endif
897  }
898  else
899  B3dMaterial.Textures[i]=0;
900  }
901  // skip other texture ids
902  for (i=0; i<n_texs_offset; ++i)
903  {
904  s32 texture_id=-1;
905  B3DFile->read(&texture_id, sizeof(s32));
906 #ifdef __BIG_ENDIAN__
907  texture_id = os::Byteswap::byteswap(texture_id);
908 #endif
909  if (ShowWarning && (texture_id != -1) && (n_texs>video::MATERIAL_MAX_TEXTURES))
910  {
911  os::Printer::log("Too many textures used in one material", B3DFile->getFileName(), ELL_WARNING);
912  ShowWarning = false;
913  }
914  }
915 
916  //Fixes problems when the lightmap is on the first texture:
917  if (B3dMaterial.Textures[0] != 0)
918  {
919  if (B3dMaterial.Textures[0]->Flags & 65536) // 65536 = secondary UV
920  {
921  SB3dTexture *TmpTexture;
922  TmpTexture = B3dMaterial.Textures[1];
923  B3dMaterial.Textures[1] = B3dMaterial.Textures[0];
924  B3dMaterial.Textures[0] = TmpTexture;
925  }
926  }
927 
928  //If a preceeding texture slot is empty move the others down:
929  for (i=num_textures; i>0; --i)
930  {
931  for (u32 j=i-1; j<num_textures-1; ++j)
932  {
933  if (B3dMaterial.Textures[j+1] != 0 && B3dMaterial.Textures[j] == 0)
934  {
935  B3dMaterial.Textures[j] = B3dMaterial.Textures[j+1];
936  B3dMaterial.Textures[j+1] = 0;
937  }
938  }
939  }
940 
941  //------ Convert blitz flags/blend to irrlicht -------
942 
943  //Two textures:
944  if (B3dMaterial.Textures[1])
945  {
946  if (B3dMaterial.alpha==1.f)
947  {
948  if (B3dMaterial.Textures[1]->Blend == 5) //(Multiply 2)
949  B3dMaterial.Material.MaterialType = video::EMT_LIGHTMAP_M2;
950  else
951  B3dMaterial.Material.MaterialType = video::EMT_LIGHTMAP;
952  B3dMaterial.Material.Lighting = false;
953  }
954  else
955  {
956  B3dMaterial.Material.MaterialType = video::EMT_TRANSPARENT_VERTEX_ALPHA;
957  B3dMaterial.Material.ZWriteEnable = video::EZW_OFF;
958  }
959  }
960  else if (B3dMaterial.Textures[0]) //One texture:
961  {
962  // Flags & 0x1 is usual SOLID, 0x8 is mipmap (handled before)
963  if (B3dMaterial.Textures[0]->Flags & 0x2) //(Alpha mapped)
964  {
965  B3dMaterial.Material.MaterialType = video::EMT_TRANSPARENT_ALPHA_CHANNEL;
966  B3dMaterial.Material.ZWriteEnable = video::EZW_OFF;
967  }
968  else if (B3dMaterial.Textures[0]->Flags & 0x4) //(Masked)
969  B3dMaterial.Material.MaterialType = video::EMT_TRANSPARENT_ALPHA_CHANNEL_REF; // TODO: create color key texture
970  else if (B3dMaterial.Textures[0]->Flags & 0x40)
971  B3dMaterial.Material.MaterialType = video::EMT_SPHERE_MAP;
972  else if (B3dMaterial.Textures[0]->Flags & 0x80)
973  B3dMaterial.Material.MaterialType = video::EMT_SPHERE_MAP; // TODO: Should be cube map
974  else if (B3dMaterial.alpha == 1.f)
975  B3dMaterial.Material.MaterialType = video::EMT_SOLID;
976  else
977  {
978  B3dMaterial.Material.MaterialType = video::EMT_TRANSPARENT_VERTEX_ALPHA;
979  B3dMaterial.Material.ZWriteEnable = video::EZW_OFF;
980  }
981  }
982  else //No texture:
983  {
984  if (B3dMaterial.alpha == 1.f)
985  B3dMaterial.Material.MaterialType = video::EMT_SOLID;
986  else
987  {
988  B3dMaterial.Material.MaterialType = video::EMT_TRANSPARENT_VERTEX_ALPHA;
989  B3dMaterial.Material.ZWriteEnable = video::EZW_OFF;
990  }
991  }
992 
993  B3dMaterial.Material.DiffuseColor = video::SColorf(B3dMaterial.red, B3dMaterial.green, B3dMaterial.blue, B3dMaterial.alpha).toSColor();
994  B3dMaterial.Material.ColorMaterial=video::ECM_NONE;
995 
996  //------ Material fx ------
997 
998  if (B3dMaterial.fx & 1) //full-bright
999  {
1000  B3dMaterial.Material.AmbientColor = video::SColor(255, 255, 255, 255);
1001  B3dMaterial.Material.Lighting = false;
1002  }
1003  else
1004  B3dMaterial.Material.AmbientColor = B3dMaterial.Material.DiffuseColor;
1005 
1006  if (B3dMaterial.fx & 2) //use vertex colors instead of brush color
1007  B3dMaterial.Material.ColorMaterial=video::ECM_DIFFUSE_AND_AMBIENT;
1008 
1009  if (B3dMaterial.fx & 4) //flatshaded
1010  B3dMaterial.Material.GouraudShading = false;
1011 
1012  if (B3dMaterial.fx & 16) //disable backface culling
1013  B3dMaterial.Material.BackfaceCulling = false;
1014 
1015  if (B3dMaterial.fx & 32) //force vertex alpha-blending
1016  {
1017  B3dMaterial.Material.MaterialType = video::EMT_TRANSPARENT_VERTEX_ALPHA;
1018  B3dMaterial.Material.ZWriteEnable = video::EZW_OFF;
1019  }
1020 
1021  B3dMaterial.Material.Shininess = B3dMaterial.shininess;
1022  }
1023 
1024  B3dStack.erase(B3dStack.size()-1);
1025 
1026  return true;
