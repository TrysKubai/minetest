107 std::string PlayerSAO::getClientInitializationData(u16 protocol_version)
108 {
109  std::ostringstream os(std::ios::binary);
110 
111  // Protocol >= 15
112  writeU8(os, 1); // version
113  os << serializeString16(m_player->getName()); // name
114  writeU8(os, 1); // is_player
115  writeS16(os, getId()); // id
116  writeV3F32(os, m_base_position);
117  writeV3F32(os, m_rotation);
118  writeU16(os, getHP());
119 
120  std::ostringstream msg_os(std::ios::binary);
121  msg_os << serializeString32(getPropertyPacket()); // message 1
122  msg_os << serializeString32(generateUpdateArmorGroupsCommand()); // 2
123  msg_os << serializeString32(generateUpdateAnimationCommand()); // 3
124  for (const auto &bone_pos : m_bone_position) {
125  msg_os << serializeString32(generateUpdateBonePositionCommand(
126  bone_pos.first, bone_pos.second.X, bone_pos.second.Y)); // 3 + N
127  }
128  msg_os << serializeString32(generateUpdateAttachmentCommand()); // 4 + m_bone_position.size
129  msg_os << serializeString32(generateUpdatePhysicsOverrideCommand()); // 5 + m_bone_position.size
130 
131  int message_count = 5 + m_bone_position.size();
132 
133  for (const auto &id : getAttachmentChildIds()) {
134  if (ServerActiveObject *obj = m_env->getActiveObject(id)) {
135  message_count++;
136  msg_os << serializeString32(obj->generateUpdateInfantCommand(
137  id, protocol_version));
138  }
139  }
140 
141  writeU8(os, message_count);
142  std::string serialized = msg_os.str();
143  os.write(serialized.c_str(), serialized.size());
144 
145  // return result
146  return os.str();
147 }

154 void PlayerSAO::step(float dtime, bool send_recommended)
155 {
156  if (!isImmortal() && m_drowning_interval.step(dtime, 2.0f)) {
157  // Get nose/mouth position, approximate with eye position
158  v3s16 p = floatToInt(getEyePosition(), BS);
159  MapNode n = m_env->getMap().getNode(p);
160  const ContentFeatures &c = m_env->getGameDef()->ndef()->get(n);
161  // If node generates drown
162  if (c.drowning > 0 && m_hp > 0) {
163  if (m_breath > 0)
164  setBreath(m_breath - 1);
165 
166  // No more breath, damage player
167  if (m_breath == 0) {
168  PlayerHPChangeReason reason(PlayerHPChangeReason::DROWNING);
169  setHP(m_hp - c.drowning, reason);
170  }
171  }
172  }
173 
174  if (m_breathing_interval.step(dtime, 0.5f) && !isImmortal()) {
175  // Get nose/mouth position, approximate with eye position
176  v3s16 p = floatToInt(getEyePosition(), BS);
177  MapNode n = m_env->getMap().getNode(p);
178  const ContentFeatures &c = m_env->getGameDef()->ndef()->get(n);
179  // If player is alive & not drowning & not in ignore & not immortal, breathe
180  if (m_breath < m_prop.breath_max && c.drowning == 0 &&
181  n.getContent() != CONTENT_IGNORE && m_hp > 0)
182  setBreath(m_breath + 1);
183  }
184 
185  if (!isImmortal() && m_node_hurt_interval.step(dtime, 1.0f)) {
186  u32 damage_per_second = 0;
187  std::string nodename;
188  v3s16 node_pos;
189  // Lowest and highest damage points are 0.1 within collisionbox
190  float dam_top = m_prop.collisionbox.MaxEdge.Y - 0.1f;
191 
192  // Sequence of damage points, starting 0.1 above feet and progressing
193  // upwards in 1 node intervals, stopping below top damage point.
194  for (float dam_height = 0.1f; dam_height < dam_top; dam_height++) {
195  v3s16 p = floatToInt(m_base_position +
196  v3f(0.0f, dam_height * BS, 0.0f), BS);
197  MapNode n = m_env->getMap().getNode(p);
198  const ContentFeatures &c = m_env->getGameDef()->ndef()->get(n);
199  if (c.damage_per_second > damage_per_second) {
200  damage_per_second = c.damage_per_second;
201  nodename = c.name;
202  node_pos = p;
203  }
204  }
205 
206  // Top damage point
207  v3s16 ptop = floatToInt(m_base_position +
208  v3f(0.0f, dam_top * BS, 0.0f), BS);
209  MapNode ntop = m_env->getMap().getNode(ptop);
210  const ContentFeatures &c = m_env->getGameDef()->ndef()->get(ntop);
211  if (c.damage_per_second > damage_per_second) {
212  damage_per_second = c.damage_per_second;
213  nodename = c.name;
214  node_pos = ptop;
215  }
216 
217  if (damage_per_second != 0 && m_hp > 0) {
218  s32 newhp = (s32)m_hp - (s32)damage_per_second;
219  PlayerHPChangeReason reason(PlayerHPChangeReason::NODE_DAMAGE, nodename, node_pos);
220  setHP(newhp, reason);
221  }
222  }
223 
224  if (!m_properties_sent) {
225  m_properties_sent = true;
226  std::string str = getPropertyPacket();
227  // create message and add to list
228  m_messages_out.emplace(getId(), true, str);
229  m_env->getScriptIface()->player_event(this, "properties_changed");
230  }
231 
232  // If attached, check that our parent is still there. If it isn't, detach.
233  if (m_attachment_parent_id && !isAttached()) {
234  // This is handled when objects are removed from the map
235  warningstream << "PlayerSAO::step() id=" << m_id <<
236  " is attached to nonexistent parent. This is a bug." << std::endl;
237  clearParentAttachment();
238  setBasePosition(m_last_good_position);
239  m_env->getGameDef()->SendMovePlayer(m_peer_id);
240  }
241 
242  //dstream<<"PlayerSAO::step: dtime: "<<dtime<<std::endl;
243 
244  // Set lag pool maximums based on estimated lag
245  const float LAG_POOL_MIN = 5.0f;
246  float lag_pool_max = m_env->getMaxLagEstimate() * 2.0f;
247  if(lag_pool_max < LAG_POOL_MIN)
248  lag_pool_max = LAG_POOL_MIN;
249  m_dig_pool.setMax(lag_pool_max);
250  m_move_pool.setMax(lag_pool_max);
251 
252  // Increment cheat prevention timers
253  m_dig_pool.add(dtime);
254  m_move_pool.add(dtime);
255  m_time_from_last_teleport += dtime;
256  m_time_from_last_punch += dtime;
257  m_nocheat_dig_time += dtime;
258  m_max_speed_override_time = MYMAX(m_max_speed_override_time - dtime, 0.0f);
259 
260  // Each frame, parent position is copied if the object is attached,
261  // otherwise it's calculated normally.
262  // If the object gets detached this comes into effect automatically from
263  // the last known origin.
264  if (auto *parent = getParent()) {
265  v3f pos = parent->getBasePosition();
266  m_last_good_position = pos;
267  setBasePosition(pos);
268 
269  if (m_player)
270  m_player->setSpeed(v3f());
271  }
272 
273  if (!send_recommended)
274  return;
275 
276  if (m_position_not_sent) {
277  m_position_not_sent = false;
278  float update_interval = m_env->getSendRecommendedInterval();
279  v3f pos;
280  // When attached, the position is only sent to clients where the
281  // parent isn't known
282  if (isAttached())
283  pos = m_last_good_position;
284  else
285  pos = m_base_position;
286 
287  std::string str = generateUpdatePositionCommand(
288  pos,
289  v3f(0.0f, 0.0f, 0.0f),
290  v3f(0.0f, 0.0f, 0.0f),
291  m_rotation,
292  true,
293  false,
294  update_interval
295  );
296  // create message and add to list
297  m_messages_out.emplace(getId(), false, str);
298  }
299 
300  if (!m_physics_override_sent) {
301  m_physics_override_sent = true;
302  // create message and add to list
303  m_messages_out.emplace(getId(), true, generateUpdatePhysicsOverrideCommand());
304  }
305 
306  sendOutdatedData();
307 }

309 std::string PlayerSAO::generateUpdatePhysicsOverrideCommand() const
310 {
311  if (!m_player) {
312  // Will output a format warning client-side
313  return "";
314  }
315 
316  const auto &phys = m_player->physics_override;
317  std::ostringstream os(std::ios::binary);
318  // command
319  writeU8(os, AO_CMD_SET_PHYSICS_OVERRIDE);
320  // parameters
321  writeF32(os, phys.speed);
322  writeF32(os, phys.jump);
323  writeF32(os, phys.gravity);
324  // MT 0.4.10 legacy: send inverted for detault `true` if the server sends nothing
325  writeU8(os, !phys.sneak);
326  writeU8(os, !phys.sneak_glitch);
327  writeU8(os, !phys.new_move);
328  return os.str();
329 }

421 u32 PlayerSAO::punch(v3f dir,
422  const ToolCapabilities *toolcap,
423  ServerActiveObject *puncher,
424  float time_from_last_punch,
425  u16 initial_wear)
426 {
427  if (!toolcap)
428  return 0;
429 
430  FATAL_ERROR_IF(!puncher, "Punch action called without SAO");
431 
432  // No effect if PvP disabled or if immortal
433  if (isImmortal() || !g_settings->getBool("enable_pvp")) {
434  if (puncher->getType() == ACTIVEOBJECT_TYPE_PLAYER) {
435  // create message and add to list
436  sendPunchCommand();
437  return 0;
438  }
439  }
440 
441  s32 old_hp = getHP();
442  HitParams hitparams = getHitParams(m_armor_groups, toolcap,
443  time_from_last_punch, initial_wear);
444 
445  PlayerSAO *playersao = m_player->getPlayerSAO();
446 
447  bool damage_handled = m_env->getScriptIface()->on_punchplayer(playersao,
448  puncher, time_from_last_punch, toolcap, dir,
449  hitparams.hp);
450 
451  if (!damage_handled) {
452  setHP((s32)getHP() - (s32)hitparams.hp,
453  PlayerHPChangeReason(PlayerHPChangeReason::PLAYER_PUNCH, puncher));
454  } else { // override client prediction
455  if (puncher->getType() == ACTIVEOBJECT_TYPE_PLAYER) {
456  // create message and add to list
457  sendPunchCommand();
458  }
459  }
460 
461  actionstream << puncher->getDescription() << " (id=" << puncher->getId() <<
462  ", hp=" << puncher->getHP() << ") punched " <<
463  getDescription() << " (id=" << m_id << ", hp=" << m_hp <<
464  "), damage=" << (old_hp - (s32)getHP()) <<
465  (damage_handled ? " (handled by Lua)" : "") << std::endl;
466 
467  return hitparams.wear;
468 }

475 void PlayerSAO::setHP(s32 target_hp, const PlayerHPChangeReason &reason, bool from_client)
476 {
477  target_hp = rangelim(target_hp, 0, U16_MAX);
478 
479  if (target_hp == m_hp)
480  return; // Nothing to do
481 
482  s32 hp_change = m_env->getScriptIface()->on_player_hpchange(this, target_hp - (s32)m_hp, reason);
483 
484  s32 hp = (s32)m_hp + std::min(hp_change, U16_MAX); // Protection against s32 overflow
485  hp = rangelim(hp, 0, U16_MAX);
486 
487  if (hp > m_prop.hp_max)
488  hp = m_prop.hp_max;
489 
490  if (hp < m_hp && isImmortal())
491  hp = m_hp; // Do not allow immortal players to be damaged
492 
493  // Update properties on death
494  if ((hp == 0) != (m_hp == 0))
495  m_properties_sent = false;
496 
497  if (hp != m_hp) {
498  m_hp = hp;
499  m_env->getGameDef()->HandlePlayerHPChange(this, reason);
500  } else if (from_client)
501  m_env->getGameDef()->SendPlayerHP(this, true);
502 }

581 bool PlayerSAO::checkMovementCheat()
582 {
583  if (m_is_singleplayer ||
584  isAttached() ||
585  g_settings->getBool("disable_anticheat")) {
586  m_last_good_position = m_base_position;
587  return false;
588  }
589 
590  bool cheated = false;
591  /*
592  Check player movements
593 
594  NOTE: Actually the server should handle player physics like the
595  client does and compare player's position to what is calculated
596  on our side. This is required when eg. players fly due to an
597  explosion. Altough a node-based alternative might be possible
598  too, and much more lightweight.
599  */
600 
601  float override_max_H, override_max_V;
602  if (m_max_speed_override_time > 0.0f) {
603  override_max_H = MYMAX(fabs(m_max_speed_override.X), fabs(m_max_speed_override.Z));
604  override_max_V = fabs(m_max_speed_override.Y);
605  } else {
606  override_max_H = override_max_V = 0.0f;
607  }
608 
609  float player_max_walk = 0; // horizontal movement
610  float player_max_jump = 0; // vertical upwards movement
611 
612  if (m_privs.count("fast") != 0)
613  player_max_walk = m_player->movement_speed_fast; // Fast speed
614  else
615  player_max_walk = m_player->movement_speed_walk; // Normal speed
616  player_max_walk *= m_player->physics_override.speed;
617  player_max_walk = MYMAX(player_max_walk, override_max_H);
618 
619  player_max_jump = m_player->movement_speed_jump * m_player->physics_override.jump;
620  // FIXME: Bouncy nodes cause practically unbound increase in Y speed,
621  // until this can be verified correctly, tolerate higher jumping speeds
622  player_max_jump *= 2.0;
623  player_max_jump = MYMAX(player_max_jump, override_max_V);
624 
625  // Don't divide by zero!
626  if (player_max_walk < 0.0001f)
627  player_max_walk = 0.0001f;
628  if (player_max_jump < 0.0001f)
