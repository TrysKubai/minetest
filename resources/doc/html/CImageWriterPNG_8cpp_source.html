62 bool CImageWriterPNG::writeImage(io::IWriteFile* file, IImage* image,u32 param) const
63 {
64  if (!file || !image)
65  return false;
66 
67  // Allocate the png write struct
68  png_structp png_ptr = png_create_write_struct(PNG_LIBPNG_VER_STRING,
69  NULL, (png_error_ptr)png_cpexcept_error, (png_error_ptr)png_cpexcept_warning);
70  if (!png_ptr)
71  {
72  os::Printer::log("PNGWriter: Internal PNG create write struct failure", file->getFileName(), ELL_ERROR);
73  return false;
74  }
75 
76  // Allocate the png info struct
77  png_infop info_ptr = png_create_info_struct(png_ptr);
78  if (!info_ptr)
79  {
80  os::Printer::log("PNGWriter: Internal PNG create info struct failure", file->getFileName(), ELL_ERROR);
81  png_destroy_write_struct(&png_ptr, NULL);
82  return false;
83  }
84 
85  // for proper error handling
86  if (setjmp(png_jmpbuf(png_ptr)))
87  {
88  png_destroy_write_struct(&png_ptr, &info_ptr);
89  return false;
90  }
91 
92  png_set_write_fn(png_ptr, file, user_write_data_fcn, NULL);
93 
94  // Set info
95  switch(image->getColorFormat())
96  {
97  case ECF_A8R8G8B8:
98  case ECF_A1R5G5B5:
99  png_set_IHDR(png_ptr, info_ptr,
100  image->getDimension().Width, image->getDimension().Height,
101  8, PNG_COLOR_TYPE_RGB_ALPHA, PNG_INTERLACE_NONE,
102  PNG_COMPRESSION_TYPE_DEFAULT, PNG_FILTER_TYPE_DEFAULT);
103  break;
104  default:
105  png_set_IHDR(png_ptr, info_ptr,
106  image->getDimension().Width, image->getDimension().Height,
107  8, PNG_COLOR_TYPE_RGB, PNG_INTERLACE_NONE,
108  PNG_COMPRESSION_TYPE_DEFAULT, PNG_FILTER_TYPE_DEFAULT);
109  }
110 
111  s32 lineWidth = image->getDimension().Width;
112  switch(image->getColorFormat())
113  {
114  case ECF_R8G8B8:
115  case ECF_R5G6B5:
116  lineWidth*=3;
117  break;
118  case ECF_A8R8G8B8:
119  case ECF_A1R5G5B5:
120  lineWidth*=4;
121  break;
122  // TODO: Error handling in case of unsupported color format
123  default:
124  break;
125  }
126  u8* tmpImage = new u8[image->getDimension().Height*lineWidth];
127  if (!tmpImage)
128  {
129  os::Printer::log("PNGWriter: Internal PNG create image failure", file->getFileName(), ELL_ERROR);
130  png_destroy_write_struct(&png_ptr, &info_ptr);
131  return false;
132  }
133 
134  u8* data = (u8*)image->getData();
135  switch(image->getColorFormat())
136  {
137  case ECF_R8G8B8:
138  CColorConverter::convert_R8G8B8toR8G8B8(data,image->getDimension().Height*image->getDimension().Width,tmpImage);
139  break;
140  case ECF_A8R8G8B8:
141  CColorConverter::convert_A8R8G8B8toA8R8G8B8(data,image->getDimension().Height*image->getDimension().Width,tmpImage);
142  break;
143  case ECF_R5G6B5:
144  CColorConverter::convert_R5G6B5toR8G8B8(data,image->getDimension().Height*image->getDimension().Width,tmpImage);
145  break;
146  case ECF_A1R5G5B5:
147  CColorConverter::convert_A1R5G5B5toA8R8G8B8(data,image->getDimension().Height*image->getDimension().Width,tmpImage);
148  break;
149  // TODO: Error handling in case of unsupported color format
150  default:
151  os::Printer::log("CImageWriterPNG does not support image format", ColorFormatNames[image->getColorFormat()], ELL_WARNING);
152  png_destroy_write_struct(&png_ptr, &info_ptr);
153  delete [] tmpImage;
154  return false;
155  }
156 
157  // Create array of pointers to rows in image data
158 
159  //Used to point to image rows
160  u8** RowPointers = new png_bytep[image->getDimension().Height];
161  if (!RowPointers)
162  {
163  os::Printer::log("PNGWriter: Internal PNG create row pointers failure", file->getFileName(), ELL_ERROR);
164  png_destroy_write_struct(&png_ptr, &info_ptr);
165  delete [] tmpImage;
166  return false;
167  }
168 
169  data=tmpImage;
170  // Fill array of pointers to rows in image data
171  for (u32 i=0; i<image->getDimension().Height; ++i)
172  {
173  RowPointers[i]=data;
174  data += lineWidth;
175  }
176  // for proper error handling
177  if (setjmp(png_jmpbuf(png_ptr)))
178  {
179  png_destroy_write_struct(&png_ptr, &info_ptr);
180  delete [] RowPointers;
181  delete [] tmpImage;
182  return false;
183  }
184 
185  png_set_rows(png_ptr, info_ptr, RowPointers);
186 
187  if (image->getColorFormat()==ECF_A8R8G8B8 || image->getColorFormat()==ECF_A1R5G5B5)
188  png_write_png(png_ptr, info_ptr, PNG_TRANSFORM_BGR, NULL);
189  else
190  {
191  png_write_png(png_ptr, info_ptr, PNG_TRANSFORM_IDENTITY, NULL);
192  }
193 
194  delete [] RowPointers;
195  delete [] tmpImage;
196  png_destroy_write_struct(&png_ptr, &info_ptr);
197  return true;
198 }
