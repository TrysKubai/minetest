131 bool GUIButton::OnEvent(const SEvent& event)
132 {
133  if (!isEnabled())
134  return IGUIElement::OnEvent(event);
135 
136  switch(event.EventType)
137  {
138  case EET_KEY_INPUT_EVENT:
139  if (event.KeyInput.PressedDown &&
140  (event.KeyInput.Key == KEY_RETURN || event.KeyInput.Key == KEY_SPACE))
141  {
142  if (!IsPushButton)
143  setPressed(true);
144  else
145  setPressed(!Pressed);
146 
147  return true;
148  }
149  if (Pressed && !IsPushButton && event.KeyInput.PressedDown && event.KeyInput.Key == KEY_ESCAPE)
150  {
151  setPressed(false);
152  return true;
153  }
154  else
155  if (!event.KeyInput.PressedDown && Pressed &&
156  (event.KeyInput.Key == KEY_RETURN || event.KeyInput.Key == KEY_SPACE))
157  {
158 
159  if (!IsPushButton)
160  setPressed(false);
161 
162  if (Parent)
163  {
164  ClickShiftState = event.KeyInput.Shift;
165  ClickControlState = event.KeyInput.Control;
166 
167  SEvent newEvent;
168  newEvent.EventType = EET_GUI_EVENT;
169  newEvent.GUIEvent.Caller = this;
170  newEvent.GUIEvent.Element = 0;
171  newEvent.GUIEvent.EventType = EGET_BUTTON_CLICKED;
172  Parent->OnEvent(newEvent);
173  }
174  return true;
175  }
176  break;
177  case EET_GUI_EVENT:
178  if (event.GUIEvent.Caller == this)
179  {
180  if (event.GUIEvent.EventType == EGET_ELEMENT_FOCUS_LOST)
181  {
182  if (!IsPushButton)
183  setPressed(false);
184  FocusTime = (u32)porting::getTimeMs();
185  }
186  else if (event.GUIEvent.EventType == EGET_ELEMENT_FOCUSED)
187  {
188  FocusTime = (u32)porting::getTimeMs();
189  }
190  else if (event.GUIEvent.EventType == EGET_ELEMENT_HOVERED || event.GUIEvent.EventType == EGET_ELEMENT_LEFT)
191  {
192  HoverTime = (u32)porting::getTimeMs();
193  }
194  }
195  break;
196  case EET_MOUSE_INPUT_EVENT:
197  if (event.MouseInput.Event == EMIE_LMOUSE_PRESSED_DOWN)
198  {
199  // Sometimes formspec elements can receive mouse events when the
200  // mouse is outside of the formspec. Thus, we test the position here.
201  if ( !IsPushButton && AbsoluteClippingRect.isPointInside(
202  core::position2d<s32>(event.MouseInput.X, event.MouseInput.Y ))) {
203  Environment->setFocus(this);
204  setPressed(true);
205  }
206 
207  return true;
208  }
209  else
210  if (event.MouseInput.Event == EMIE_LMOUSE_LEFT_UP)
211  {
212  bool wasPressed = Pressed;
213 
214  if ( !AbsoluteClippingRect.isPointInside( core::position2d<s32>(event.MouseInput.X, event.MouseInput.Y ) ) )
215  {
216  if (!IsPushButton)
217  setPressed(false);
218  return true;
219  }
220 
221  if (!IsPushButton)
222  setPressed(false);
223  else
224  {
225  setPressed(!Pressed);
226  }
227 
228  if ((!IsPushButton && wasPressed && Parent) ||
229  (IsPushButton && wasPressed != Pressed))
230  {
231  ClickShiftState = event.MouseInput.Shift;
232  ClickControlState = event.MouseInput.Control;
233 
234  SEvent newEvent;
235  newEvent.EventType = EET_GUI_EVENT;
236  newEvent.GUIEvent.Caller = this;
237  newEvent.GUIEvent.Element = 0;
238  newEvent.GUIEvent.EventType = EGET_BUTTON_CLICKED;
239  Parent->OnEvent(newEvent);
240  }
241 
242  return true;
243  }
244  break;
245  default:
246  break;
247  }
248 
249  return Parent ? Parent->OnEvent(event) : false;
250 }

254 void GUIButton::draw()
255 {
256  if (!IsVisible)
257  return;
258 
259  // PATCH
260  // Track hovered state, if it has changed then we need to update the style.
261  bool hovered = isHovered();
262  bool focused = isFocused();
263  if (hovered != WasHovered || focused != WasFocused) {
264  WasHovered = hovered;
265  WasFocused = focused;
266  setFromState();
267  }
268 
269  GUISkin* skin = dynamic_cast<GUISkin*>(Environment->getSkin());
270  video::IVideoDriver* driver = Environment->getVideoDriver();
271  // END PATCH
272 
273  if (DrawBorder)
274  {
275  if (!Pressed)
276  {
277  // PATCH
278  skin->drawColored3DButtonPaneStandard(this, AbsoluteRect,
279  &AbsoluteClippingRect, Colors);
280  // END PATCH
281  }
282  else
283  {
284  // PATCH
285  skin->drawColored3DButtonPanePressed(this, AbsoluteRect,
286  &AbsoluteClippingRect, Colors);
287  // END PATCH
288  }
289  }
290 
291  const core::position2di buttonCenter(AbsoluteRect.getCenter());
292  // PATCH
293  // The image changes based on the state, so we use the default every time.
294  EGUI_BUTTON_IMAGE_STATE imageState = EGBIS_IMAGE_UP;
295  // END PATCH
296  if ( ButtonImages[(u32)imageState].Texture )
297  {
298  core::position2d<s32> pos(buttonCenter);
299  core::rect<s32> sourceRect(ButtonImages[(u32)imageState].SourceRect);
300  if ( sourceRect.getWidth() == 0 && sourceRect.getHeight() == 0 )
301  sourceRect = core::rect<s32>(core::position2di(0,0), ButtonImages[(u32)imageState].Texture->getOriginalSize());
302 
303  pos.X -= sourceRect.getWidth() / 2;
304  pos.Y -= sourceRect.getHeight() / 2;
305 
306  if ( Pressed )
307  {
308  // Create a pressed-down effect by moving the image when it looks identical to the unpressed state image
309  EGUI_BUTTON_IMAGE_STATE unpressedState = getImageState(false);
310  if ( unpressedState == imageState || ButtonImages[(u32)imageState] == ButtonImages[(u32)unpressedState] )
311  {
312  pos.X += skin->getSize(EGDS_BUTTON_PRESSED_IMAGE_OFFSET_X);
313  pos.Y += skin->getSize(EGDS_BUTTON_PRESSED_IMAGE_OFFSET_Y);
314  }
315  }
316 
317  // PATCH
318  video::ITexture* texture = ButtonImages[(u32)imageState].Texture;
319  video::SColor image_colors[] = { BgColor, BgColor, BgColor, BgColor };
320  if (BgMiddle.getArea() == 0) {
321  driver->draw2DImage(texture,
322  ScaleImage? AbsoluteRect : core::rect<s32>(pos, sourceRect.getSize()),
323  sourceRect, &AbsoluteClippingRect,
324  image_colors, UseAlphaChannel);
325  } else {
326  draw2DImage9Slice(driver, texture,
327  ScaleImage ? AbsoluteRect : core::rect<s32>(pos, sourceRect.getSize()),
328  sourceRect, BgMiddle, &AbsoluteClippingRect, image_colors);
329  }
330  // END PATCH
331  }
332 
333  if (SpriteBank)
334  {
335  if (isEnabled())
336  {
337  core::position2di pos(buttonCenter);
338  // pressed / unpressed animation
339  EGUI_BUTTON_STATE state = Pressed ? EGBS_BUTTON_DOWN : EGBS_BUTTON_UP;
340  drawSprite(state, ClickTime, pos);
341 
342  // focused / unfocused animation
343  state = Environment->hasFocus(this) ? EGBS_BUTTON_FOCUSED : EGBS_BUTTON_NOT_FOCUSED;
344  drawSprite(state, FocusTime, pos);
345 
346  // mouse over / off animation
347  state = isHovered() ? EGBS_BUTTON_MOUSE_OVER : EGBS_BUTTON_MOUSE_OFF;
348  drawSprite(state, HoverTime, pos);
349  }
350  else
351  {
352  // draw disabled
353 // drawSprite(EGBS_BUTTON_DISABLED, 0, pos);
354  }
355  }
356 
357  IGUIElement::draw();
358 }

360 void GUIButton::drawSprite(EGUI_BUTTON_STATE state, u32 startTime, const core::position2di& center)
361 {
362  u32 stateIdx = (u32)state;
363 
364  if (ButtonSprites[stateIdx].Index != -1)
365  {
366  if ( ButtonSprites[stateIdx].Scale )
367  {
368  const video::SColor colors[] = {ButtonSprites[stateIdx].Color,ButtonSprites[stateIdx].Color,ButtonSprites[stateIdx].Color,ButtonSprites[stateIdx].Color};
369  SpriteBank->draw2DSprite(ButtonSprites[stateIdx].Index, AbsoluteRect.UpperLeftCorner,
370  &AbsoluteClippingRect, colors[0], // FIXME: remove [0]
371  porting::getTimeMs()-startTime, ButtonSprites[stateIdx].Loop);
372  }
373  else
374  {
375  SpriteBank->draw2DSprite(ButtonSprites[stateIdx].Index, center,
376  &AbsoluteClippingRect, ButtonSprites[stateIdx].Color, startTime, porting::getTimeMs(),
377  ButtonSprites[stateIdx].Loop, true);
378  }
379  }

389 EGUI_BUTTON_IMAGE_STATE GUIButton::getImageState(bool pressed, const ButtonImage* images) const
390 {
391  // figure state we should have
392  EGUI_BUTTON_IMAGE_STATE state = EGBIS_IMAGE_DISABLED;
393  bool focused = isFocused();
394  bool mouseOver = isHovered();
395  if (isEnabled())
396  {
397  if ( pressed )
398  {
399  if ( focused && mouseOver )
400  state = EGBIS_IMAGE_DOWN_FOCUSED_MOUSEOVER;
401  else if ( focused )
402  state = EGBIS_IMAGE_DOWN_FOCUSED;
403  else if ( mouseOver )
404  state = EGBIS_IMAGE_DOWN_MOUSEOVER;
405  else
406  state = EGBIS_IMAGE_DOWN;
407  }
408  else // !pressed
409  {
410  if ( focused && mouseOver )
411  state = EGBIS_IMAGE_UP_FOCUSED_MOUSEOVER;
412  else if ( focused )
413  state = EGBIS_IMAGE_UP_FOCUSED;
414  else if ( mouseOver )
415  state = EGBIS_IMAGE_UP_MOUSEOVER;
416  else
417  state = EGBIS_IMAGE_UP;
418  }
419  }
420 
421  // find a compatible state that has images
422  while ( state != EGBIS_IMAGE_UP && !images[(u32)state].Texture )
423  {
424  // PATCH
425  switch ( state )
426  {
427  case EGBIS_IMAGE_UP_FOCUSED:
428  state = EGBIS_IMAGE_UP;
429  break;
430  case EGBIS_IMAGE_UP_FOCUSED_MOUSEOVER:
431  state = EGBIS_IMAGE_UP_FOCUSED;
432  break;
433  case EGBIS_IMAGE_DOWN_MOUSEOVER:
434  state = EGBIS_IMAGE_DOWN;
435  break;
436  case EGBIS_IMAGE_DOWN_FOCUSED:
437  state = EGBIS_IMAGE_DOWN;
438  break;
439  case EGBIS_IMAGE_DOWN_FOCUSED_MOUSEOVER:
440  state = EGBIS_IMAGE_DOWN_FOCUSED;
441  break;
442  case EGBIS_IMAGE_DISABLED:
443  if ( pressed )
444  state = EGBIS_IMAGE_DOWN;
445  else
446  state = EGBIS_IMAGE_UP;
447  break;
448  default:
449  state = EGBIS_IMAGE_UP;
450  }
451  // END PATCH
452  }
453 
454  return state;
455 }

681 void GUIButton::setFromStyle(const StyleSpec& style)
682 {
683  bool hovered = (style.getState() & StyleSpec::STATE_HOVERED) != 0;
684  bool pressed = (style.getState() & StyleSpec::STATE_PRESSED) != 0;
685 
686  if (style.isNotDefault(StyleSpec::BGCOLOR)) {
687  setColor(style.getColor(StyleSpec::BGCOLOR));
688 
689  // If we have a propagated hover/press color, we need to automatically
690  // lighten/darken it
691  if (!Styles[style.getState()].isNotDefault(StyleSpec::BGCOLOR)) {
692  if (pressed) {
693  BgColor = multiplyColorValue(BgColor, COLOR_PRESSED_MOD);
694 
695  for (size_t i = 0; i < 4; i++)
696  Colors[i] = multiplyColorValue(Colors[i], COLOR_PRESSED_MOD);
697  } else if (hovered) {
698  BgColor = multiplyColorValue(BgColor, COLOR_HOVERED_MOD);
699 
700  for (size_t i = 0; i < 4; i++)
701  Colors[i] = multiplyColorValue(Colors[i], COLOR_HOVERED_MOD);
702  }
703  }
704 
705  } else {
706  BgColor = video::SColor(255, 255, 255, 255);
707  for (size_t i = 0; i < 4; i++) {
708  video::SColor base =
709  Environment->getSkin()->getColor((gui::EGUI_DEFAULT_COLOR)i);
710  if (pressed) {
711  Colors[i] = multiplyColorValue(base, COLOR_PRESSED_MOD);
712  } else if (hovered) {
713  Colors[i] = multiplyColorValue(base, COLOR_HOVERED_MOD);
714  } else {
715  Colors[i] = base;
716  }
717  }
718  }
719 
720  if (style.isNotDefault(StyleSpec::TEXTCOLOR)) {
721  setOverrideColor(style.getColor(StyleSpec::TEXTCOLOR));
722  } else {
723  setOverrideColor(video::SColor(255,255,255,255));
724  OverrideColorEnabled = false;
725  }
726  setNotClipped(style.getBool(StyleSpec::NOCLIP, false));
727  setDrawBorder(style.getBool(StyleSpec::BORDER, true));
728  setUseAlphaChannel(style.getBool(StyleSpec::ALPHA, true));
729  setOverrideFont(style.getFont());
730 
731  if (style.isNotDefault(StyleSpec::BGIMG)) {
732  video::ITexture *texture = style.getTexture(StyleSpec::BGIMG,
733  getTextureSource());
734  setImage(guiScalingImageButton(
735  Environment->getVideoDriver(), texture,
736  AbsoluteRect.getWidth(), AbsoluteRect.getHeight()));
737  setScaleImage(true);
738  } else {
739  setImage(nullptr);
740  }
741 
742  BgMiddle = style.getRect(StyleSpec::BGIMG_MIDDLE, BgMiddle);
743 
744  // Child padding and offset
745  Padding = style.getRect(StyleSpec::PADDING, core::rect<s32>());
746  Padding = core::rect<s32>(
747  Padding.UpperLeftCorner + BgMiddle.UpperLeftCorner,
748  Padding.LowerRightCorner + BgMiddle.LowerRightCorner);
749 
750  GUISkin* skin = dynamic_cast<GUISkin*>(Environment->getSkin());
751  core::vector2d<s32> defaultPressOffset(
752  skin->getSize(irr::gui::EGDS_BUTTON_PRESSED_IMAGE_OFFSET_X),
753  skin->getSize(irr::gui::EGDS_BUTTON_PRESSED_IMAGE_OFFSET_Y));
754  ContentOffset = style.getVector2i(StyleSpec::CONTENT_OFFSET, isPressed()
755  ? defaultPressOffset
