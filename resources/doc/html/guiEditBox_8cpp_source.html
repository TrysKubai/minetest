211 bool GUIEditBox::OnEvent(const SEvent &event)
212 {
213  if (isEnabled()) {
214  switch (event.EventType) {
215  case EET_GUI_EVENT:
216  if (event.GUIEvent.EventType == EGET_ELEMENT_FOCUS_LOST) {
217  if (event.GUIEvent.Caller == this) {
218  m_mouse_marking = false;
219  setTextMarkers(0, 0);
220  }
221  }
222  break;
223  case EET_KEY_INPUT_EVENT:
224  if (processKey(event))
225  return true;
226  break;
227  case EET_MOUSE_INPUT_EVENT:
228  if (processMouse(event))
229  return true;
230  break;
231  case EET_STRING_INPUT_EVENT:
232  inputString(*event.StringInput.Str);
233  return true;
234  default:
235  break;
236  }
237  }
238 
239  return IGUIElement::OnEvent(event);
240 }

242 bool GUIEditBox::processKey(const SEvent &event)
243 {
244  if (!event.KeyInput.PressedDown)
245  return false;
246 
247  bool text_changed = false;
248  s32 new_mark_begin = m_mark_begin;
249  s32 new_mark_end = m_mark_end;
250 
251  // control shortcut handling
252  if (event.KeyInput.Control) {
253 
254  // german backlash '\' entered with control + '?'
255  if (event.KeyInput.Char == '\\') {
256  inputChar(event.KeyInput.Char);
257  return true;
258  }
259 
260  switch (event.KeyInput.Key) {
261  case KEY_KEY_A:
262  // select all
263  new_mark_begin = 0;
264  new_mark_end = Text.size();
265  break;
266  case KEY_KEY_C:
267  onKeyControlC(event);
268  break;
269  case KEY_KEY_X:
270  text_changed = onKeyControlX(event, new_mark_begin, new_mark_end);
271  break;
272  case KEY_KEY_V:
273  text_changed = onKeyControlV(event, new_mark_begin, new_mark_end);
274  break;
275  case KEY_HOME:
276  // move/highlight to start of text
277  if (event.KeyInput.Shift) {
278  new_mark_end = m_cursor_pos;
279  new_mark_begin = 0;
280  m_cursor_pos = 0;
281  } else {
282  m_cursor_pos = 0;
283  new_mark_begin = 0;
284  new_mark_end = 0;
285  }
286  break;
287  case KEY_END:
288  // move/highlight to end of text
289  if (event.KeyInput.Shift) {
290  new_mark_begin = m_cursor_pos;
291  new_mark_end = Text.size();
292  m_cursor_pos = 0;
293  } else {
294  m_cursor_pos = Text.size();
295  new_mark_begin = 0;
296  new_mark_end = 0;
297  }
298  break;
299  default:
300  return false;
301  }
302  } else {
303  switch (event.KeyInput.Key) {
304  case KEY_END: {
305  s32 p = Text.size();
306  if (m_word_wrap || m_multiline) {
307  p = getLineFromPos(m_cursor_pos);
308  p = m_broken_text_positions[p] +
309  (s32)m_broken_text[p].size();
310  if (p > 0 && (Text[p - 1] == L'\r' ||
311  Text[p - 1] == L'\n'))
312  p -= 1;
313  }
314 
315  if (event.KeyInput.Shift) {
316  if (m_mark_begin == m_mark_end)
317  new_mark_begin = m_cursor_pos;
318 
319  new_mark_end = p;
320  } else {
321  new_mark_begin = 0;
322  new_mark_end = 0;
323  }
324  m_cursor_pos = p;
325  m_blink_start_time = porting::getTimeMs();
326  } break;
327  case KEY_HOME: {
328 
329  s32 p = 0;
330  if (m_word_wrap || m_multiline) {
331  p = getLineFromPos(m_cursor_pos);
332  p = m_broken_text_positions[p];
333  }
334 
335  if (event.KeyInput.Shift) {
336  if (m_mark_begin == m_mark_end)
337  new_mark_begin = m_cursor_pos;
338  new_mark_end = p;
339  } else {
340  new_mark_begin = 0;
341  new_mark_end = 0;
342  }
343  m_cursor_pos = p;
344  m_blink_start_time = porting::getTimeMs();
345  } break;
346  case KEY_RETURN:
347  if (m_multiline) {
348  inputChar(L'\n');
349  } else {
350  calculateScrollPos();
351  sendGuiEvent(EGET_EDITBOX_ENTER);
352  }
353  return true;
354  case KEY_LEFT:
355  if (event.KeyInput.Shift) {
356  if (m_cursor_pos > 0) {
357  if (m_mark_begin == m_mark_end)
358  new_mark_begin = m_cursor_pos;
359 
360  new_mark_end = m_cursor_pos - 1;
361  }
362  } else {
363  new_mark_begin = 0;
364  new_mark_end = 0;
365  }
366 
367  if (m_cursor_pos > 0)
368  m_cursor_pos--;
369  m_blink_start_time = porting::getTimeMs();
370  break;
371  case KEY_RIGHT:
372  if (event.KeyInput.Shift) {
373  if (Text.size() > (u32)m_cursor_pos) {
374  if (m_mark_begin == m_mark_end)
375  new_mark_begin = m_cursor_pos;
376 
377  new_mark_end = m_cursor_pos + 1;
378  }
379  } else {
380  new_mark_begin = 0;
381  new_mark_end = 0;
382  }
383 
384  if (Text.size() > (u32)m_cursor_pos)
385  m_cursor_pos++;
386  m_blink_start_time = porting::getTimeMs();
387  break;
388  case KEY_UP:
389  if (!onKeyUp(event, new_mark_begin, new_mark_end)) {
390  return false;
391  }
392  break;
393  case KEY_DOWN:
394  if (!onKeyDown(event, new_mark_begin, new_mark_end)) {
395  return false;
396  }
397  break;
398  case KEY_BACK:
399  text_changed = onKeyBack(event, new_mark_begin, new_mark_end);
400  break;
401 
402  case KEY_DELETE:
403  text_changed = onKeyDelete(event, new_mark_begin, new_mark_end);
404  break;
405 
406  case KEY_ESCAPE:
407  case KEY_TAB:
408  case KEY_SHIFT:
409  case KEY_F1:
410  case KEY_F2:
411  case KEY_F3:
412  case KEY_F4:
413  case KEY_F5:
414  case KEY_F6:
415  case KEY_F7:
416  case KEY_F8:
417  case KEY_F9:
418  case KEY_F10:
419  case KEY_F11:
420  case KEY_F12:
421  case KEY_F13:
422  case KEY_F14:
423  case KEY_F15:
424  case KEY_F16:
425  case KEY_F17:
426  case KEY_F18:
427  case KEY_F19:
428  case KEY_F20:
429  case KEY_F21:
430  case KEY_F22:
431  case KEY_F23:
432  case KEY_F24:
433  // ignore these keys
434  return false;
435 
436  default:
437  inputChar(event.KeyInput.Char);
438  return true;
439  }
440  }
441 
442  // Set new text markers
443  setTextMarkers(new_mark_begin, new_mark_end);
444 
445  // break the text if it has changed
446  if (text_changed) {
447  breakText();
448  sendGuiEvent(EGET_EDITBOX_CHANGED);
449  }
450 
451  calculateScrollPos();
452 
453  return true;
454 }

456 bool GUIEditBox::onKeyUp(const SEvent &event, s32 &mark_begin, s32 &mark_end)
457 {
458  // clang-format off
459  if (m_multiline || (m_word_wrap && m_broken_text.size() > 1)) {
460  s32 lineNo = getLineFromPos(m_cursor_pos);
461  s32 mb = (m_mark_begin == m_mark_end) ? m_cursor_pos :
462  (m_mark_begin > m_mark_end ? m_mark_begin : m_mark_end);
463  if (lineNo > 0) {
464  s32 cp = m_cursor_pos - m_broken_text_positions[lineNo];
465  if ((s32)m_broken_text[lineNo - 1].size() < cp) {
466  m_cursor_pos = m_broken_text_positions[lineNo - 1] +
467  core::max_((u32)1, m_broken_text[lineNo - 1].size()) - 1;
468  }
469  else
470  m_cursor_pos = m_broken_text_positions[lineNo - 1] + cp;
471  }
472 
473  if (event.KeyInput.Shift) {
474  mark_begin = mb;
475  mark_end = m_cursor_pos;
476  } else {
477  mark_begin = 0;
478  mark_end = 0;
479  }
480 
481  return true;
482  }
483 
484  // clang-format on
485  return false;
486 }

488 bool GUIEditBox::onKeyDown(const SEvent &event, s32 &mark_begin, s32 &mark_end)
489 {
490  // clang-format off
491  if (m_multiline || (m_word_wrap && m_broken_text.size() > 1)) {
492  s32 lineNo = getLineFromPos(m_cursor_pos);
493  s32 mb = (m_mark_begin == m_mark_end) ? m_cursor_pos :
494  (m_mark_begin < m_mark_end ? m_mark_begin : m_mark_end);
495  if (lineNo < (s32)m_broken_text.size() - 1) {
496  s32 cp = m_cursor_pos - m_broken_text_positions[lineNo];
497  if ((s32)m_broken_text[lineNo + 1].size() < cp) {
498  m_cursor_pos = m_broken_text_positions[lineNo + 1] +
499  core::max_((u32)1, m_broken_text[lineNo + 1].size()) - 1;
500  }
501  else
502  m_cursor_pos = m_broken_text_positions[lineNo + 1] + cp;
503  }
504 
505  if (event.KeyInput.Shift) {
506  mark_begin = mb;
507  mark_end = m_cursor_pos;
508  } else {
509  mark_begin = 0;
510  mark_end = 0;
511  }
512 
513  return true;
514  }
515 
516  // clang-format on
517  return false;

533 bool GUIEditBox::onKeyControlX(const SEvent &event, s32 &mark_begin, s32 &mark_end)
534 {
535  // First copy to clipboard
536  onKeyControlC(event);
537 
538  if (!m_writable)
539  return false;
540 
541  if (m_passwordbox || !m_operator || m_mark_begin == m_mark_end)
542  return false;
543 
544  const s32 realmbgn = m_mark_begin < m_mark_end ? m_mark_begin : m_mark_end;
545  const s32 realmend = m_mark_begin < m_mark_end ? m_mark_end : m_mark_begin;
546 
547  // Now remove from box if enabled
548  if (isEnabled()) {
549  // delete
550  core::stringw s;
551  s = Text.subString(0, realmbgn);
552  s.append(Text.subString(realmend, Text.size() - realmend));
553  Text = s;
554 
555  m_cursor_pos = realmbgn;
556  mark_begin = 0;
557  mark_end = 0;
558  return true;
559  }
560 
561  return false;

564 bool GUIEditBox::onKeyControlV(const SEvent &event, s32 &mark_begin, s32 &mark_end)
565 {
566  if (!isEnabled() || !m_writable)
567  return false;
568 
569  // paste from the clipboard
570  if (!m_operator)
571  return false;
572 
573  const s32 realmbgn = m_mark_begin < m_mark_end ? m_mark_begin : m_mark_end;
574  const s32 realmend = m_mark_begin < m_mark_end ? m_mark_end : m_mark_begin;
575 
576  // add new character
577  if (const c8 *p = m_operator->getTextFromClipboard()) {
578  core::stringw inserted_text = utf8_to_stringw(p);
579  if (m_mark_begin == m_mark_end) {
580  // insert text
581  core::stringw s = Text.subString(0, m_cursor_pos);
582  s.append(inserted_text);
583  s.append(Text.subString(
584  m_cursor_pos, Text.size() - m_cursor_pos));
585 
586  if (!m_max || s.size() <= m_max) {
587  Text = s;
588  m_cursor_pos += inserted_text.size();
589  }
590  } else {
591  // replace text
592 
593  core::stringw s = Text.subString(0, realmbgn);
594  s.append(inserted_text);
595  s.append(Text.subString(realmend, Text.size() - realmend));
596 
597  if (!m_max || s.size() <= m_max) {
598  Text = s;
599  m_cursor_pos = realmbgn + inserted_text.size();
600  }
601  }
602  }
603 
604  mark_begin = 0;
605  mark_end = 0;
606  return true;
607 }

609 bool GUIEditBox::onKeyBack(const SEvent &event, s32 &mark_begin, s32 &mark_end)
610 {
611  if (!isEnabled() || Text.empty() || !m_writable)
612  return false;
613 
614  core::stringw s;
615 
616  if (m_mark_begin != m_mark_end) {
617  // delete marked text
618  const s32 realmbgn =
619  m_mark_begin < m_mark_end ? m_mark_begin : m_mark_end;
620  const s32 realmend =
621  m_mark_begin < m_mark_end ? m_mark_end : m_mark_begin;
622 
623  s = Text.subString(0, realmbgn);
624  s.append(Text.subString(realmend, Text.size() - realmend));
625  Text = s;
626 
627  m_cursor_pos = realmbgn;
628  } else {
629  // delete text behind cursor
630  if (m_cursor_pos > 0)
631  s = Text.subString(0, m_cursor_pos - 1);
632  else
633  s = L"";
634  s.append(Text.subString(m_cursor_pos, Text.size() - m_cursor_pos));
635  Text = s;
636  --m_cursor_pos;
637  }
638 
639  if (m_cursor_pos < 0)
640  m_cursor_pos = 0;
641  m_blink_start_time = porting::getTimeMs(); // os::Timer::getTime();
642  mark_begin = 0;
643  mark_end = 0;
644  return true;
645 }

691 void GUIEditBox::inputString(const core::stringw &str)
692 {
693  if (!isEnabled() || !m_writable)
694  return;
695 
696  u32 len = str.size();
697  if (Text.size()+len <= m_max || m_max == 0) {
698  core::stringw s;
699  if (m_mark_begin != m_mark_end) {
700  // replace marked text
701  s32 real_begin = m_mark_begin < m_mark_end ? m_mark_begin : m_mark_end;
702  s32 real_end = m_mark_begin < m_mark_end ? m_mark_end : m_mark_begin;
703 
704  s = Text.subString(0, real_begin);
705  s.append(str);
706  s.append(Text.subString(real_end, Text.size() - real_end));
707  Text = s;
708  m_cursor_pos = real_begin + len;
709  } else {
710  // append string
711  s = Text.subString(0, m_cursor_pos);
712  s.append(str);
713  s.append(Text.subString(m_cursor_pos,
714  Text.size() - m_cursor_pos));
715  Text = s;
716  m_cursor_pos += len;
717  }
718 
719  m_blink_start_time = porting::getTimeMs();
720  setTextMarkers(0, 0);
721  }
722 
723  breakText();
724  sendGuiEvent(EGET_EDITBOX_CHANGED);
725  calculateScrollPos();
726 }

728 bool GUIEditBox::processMouse(const SEvent &event)
729 {
730  switch (event.MouseInput.Event) {
731  case irr::EMIE_LMOUSE_LEFT_UP:
732  if (Environment->hasFocus(this)) {
733  m_cursor_pos = getCursorPos(
734  event.MouseInput.X, event.MouseInput.Y);
735  if (m_mouse_marking) {
736  setTextMarkers(m_mark_begin, m_cursor_pos);
737  }
738  m_mouse_marking = false;
739  calculateScrollPos();
740  return true;
741  }
742  break;
743  case irr::EMIE_MOUSE_MOVED: {
