110 void *ServerThread::run()
111 {
112  BEGIN_DEBUG_EXCEPTION_HANDLER
113 
114  /*
115  * The real business of the server happens on the ServerThread.
116  * How this works:
117  * AsyncRunStep() runs an actual server step as soon as enough time has
118  * passed (dedicated_server_loop keeps track of that).
119  * Receive() blocks at least(!) 30ms waiting for a packet (so this loop
120  * doesn't busy wait) and will process any remaining packets.
121  */
122 
123  try {
124  m_server->AsyncRunStep(true);
125  } catch (con::ConnectionBindFailed &e) {
126  m_server->setAsyncFatalError(e.what());
127  } catch (LuaError &e) {
128  m_server->setAsyncFatalError(e);
129  }
130 
131  while (!stopRequested()) {
132  ScopeProfiler spm(g_profiler, "Server::RunStep() (max)", SPT_MAX);
133  try {
134  m_server->AsyncRunStep();
135 
136  m_server->Receive();
137 
138  } catch (con::PeerNotFoundException &e) {
139  infostream<<"Server: PeerNotFoundException"<<std::endl;
140  } catch (ClientNotFoundException &e) {
141  } catch (con::ConnectionBindFailed &e) {
142  m_server->setAsyncFatalError(e.what());
143  } catch (LuaError &e) {
144  m_server->setAsyncFatalError(e);
145  }
146  }
147 
148  END_DEBUG_EXCEPTION_HANDLER
149 
150  return nullptr;
151 }

196 void Server::ShutdownState::tick(float dtime, Server *server)
197 {
198  if (m_timer <= 0.0f)
199  return;
200 
201  // Timed shutdown
202  static const float shutdown_msg_times[] =
203  {
204  1, 2, 3, 4, 5, 10, 20, 40, 60, 120, 180, 300, 600, 1200, 1800, 3600
205  };
206 
207  // Automated messages
208  if (m_timer < shutdown_msg_times[ARRLEN(shutdown_msg_times) - 1]) {
209  for (float t : shutdown_msg_times) {
210  // If shutdown timer matches an automessage, shot it
211  if (m_timer > t && m_timer - dtime < t) {
212  std::wstring periodicMsg = getShutdownTimerMessage();
213 
214  infostream << wide_to_utf8(periodicMsg).c_str() << std::endl;
215  server->SendChatMessage(PEER_ID_INEXISTENT, periodicMsg);
216  break;
217  }
218  }
219  }
220 
221  m_timer -= dtime;
222  if (m_timer < 0.0f) {
223  m_timer = 0.0f;
224  is_requested = true;
225  }
226 }

601 void Server::AsyncRunStep(bool initial_step)
602 {
603 
604  float dtime;
605  {
606  MutexAutoLock lock1(m_step_dtime_mutex);
607  dtime = m_step_dtime;
608  }
609 
610  {
611  // Send blocks to clients
612  SendBlocks(dtime);
613  }
614 
615  if((dtime < 0.001) && !initial_step)
616  return;
617 
618  ScopeProfiler sp(g_profiler, "Server::AsyncRunStep()", SPT_AVG);
619 
620  {
621  MutexAutoLock lock1(m_step_dtime_mutex);
622  m_step_dtime -= dtime;
623  }
624 
625  /*
626  Update uptime
627  */
628  m_uptime_counter->increment(dtime);
629 
630  handlePeerChanges();
631 
632  /*
633  Update time of day and overall game time
634  */
635  m_env->setTimeOfDaySpeed(g_settings->getFloat("time_speed"));
636 
637  /*
638  Send to clients at constant intervals
639  */
640 
641  m_time_of_day_send_timer -= dtime;
642  if (m_time_of_day_send_timer < 0.0) {
643  m_time_of_day_send_timer = g_settings->getFloat("time_send_interval");
644  u16 time = m_env->getTimeOfDay();
645  float time_speed = g_settings->getFloat("time_speed");
646  SendTimeOfDay(PEER_ID_INEXISTENT, time, time_speed);
647 
648  m_timeofday_gauge->set(time);
649  }
650 
651  {
652  MutexAutoLock lock(m_env_mutex);
653  // Figure out and report maximum lag to environment
654  float max_lag = m_env->getMaxLagEstimate();
655  max_lag *= 0.9998; // Decrease slowly (about half per 5 minutes)
656  if(dtime > max_lag){
657  if(dtime > 0.1 && dtime > max_lag * 2.0)
658  infostream<<"Server: Maximum lag peaked to "<<dtime
659  <<" s"<<std::endl;
660  max_lag = dtime;
661  }
662  m_env->reportMaxLagEstimate(max_lag);
663 
664  // Step environment
665  m_env->step(dtime);
666  }
667 
668  static const float map_timer_and_unload_dtime = 2.92;
669  if(m_map_timer_and_unload_interval.step(dtime, map_timer_and_unload_dtime))
670  {
671  MutexAutoLock lock(m_env_mutex);
672  // Run Map's timers and unload unused data
673  ScopeProfiler sp(g_profiler, "Server: map timer and unload");
674  m_env->getMap().timerUpdate(map_timer_and_unload_dtime,
675  std::max(g_settings->getFloat("server_unload_unused_data_timeout"), 0.0f),
676  -1);
677  }
678 
679  /*
680  Note: Orphan MapBlock ptrs become dangling after this call.
681  */
682  m_env->getServerMap().step();
683 
684  /*
685  Listen to the admin chat, if available

1176 void Server::ProcessData(NetworkPacket *pkt)
1177 {
1178  // Environment is locked first.
1179  MutexAutoLock envlock(m_env_mutex);
1180 
1181  ScopeProfiler sp(g_profiler, "Server: Process network packet (sum)");
1182  u32 peer_id = pkt->getPeerId();
1183 
1184  try {
1185  Address address = getPeerAddress(peer_id);
1186  std::string addr_s = address.serializeString();
1187 
1188  // FIXME: Isn't it a bit excessive to check this for every packet?
1189  if (m_banmanager->isIpBanned(addr_s)) {
1190  std::string ban_name = m_banmanager->getBanName(addr_s);
1191  infostream << "Server: A banned client tried to connect from "
1192  << addr_s << "; banned name was " << ban_name << std::endl;
1193  DenyAccess(peer_id, SERVER_ACCESSDENIED_CUSTOM_STRING,
1194  "Your IP is banned. Banned name was " + ban_name);
1195  return;
1196  }
1197  } catch (con::PeerNotFoundException &e) {
1198  /*
1199  * no peer for this packet found
1200  * most common reason is peer timeout, e.g. peer didn't
1201  * respond for some time, your server was overloaded or
1202  * things like that.
1203  */
1204  infostream << "Server::ProcessData(): Canceling: peer "
1205  << peer_id << " not found" << std::endl;
1206  return;
1207  }
1208 
1209  try {
1210  ToServerCommand command = (ToServerCommand) pkt->getCommand();
1211 
1212  // Command must be handled into ToServerCommandHandler
1213  if (command >= TOSERVER_NUM_MSG_TYPES) {
1214  infostream << "Server: Ignoring unknown command "
1215  << command << std::endl;
1216  return;
1217  }
1218 
1219  if (toServerCommandTable[command].state == TOSERVER_STATE_NOT_CONNECTED) {
1220  handleCommand(pkt);
1221  return;
1222  }
1223 
1224  u8 peer_ser_ver = getClient(peer_id, CS_InitDone)->serialization_version;
1225 
1226  if(peer_ser_ver == SER_FMT_VER_INVALID) {
1227  errorstream << "Server::ProcessData(): Cancelling: Peer"
1228  " serialization format invalid or not initialized."
1229  " Skipping incoming command=" << command << std::endl;
1230  return;
1231  }
1232 
1233  /* Handle commands related to client startup */
1234  if (toServerCommandTable[command].state == TOSERVER_STATE_STARTUP) {
1235  handleCommand(pkt);
1236  return;
1237  }
1238 
1239  if (m_clients.getClientState(peer_id) < CS_Active) {
1240  if (command == TOSERVER_PLAYERPOS) return;
1241 
1242  errorstream << "Got packet command: " << command << " for peer id "
1243  << peer_id << " but client isn't active yet. Dropping packet "
1244  << std::endl;
1245  return;
1246  }
1247 
1248  handleCommand(pkt);
1249  } catch (SendFailedException &e) {
1250  errorstream << "Server::ProcessData(): SendFailedException: "
1251  << "what=" << e.what()
1252  << std::endl;
1253  } catch (PacketError &e) {
1254  actionstream << "Server::ProcessData(): PacketError: "
1255  << "what=" << e.what()
1256  << std::endl;
1257  }
1258 }

3768 {

3770  v3f nodeposf;
3771  if (g_settings->getV3FNoEx("static_spawnpoint", nodeposf))
3772  return nodeposf * BS;

3775  // Limit spawn range to mapgen edges (determined by 'mapgen_limit')
3776  s32 range_max = map.getMapgenParams()->getSpawnRangeMax();
3777 
3778  // Try to find a good place a few times
3779  for (s32 i = 0; i < 4000 && !is_good; i++) {
3780  s32 range = MYMIN(1 + i, range_max);
3781  // We're going to try to throw the player to this position
3782  v2s16 nodepos2d = v2s16(
3783  -range + myrand_range(0, range*2),
3784  -range + myrand_range(0, range*2));

3788  // signify an unsuitable spawn position, or if outside limits.
3789  if (spawn_level >= MAX_MAP_GENERATION_LIMIT ||
3790  spawn_level <= -MAX_MAP_GENERATION_LIMIT)
3791  continue;
3792

3794  // Consecutive empty nodes
3795  s32 air_count = 0;
3796

3798  // avoid obstructions in already-generated mapblocks.
3799  // In ungenerated mapblocks consisting of 'ignore' nodes, there will be
3800  // no obstructions, but mapgen decorations are generated after spawn so
3801  // the player may end up inside one.
3802  for (s32 i = 0; i < 8; i++) {

3818  break;

3821  is_good = true;
3822  break;
3823  }
3824  } else {
3825  air_count = 0;
3826  }
3827  nodepos.Y++;
3828  }
3829  }

3832  return nodeposf;
