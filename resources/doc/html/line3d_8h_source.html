89  vector3d<T> getClosestPoint(const vector3d<T>& point) const
90  {
91  vector3d<T> c = point - start;
92  vector3d<T> v = end - start;
93  T d = (T)v.getLength();
94  v /= d;
95  T t = v.dotProduct(c);
96 
97  if (t < (T)0.0)
98  return start;
99  if (t > d)
100  return end;
101 
102  v *= t;
103  return start + v;
104  }

114  {

116  T c = q.getLength();
