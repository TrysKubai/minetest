189 bool CGUIComboBox::OnEvent(const SEvent& event)
190 {
191  if (isEnabled())
192  {
193  switch(event.EventType)
194  {
195 
196  case EET_KEY_INPUT_EVENT:
197  if (ListBox && event.KeyInput.PressedDown && event.KeyInput.Key == KEY_ESCAPE)
198  {
199  // hide list box
200  openCloseMenu();
201  return true;
202  }
203  else
204  if (event.KeyInput.Key == KEY_RETURN || event.KeyInput.Key == KEY_SPACE)
205  {
206  if (!event.KeyInput.PressedDown)
207  {
208  openCloseMenu();
209  }
210 
211  ListButton->setPressed(ListBox == 0);
212 
213  return true;
214  }
215  else
216  if (event.KeyInput.PressedDown)
217  {
218  s32 oldSelected = Selected;
219  bool absorb = true;
220  switch (event.KeyInput.Key)
221  {
222  case KEY_DOWN:
223  setSelected(Selected+1);
224  break;
225  case KEY_UP:
226  setSelected(Selected-1);
227  break;
228  case KEY_HOME:
229  case KEY_PRIOR:
230  setSelected(0);
231  break;
232  case KEY_END:
233  case KEY_NEXT:
234  setSelected((s32)Items.size()-1);
235  break;
236  default:
237  absorb = false;
238  }
239 
240  if (Selected <0)
241  setSelected(0);
242 
243  if (Selected >= (s32)Items.size())
244  setSelected((s32)Items.size() -1);
245 
246  if (Selected != oldSelected)
247  {
248  sendSelectionChangedEvent();
249  return true;
250  }
251 
252  if (absorb)
253  return true;
254  }
255  break;
256 
257  case EET_GUI_EVENT:
258 
259  switch(event.GUIEvent.EventType)
260  {
261  case EGET_ELEMENT_FOCUS_LOST:
262  if (ListBox &&
263  (Environment->hasFocus(ListBox) || ListBox->isMyChild(event.GUIEvent.Caller) ) &&
264  event.GUIEvent.Element != this &&
265  !isMyChild(event.GUIEvent.Element) &&
266  !ListBox->isMyChild(event.GUIEvent.Element))
267  {
268  openCloseMenu();
269  }
270  break;
271  case EGET_BUTTON_CLICKED:
272  if (event.GUIEvent.Caller == ListButton)
273  {
274  openCloseMenu();
275  return true;
276  }
277  break;
278  case EGET_LISTBOX_SELECTED_AGAIN:
279  case EGET_LISTBOX_CHANGED:
280  if (event.GUIEvent.Caller == ListBox)
281  {
282  setSelected(ListBox->getSelected());
283  if (Selected <0 || Selected >= (s32)Items.size())
284  setSelected(-1);
285  openCloseMenu();
286 
287  sendSelectionChangedEvent();
288  }
289  return true;
290  default:
291  break;
292  }
293  break;
294  case EET_MOUSE_INPUT_EVENT:
295 
296  switch(event.MouseInput.Event)
297  {
298  case EMIE_LMOUSE_PRESSED_DOWN:
299  {
300  core::position2d<s32> p(event.MouseInput.X, event.MouseInput.Y);
301 
302  // send to list box
303  if (ListBox && ListBox->isPointInside(p) && ListBox->OnEvent(event))
304  return true;
305 
306  return true;
307  }
308  case EMIE_LMOUSE_LEFT_UP:
309  {
310  core::position2d<s32> p(event.MouseInput.X, event.MouseInput.Y);
311 
312  // send to list box
313  if (!(ListBox &&
314  ListBox->getAbsolutePosition().isPointInside(p) &&
315  ListBox->OnEvent(event)))
316  {
317  openCloseMenu();
318  }
319  return true;
320  }
321  case EMIE_MOUSE_WHEEL:
322  {
323  // Try scrolling parent first
324  if (IGUIElement::OnEvent(event))
325  return true;
326 
327  s32 oldSelected = Selected;
328  setSelected( Selected + ((event.MouseInput.Wheel < 0) ? 1 : -1));
329 
330  if (Selected <0)
331  setSelected(0);
332 
333  if (Selected >= (s32)Items.size())
334  setSelected((s32)Items.size() -1);
335 
336  if (Selected != oldSelected) {
337  sendSelectionChangedEvent();
338  return true;
339  }
340 
341  return false;
342  }
343  default:
344  break;
345  }
346  break;
347  default:
348  break;
349  }
350  }
351 
352  return IGUIElement::OnEvent(event);
353 }

370 void CGUIComboBox::updateListButtonWidth(s32 width)
371 {
372  if (ListButton->getRelativePosition().getWidth() != width)
373  {
374  core::rect<s32> r;
375  r.UpperLeftCorner.X = RelativeRect.getWidth() - width - 2;
376  r.LowerRightCorner.X = RelativeRect.getWidth() - 2;
377  r.UpperLeftCorner.Y = 2;
378  r.LowerRightCorner.Y = RelativeRect.getHeight() - 2;
379  ListButton->setRelativePosition(r);
380 
381  r.UpperLeftCorner.X = 2;
382  r.UpperLeftCorner.Y = 2;
383  r.LowerRightCorner.X = RelativeRect.getWidth() - (width + 2);
384  r.LowerRightCorner.Y = RelativeRect.getHeight() - 2;
385  SelectedText->setRelativePosition(r);
386  }

390 void CGUIComboBox::draw()
391 {
392  if (!IsVisible)
393  return;
394 
395  IGUISkin* skin = Environment->getSkin();
396 
397  updateListButtonWidth(skin->getSize(EGDS_SCROLLBAR_SIZE));
398 
399  // font changed while the listbox is open?
400  if ( ActiveFont != skin->getFont() && ListBox )
401  {
402  // close and re-open to use new font-size
403  openCloseMenu();
404  openCloseMenu();
405  }
406 
407 
408  IGUIElement *currentFocus = Environment->getFocus();
409  if (currentFocus != LastFocus)
410  {
411  HasFocus = currentFocus == this || isMyChild(currentFocus);
412  LastFocus = currentFocus;
413  }
414 
415  // set colors each time as skin-colors can be changed
416  SelectedText->setBackgroundColor(skin->getColor(EGDC_HIGH_LIGHT));
417  if(isEnabled())
418  {
419  SelectedText->setDrawBackground(HasFocus);
420  SelectedText->setOverrideColor(skin->getColor(HasFocus ? EGDC_HIGH_LIGHT_TEXT : EGDC_BUTTON_TEXT));
421  }
422  else
423  {
424  SelectedText->setDrawBackground(false);
425  SelectedText->setOverrideColor(skin->getColor(EGDC_GRAY_TEXT));
426  }
427  ListButton->setSprite(EGBS_BUTTON_UP, skin->getIcon(EGDI_CURSOR_DOWN), skin->getColor(isEnabled() ? EGDC_WINDOW_SYMBOL : EGDC_GRAY_WINDOW_SYMBOL));
428  ListButton->setSprite(EGBS_BUTTON_DOWN, skin->getIcon(EGDI_CURSOR_DOWN), skin->getColor(isEnabled() ? EGDC_WINDOW_SYMBOL : EGDC_GRAY_WINDOW_SYMBOL));
429 
430 
431  core::rect<s32> frameRect(AbsoluteRect);
432 
433  // draw the border
434 
435  skin->draw3DSunkenPane(this, skin->getColor(EGDC_3D_HIGH_LIGHT),
436  true, true, frameRect, &AbsoluteClippingRect);
437 
438  // draw children
439  IGUIElement::draw();
440 }

443 void CGUIComboBox::openCloseMenu()
444 {
445  if (ListBox)
446  {
447  // close list box
448  Environment->setFocus(this);
449  ListBox->remove();
450  ListBox = 0;
451  }
452  else
453  {
454  if (Parent)
455  Parent->bringToFront(this);
456 
457  IGUISkin* skin = Environment->getSkin();
458  u32 h = Items.size();
459 
460  if (h > getMaxSelectionRows())
461  h = getMaxSelectionRows();
462  if (h == 0)
463  h = 1;
464 
465  ActiveFont = skin->getFont();
466  if (ActiveFont)
467  h *= (ActiveFont->getDimension(L"A").Height + 4);
468 
469  // open list box
470  core::rect<s32> r(0, AbsoluteRect.getHeight(),
471  AbsoluteRect.getWidth(), AbsoluteRect.getHeight() + h);
472 
473  ListBox = new CGUIListBox(Environment, this, -1, r, false, true, true);
474  ListBox->setSubElement(true);
475  ListBox->setNotClipped(true);
476  ListBox->drop();
477 
478  // ensure that list box is always completely visible
479  if (ListBox->getAbsolutePosition().LowerRightCorner.Y > Environment->getRootGUIElement()->getAbsolutePosition().getHeight())
480  ListBox->setRelativePosition( core::rect<s32>(0, -ListBox->getAbsolutePosition().getHeight(), AbsoluteRect.getWidth(), 0) );
481 
482  for (s32 i=0; i<(s32)Items.size(); ++i)
483  ListBox->addItem(Items[i].Name.c_str());
484 
485  ListBox->setSelected(Selected);
486 
487  // set focus
488  Environment->setFocus(ListBox);
489  }
490 }
