108 s32 CGUISpriteBank::addTextureAsSprite(video::ITexture* texture)
109 {
110  if ( !texture )
111  return -1;
112 
113  addTexture(texture);
114  u32 textureIndex = getTextureCount() - 1;
115 
116  u32 rectangleIndex = Rectangles.size();
117  Rectangles.push_back( core::rect<s32>(0,0, texture->getOriginalSize().Width, texture->getOriginalSize().Height) );
118 
119  SGUISprite sprite;
120  sprite.frameTime = 0;
121 
122  SGUISpriteFrame frame;
123  frame.textureNumber = textureIndex;
124  frame.rectNumber = rectangleIndex;
125  sprite.Frames.push_back( frame );
126 
127  Sprites.push_back( sprite );
128 
129  return Sprites.size() - 1;
130 }

132 // get FrameNr for time. return true on exisiting frame
133 inline bool CGUISpriteBank::getFrameNr(u32& frame,u32 index, u32 time, bool loop) const
134 {
135  frame = 0;
136  if (index >= Sprites.size())
137  return false;
138 
139  const SGUISprite& sprite = Sprites[index];
140  const u32 frameSize = sprite.Frames.size();
141  if (frameSize < 1)
142  return false;
143 
144  if (sprite.frameTime)
145  {
146  u32 f = (time / sprite.frameTime);
147  if (loop)
148  frame = f % frameSize;
149  else
150  frame = (f >= frameSize) ? frameSize - 1 : f;
151  }
152  return true;
153 }

156 void CGUISpriteBank::draw2DSprite(u32 index, const core::position2di& pos,
157  const core::rect<s32>* clip, const video::SColor& color,
158  u32 starttime, u32 currenttime, bool loop, bool center)
159 {
160  u32 frame = 0;
161  if (!getFrameNr(frame, index, currenttime - starttime, loop))
162  return;
163 
164  const video::ITexture* tex = getTexture(Sprites[index].Frames[frame].textureNumber);
165  if (!tex)
166  return;
167 
168  const u32 rn = Sprites[index].Frames[frame].rectNumber;
169  if (rn >= Rectangles.size())
170  return;
171 
172  const core::rect<s32>& r = Rectangles[rn];
173  core::position2di p(pos);
174  if (center)
175  {
176  p -= r.getSize() / 2;
177  }
178  Driver->draw2DImage(tex, p, r, clip, color, true);

200 void CGUISpriteBank::draw2DSpriteBatch( const core::array<u32>& indices,
201  const core::array<core::position2di>& pos,
202  const core::rect<s32>* clip,
203  const video::SColor& color,
204  u32 starttime, u32 currenttime,
205  bool loop, bool center)
206 {
207  const irr::u32 drawCount = core::min_<u32>(indices.size(), pos.size());
208 
209  if (!getTextureCount())
210  return;
211  core::array<SDrawBatch> drawBatches(getTextureCount());
212  for (u32 i=0; i < Textures.size(); ++i)
213  {
214  drawBatches.push_back(SDrawBatch());
215  drawBatches[i].positions.reallocate(drawCount);
216  drawBatches[i].sourceRects.reallocate(drawCount);
217  }
218 
219  for (u32 i = 0; i < drawCount; ++i)
220  {
221  const u32 index = indices[i];
222 
223  // work out frame number
224  u32 frame = 0;
225  if (!getFrameNr(frame, index, currenttime - starttime, loop))
226  return;
227 
228  const u32 texNum = Sprites[index].Frames[frame].textureNumber;
229  if (texNum >= drawBatches.size())
230  {
231  continue;
232  }
233  SDrawBatch& currentBatch = drawBatches[texNum];
234 
235  const u32 rn = Sprites[index].Frames[frame].rectNumber;
236  if (rn >= Rectangles.size())
237  return;
238 
239  const core::rect<s32>& r = Rectangles[rn];
240 
241  if (center)
242  {
243  core::position2di p = pos[i];
244  p -= r.getSize() / 2;
245 
246  currentBatch.positions.push_back(p);
247  currentBatch.sourceRects.push_back(r);
248  }
249  else
250  {
251  currentBatch.positions.push_back(pos[i]);
252  currentBatch.sourceRects.push_back(r);
253  }
254  }
255 
256  for(u32 i = 0;i < drawBatches.size();i++)
257  {
258  if(!drawBatches[i].positions.empty() && !drawBatches[i].sourceRects.empty())
259  Driver->draw2DImageBatch(getTexture(i), drawBatches[i].positions,
260  drawBatches[i].sourceRects, clip, color, true);
261  }
262 }
