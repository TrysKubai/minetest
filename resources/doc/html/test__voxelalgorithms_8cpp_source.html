49 void TestVoxelAlgorithms::testVoxelLineIterator()
50 {
51  // Test some lines
52  // Do not test lines that start or end on the border of
53  // two voxels as rounding errors can make the test fail!
54  std::vector<core::line3d<f32> > lines;
55  for (f32 x = -9.1; x < 9; x += 3.124) {
56  for (f32 y = -9.2; y < 9; y += 3.123) {
57  for (f32 z = -9.3; z < 9; z += 3.122) {
58  lines.emplace_back(-x, -y, -z, x, y, z);
59  }
60  }
61  }
62  lines.emplace_back(0, 0, 0, 0, 0, 0);
63  // Test every line
64  std::vector<core::line3d<f32> >::iterator it = lines.begin();
65  for (; it < lines.end(); it++) {
66  core::line3d<f32> l = *it;
67 
68  // Initialize test
69  voxalgo::VoxelLineIterator iterator(l.start, l.getVector());
70 
71  //Test the first voxel
72  v3s16 start_voxel = floatToInt(l.start, 1);
73  UASSERT(iterator.m_current_node_pos == start_voxel);
74 
75  // Values for testing
76  v3s16 end_voxel = floatToInt(l.end, 1);
77  v3s16 voxel_vector = end_voxel - start_voxel;
78  int nodecount = abs(voxel_vector.X) + abs(voxel_vector.Y)
79  + abs(voxel_vector.Z);
80  int actual_nodecount = 0;
81  v3s16 old_voxel = iterator.m_current_node_pos;
82 
83  while (iterator.hasNext()) {
84  iterator.next();
85  actual_nodecount++;
86  v3s16 new_voxel = iterator.m_current_node_pos;
87  // This must be a neighbor of the old voxel
88  UASSERTEQ(f32, (new_voxel - old_voxel).getLengthSQ(), 1);
89  // The line must intersect with the voxel
90  v3f voxel_center = intToFloat(iterator.m_current_node_pos, 1);
91  aabb3f box(voxel_center - v3f(0.5, 0.5, 0.5),
92  voxel_center + v3f(0.5, 0.5, 0.5));
93  UASSERT(box.intersectsWithLine(l));
94  // Update old voxel
95  old_voxel = new_voxel;
96  }
97 
98  // Test last node
99  UASSERT(iterator.m_current_node_pos == end_voxel);
100  // Test node count
101  UASSERTEQ(int, actual_nodecount, nodecount);
102  }
103 }

105 void TestVoxelAlgorithms::testLighting(IGameDef *gamedef)
106 {
107  v3s16 pmin(-32, -32, -32);
108  v3s16 pmax(31, 31, 31);
109  v3s16 bpmin = getNodeBlockPos(pmin), bpmax = getNodeBlockPos(pmax);
110  DummyMap map(gamedef, bpmin, bpmax);
111 
112  // Make a 21x21x21 hollow box centered at the origin.
113  {
114  std::map<v3s16, MapBlock*> modified_blocks;
115  MMVManip vm(&map);
116  vm.initialEmerge(bpmin, bpmax, false);
117  s32 volume = vm.m_area.getVolume();
118  for (s32 i = 0; i < volume; i++)
119  vm.m_data[i] = MapNode(CONTENT_AIR);
120  for (s16 z = -10; z <= 10; z++)
121  for (s16 y = -10; y <= 10; y++)
122  for (s16 x = -10; x <= 10; x++)
123  vm.setNodeNoEmerge(v3s16(x, y, z), MapNode(t_CONTENT_STONE));
124  for (s16 z = -9; z <= 9; z++)
125  for (s16 y = -9; y <= 9; y++)
126  for (s16 x = -9; x <= 9; x++)
127  vm.setNodeNoEmerge(v3s16(x, y, z), MapNode(CONTENT_AIR));
128  voxalgo::blit_back_with_light(&map, &vm, &modified_blocks);
129  }
130 
131  // Place two holes on the edges a torch in the center.
132  {
133  std::map<v3s16, MapBlock*> modified_blocks;
134  map.addNodeAndUpdate(v3s16(-10, 0, 0), MapNode(CONTENT_AIR), modified_blocks);
135  map.addNodeAndUpdate(v3s16(9, 10, -9), MapNode(t_CONTENT_WATER), modified_blocks);
136  map.addNodeAndUpdate(v3s16(0, 0, 0), MapNode(t_CONTENT_TORCH), modified_blocks);
137  map.addNodeAndUpdate(v3s16(-10, 1, 0), MapNode(t_CONTENT_STONE, 153), modified_blocks);
138  }
139 
140  const NodeDefManager *ndef = gamedef->ndef();
141  {
142  MapNode n = map.getNode(v3s16(9, 9, -9));
143  UASSERTEQ(int, n.getLight(LIGHTBANK_NIGHT, ndef->getLightingFlags(n)), 0);
144  UASSERTEQ(int, n.getLight(LIGHTBANK_DAY, ndef->getLightingFlags(n)), 13);
145  }
146  {
147  MapNode n = map.getNode(v3s16(0, 1, 0));
148  UASSERTEQ(int, n.getLight(LIGHTBANK_NIGHT, ndef->getLightingFlags(n)), 12);
149  UASSERTEQ(int, n.getLight(LIGHTBANK_DAY, ndef->getLightingFlags(n)), 12);
150  }
151  {
152  MapNode n = map.getNode(v3s16(-9, -1, 0));
153  UASSERTEQ(int, n.getLight(LIGHTBANK_NIGHT, ndef->getLightingFlags(n)), 3);
154  UASSERTEQ(int, n.getLight(LIGHTBANK_DAY, ndef->getLightingFlags(n)), 12);
155  }
156  {
157  MapNode n = map.getNode(v3s16(-10, 0, 0));
158  UASSERTEQ(int, n.getLight(LIGHTBANK_NIGHT, ndef->getLightingFlags(n)), 3);
159  UASSERTEQ(int, n.getLight(LIGHTBANK_DAY, ndef->getLightingFlags(n)), 14);
160  }
161  {
162  MapNode n = map.getNode(v3s16(-11, 0, 0));
163  UASSERTEQ(int, n.getLight(LIGHTBANK_NIGHT, ndef->getLightingFlags(n)), 2);
164  UASSERTEQ(int, n.getLight(LIGHTBANK_DAY, ndef->getLightingFlags(n)), 15);
165  }
166  {
167  // Test that irrelevant param1 values are not clobbered.
168  MapNode n = map.getNode(v3s16(-10, 1, 0));
169  UASSERTEQ(int, n.getParam1(), 153);
170  }
171 }
