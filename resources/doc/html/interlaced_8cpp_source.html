32 void InitInterlacedMaskStep::run(PipelineContext &context)
33 {
34  video::ITexture *mask = buffer->getTexture(index);
35  if (!mask)
36  return;
37  if (mask == last_mask)
38  return;
39  last_mask = mask;
40 
41  auto size = mask->getSize();
42  u8 *data = reinterpret_cast<u8 *>(mask->lock());
43  for (u32 j = 0; j < size.Height; j++) {
44  u8 val = j % 2 ? 0xff : 0x00;
45  memset(data, val, 4 * size.Width);
46  data += 4 * size.Width;
47  }
48  mask->unlock();
49 }

59  buffer->setTexture(TEXTURE_RIGHT, v2f(1.0f, 0.5f), "3d_render_right", video::ECF_A8R8G8B8);

61 
62  pipeline->addStep<InitInterlacedMaskStep>(buffer, TEXTURE_MASK);
63 
64  auto step3D = pipeline->own(create3DStage(client, v2f(1.0f, 0.5f)));
65 
66  // eyes
67  for (bool right : { false, true }) {
68  pipeline->addStep<OffsetCameraStep>(right);
69  auto output = pipeline->createOwned<TextureBufferOutput>(buffer, right ? TEXTURE_RIGHT : TEXTURE_LEFT);
70  pipeline->addStep<SetRenderTargetStep>(step3D, output);
71  pipeline->addStep(step3D);
72  pipeline->addStep<DrawWield>();
73  pipeline->addStep<MapPostFxStep>();
74  }
75 
76  pipeline->addStep<OffsetCameraStep>(0.0f);
77  IShaderSource *s = client->getShaderSource();
78  u32 shader = s->getShader("3d_interlaced_merge", TILE_MATERIAL_BASIC);
79  video::E_MATERIAL_TYPE material = s->getShaderInfo(shader).material;
80  auto texture_map = { TEXTURE_LEFT, TEXTURE_RIGHT, TEXTURE_MASK };
81  auto merge = pipeline->addStep<PostProcessingStep>(material, texture_map);

85 }
