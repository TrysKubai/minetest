66 void populateAnaglyphPipeline(RenderPipeline *pipeline, Client *client)
67 {
68  // clear depth buffer every time 3D is rendered
69  auto step3D = pipeline->own(create3DStage(client, v2f(1.0)));
70  auto screen = pipeline->createOwned<ScreenTarget>();
71  auto clear_depth = pipeline->createOwned<ClearDepthBufferTarget>(screen);
72  auto enable_override_material = pipeline->createOwned<ConfigureOverrideMaterialTarget>(clear_depth, true);
73  step3D->setRenderTarget(enable_override_material);
74 
75  // left eye
76  pipeline->addStep<OffsetCameraStep>(false);
77  pipeline->addStep<SetColorMaskStep>(video::ECP_RED);
78  pipeline->addStep(step3D);
79 
80  // right eye
81  pipeline->addStep<OffsetCameraStep>(true);
82  pipeline->addStep<SetColorMaskStep>(video::ECP_GREEN | video::ECP_BLUE);
83  pipeline->addStep(step3D);
84 
85  // reset
86  pipeline->addStep<OffsetCameraStep>(0.0f);
87  pipeline->addStep<SetColorMaskStep>(video::ECP_ALL);
88 
89  pipeline->addStep<DrawWield>();
90  pipeline->addStep<MapPostFxStep>();
91  pipeline->addStep<DrawHUD>();
92 }
