39 core::line3d<f32> CSceneCollisionManager::getRayFromScreenCoordinates(
40  const core::position2d<s32> & pos, const ICameraSceneNode* camera)
41 {
42  core::line3d<f32> ln(0,0,0,0,0,0);
43 
44  if (!SceneManager)
45  return ln;
46 
47  if (!camera)
48  camera = SceneManager->getActiveCamera();
49 
50  if (!camera)
51  return ln;
52 
53  const scene::SViewFrustum* f = camera->getViewFrustum();
54 
55  core::vector3df farLeftUp = f->getFarLeftUp();
56  core::vector3df lefttoright = f->getFarRightUp() - farLeftUp;
57  core::vector3df uptodown = f->getFarLeftDown() - farLeftUp;
58 
59  const core::rect<s32>& viewPort = Driver->getViewPort();
60  core::dimension2d<u32> screenSize(viewPort.getWidth(), viewPort.getHeight());
61 
62  f32 dx = pos.X / (f32)screenSize.Width;
63  f32 dy = pos.Y / (f32)screenSize.Height;
64 
65  if (camera->isOrthogonal())
66  ln.start = f->cameraPosition + (lefttoright * (dx-0.5f)) + (uptodown * (dy-0.5f));
67  else
68  ln.start = f->cameraPosition;
69 
70  ln.end = farLeftUp + (lefttoright * dx) + (uptodown * dy);
71 
72  return ln;
73 }
