32 Inventory *ServerInventoryManager::getInventory(const InventoryLocation &loc)
33 {
34  // No m_env check here: allow creation and modification of detached inventories
35 
36  switch (loc.type) {
37  case InventoryLocation::UNDEFINED:
38  case InventoryLocation::CURRENT_PLAYER:
39  break;
40  case InventoryLocation::PLAYER: {
41  if (!m_env)
42  return nullptr;
43 
44  RemotePlayer *player = m_env->getPlayer(loc.name.c_str());
45  if (!player)
46  return nullptr;
47 
48  PlayerSAO *playersao = player->getPlayerSAO();
49  return playersao ? playersao->getInventory() : nullptr;
50  } break;
51  case InventoryLocation::NODEMETA: {
52  if (!m_env)
53  return nullptr;
54 
55  NodeMetadata *meta = m_env->getMap().getNodeMetadata(loc.p);
56  return meta ? meta->getInventory() : nullptr;
57  } break;
58  case InventoryLocation::DETACHED: {
59  auto it = m_detached_inventories.find(loc.name);
60  if (it == m_detached_inventories.end())
61  return nullptr;
62  return it->second.inventory.get();
63  } break;
64  default:
65  sanity_check(false); // abort
66  break;
67  }
68  return nullptr;
69 }

71 void ServerInventoryManager::setInventoryModified(const InventoryLocation &loc)
72 {
73  switch (loc.type) {
74  case InventoryLocation::UNDEFINED:
75  break;
76  case InventoryLocation::PLAYER: {
77 
78  RemotePlayer *player = m_env->getPlayer(loc.name.c_str());
79 
80  if (!player)
81  return;
82 
83  player->setModified(true);
84  player->inventory.setModified(true);
85  // Updates are sent in ServerEnvironment::step()
86  } break;
87  case InventoryLocation::NODEMETA: {
88  MapEditEvent event;
89  event.type = MEET_BLOCK_NODE_METADATA_CHANGED;
90  event.setPositionModified(loc.p);
91  m_env->getMap().dispatchEvent(event);
92  } break;
93  case InventoryLocation::DETACHED: {
94  // Updates are sent in ServerEnvironment::step()
95  } break;
96  default:
97  sanity_check(false); // abort
98  break;
99  }
100 }

102 Inventory *ServerInventoryManager::createDetachedInventory(
103  const std::string &name, IItemDefManager *idef, const std::string &player)
104 {
105  if (m_detached_inventories.count(name) > 0) {
106  infostream << "Server clearing detached inventory \"" << name << "\""
107  << std::endl;
108  m_detached_inventories[name].inventory.reset();
109  } else {
110  infostream << "Server creating detached inventory \"" << name << "\""
111  << std::endl;
112  }
113 
114  auto inv_u = std::make_unique<Inventory>(idef);
115  auto inv = inv_u.get();
116  sanity_check(inv);
117  m_detached_inventories[name].inventory = std::move(inv_u);
118  if (!player.empty()) {
119  m_detached_inventories[name].owner = player;
120 
121  if (!m_env)
122  return inv; // Mods are not loaded yet, ignore
123 
124  RemotePlayer *p = m_env->getPlayer(name.c_str());
125 
126  // if player is connected, send him the inventory
127  if (p && p->getPeerId() != PEER_ID_INEXISTENT) {
128  m_env->getGameDef()->sendDetachedInventory(
129  inv, name, p->getPeerId());
130  }
131  } else {
132  if (!m_env)
133  return inv; // Mods are not loaded yet, don't send
134 
135  // Inventory is for everybody, broadcast
136  m_env->getGameDef()->sendDetachedInventory(inv, name, PEER_ID_INEXISTENT);
137  }
138 
139  return inv;
140 }

142 bool ServerInventoryManager::removeDetachedInventory(const std::string &name)
143 {
144  const auto &inv_it = m_detached_inventories.find(name);
145  if (inv_it == m_detached_inventories.end())
146  return false;
147 
148  inv_it->second.inventory.reset();
149  const std::string &owner = inv_it->second.owner;
150 
151  if (!owner.empty()) {
152  if (m_env) {
153  RemotePlayer *player = m_env->getPlayer(owner.c_str());
154 
155  if (player && player->getPeerId() != PEER_ID_INEXISTENT)
156  m_env->getGameDef()->sendDetachedInventory(
157  nullptr, name, player->getPeerId());
158  }
159  } else if (m_env) {
160  // Notify all players about the change as soon ServerEnv exists
161  m_env->getGameDef()->sendDetachedInventory(
162  nullptr, name, PEER_ID_INEXISTENT);
163  }
164 
165  m_detached_inventories.erase(inv_it);
166 
167  return true;
168 }

170 bool ServerInventoryManager::checkDetachedInventoryAccess(

182 void ServerInventoryManager::sendDetachedInventories(const std::string &peer_name,
183  bool incremental,
184  std::function<void(const std::string &, Inventory *)> apply_cb)
185 {
186  for (const auto &detached_inventory : m_detached_inventories) {
187  const DetachedInventory &dinv = detached_inventory.second;
188  if (incremental) {
189  if (!dinv.inventory || !dinv.inventory->checkModified())
190  continue;
191  }
192 
193  // if we are pushing inventories to a specific player
194  // we should filter to send only the right inventories
195  if (!peer_name.empty()) {
196  const std::string &attached_player = dinv.owner;
197  if (!attached_player.empty() && peer_name != attached_player)
198  continue;
199  }
200 
201  apply_cb(detached_inventory.first, detached_inventory.second.inventory.get());
202  }
