49 TextureOverrideSource::TextureOverrideSource(std::string filepath)
50 {
51  std::ifstream infile(filepath.c_str());
52  std::string line;
53  int line_index = 0;
54  while (std::getline(infile, line)) {
55  line_index++;
56 
57  // Also trim '\r' on DOS-style files
58  line = trim(line);
59 
60  // Ignore empty lines and comments
61  if (line.empty() || line[0] == '#')
62  continue;
63 
64  // Format: mod_name:item_name target1[,...] texture_name.png
65  std::vector<std::string> splitted = str_split(line, ' ');
66  if (splitted.size() < 3) {
67  warningstream << filepath << ":" << line_index
68  << " Syntax error in texture override \"" << line
69  << "\": Expected 3 arguments, got " << splitted.size()
70  << std::endl;
71  continue;
72  }
73 
74  TextureOverride texture_override = {};
75  texture_override.id = splitted[0];
76  texture_override.texture = splitted[2];
77 
78  // Parse the target mask
79  std::vector<std::string> targets = str_split(splitted[1], ',');
80  for (const std::string &target : targets) {
81  std::vector<std::string> kvpair = str_split(target, '=');
82  if (kvpair.size() == 2) {
83  // Key-value pairs
84  if (kvpair[0] == "align_world") {
85  // Global textures
86  texture_override.world_scale = stoi(kvpair[1], 0, U8_MAX);
87  continue;
88  }
89  }
90  if (kvpair.size() == 1) {
91  // Regular override flags
92  auto pair = override_LUT.find(target);
93 
94  if (pair != override_LUT.end()) {
95  texture_override.target |= override_cast(pair->second);
96  continue;
97  }
98  }
99 
100  // Report invalid target
101  warningstream << filepath << ":" << line_index
102  << " Syntax error in texture override \"" << line
103  << "\": Unknown target \"" << target << "\""
104  << std::endl;
105 
106  }
107 
108  // If there are no valid targets, skip adding this override
109  if (texture_override.target == override_cast(OverrideTarget::INVALID)) {
110  continue;
111  }
112 
113  m_overrides.push_back(texture_override);
114  }
115 }

118 std::vector<TextureOverride> TextureOverrideSource::getItemTextureOverrides()
119 {

121 
122  for (const TextureOverride &texture_override : m_overrides) {
123  if (texture_override.hasTarget(OverrideTarget::ITEM_TARGETS))
