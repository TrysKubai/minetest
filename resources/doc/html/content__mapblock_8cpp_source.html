242 enum class QuadDiagonal {
243  Diag02,
244  Diag13,
245 };

247 // Create a cuboid with custom lighting.
248 // tiles - the tiles (materials) to use (for all 6 faces)
249 // tilecount - number of entries in tiles, 1<=tilecount<=6
250 // txc - texture coordinates - this is a list of texture coordinates
251 // for the opposite corners of each face - therefore, there
252 // should be (2+2)*6=24 values in the list. The order of
253 // the faces in the list is up-down-right-left-back-front
254 // (compatible with ContentFeatures).
255 // mask - a bit mask that suppresses drawing of tiles.
256 // tile i will not be drawn if mask & (1 << i) is 1
257 // face_lighter(int face, video::S3DVertex vertices[4]) -> QuadDiagonal -
258 // a callback that will be called for each face drawn to setup vertex colors,
259 // and to choose diagonal to split the quad at.
260 template <typename Fn>
261 void MapblockMeshGenerator::drawCuboid(const aabb3f &box,
262  TileSpec *tiles, int tilecount, const f32 *txc, u8 mask, Fn &&face_lighter)
263 {
264  assert(tilecount >= 1 && tilecount <= 6); // pre-condition
265 
266  auto vertices = setupCuboidVertices(box, txc, tiles, tilecount);
267 
268  for (int k = 0; k < 6; ++k) {
269  if (mask & (1 << k))
270  continue;
271  QuadDiagonal diagonal = face_lighter(k, &vertices[4 * k]);
272  const u16 *indices = diagonal == QuadDiagonal::Diag13 ? quad_indices_13 : quad_indices_02;
273  int tileindex = MYMIN(k, tilecount - 1);
274  collector->append(tiles[tileindex], &vertices[4 * k], 4, indices, 6);
275  }
276 }

278 // Gets the base lighting values for a node
279 void MapblockMeshGenerator::getSmoothLightFrame()
280 {
281  for (int k = 0; k < 8; ++k)
282  frame.sunlight[k] = false;
283  for (int k = 0; k < 8; ++k) {
284  LightPair light(getSmoothLightTransparent(blockpos_nodes + p, light_dirs[k], data));
285  frame.lightsDay[k] = light.lightDay;
286  frame.lightsNight[k] = light.lightNight;
287  // If there is direct sunlight and no ambient occlusion at some corner,
288  // mark the vertical edge (top and bottom corners) containing it.
289  if (light.lightDay == 255) {
290  frame.sunlight[k] = true;
291  frame.sunlight[k ^ 2] = true;
292  }
293  }
294 }

296 // Calculates vertex light level
297 // vertex_pos - vertex position in the node (coordinates are clamped to [0.0, 1.0] or so)
298 LightInfo MapblockMeshGenerator::blendLight(const v3f &vertex_pos)
299 {
300  // Light levels at (logical) node corners are known. Here,
301  // trilinear interpolation is used to calculate light level
302  // at a given point in the node.
303  f32 x = core::clamp(vertex_pos.X / BS + 0.5, 0.0 - SMOOTH_LIGHTING_OVERSIZE, 1.0 + SMOOTH_LIGHTING_OVERSIZE);
304  f32 y = core::clamp(vertex_pos.Y / BS + 0.5, 0.0 - SMOOTH_LIGHTING_OVERSIZE, 1.0 + SMOOTH_LIGHTING_OVERSIZE);
305  f32 z = core::clamp(vertex_pos.Z / BS + 0.5, 0.0 - SMOOTH_LIGHTING_OVERSIZE, 1.0 + SMOOTH_LIGHTING_OVERSIZE);
306  f32 lightDay = 0.0; // daylight
307  f32 lightNight = 0.0;
308  f32 lightBoosted = 0.0; // daylight + direct sunlight, if any
309  for (int k = 0; k < 8; ++k) {
310  f32 dx = (k & 4) ? x : 1 - x;
311  f32 dy = (k & 2) ? y : 1 - y;
312  f32 dz = (k & 1) ? z : 1 - z;
313  // Use direct sunlight (255), if any; use daylight otherwise.
314  f32 light_boosted = frame.sunlight[k] ? 255 : frame.lightsDay[k];
315  lightDay += dx * dy * dz * frame.lightsDay[k];
316  lightNight += dx * dy * dz * frame.lightsNight[k];
317  lightBoosted += dx * dy * dz * light_boosted;
318  }
319  return LightInfo{lightDay, lightNight, lightBoosted};
320 }

341 void MapblockMeshGenerator::generateCuboidTextureCoords(const aabb3f &box, f32 *coords)
342 {
343  f32 tx1 = (box.MinEdge.X / BS) + 0.5;
344  f32 ty1 = (box.MinEdge.Y / BS) + 0.5;
345  f32 tz1 = (box.MinEdge.Z / BS) + 0.5;
346  f32 tx2 = (box.MaxEdge.X / BS) + 0.5;
347  f32 ty2 = (box.MaxEdge.Y / BS) + 0.5;
348  f32 tz2 = (box.MaxEdge.Z / BS) + 0.5;
349  f32 txc[24] = {
350  tx1, 1 - tz2, tx2, 1 - tz1, // up
351  tx1, tz1, tx2, tz2, // down
352  tz1, 1 - ty2, tz2, 1 - ty1, // right
353  1 - tz2, 1 - ty2, 1 - tz1, 1 - ty1, // left
354  1 - tx2, 1 - ty2, 1 - tx1, 1 - ty1, // back
355  tx1, 1 - ty2, tx2, 1 - ty1, // front
356  };
357  for (int i = 0; i != 24; ++i)
358  coords[i] = txc[i];
359 }

432 void MapblockMeshGenerator::drawSolidNode()
433 {
434  u8 faces = 0; // k-th bit will be set if k-th face is to be drawn.
435  static const v3s16 tile_dirs[6] = {
436  v3s16(0, 1, 0),
437  v3s16(0, -1, 0),
438  v3s16(1, 0, 0),
439  v3s16(-1, 0, 0),
440  v3s16(0, 0, 1),
441  v3s16(0, 0, -1)
442  };
443  TileSpec tiles[6];
444  u16 lights[6];
445  content_t n1 = n.getContent();
446  for (int face = 0; face < 6; face++) {
447  v3s16 p2 = blockpos_nodes + p + tile_dirs[face];
448  MapNode neighbor = data->m_vmanip.getNodeNoEx(p2);
449  content_t n2 = neighbor.getContent();
450  bool backface_culling = f->drawtype == NDT_NORMAL;
451  if (n2 == n1)
452  continue;
453  if (n2 == CONTENT_IGNORE)
454  continue;
455  if (n2 != CONTENT_AIR) {
456  const ContentFeatures &f2 = nodedef->get(n2);
457  if (f2.solidness == 2)
458  continue;
459  if (f->drawtype == NDT_LIQUID) {
460  if (n2 == nodedef->getId(f->liquid_alternative_flowing))
461  continue;
462  if (n2 == nodedef->getId(f->liquid_alternative_source))
463  continue;
464  backface_culling = f2.solidness >= 1;
465  }
466  }
467  faces |= 1 << face;
468  getTile(tile_dirs[face], &tiles[face]);
469  for (auto &layer : tiles[face].layers) {
470  if (backface_culling)
471  layer.material_flags |= MATERIAL_FLAG_BACKFACE_CULLING;
472  else
473  layer.material_flags &= ~MATERIAL_FLAG_BACKFACE_CULLING;
474  layer.material_flags |= MATERIAL_FLAG_TILEABLE_HORIZONTAL;
475  layer.material_flags |= MATERIAL_FLAG_TILEABLE_VERTICAL;
476  }
477  if (!data->m_smooth_lighting) {
478  lights[face] = getFaceLight(n, neighbor, nodedef);
479  }
480  }
481  if (!faces)
482  return;
483  u8 mask = faces ^ 0b0011'1111; // k-th bit is set if k-th face is to be *omitted*, as expected by cuboid drawing functions.
484  origin = intToFloat(p, BS);
485  auto box = aabb3f(v3f(-0.5 * BS), v3f(0.5 * BS));
486  f32 texture_coord_buf[24];
487  box.MinEdge += origin;
488  box.MaxEdge += origin;
489  generateCuboidTextureCoords(box, texture_coord_buf);
490  if (data->m_smooth_lighting) {
491  LightPair lights[6][4];
492  for (int face = 0; face < 6; ++face) {
493  for (int k = 0; k < 4; k++) {
494  v3s16 corner = light_dirs[light_indices[face][k]];
495  lights[face][k] = LightPair(getSmoothLightSolid(blockpos_nodes + p, tile_dirs[face], corner, data));
496  }
497  }

500  auto final_lights = lights[face];
501  for (int j = 0; j < 4; j++) {
502  video::S3DVertex &vertex = vertices[j];
503  vertex.Color = encode_light(final_lights[j], f->light_source);

525 u8 MapblockMeshGenerator::getNodeBoxMask(aabb3f box, u8 solid_neighbors, u8 sametype_neighbors) const
526 {
527  const f32 NODE_BOUNDARY = 0.5 * BS;
528 
529  // For an oversized nodebox, return immediately
530  if (box.MaxEdge.X > NODE_BOUNDARY ||
531  box.MinEdge.X < -NODE_BOUNDARY ||
532  box.MaxEdge.Y > NODE_BOUNDARY ||
533  box.MinEdge.Y < -NODE_BOUNDARY ||
534  box.MaxEdge.Z > NODE_BOUNDARY ||
535  box.MinEdge.Z < -NODE_BOUNDARY)
536  return 0;
537 
538  // We can skip faces at node boundary if the matching neighbor is solid
539  u8 solid_mask =
540  (box.MaxEdge.Y == NODE_BOUNDARY ? 1 : 0) |
541  (box.MinEdge.Y == -NODE_BOUNDARY ? 2 : 0) |
542  (box.MaxEdge.X == NODE_BOUNDARY ? 4 : 0) |
543  (box.MinEdge.X == -NODE_BOUNDARY ? 8 : 0) |
544  (box.MaxEdge.Z == NODE_BOUNDARY ? 16 : 0) |
545  (box.MinEdge.Z == -NODE_BOUNDARY ? 32 : 0);
546 
547  u8 sametype_mask = 0;
548  if (f->alpha == AlphaMode::ALPHAMODE_OPAQUE) {
549  // In opaque nodeboxes, faces on opposite sides can cancel
550  // each other out if there is a matching neighbor of the same type
551  sametype_mask =
552  ((solid_mask & 3) == 3 ? 3 : 0) |
553  ((solid_mask & 12) == 12 ? 12 : 0) |
554  ((solid_mask & 48) == 48 ? 48 : 0);
555  }
556 
557  // Combine masks with actual neighbors to get the faces to be skipped
558  return (solid_mask & solid_neighbors) | (sametype_mask & sametype_neighbors);

643 f32 MapblockMeshGenerator::getCornerLevel(int i, int k)
644 {
645  float sum = 0;
646  int count = 0;
647  int air_count = 0;
648  for (int dk = 0; dk < 2; dk++)
649  for (int di = 0; di < 2; di++) {
650  NeighborData &neighbor_data = liquid_neighbors[k + dk][i + di];
651  content_t content = neighbor_data.content;
652 
653  // If top is liquid, draw starting from top of node
654  if (neighbor_data.top_is_same_liquid)
655  return 0.5 * BS;
656 
657  // Source always has the full height
658  if (content == c_source)
659  return 0.5 * BS;
660

662  if (content == c_flowing) {
663  sum += neighbor_data.level;
664  count++;
665  } else if (content == CONTENT_AIR) {
666  air_count++;
667  }
668  }
669  if (air_count >= 2)
670  return -0.5 * BS + 0.2;
671  if (count > 0)
672  return sum / count;
673  return 0;
