186  void COpenGL3DriverBase::initQuadsIndices(int max_vertex_count)
187  {
188  int max_quad_count = max_vertex_count / 4;
189  std::vector<GLushort> QuadsIndices;
190  QuadsIndices.reserve(6 * max_quad_count);
191  for (int k = 0; k < max_quad_count; k++) {
192  QuadsIndices.push_back(4 * k + 0);
193  QuadsIndices.push_back(4 * k + 1);
194  QuadsIndices.push_back(4 * k + 2);
195  QuadsIndices.push_back(4 * k + 0);
196  QuadsIndices.push_back(4 * k + 2);
197  QuadsIndices.push_back(4 * k + 3);
198  }
199  glGenBuffers(1, &QuadIndexBuffer);
200  glBindBuffer(GL_ARRAY_BUFFER, QuadIndexBuffer);
201  glBufferData(GL_ARRAY_BUFFER, sizeof(QuadsIndices[0]) * QuadsIndices.size(), QuadsIndices.data(), GL_STATIC_DRAW);
202  glBindBuffer(GL_ARRAY_BUFFER, 0);
203  QuadIndexCount = QuadsIndices.size();
204  }

283  void COpenGL3DriverBase::loadShaderData(const io::path& vertexShaderName, const io::path& fragmentShaderName, c8** vertexShaderData, c8** fragmentShaderData)
284  {
285  io::path vsPath(OGLES2ShaderPath);
286  vsPath += vertexShaderName;
287 
288  io::path fsPath(OGLES2ShaderPath);
289  fsPath += fragmentShaderName;
290 
291  *vertexShaderData = 0;
292  *fragmentShaderData = 0;
293 
294  io::IReadFile* vsFile = FileSystem->createAndOpenFile(vsPath);
295  if ( !vsFile )
296  {
297  core::stringw warning(L"Warning: Missing shader files needed to simulate fixed function materials:\n");
298  warning += core::stringw(vsPath) + L"\n";
299  warning += L"Shaderpath can be changed in SIrrCreationParamters::OGLES2ShaderPath";
300  os::Printer::log(warning.c_str(), ELL_WARNING);
301  return;
302  }
303 
304  io::IReadFile* fsFile = FileSystem->createAndOpenFile(fsPath);
305  if ( !fsFile )
306  {
307  core::stringw warning(L"Warning: Missing shader files needed to simulate fixed function materials:\n");
308  warning += core::stringw(fsPath) + L"\n";
309  warning += L"Shaderpath can be changed in SIrrCreationParamters::OGLES2ShaderPath";
310  os::Printer::log(warning.c_str(), ELL_WARNING);
311  return;
312  }
313 
314  long size = vsFile->getSize();
315  if (size)
316  {
317  *vertexShaderData = new c8[size+1];
318  vsFile->read(*vertexShaderData, size);
319  (*vertexShaderData)[size] = 0;
320  }
321 
322  size = fsFile->getSize();
323  if (size)
324  {
325  // if both handles are the same we must reset the file
326  if (fsFile == vsFile)
327  fsFile->seek(0);
328 
329  *fragmentShaderData = new c8[size+1];
330  fsFile->read(*fragmentShaderData, size);
331  (*fragmentShaderData)[size] = 0;
332  }
333 
334  vsFile->drop();
335  fsFile->drop();
336  }

470  bool COpenGL3DriverBase::updateVertexHardwareBuffer(SHWBufferLink_opengl *HWBuffer)
471  {
472  if (!HWBuffer)
473  return false;
474 
475  const scene::IMeshBuffer* mb = HWBuffer->MeshBuffer;
476  const void* vertices = mb->getVertices();
477  const u32 vertexCount = mb->getVertexCount();
478  const E_VERTEX_TYPE vType = mb->getVertexType();
479  const u32 vertexSize = getVertexPitchFromType(vType);
480 
481  const void *buffer = vertices;
482  size_t bufferSize = vertexSize * vertexCount;
483 
484  //get or create buffer
485  bool newBuffer = false;
486  if (!HWBuffer->vbo_verticesID)
487  {
488  glGenBuffers(1, &HWBuffer->vbo_verticesID);
489  if (!HWBuffer->vbo_verticesID) return false;
490  newBuffer = true;
491  }
492  else if (HWBuffer->vbo_verticesSize < bufferSize)
493  {
494  newBuffer = true;
495  }
496 
497  glBindBuffer(GL_ARRAY_BUFFER, HWBuffer->vbo_verticesID);
498 
499  // copy data to graphics card
500  if (!newBuffer)
501  glBufferSubData(GL_ARRAY_BUFFER, 0, bufferSize, buffer);
502  else
503  {
504  HWBuffer->vbo_verticesSize = bufferSize;
505 
506  if (HWBuffer->Mapped_Vertex == scene::EHM_STATIC)
507  glBufferData(GL_ARRAY_BUFFER, bufferSize, buffer, GL_STATIC_DRAW);
508  else
509  glBufferData(GL_ARRAY_BUFFER, bufferSize, buffer, GL_DYNAMIC_DRAW);
510  }
511 
512  glBindBuffer(GL_ARRAY_BUFFER, 0);
513 
514  return (!testGLError(__LINE__));
515  }

582  bool COpenGL3DriverBase::updateHardwareBuffer(SHWBufferLink *HWBuffer)
583  {
584  if (!HWBuffer)
585  return false;
586 
587  if (HWBuffer->Mapped_Vertex != scene::EHM_NEVER)
588  {
589  if (HWBuffer->ChangedID_Vertex != HWBuffer->MeshBuffer->getChangedID_Vertex()
590  || !static_cast<SHWBufferLink_opengl*>(HWBuffer)->vbo_verticesID)
591  {
592

594 
595  if (!updateVertexHardwareBuffer(static_cast<SHWBufferLink_opengl*>(HWBuffer)))
596  return false;
597  }
598  }
599 
600  if (HWBuffer->Mapped_Index != scene::EHM_NEVER)
601  {
602  if (HWBuffer->ChangedID_Index != HWBuffer->MeshBuffer->getChangedID_Index()
603  || !static_cast<SHWBufferLink_opengl*>(HWBuffer)->vbo_indicesID)

1462  void COpenGL3DriverBase::setTextureRenderStates(const SMaterial& material, bool resetAllRenderstates)
1463  {

1465

1467  {

1469 
1470  if (!tmpTexture)
1471  continue;
1472

1474 
1475  CacheHandler->setActiveTexture(GL_TEXTURE0 + i);
1476

1478  tmpTexture->getStatesCache().IsCached = false;

1481  material.TextureLayer[i].TrilinearFilter != tmpTexture->getStatesCache().TrilinearFilter)
1482  {
1483  glTexParameteri(tmpTextureType, GL_TEXTURE_MAG_FILTER,
1484  (material.TextureLayer[i].BilinearFilter || material.TextureLayer[i].TrilinearFilter) ? GL_LINEAR : GL_NEAREST);

1497  material.TextureLayer[i].BilinearFilter ? GL_LINEAR_MIPMAP_NEAREST :
1498  GL_NEAREST_MIPMAP_NEAREST);

1500  tmpTexture->getStatesCache().BilinearFilter = material.TextureLayer[i].BilinearFilter;

1503  }
1504  }
