26 // The writer uses a 4k buffer and flushes to disk each time it's filled

42 {

49 // flush to disk and reset buffer
50 static boolean jpeg_empty_output_buffer(j_compress_ptr cinfo)
51 {
52  mem_dest_ptr dest = (mem_dest_ptr) cinfo->dest;
53 
54  // for now just exit upon file error
55  if (dest->file->write(dest->buffer, OUTPUT_BUF_SIZE) != OUTPUT_BUF_SIZE)
56  ERREXIT (cinfo, JERR_FILE_WRITE);
57 
58  dest->pub.next_output_byte = dest->buffer;
59  dest->pub.free_in_buffer = OUTPUT_BUF_SIZE;
60 
61  return TRUE;
62 }

65 static void jpeg_term_destination(j_compress_ptr cinfo)
66 {
67  mem_dest_ptr dest = (mem_dest_ptr) cinfo->dest;
68  const size_t datacount = OUTPUT_BUF_SIZE - dest->pub.free_in_buffer;
69  // for now just exit upon file error
70  if (dest->file->write(dest->buffer, datacount) != datacount)
71  ERREXIT (cinfo, JERR_FILE_WRITE);
72 }

75 // set up buffer data
76 static void jpeg_file_dest(j_compress_ptr cinfo, io::IWriteFile* file)
77 {
78  if (cinfo->dest == NULL)
79  { /* first time for this JPEG object? */
80  cinfo->dest = (struct jpeg_destination_mgr *)
81  (*cinfo->mem->alloc_small) ((j_common_ptr) cinfo,
82  JPOOL_PERMANENT,
83  sizeof(mem_destination_mgr));
84  }
85 
86  mem_dest_ptr dest = (mem_dest_ptr) cinfo->dest; /* for casting */
87 
88  /* Initialize method pointers */
89  dest->pub.init_destination = jpeg_init_destination;
90  dest->pub.empty_output_buffer = jpeg_empty_output_buffer;
91  dest->pub.term_destination = jpeg_term_destination;
92 
93  /* Initialize private member */
94  dest->file = file;
95 }

98 /* write_JPEG_memory: store JPEG compressed image into memory.
99 */
100 static bool writeJPEGFile(io::IWriteFile* file, IImage* image, u32 quality)
101 {
102  void (*format)(const void*, s32, void*) = 0;
103  switch( image->getColorFormat () )
104  {
105  case ECF_R8G8B8:
106  format = CColorConverter::convert_R8G8B8toR8G8B8;
107  break;
108  case ECF_A8R8G8B8:
109  format = CColorConverter::convert_A8R8G8B8toR8G8B8;
110  break;
111  case ECF_A1R5G5B5:
112  format = CColorConverter::convert_A1R5G5B5toB8G8R8;
113  break;
114  case ECF_R5G6B5:
115  format = CColorConverter::convert_R5G6B5toR8G8B8;
116  break;
117  default:
118  os::Printer::log("writeJPEGFile does not support image format", ColorFormatNames[image->getColorFormat()], ELL_WARNING);
119  break;
120  }
121 
122  // couldn't find a color converter
123  if ( 0 == format )
124  return false;
125 
126  const core::dimension2du dim = image->getDimension();
127 
128  struct jpeg_compress_struct cinfo;
129  struct jpeg_error_mgr jerr;
130  cinfo.err = jpeg_std_error(&jerr);
131 
132  jpeg_create_compress(&cinfo);
133  jpeg_file_dest(&cinfo, file);
134  cinfo.image_width = dim.Width;
135  cinfo.image_height = dim.Height;
136  cinfo.input_components = 3;
137  cinfo.in_color_space = JCS_RGB;
138 
139  jpeg_set_defaults(&cinfo);
140 
141  if ( 0 == quality )
142  quality = 75;
143 
144  jpeg_set_quality(&cinfo, quality, TRUE);
145  jpeg_start_compress(&cinfo, TRUE);
146 
147  u8 * dest = new u8[dim.Width*3];
148 
149  if (dest)
150  {
151  const u32 pitch = image->getPitch();
152  JSAMPROW row_pointer[1]; /* pointer to JSAMPLE row[s] */
153  row_pointer[0] = dest;
154 
155  u8* src = (u8*)image->getData();
156 
157  while (cinfo.next_scanline < cinfo.image_height)
158  {
159  // convert next line
160  format( src, dim.Width, dest );
161  src += pitch;
162  jpeg_write_scanlines(&cinfo, row_pointer, 1);
163  }
164 
165  delete [] dest;
166 
167  /* Step 6: Finish compression */
168  jpeg_finish_compress(&cinfo);
169  }
170 
171  /* Step 7: Destroy */
172  jpeg_destroy_compress(&cinfo);
173 
174  return (dest != 0);
175 }
