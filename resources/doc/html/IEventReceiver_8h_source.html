464 
465  AXIS_X = 0, // e.g. analog stick 1 left to right
466  AXIS_Y, // e.g. analog stick 1 top to bottom

468  AXIS_R, // e.g. rudder, or analog 2 stick 2 top to bottom

472  };

477 
486  s16 Axis[NUMBER_OF_AXES];
487 
493  u16 POV;

496

501  bool IsButtonPressed(u32 button) const
502  {
503  if(button >= (u32)NUMBER_OF_BUTTONS)
504  return false;
505 
506  return (ButtonStates & (1 << button)) ? true : false;
