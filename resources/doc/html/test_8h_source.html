74 #define UASSERTCMP(T, CMP, actual, expected) { \
75  T a = (actual); \
76  T e = (expected); \
77  if (!(a CMP e)) { \

83  << " actual : " << a << std::endl << " expected: " \
84  << e << std::endl; \

86  } \
