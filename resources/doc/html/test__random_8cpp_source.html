138 void TestRandom::testPcgRandomNormalDist()
139 {
140  static const int max = 120;
141  static const int min = -120;
142  static const int num_trials = 20;
143  static const u32 num_samples = 61000;
144  s32 bins[max - min + 1];
145  memset(bins, 0, sizeof(bins));
146 
147  PcgRandom r(486179 + (int)time(NULL));
148 
149  for (u32 i = 0; i != num_samples; i++) {
150  s32 randval = r.randNormalDist(min, max, num_trials);
151  UASSERT(randval <= max);
152  UASSERT(randval >= min);
153  bins[randval - min]++;
154  }
155 
156  // Note that here we divide variance by the number of trials;
157  // this is because variance is a biased estimator.
158  int range = (max - min + 1);
159  float mean = (max + min) / 2;
160  float variance = ((range * range - 1) / 12) / num_trials;
161  float stddev = std::sqrt(variance);
162 
163  static const float prediction_intervals[] = {
164  0.68269f, // 1.0
165  0.86639f, // 1.5
166  0.95450f, // 2.0
167  0.98758f, // 2.5
168  0.99730f, // 3.0
169  };
170 
172  for (u32 i = 0; i != ARRLEN(prediction_intervals); i++) {
173  float deviations = i / 2.f + 1.f;
174  int lbound = myround(mean - deviations * stddev);
175  int ubound = myround(mean + deviations * stddev);
176  UASSERT(lbound >= min);
177  UASSERT(ubound <= max);
178 
179  int accum = 0;
180  for (int j = lbound; j != ubound; j++)
181  accum += bins[j - min];
182 
183  float actual = (float)accum / num_samples;
184  UASSERT(std::fabs(actual - prediction_intervals[i]) < 0.02f);
185  }
186 }
