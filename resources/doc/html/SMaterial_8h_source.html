120  inline f32 pack_textureBlendFunc(const E_BLEND_FACTOR srcFact, const E_BLEND_FACTOR dstFact,
121  const E_MODULATE_FUNC modulate=EMFN_MODULATE_1X, const u32 alphaSource=EAS_TEXTURE)
122  {
123  const u32 tmp = (alphaSource << 20) | (modulate << 16) | (srcFact << 12) | (dstFact << 8) | (srcFact << 4) | dstFact;
124  return FR(tmp);
125  }

139  inline void unpack_textureBlendFunc(E_BLEND_FACTOR &srcFact, E_BLEND_FACTOR &dstFact,
140  E_MODULATE_FUNC &modulo, u32& alphaSource, const f32 param)
141  {
142  const u32 state = IR(param);
143  alphaSource = (state & 0x00F00000) >> 20;
144  modulo = E_MODULATE_FUNC( ( state & 0x000F0000 ) >> 16 );
145  srcFact = E_BLEND_FACTOR ( ( state & 0x000000F0 ) >> 4 );
146  dstFact = E_BLEND_FACTOR ( ( state & 0x0000000F ) );
147  }

520  void setFlag(E_MATERIAL_FLAG flag, bool value)
521  {
522  switch (flag)
523  {
524  case EMF_WIREFRAME:
525  Wireframe = value; break;
526  case EMF_POINTCLOUD:
527  PointCloud = value; break;
528  case EMF_GOURAUD_SHADING:
529  GouraudShading = value; break;
530  case EMF_LIGHTING:
531  Lighting = value; break;
532  case EMF_ZBUFFER:
533  ZBuffer = value; break;
534  case EMF_ZWRITE_ENABLE:
535  ZWriteEnable = value ? EZW_AUTO : EZW_OFF; break;
536  case EMF_BACK_FACE_CULLING:
537  BackfaceCulling = value; break;
538  case EMF_FRONT_FACE_CULLING:
539  FrontfaceCulling = value; break;
540  case EMF_BILINEAR_FILTER:
541  {
542  for (u32 i=0; i<MATERIAL_MAX_TEXTURES; ++i)
543  TextureLayer[i].BilinearFilter = value;
544  }
545  break;
546  case EMF_TRILINEAR_FILTER:
547  {
548  for (u32 i=0; i<MATERIAL_MAX_TEXTURES; ++i)
549  TextureLayer[i].TrilinearFilter = value;
550  }
551  break;
552  case EMF_ANISOTROPIC_FILTER:
553  {
554  if (value)
555  for (u32 i=0; i<MATERIAL_MAX_TEXTURES; ++i)
556  TextureLayer[i].AnisotropicFilter = 0xFF;
557  else
558  for (u32 i=0; i<MATERIAL_MAX_TEXTURES; ++i)
559  TextureLayer[i].AnisotropicFilter = 0;
560  }
561  break;
562  case EMF_FOG_ENABLE:
563  FogEnable = value; break;
564  case EMF_NORMALIZE_NORMALS:
565  NormalizeNormals = value; break;
566  case EMF_TEXTURE_WRAP:
567  {
568  for (u32 i=0; i<MATERIAL_MAX_TEXTURES; ++i)
569  {
570  TextureLayer[i].TextureWrapU = (E_TEXTURE_CLAMP)value;
571  TextureLayer[i].TextureWrapV = (E_TEXTURE_CLAMP)value;
572  TextureLayer[i].TextureWrapW = (E_TEXTURE_CLAMP)value;
573  }
574  }
575  break;
576  case EMF_ANTI_ALIASING:
577  AntiAliasing = value?EAAM_SIMPLE:EAAM_OFF; break;
578  case EMF_COLOR_MASK:
579  ColorMask = value?ECP_ALL:ECP_NONE; break;
580  case EMF_COLOR_MATERIAL:
581  ColorMaterial = value?ECM_DIFFUSE:ECM_NONE; break;
582  case EMF_USE_MIP_MAPS:
583  UseMipMaps = value; break;
584  case EMF_BLEND_OPERATION:
585  BlendOperation = value?EBO_ADD:EBO_NONE; break;
586  case EMF_BLEND_FACTOR:
587  break;
588  case EMF_POLYGON_OFFSET:
589  PolygonOffsetFactor = value?1:0;
590  PolygonOffsetDirection = EPO_BACK;
591  PolygonOffsetSlopeScale = value?1.f:0.f;
592  PolygonOffsetDepthBias = value?1.f:0.f;
593  default:
594  break;
595  }
596  }

601  bool getFlag(E_MATERIAL_FLAG flag) const
602  {
603  switch (flag)
604  {
605  case EMF_WIREFRAME:
606  return Wireframe;
607  case EMF_POINTCLOUD:
608  return PointCloud;
609  case EMF_GOURAUD_SHADING:
610  return GouraudShading;
611  case EMF_LIGHTING:
612  return Lighting;
613  case EMF_ZBUFFER:
614  return ZBuffer!=ECFN_DISABLED;
615  case EMF_ZWRITE_ENABLE:
616  return ZWriteEnable != EZW_OFF;
617  case EMF_BACK_FACE_CULLING:
618  return BackfaceCulling;
619  case EMF_FRONT_FACE_CULLING:
620  return FrontfaceCulling;
621  case EMF_BILINEAR_FILTER:
622  return TextureLayer[0].BilinearFilter;
623  case EMF_TRILINEAR_FILTER:
624  return TextureLayer[0].TrilinearFilter;
625  case EMF_ANISOTROPIC_FILTER:
626  return TextureLayer[0].AnisotropicFilter!=0;
627  case EMF_FOG_ENABLE:
628  return FogEnable;
629  case EMF_NORMALIZE_NORMALS:
630  return NormalizeNormals;
631  case EMF_TEXTURE_WRAP:
632  return !(TextureLayer[0].TextureWrapU ||
633  TextureLayer[0].TextureWrapV ||
634  TextureLayer[0].TextureWrapW);
635  case EMF_ANTI_ALIASING:
636  return (AntiAliasing==1);
637  case EMF_COLOR_MASK:
638  return (ColorMask!=ECP_NONE);
639  case EMF_COLOR_MATERIAL:
640  return (ColorMaterial != ECM_NONE);
641  case EMF_USE_MIP_MAPS:
642  return UseMipMaps;
643  case EMF_BLEND_OPERATION:
644  return BlendOperation != EBO_NONE;
645  case EMF_BLEND_FACTOR:
646  return BlendFactor != 0.f;
647  case EMF_POLYGON_OFFSET:
648  return PolygonOffsetFactor != 0 || PolygonOffsetDepthBias != 0.f;
649  }
650 
651  return false;
652  }

704  bool isAlphaBlendOperation() const
705  {
706  if (BlendOperation != EBO_NONE && BlendFactor != 0.f)
707  {
708  E_BLEND_FACTOR srcRGBFact = EBF_ZERO;
709  E_BLEND_FACTOR dstRGBFact = EBF_ZERO;
710  E_BLEND_FACTOR srcAlphaFact = EBF_ZERO;
711  E_BLEND_FACTOR dstAlphaFact = EBF_ZERO;
712  E_MODULATE_FUNC modulo = EMFN_MODULATE_1X;
713  u32 alphaSource = 0;
714 
715  unpack_textureBlendFuncSeparate(srcRGBFact, dstRGBFact, srcAlphaFact, dstAlphaFact, modulo, alphaSource, BlendFactor);
716 
717  if (textureBlendFunc_hasAlpha(srcRGBFact) || textureBlendFunc_hasAlpha(dstRGBFact) ||
718  textureBlendFunc_hasAlpha(srcAlphaFact) || textureBlendFunc_hasAlpha(dstAlphaFact))
719  {
720  return true;
721  }
722  }
723  return false;
724  }
