51 void CGUIStaticText::draw()
52 {
53  if (!IsVisible)
54  return;
55 
56  IGUISkin* skin = Environment->getSkin();
57  if (!skin)
58  return;
59  video::IVideoDriver* driver = Environment->getVideoDriver();
60 
61  core::rect<s32> frameRect(AbsoluteRect);
62 
63  // draw background
64 
65  if (Background)
66  {
67  if ( !OverrideBGColorEnabled ) // skin-colors can change
68  BGColor = skin->getColor(gui::EGDC_3D_FACE);
69 
70  driver->draw2DRectangle(BGColor, frameRect, &AbsoluteClippingRect);
71  }
72 
73  // draw the border
74 
75  if (Border)
76  {
77  skin->draw3DSunkenPane(this, 0, true, false, frameRect, &AbsoluteClippingRect);
78  frameRect.UpperLeftCorner.X += skin->getSize(EGDS_TEXT_DISTANCE_X);
79  }
80 
81  // draw the text
82  if (Text.size())
83  {
84  IGUIFont* font = getActiveFont();
85 
86  if (font)
87  {
88  if (!WordWrap)
89  {
90  if (VAlign == EGUIA_LOWERRIGHT)
91  {
92  frameRect.UpperLeftCorner.Y = frameRect.LowerRightCorner.Y -
93  font->getDimension(L"A").Height - font->getKerningHeight();
94  }
95  if (HAlign == EGUIA_LOWERRIGHT)
96  {
97  frameRect.UpperLeftCorner.X = frameRect.LowerRightCorner.X -
98  font->getDimension(Text.c_str()).Width;
99  }
100 
101  font->draw(Text.c_str(), frameRect,
102  getActiveColor(),
103  HAlign == EGUIA_CENTER, VAlign == EGUIA_CENTER, (RestrainTextInside ? &AbsoluteClippingRect : NULL));
104  }
105  else
106  {
107  if (font != LastBreakFont)
108  breakText();
109 
110  core::rect<s32> r = frameRect;
111  s32 height = font->getDimension(L"A").Height + font->getKerningHeight();
112  s32 totalHeight = height * BrokenText.size();
113  if (VAlign == EGUIA_CENTER)
114  {
115  r.UpperLeftCorner.Y = r.getCenter().Y - (totalHeight / 2);
116  }
117  else if (VAlign == EGUIA_LOWERRIGHT)
118  {
119  r.UpperLeftCorner.Y = r.LowerRightCorner.Y - totalHeight;
120  }
121 
122  for (u32 i=0; i<BrokenText.size(); ++i)
123  {
124  if (HAlign == EGUIA_LOWERRIGHT)
125  {
126  r.UpperLeftCorner.X = frameRect.LowerRightCorner.X -
127  font->getDimension(BrokenText[i].c_str()).Width;
128  }
129 
130  font->draw(BrokenText[i].c_str(), r,
131  getActiveColor(),
132  HAlign == EGUIA_CENTER, false, (RestrainTextInside ? &AbsoluteClippingRect : NULL));
133 
134  r.LowerRightCorner.Y += height;
135  r.UpperLeftCorner.Y += height;
136  }
137  }
138  }
139  }
140 
141  IGUIElement::draw();
142 }

312 void CGUIStaticText::breakText()
313 {
314  if (!WordWrap)
315  return;
316 
317  BrokenText.clear();
318 
319  IGUISkin* skin = Environment->getSkin();
320  IGUIFont* font = getActiveFont();
321  if (!font)
322  return;
323 
324  LastBreakFont = font;
325 
326  core::stringw line;
327  core::stringw word;
328  core::stringw whitespace;
329  s32 size = Text.size();
330  s32 length = 0;
331  s32 elWidth = RelativeRect.getWidth();
332  if (Border)
333  elWidth -= 2*skin->getSize(EGDS_TEXT_DISTANCE_X);
334  wchar_t c;
335 
336  // We have to deal with right-to-left and left-to-right differently
337  // However, most parts of the following code is the same, it's just
338  // some order and boundaries which change.
339  if (!RightToLeft)
340  {
341  // regular (left-to-right)
342  for (s32 i=0; i<size; ++i)
343  {
344  c = Text[i];
345  bool lineBreak = false;
346 
347  if (c == L'\r') // Mac or Windows breaks
348  {
349  lineBreak = true;
350  if (Text[i+1] == L'\n') // Windows breaks
351  {
352  Text.erase(i+1);
353  --size;
354  }
355  c = '\0';
356  }
357  else if (c == L'\n') // Unix breaks
358  {
359  lineBreak = true;
360  c = '\0';
361  }
362 
363  bool isWhitespace = (c == L' ' || c == 0);
364  if ( !isWhitespace )
365  {
366  // part of a word
367  word += c;
368  }
369 
370  if ( isWhitespace || i == (size-1))
371  {
372  if (word.size())
373  {
374  // here comes the next whitespace, look if
375  // we must break the last word to the next line.
376  const s32 whitelgth = font->getDimension(whitespace.c_str()).Width;
377  const s32 wordlgth = font->getDimension(word.c_str()).Width;
378 
379  if (wordlgth > elWidth)
380  {
381  // This word is too long to fit in the available space, look for
382  // the Unicode Soft HYphen (SHY / 00AD) character for a place to
383  // break the word at
384  int where = word.findFirst( wchar_t(0x00AD) );
385  if (where != -1)
386  {
387  core::stringw first = word.subString(0, where);
388  core::stringw second = word.subString(where, word.size() - where);
389  BrokenText.push_back(line + first + L"-");
390  const s32 secondLength = font->getDimension(second.c_str()).Width;
391 
392  length = secondLength;
393  line = second;
394  }
395  else
396  {
397  // No soft hyphen found, so there's nothing more we can do
398  // break to next line
399  if (length)
400  BrokenText.push_back(line);
401  length = wordlgth;
402  line = word;
403  }
404  }
405  else if (length && (length + wordlgth + whitelgth > elWidth))
406  {
407  // break to next line
408  BrokenText.push_back(line);
409  length = wordlgth;
410  line = word;
411  }
412  else
413  {
414  // add word to line
415  line += whitespace;
416  line += word;
417  length += whitelgth + wordlgth;
418  }
419 
420  word = L"";
421  whitespace = L"";
422  }
423 
424  if ( isWhitespace )
425  {
426  whitespace += c;
427  }
428 
429  // compute line break
430  if (lineBreak)
431  {
432  line += whitespace;
433  line += word;
434  BrokenText.push_back(line);
435  line = L"";
436  word = L"";
437  whitespace = L"";
438  length = 0;
439  }
440  }
441  }
442 
443  line += whitespace;
444  line += word;
445  BrokenText.push_back(line);
446  }
447  else
448  {
449  // right-to-left
450  for (s32 i=size; i>=0; --i)
451  {
452  c = Text[i];
453  bool lineBreak = false;
454 
455  if (c == L'\r') // Mac or Windows breaks
456  {
457  lineBreak = true;
458  if ((i>0) && Text[i-1] == L'\n') // Windows breaks
459  {
460  Text.erase(i-1);
461  --size;
462  }
463  c = '\0';
464  }
465  else if (c == L'\n') // Unix breaks
466  {
467  lineBreak = true;
468  c = '\0';
469  }
470 
471  if (c==L' ' || c==0 || i==0)
472  {
473  if (word.size())
474  {
475  // here comes the next whitespace, look if
476  // we must break the last word to the next line.
477  const s32 whitelgth = font->getDimension(whitespace.c_str()).Width;
478  const s32 wordlgth = font->getDimension(word.c_str()).Width;
479 
480  if (length && (length + wordlgth + whitelgth > elWidth))
481  {
482  // break to next line
483  BrokenText.push_back(line);
484  length = wordlgth;
485  line = word;
486  }
487  else
488  {
489  // add word to line
490  line = whitespace + line;
491  line = word + line;
492  length += whitelgth + wordlgth;
493  }
494 
495  word = L"";
496  whitespace = L"";
497  }
498 
499  if (c != 0)
500  whitespace = core::stringw(&c, 1) + whitespace;
501 
502  // compute line break
503  if (lineBreak)
504  {
505  line = whitespace + line;
506  line = word + line;
507  BrokenText.push_back(line);
508  line = L"";
509  word = L"";
510  whitespace = L"";
511  length = 0;
512  }
513  }
514  else
515  {
516  // yippee this is a word..
517  word = core::stringw(&c, 1) + word;
518  }
519  }
520 
521  line = whitespace + line;
522  line = word + line;
523  BrokenText.push_back(line);
524  }
525 }

544 s32 CGUIStaticText::getTextHeight() const
545 {
546  IGUIFont* font = getActiveFont();
547  if (!font)
548  return 0;
549 
550  if (WordWrap)
551  {
552  s32 height = font->getDimension(L"A").Height + font->getKerningHeight();
553  return height* BrokenText.size();
554  }
555  else
556  {
557  // TODO: Text can have multiple lines which are not in BrokenText
558  // This is likely not correct. But as I have no time for further
559  // investigation I just fix it for now by return the true height here.
560  return font->getDimension(Text.c_str()).Height;
561  }

565 s32 CGUIStaticText::getTextWidth() const
566 {
567  IGUIFont * font = getActiveFont();
568  if(!font)
569  return 0;
570 
571  if(WordWrap)
572  {
573  s32 widest = 0;
574 
575  for(u32 line = 0; line < BrokenText.size(); ++line)
576  {
577  s32 width = font->getDimension(BrokenText[line].c_str()).Width;
578 
579  if(width > widest)
580  widest = width;
581  }
582 
583  return widest;
584  }
585  else
586  {
587  return font->getDimension(Text.c_str()).Width;
588  }
