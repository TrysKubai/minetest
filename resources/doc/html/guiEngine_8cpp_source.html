70 /******************************************************************************/
71 video::ITexture *MenuTextureSource::getTexture(const std::string &name, u32 *id)
72 {
73  if (id)
74  *id = 0;
75 
76  if (name.empty())
77  return NULL;
78 
79  // return if already loaded
80  video::ITexture *retval = m_driver->findTexture(name.c_str());
81  if (retval)
82  return retval;
83 
84  video::IImage *image = m_driver->createImageFromFile(name.c_str());
85  if (!image)
86  return NULL;
87 
88  image = Align2Npot2(image, m_driver);
89  retval = m_driver->addTexture(name.c_str(), image);
90  image->drop();
91 
92  if (retval)
93  m_to_delete.push_back(retval);
94  return retval;
95 }

241 /******************************************************************************/
242 void GUIEngine::run()
243 {
244  // Always create clouds because they may or may not be
245  // needed based on the game selected
246  video::IVideoDriver *driver = m_rendering_engine->get_video_driver();
247 
248  cloudInit();
249 
250  unsigned int text_height = g_fontengine->getTextHeight();
251 
252  // Reset fog color
253  {
254  video::SColor fog_color;
255  video::E_FOG_TYPE fog_type = video::EFT_FOG_LINEAR;
256  f32 fog_start = 0;
257  f32 fog_end = 0;
258  f32 fog_density = 0;
259  bool fog_pixelfog = false;
260  bool fog_rangefog = false;
261  driver->getFog(fog_color, fog_type, fog_start, fog_end, fog_density,
262  fog_pixelfog, fog_rangefog);
263 
264  driver->setFog(RenderingEngine::MENU_SKY_COLOR, fog_type, fog_start,
265  fog_end, fog_density, fog_pixelfog, fog_rangefog);
266  }
267 
268  const irr::core::dimension2d<u32> initial_screen_size(
269  g_settings->getU16("screen_w"),
270  g_settings->getU16("screen_h")
271  );
272  const bool initial_window_maximized = g_settings->getBool("window_maximized");
273 
274  while (m_rendering_engine->run() && (!m_startgame) && (!m_kill)) {
275 
276  //check if we need to update the "upper left corner"-text
277  if (text_height != g_fontengine->getTextHeight()) {
278  updateTopLeftTextSize();
279  text_height = g_fontengine->getTextHeight();
280  }
281 
282  driver->beginScene(true, true, RenderingEngine::MENU_SKY_COLOR);
283 
284  if (m_clouds_enabled)
285  {
286  cloudPreProcess();
287  drawOverlay(driver);
288  }
289  else
290  drawBackground(driver);
291 
292  drawHeader(driver);
293  drawFooter(driver);
294 
295  m_rendering_engine->get_gui_env()->drawAll();
296 
297  driver->endScene();
298 
299  IrrlichtDevice *device = m_rendering_engine->get_raw_device();
300  u32 frametime_min = 1000 / (device->isWindowFocused()
301  ? g_settings->getFloat("fps_max")
302  : g_settings->getFloat("fps_max_unfocused"));
303  if (m_clouds_enabled)
304  cloudPostProcess(frametime_min, device);
305  else
306  sleep_ms(frametime_min);
307 
308  m_script->step();
309 
310 #ifdef __ANDROID__
311  m_menu->getAndroidUIInput();
312 #endif
313  }
314 
315  RenderingEngine::autosaveScreensizeAndCo(initial_screen_size, initial_window_maximized);
316 }

399 /******************************************************************************/
400 void GUIEngine::drawBackground(video::IVideoDriver *driver)
401 {
402  v2u32 screensize = driver->getScreenSize();
403 
404  video::ITexture* texture = m_textures[TEX_LAYER_BACKGROUND].texture;
405 
406  /* If no texture, draw background of solid color */
407  if(!texture){
408  video::SColor color(255,80,58,37);
409  core::rect<s32> rect(0, 0, screensize.X, screensize.Y);
410  driver->draw2DRectangle(color, rect, NULL);
411  return;
412  }
413 
414  v2u32 sourcesize = texture->getOriginalSize();
415 
416  if (m_textures[TEX_LAYER_BACKGROUND].tile)
417  {
418  v2u32 tilesize(
419  MYMAX(sourcesize.X,m_textures[TEX_LAYER_BACKGROUND].minsize),
420  MYMAX(sourcesize.Y,m_textures[TEX_LAYER_BACKGROUND].minsize));
421  for (unsigned int x = 0; x < screensize.X; x += tilesize.X )
422  {
423  for (unsigned int y = 0; y < screensize.Y; y += tilesize.Y )
424  {
425  draw2DImageFilterScaled(driver, texture,
426  core::rect<s32>(x, y, x+tilesize.X, y+tilesize.Y),
427  core::rect<s32>(0, 0, sourcesize.X, sourcesize.Y),
428  NULL, NULL, true);
429  }
430  }
431  return;
432  }
433 
434  // Chop background image to the smaller screen dimension
435  v2u32 bg_size = screensize;
436  v2f32 scale(
437  (f32) bg_size.X / sourcesize.X,
438  (f32) bg_size.Y / sourcesize.Y);
439  if (scale.X < scale.Y)
440  bg_size.X = (int) (scale.Y * sourcesize.X);
441  else
442  bg_size.Y = (int) (scale.X * sourcesize.Y);
443  v2s32 offset = v2s32(
444  (s32) screensize.X - (s32) bg_size.X,
445  (s32) screensize.Y - (s32) bg_size.Y
446  ) / 2;
447  /* Draw background texture */
448  draw2DImageFilterScaled(driver, texture,
449  core::rect<s32>(offset.X, offset.Y, bg_size.X + offset.X, bg_size.Y + offset.Y),
450  core::rect<s32>(0, 0, sourcesize.X, sourcesize.Y),
451  NULL, NULL, true);
452 }

473 /******************************************************************************/
474 void GUIEngine::drawHeader(video::IVideoDriver *driver)
475 {
476  core::dimension2d<u32> screensize = driver->getScreenSize();
477 
478  video::ITexture* texture = m_textures[TEX_LAYER_HEADER].texture;
479 
480  /* If no texture, draw nothing */
481  if(!texture)
482  return;
483 
484  f32 mult = (((f32)screensize.Width / 2.0)) /
485  ((f32)texture->getOriginalSize().Width);
486 
487  v2s32 splashsize(((f32)texture->getOriginalSize().Width) * mult,
488  ((f32)texture->getOriginalSize().Height) * mult);
489 
490  // Don't draw the header if there isn't enough room
491  s32 free_space = (((s32)screensize.Height)-320)/2;
492 
493  if (free_space > splashsize.Y) {
494  core::rect<s32> splashrect(0, 0, splashsize.X, splashsize.Y);
495  splashrect += v2s32((screensize.Width/2)-(splashsize.X/2),
496  ((free_space/2)-splashsize.Y/2)+10);
497 
498  draw2DImageFilterScaled(driver, texture, splashrect,
499  core::rect<s32>(core::position2d<s32>(0,0),
500  core::dimension2di(texture->getOriginalSize())),
501  NULL, NULL, true);
502  }
503 }

560  return true;
