260 bool CIrrDeviceLinux::switchToFullscreen()
261 {
262  if (!CreationParams.Fullscreen)
263  return true;
264 
265  if (!HasNetWM)
266  {
267  os::Printer::log("NetWM support is required to allow Irrlicht to switch "
268  "to fullscreen mode. Running in windowed mode instead.", ELL_WARNING);
269  CreationParams.Fullscreen = false;
270  return false;
271  }
272 
273  XEvent ev = {0};
274 
275  ev.type = ClientMessage;
276  ev.xclient.window = XWindow;
277  ev.xclient.message_type = X_ATOM_NETWM_STATE;
278  ev.xclient.format = 32;
279  ev.xclient.data.l[0] = 1; // _NET_WM_STATE_ADD
280  ev.xclient.data.l[1] = X_ATOM_NETWM_STATE_FULLSCREEN;
281 
282  XSendEvent(XDisplay, DefaultRootWindow(XDisplay), false,
283  SubstructureNotifyMask | SubstructureRedirectMask, &ev);
284 
285  return true;
286 }

289 void CIrrDeviceLinux::setupTopLevelXorgWindow()
290 {
291 #ifdef _IRR_COMPILE_WITH_X11_
292  if (CreationParams.DriverType == video::EDT_NULL)
293  return; // no display and window
294 
295  os::Printer::log("Configuring X11-specific top level window properties", ELL_DEBUG);
296 
297  // Set application name and class hints. For now name and class are the same.
298  // Note: SDL uses the executable name here (i.e. "minetest").
299  XClassHint *classhint = XAllocClassHint();
300  classhint->res_name = const_cast<char *>("Minetest");
301  classhint->res_class = const_cast<char *>("Minetest");
302 
303  XSetClassHint(XDisplay, XWindow, classhint);
304  XFree(classhint);
305 
306  // FIXME: In the future WMNormalHints should be set ... e.g see the
307  // gtk/gdk code (gdk/x11/gdksurface-x11.c) for the setup_top_level
308  // method. But for now (as it would require some significant changes)
309  // leave the code as is.
310 
311  // The following is borrowed from the above gdk source for setting top
312  // level windows. The source indicates and the Xlib docs suggest that
313  // this will set the WM_CLIENT_MACHINE and WM_LOCAL_NAME. This will not
314  // set the WM_CLIENT_MACHINE to a Fully Qualified Domain Name (FQDN) which is
315  // required by the Extended Window Manager Hints (EWMH) spec when setting
316  // the _NET_WM_PID (see further down) but running Minetest in an env
317  // where the window manager is on another machine from Minetest (therefore
318  // making the PID useless) is not expected to be a problem. Further
319  // more, using gtk/gdk as the model it would seem that not using a FQDN is
320  // not an issue for modern Xorg window managers.
321 
322  os::Printer::log("Setting Xorg window manager Properties", ELL_DEBUG);
323 
324  XSetWMProperties (XDisplay, XWindow, NULL, NULL, NULL, 0, NULL, NULL, NULL);
325 
326  // Set the _NET_WM_PID window property according to the EWMH spec. _NET_WM_PID
327  // (in conjunction with WM_CLIENT_MACHINE) can be used by window managers to
328  // force a shutdown of an application if it doesn't respond to the destroy
329  // window message.
330 
331  os::Printer::log("Setting Xorg _NET_WM_PID extended window manager property", ELL_DEBUG);
332 
333  Atom NET_WM_PID = XInternAtom(XDisplay, "_NET_WM_PID", false);
334 
335  pid_t pid = getpid();
336 
337  XChangeProperty(XDisplay, XWindow, NET_WM_PID,
338  XA_CARDINAL, 32, PropModeReplace,
339  reinterpret_cast<unsigned char *>(&pid),1);
340 
341  // Set the WM_CLIENT_LEADER window property here. Minetest has only one
342  // window and that window will always be the leader.
343 
344  os::Printer::log("Setting Xorg WM_CLIENT_LEADER property", ELL_DEBUG);
345 
346  Atom WM_CLIENT_LEADER = XInternAtom(XDisplay, "WM_CLIENT_LEADER", false);
347 
348  XChangeProperty (XDisplay, XWindow, WM_CLIENT_LEADER,
349  XA_WINDOW, 32, PropModeReplace,
350  reinterpret_cast<unsigned char *>(&XWindow), 1);
351 #endif
352 }

386 bool CIrrDeviceLinux::createWindow()
387 {
388 #ifdef _IRR_COMPILE_WITH_X11_
389 #ifdef _DEBUG
390  os::Printer::log("Creating X window...", ELL_INFORMATION);
391  XSetErrorHandler(IrrPrintXError);
392 #endif
393 
394  XDisplay = XOpenDisplay(0);
395  if (!XDisplay)
396  {
397  os::Printer::log("Error: Need running XServer to start Irrlicht Engine.", ELL_ERROR);
398  if (XDisplayName(0)[0])
399  os::Printer::log("Could not open display", XDisplayName(0), ELL_ERROR);
400  else
401  os::Printer::log("Could not open display, set DISPLAY variable", ELL_ERROR);
402  return false;
403  }
404 
405  Screennr = DefaultScreen(XDisplay);
406 
407  initXAtoms();
408 
409  // check netwm support
410  Atom WMCheck = XInternAtom(XDisplay, "_NET_SUPPORTING_WM_CHECK", True);
411  if (WMCheck != None)
412  HasNetWM = true;
413 
414 #if defined(_IRR_COMPILE_WITH_OPENGL_)
415  // don't use the XVisual with OpenGL, because it ignores all requested
416  // properties of the CreationParams
417  if (CreationParams.DriverType == video::EDT_OPENGL)
418  {
419  video::SExposedVideoData data;
420  data.OpenGLLinux.X11Display = XDisplay;
421  ContextManager = new video::CGLXManager(CreationParams, data, Screennr);
422  VisualInfo = ((video::CGLXManager*)ContextManager)->getVisual();
423  }
424 #endif
425 
426  if (!VisualInfo)
427  {
428  // create visual with standard X methods
429  os::Printer::log("Using plain X visual");
430  XVisualInfo visTempl; //Template to hold requested values
431  int visNumber; // Return value of available visuals
432 
433  visTempl.screen = Screennr;
434  // ARGB visuals should be avoided for usual applications
435  visTempl.depth = CreationParams.WithAlphaChannel?32:24;
436  while ((!VisualInfo) && (visTempl.depth>=16))
437  {
438  VisualInfo = XGetVisualInfo(XDisplay, VisualScreenMask|VisualDepthMask,
439  &visTempl, &visNumber);
440  visTempl.depth -= 8;
441  }
442  }
443 
444  if (!VisualInfo)
445  {
446  os::Printer::log("Fatal error, could not get visual.", ELL_ERROR);
447  XCloseDisplay(XDisplay);
448  XDisplay=0;
449  return false;
450  }
451 #ifdef _DEBUG
452  else
453  os::Printer::log("Visual chosen", core::stringc(static_cast<u32>(VisualInfo->visualid)).c_str(), ELL_DEBUG);
454 #endif
455 
456  // create color map
457  Colormap colormap;
458  colormap = XCreateColormap(XDisplay,
459  RootWindow(XDisplay, VisualInfo->screen),
460  VisualInfo->visual, AllocNone);
461 
462  WndAttributes.colormap = colormap;
463  WndAttributes.border_pixel = 0;
464  WndAttributes.event_mask = StructureNotifyMask | FocusChangeMask | ExposureMask;
465  if (!CreationParams.IgnoreInput)
466  WndAttributes.event_mask |= PointerMotionMask |
467  ButtonPressMask | KeyPressMask |
468  ButtonReleaseMask | KeyReleaseMask;
469 
470  if (!CreationParams.WindowId)
471  {
472  int x = 0;
473  int y = 0;
474 
475  if (!CreationParams.Fullscreen)
476  {
477  if (CreationParams.WindowPosition.X > 0)
478  x = CreationParams.WindowPosition.X;
479  if (CreationParams.WindowPosition.Y > 0)
480  y = CreationParams.WindowPosition.Y;
481  }
482 
483  // create new Window
484  // Remove window manager decoration in fullscreen
485  XWindow = XCreateWindow(XDisplay,
486  RootWindow(XDisplay, VisualInfo->screen),
487  x, y, Width, Height, 0, VisualInfo->depth,
488  InputOutput, VisualInfo->visual,
489  CWBorderPixel | CWColormap | CWEventMask,
490  &WndAttributes);
491 
492  XMapRaised(XDisplay, XWindow);
493  CreationParams.WindowId = (void*)XWindow;
494  X_ATOM_WM_DELETE_WINDOW = XInternAtom(XDisplay, "WM_DELETE_WINDOW", True);
495  XSetWMProtocols(XDisplay, XWindow, &X_ATOM_WM_DELETE_WINDOW, 1);
496 
497  if (CreationParams.Fullscreen)
498  {
499  // Don't try to set window position
500  }
501  else if (CreationParams.WindowPosition.X >= 0 || CreationParams.WindowPosition.Y >= 0) // default is -1, -1
502  {
503  // Window managers are free to ignore positions above, so give it another shot
504  XMoveWindow(XDisplay,XWindow,x,y);
505  }
506  }
507  else
508  {
509  // attach external window
510  XWindow = (Window)CreationParams.WindowId;
511  if (!CreationParams.IgnoreInput)
512  {
513  // Note: This might be further improved by using a InputOnly window instead of InputOutput.
514  // I think then it should be possible to render into the given parent window instead of
515  // creating a child-window.
516  // That could also be a third option for IgnoreInput in the CreationParams.
517  // But we need another window variable then and have to split input/output in
518  // the rest of the device code.
519  // Also... this does possibly leak.
520  Window child_window = XCreateWindow(XDisplay,
521  XWindow,
522  0, 0, Width, Height, 0, VisualInfo->depth,
523  InputOutput, VisualInfo->visual,
524  CWBorderPixel | CWColormap | CWEventMask,
525  &WndAttributes);
526 
527  // do not forget to map new window
528  XMapWindow(XDisplay, child_window);
529 
530  // overwrite device window id
531  XWindow = child_window;
532  }
533  XWindowAttributes wa;
534  XGetWindowAttributes(XDisplay, XWindow, &wa);
535  CreationParams.WindowSize.Width = wa.width;
536  CreationParams.WindowSize.Height = wa.height;
537  CreationParams.Fullscreen = false;
538  ExternalWindow = true;
539  }
540 
541  switchToFullscreen();
542 
543  WindowMinimized=false;
544  XkbSetDetectableAutoRepeat(XDisplay, True, &AutorepeatSupport);
545 
546  Window tmp;
547  u32 borderWidth;
548  int x,y;
549  unsigned int bits;
550 
551  XGetGeometry(XDisplay, XWindow, &tmp, &x, &y, &Width, &Height, &borderWidth, &bits);
552  CreationParams.Bits = bits;
553  CreationParams.WindowSize.Width = Width;
554  CreationParams.WindowSize.Height = Height;
555 
556  StdHints = XAllocSizeHints();
557  long num;
558  XGetWMNormalHints(XDisplay, XWindow, StdHints, &num);
559 
560  initXInput2();
561 
562 #endif // #ifdef _IRR_COMPILE_WITH_X11_
563  return true;
564 }

658 #ifdef _IRR_COMPILE_WITH_X11_
659 bool CIrrDeviceLinux::createInputContext()
660 {
661  // One one side it would be nicer to let users do that - on the other hand
662  // not setting the environment locale will not work when using i18n X11 functions.
663  // So users would have to call it always or their input is broken badly.
664  // We can restore immediately - so won't mess with anything in users apps.
665  core::stringc oldLocale(setlocale(LC_CTYPE, NULL));
666  setlocale(LC_CTYPE, ""); // use environment locale
667 
668  if ( !XSupportsLocale() )
669  {
670  os::Printer::log("Locale not supported. Falling back to non-i18n input.", ELL_WARNING);
671  setlocale(LC_CTYPE, oldLocale.c_str());
672  return false;
673  }
674 
675  // Load XMODIFIERS (e.g. for IMEs)
676  if (!XSetLocaleModifiers(""))
677  {
