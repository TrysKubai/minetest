33 class LagPool
34 {
35  float m_pool = 15.0f;
36  float m_max = 15.0f;
37 
38 public:
39  LagPool() = default;
40 
41  void setMax(float new_max)
42  {
43  m_max = new_max;
44  if (m_pool > new_max)
45  m_pool = new_max;
46  }
47 
48  void add(float dtime)
49  {
50  m_pool -= dtime;
51  if (m_pool < 0)
52  m_pool = 0;
53  }
54 
55  void empty() { m_pool = m_max; }
56 
57  bool grab(float dtime)
58  {
59  if (dtime <= 0)
60  return true;
61  if (m_pool + dtime > m_max)
62  return false;
63  m_pool += dtime;
64  return true;
65  }
66 };

256 
257  bool setTypeFromString(const std::string &typestr)
258  {
259  if (typestr == "set_hp")
260  type = SET_HP;
261  else if (typestr == "punch")
262  type = PLAYER_PUNCH;
263  else if (typestr == "fall")
264  type = FALL;
265  else if (typestr == "node_damage")
266  type = NODE_DAMAGE;
267  else if (typestr == "drown")
268  type = DROWNING;
269  else if (typestr == "respawn")
270  type = RESPAWN;
271  else
272  return false;
273 
274  return true;
275  }
276 
277  std::string getTypeAsString() const
278  {
279  switch (type) {
280  case PlayerHPChangeReason::SET_HP:
281  case PlayerHPChangeReason::SET_HP_MAX:
