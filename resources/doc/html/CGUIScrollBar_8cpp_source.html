59 bool CGUIScrollBar::OnEvent(const SEvent& event)
60 {
61  if (isEnabled())
62  {
63 
64  switch(event.EventType)
65  {
66  case EET_KEY_INPUT_EVENT:
67  if (event.KeyInput.PressedDown)
68  {
69  const s32 oldPos = Pos;
70  bool absorb = true;
71  switch (event.KeyInput.Key)
72  {
73  case KEY_LEFT:
74  case KEY_UP:
75  setPos(Pos-SmallStep);
76  break;
77  case KEY_RIGHT:
78  case KEY_DOWN:
79  setPos(Pos+SmallStep);
80  break;
81  case KEY_HOME:
82  setPos(Min);
83  break;
84  case KEY_PRIOR:
85  setPos(Pos-LargeStep);
86  break;
87  case KEY_END:
88  setPos(Max);
89  break;
90  case KEY_NEXT:
91  setPos(Pos+LargeStep);
92  break;
93  default:
94  absorb = false;
95  }
96 
97  if (Pos != oldPos)
98  {
99  SEvent newEvent;
100  newEvent.EventType = EET_GUI_EVENT;
101  newEvent.GUIEvent.Caller = this;
102  newEvent.GUIEvent.Element = 0;
103  newEvent.GUIEvent.EventType = EGET_SCROLL_BAR_CHANGED;
104  Parent->OnEvent(newEvent);
105  }
106  if (absorb)
107  return true;
108  }
109  break;
110  case EET_GUI_EVENT:
111  if (event.GUIEvent.EventType == EGET_BUTTON_CLICKED)
112  {
113  if (event.GUIEvent.Caller == UpButton)
114  setPos(Pos-SmallStep);
115  else
116  if (event.GUIEvent.Caller == DownButton)
117  setPos(Pos+SmallStep);
118 
119  SEvent newEvent;
120  newEvent.EventType = EET_GUI_EVENT;
121  newEvent.GUIEvent.Caller = this;
122  newEvent.GUIEvent.Element = 0;
123  newEvent.GUIEvent.EventType = EGET_SCROLL_BAR_CHANGED;
124  Parent->OnEvent(newEvent);
125 
126  return true;
127  }
128  else
129  if (event.GUIEvent.EventType == EGET_ELEMENT_FOCUS_LOST)
130  {
131  if (event.GUIEvent.Caller == this)
132  Dragging = false;
133  }
134  break;
135  case EET_MOUSE_INPUT_EVENT:
136  {
137  const core::position2di p(event.MouseInput.X, event.MouseInput.Y);
138  bool isInside = isPointInside ( p );
139  switch(event.MouseInput.Event)
140  {
141  case EMIE_MOUSE_WHEEL:
142  if (Environment->hasFocus(this))
143  {
144  // thanks to a bug report by REAPER
145  // thanks to tommi by tommi for another bugfix
146  // everybody needs a little thanking. hallo niko!;-)
147  setPos( getPos() +
148  ( (event.MouseInput.Wheel < 0 ? -1 : 1) * SmallStep * (Horizontal ? 1 : -1 ) )
149  );
150 
151  SEvent newEvent;
152  newEvent.EventType = EET_GUI_EVENT;
153  newEvent.GUIEvent.Caller = this;
154  newEvent.GUIEvent.Element = 0;
155  newEvent.GUIEvent.EventType = EGET_SCROLL_BAR_CHANGED;
156  Parent->OnEvent(newEvent);
157  return true;
158  }
159  break;
160  case EMIE_LMOUSE_PRESSED_DOWN:
161  {
162  if (isInside)
163  {
164  Dragging = true;
165  DraggedBySlider = SliderRect.isPointInside(p);
166  TrayClick = !DraggedBySlider;
167  DesiredPos = getPosFromMousePos(p);
168  return true;
169  }
170  break;
171  }
172  case EMIE_LMOUSE_LEFT_UP:
173  case EMIE_MOUSE_MOVED:
174  {
175  if ( !event.MouseInput.isLeftPressed () )
176  Dragging = false;
177 
178  if ( !Dragging )
179  {
180  if ( event.MouseInput.Event == EMIE_MOUSE_MOVED )
181  break;
182  return isInside;
183  }
184 
185  if ( event.MouseInput.Event == EMIE_LMOUSE_LEFT_UP )
186  Dragging = false;
187 
188  const s32 newPos = getPosFromMousePos(p);
189  const s32 oldPos = Pos;
190 
191  if (!DraggedBySlider)
192  {
193  if ( isInside )
194  {
195  DraggedBySlider = SliderRect.isPointInside(p);
196  TrayClick = !DraggedBySlider;
197  }
198 
199  if (DraggedBySlider)
200  {
201  setPos(newPos);
202  }
203  else
204  {
205  TrayClick = false;
206  if (event.MouseInput.Event == EMIE_MOUSE_MOVED)
207  return isInside;
208  }
209  }
210 
211  if (DraggedBySlider)
212  {
213  setPos(newPos);
214  }
215  else
216  {
217  DesiredPos = newPos;
218  }
219 
220  if (Pos != oldPos && Parent)
221  {
222  SEvent newEvent;
223  newEvent.EventType = EET_GUI_EVENT;
224  newEvent.GUIEvent.Caller = this;
225  newEvent.GUIEvent.Element = 0;
226  newEvent.GUIEvent.EventType = EGET_SCROLL_BAR_CHANGED;
227  Parent->OnEvent(newEvent);
228  }
229  return isInside;
230  } break;
231 
232  default:
233  break;
234  }
235  } break;
236  default:
237  break;
238  }
239  }
240 
241  return IGUIElement::OnEvent(event);
242 }

244 void CGUIScrollBar::OnPostRender(u32 timeMs)
245 {
246  if (Dragging && !DraggedBySlider && TrayClick && timeMs > LastChange + 200)
247  {
248  LastChange = timeMs;
249 
250  const s32 oldPos = Pos;
251 
252  if (DesiredPos >= Pos + LargeStep)
253  setPos(Pos + LargeStep);
254  else
255  if (DesiredPos <= Pos - LargeStep)
256  setPos(Pos - LargeStep);
257  else
258  if (DesiredPos >= Pos - LargeStep && DesiredPos <= Pos + LargeStep)
259  setPos(DesiredPos);
260 
261  if (Pos != oldPos && Parent)
262  {
263  SEvent newEvent;
264  newEvent.EventType = EET_GUI_EVENT;
265  newEvent.GUIEvent.Caller = this;
266  newEvent.GUIEvent.Element = 0;
267  newEvent.GUIEvent.EventType = EGET_SCROLL_BAR_CHANGED;
268  Parent->OnEvent(newEvent);
269  }
270  }

275 void CGUIScrollBar::draw()
276 {
277  if (!IsVisible)
278  return;
279 
280  IGUISkin* skin = Environment->getSkin();
281  if (!skin)
282  return;
283 
284 
285  video::SColor iconColor = skin->getColor(isEnabled() ? EGDC_WINDOW_SYMBOL : EGDC_GRAY_WINDOW_SYMBOL);
286  if ( iconColor != CurrentIconColor )
287  {
288  refreshControls();
289  }
290 
291 
292  SliderRect = AbsoluteRect;
293 
294  // draws the background
295  skin->draw2DRectangle(this, skin->getColor(EGDC_SCROLLBAR), SliderRect, &AbsoluteClippingRect);
296 
297  if ( core::isnotzero ( range() ) )
298  {
299  // recalculate slider rectangle
300  if (Horizontal)
301  {
302  SliderRect.UpperLeftCorner.X = AbsoluteRect.UpperLeftCorner.X + DrawPos + RelativeRect.getHeight() - DrawHeight/2;
303  SliderRect.LowerRightCorner.X = SliderRect.UpperLeftCorner.X + DrawHeight;
304  }
305  else
306  {
307  SliderRect.UpperLeftCorner.Y = AbsoluteRect.UpperLeftCorner.Y + DrawPos + RelativeRect.getWidth() - DrawHeight/2;
308  SliderRect.LowerRightCorner.Y = SliderRect.UpperLeftCorner.Y + DrawHeight;
309  }
310 
311  skin->draw3DButtonPaneStandard(this, SliderRect, &AbsoluteClippingRect);
312  }
313 
314  // draw buttons
315  IGUIElement::draw();
316 }

328 s32 CGUIScrollBar::getPosFromMousePos(const core::position2di &pos) const
329 {
330  f32 w, p;
331  if (Horizontal)
332  {
333  w = RelativeRect.getWidth() - f32(RelativeRect.getHeight())*3.0f;
334  p = pos.X - AbsoluteRect.UpperLeftCorner.X - RelativeRect.getHeight()*1.5f;
335  }
336  else
337  {
338  w = RelativeRect.getHeight() - f32(RelativeRect.getWidth())*3.0f;
339  p = pos.Y - AbsoluteRect.UpperLeftCorner.Y - RelativeRect.getWidth()*1.5f;
340  }
341  return (s32) ( p/w * range() ) + Min;
342 }

346 void CGUIScrollBar::setPos(s32 pos)
347 {
348  Pos = core::s32_clamp ( pos, Min, Max );
349 
350  if ( core::isnotzero ( range() ) )
351  {
352  if (Horizontal)
353  {
354  f32 f = (RelativeRect.getWidth() - ((f32)RelativeRect.getHeight()*3.0f)) / range();
355  DrawPos = (s32)( ( ( Pos - Min ) * f) + ((f32)RelativeRect.getHeight() * 0.5f));
356  DrawHeight = RelativeRect.getHeight();
357  }
358  else
359  {
360  f32 f = (RelativeRect.getHeight() - ((f32)RelativeRect.getWidth()*3.0f)) / range();
361 
362  DrawPos = (s32)( ( ( Pos - Min ) * f) + ((f32)RelativeRect.getWidth() * 0.5f));
363  DrawHeight = RelativeRect.getWidth();
364  }
365  }
366 }

453 void CGUIScrollBar::refreshControls()
454 {
455  CurrentIconColor = video::SColor(255,255,255,255);
456 
457  IGUISkin* skin = Environment->getSkin();
458  IGUISpriteBank* sprites = 0;
459 
460  if (skin)
461  {
462  sprites = skin->getSpriteBank();
463  CurrentIconColor = skin->getColor(isEnabled() ? EGDC_WINDOW_SYMBOL : EGDC_GRAY_WINDOW_SYMBOL);
464  }
465 
466  if (Horizontal)
467  {
468  const s32 h = RelativeRect.getHeight();
469  const s32 w = (h < RelativeRect.getWidth() / 2) ? h : RelativeRect.getWidth() / 2;
470  if (!UpButton)
471  {
472  UpButton = new CGUIButton(Environment, this, -1, core::rect<s32>(0,0, w, h), NoClip);
473  UpButton->setSubElement(true);
474  UpButton->setTabStop(false);
475  }
476  if (sprites)
477  {
478  UpButton->setSpriteBank(sprites);
479  UpButton->setSprite(EGBS_BUTTON_UP, skin->getIcon(EGDI_CURSOR_LEFT), CurrentIconColor);
480  UpButton->setSprite(EGBS_BUTTON_DOWN, skin->getIcon(EGDI_CURSOR_LEFT), CurrentIconColor);
481  }
482  UpButton->setRelativePosition(core::rect<s32>(0,0, w, h));
483  UpButton->setAlignment(EGUIA_UPPERLEFT, EGUIA_UPPERLEFT, EGUIA_UPPERLEFT, EGUIA_LOWERRIGHT);
484  if (!DownButton)
485  {
486  DownButton = new CGUIButton(Environment, this, -1, core::rect<s32>(RelativeRect.getWidth()-w, 0, RelativeRect.getWidth(), h), NoClip);
487  DownButton->setSubElement(true);
488  DownButton->setTabStop(false);
489  }
490  if (sprites)
491  {
492  DownButton->setSpriteBank(sprites);
493  DownButton->setSprite(EGBS_BUTTON_UP, skin->getIcon(EGDI_CURSOR_RIGHT), CurrentIconColor);
494  DownButton->setSprite(EGBS_BUTTON_DOWN, skin->getIcon(EGDI_CURSOR_RIGHT), CurrentIconColor);
495  }
496  DownButton->setRelativePosition(core::rect<s32>(RelativeRect.getWidth()-w, 0, RelativeRect.getWidth(), h));
497  DownButton->setAlignment(EGUIA_LOWERRIGHT, EGUIA_LOWERRIGHT, EGUIA_UPPERLEFT, EGUIA_LOWERRIGHT);
498  }
499  else
500  {
501  const s32 w = RelativeRect.getWidth();
502  const s32 h = (w < RelativeRect.getHeight() / 2) ? w : RelativeRect.getHeight() / 2;
503  if (!UpButton)
504  {
505  UpButton = new CGUIButton(Environment, this, -1, core::rect<s32>(0,0, w, h), NoClip);
506  UpButton->setSubElement(true);
507  UpButton->setTabStop(false);
508  }
509  if (sprites)
510  {
511  UpButton->setSpriteBank(sprites);
512  UpButton->setSprite(EGBS_BUTTON_UP, skin->getIcon(EGDI_CURSOR_UP), CurrentIconColor);
513  UpButton->setSprite(EGBS_BUTTON_DOWN, skin->getIcon(EGDI_CURSOR_UP), CurrentIconColor);
514  }
515  UpButton->setRelativePosition(core::rect<s32>(0,0, w, h));
516  UpButton->setAlignment(EGUIA_UPPERLEFT, EGUIA_LOWERRIGHT, EGUIA_UPPERLEFT, EGUIA_UPPERLEFT);
517  if (!DownButton)
518  {
519  DownButton = new CGUIButton(Environment, this, -1, core::rect<s32>(0,RelativeRect.getHeight()-h, w, RelativeRect.getHeight()), NoClip);
520  DownButton->setSubElement(true);

531  }
