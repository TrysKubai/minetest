36 bool ModChannel::removeConsumer(session_t peer_id)
37 {
38  bool found = false;

40  if (p == peer_id)
41  found = true;

43  return p == peer_id;
44  };
45 
46  m_client_consumers.erase(
47  std::remove_if(m_client_consumers.begin(),
