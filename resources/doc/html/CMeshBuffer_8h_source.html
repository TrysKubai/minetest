186  void append(const void* const vertices, u32 numVertices, const u16* const indices, u32 numIndices) override
187  {
188  if (vertices == getVertices())
189  return;
190 
191  const u32 vertexCount = getVertexCount();
192  u32 i;
193 
194  Vertices.reallocate(vertexCount+numVertices);
195  for (i=0; i<numVertices; ++i)
196  {
197  Vertices.push_back(static_cast<const T*>(vertices)[i]);
198  BoundingBox.addInternalPoint(static_cast<const T*>(vertices)[i].Pos);
199  }
200 
201  Indices.reallocate(getIndexCount()+numIndices);
202  for (i=0; i<numIndices; ++i)
203  {
204  Indices.push_back(indices[i]+vertexCount);
205  }
206  }
