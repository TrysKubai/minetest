221  if (!Null)
222  {
223  if (UseReferenceRect)
224  {
225 // NOTE: XIWarpPointer works when X11 has set a coordinate transformation matrix for the mouse unlike XWarpPointer
226 // which runs into a bug mentioned here: https://gitlab.freedesktop.org/xorg/xserver/-/issues/600
227 // So also workaround for Irrlicht bug #450
228 #ifdef _IRR_LINUX_X11_XINPUT2_
229  if ( DeviceId != 0)
230  {
231  XIWarpPointer(Device->XDisplay,
232  DeviceId,
233  None,
234  Device->XWindow, 0, 0,
235  Device->Width,
236  Device->Height,
237  ReferenceRect.UpperLeftCorner.X + x,
238  ReferenceRect.UpperLeftCorner.Y + y);
239  }
240  else
241 #endif
242  {
243  XWarpPointer(Device->XDisplay,
244  None,
245  Device->XWindow, 0, 0,
246  Device->Width,
247  Device->Height,
248  ReferenceRect.UpperLeftCorner.X + x,
249  ReferenceRect.UpperLeftCorner.Y + y);
250  }
251  }
252  else
253  {
254 #ifdef _IRR_LINUX_X11_XINPUT2_
255  if ( DeviceId != 0)
256  {
257  XIWarpPointer(Device->XDisplay,
258  DeviceId,
259  None,
260  Device->XWindow, 0, 0,
261  Device->Width,
262  Device->Height, x, y);
263  }
264  else
265 #endif
266  {
267  XWarpPointer(Device->XDisplay,
268  None,
269  Device->XWindow, 0, 0,
270  Device->Width,
271  Device->Height, x, y);

275  }

354  void updateCursorPos()
355  {
356 #ifdef _IRR_COMPILE_WITH_X11_
357  if (Null)
358  return;
359 
360  if ( PlatformBehavior&gui::ECPB_X11_CACHE_UPDATES && !os::Timer::isStopped() )
361  {
362  u32 now = os::Timer::getTime();
363  if (now <= LastQuery)
364  return;
365  LastQuery = now;
366  }
367 
368  Window tmp;
369  int itmp1, itmp2;
370  unsigned int maskreturn;
371  XQueryPointer(Device->XDisplay, Device->XWindow,
372  &tmp, &tmp,
373  &itmp1, &itmp2,
374  &CursorPos.X, &CursorPos.Y, &maskreturn);
375 #endif
