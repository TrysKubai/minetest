178  vector2d<T>& rotateBy(f64 degrees, const vector2d<T>& center=vector2d<T>())
179  {
180  degrees *= DEGTORAD64;
181  const f64 cs = cos(degrees);
182  const f64 sn = sin(degrees);
183 
184  X -= center.X;
185  Y -= center.Y;
186 
187  set((T)(X*cs - Y*sn), (T)(X*sn + Y*cs));
188 
189  X += center.X;
190  Y += center.Y;
191  return *this;
192  }

235  inline f64 getAngle() const
236  {
237  if (Y == 0) // corrected thanks to a suggestion by Jox
238  return X < 0 ? 180 : 0;
239  else if (X == 0)
240  return Y < 0 ? 90 : 270;
241 
242  // don't use getLength here to avoid precision loss with s32 vectors
243  // avoid floating-point trouble as sqrt(y*y) is occasionally larger than y, so clamp
244  const f64 tmp = core::clamp(Y / sqrt((f64)(X*X + Y*Y)), -1.0, 1.0);
245  const f64 angle = atan( core::squareroot(1 - tmp*tmp) / tmp) * RADTODEG64;
246 
247  if (X>0 && Y>0)
248  return angle + 270;
249  else
250  if (X>0 && Y<0)
251  return angle + 90;
252  else
253  if (X<0 && Y<0)
254  return 90 - angle;
255  else
256  if (X<0 && Y>0)
257  return 270 - angle;
258 
259  return angle;
260  }

265  inline f64 getAngleWith(const vector2d<T>& b) const
266  {
267  f64 tmp = (f64)(X*b.X + Y*b.Y);
268 
269  if (tmp == 0.0)
270  return 90.0;
271 
272  tmp = tmp / core::squareroot((f64)((X*X + Y*Y) * (b.X*b.X + b.Y*b.Y)));
273  if (tmp < 0.0)
274  tmp = -tmp;
275  if ( tmp > 1.0 ) // avoid floating-point trouble
276  tmp = 1.0;
277 
278  return atan(sqrt(1 - tmp*tmp) / tmp) * RADTODEG64;
279  }

326  vector2d<T> getInterpolated_quadratic(const vector2d<T>& v2, const vector2d<T>& v3, f64 d) const
327  {
328  // this*(1-d)*(1-d) + 2 * v2 * (1-d) + v3 * d * d;
329  const f64 inv = 1.0f - d;
330  const f64 mul0 = inv * inv;
331  const f64 mul1 = 2.0f * d * inv;
332  const f64 mul2 = d * d;
333 
334  return vector2d<T> ( (T)(X * mul0 + v2.X * mul1 + v3.X * mul2),
335  (T)(Y * mul0 + v2.Y * mul1 + v3.Y * mul2));
336  }

343  s32 checkOrientation( const vector2d<T> & b, const vector2d<T> & c) const
344  {
345  // Example of clockwise points
346  //
347  // ^ Y
348  // | A
349  // | . .
350  // | . .
351  // | C.....B
352  // +---------------> X
353 
354  T val = (b.Y - Y) * (c.X - b.X) -
355  (b.X - X) * (c.Y - b.Y);
356 
357  if (val == 0) return 0; // colinear
358 
359  return (val > 0) ? 1 : 2; // clock or counterclock wise
360  }
