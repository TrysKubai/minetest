148  lua_pushstring(m_luastack, "/");

234 void ScriptApiBase::loadScript(const std::string &script_path)
235 {
236  verbosestream << "Loading and running script from " << script_path << std::endl;
237 
238  lua_State *L = getStack();
239 
240  int error_handler = PUSH_ERROR_HANDLER(L);
241 
242  bool ok;
243  if (m_secure) {
244  ok = ScriptApiSecurity::safeLoadFile(L, script_path.c_str());
245  } else {
246  ok = !luaL_loadfile(L, script_path.c_str());
247  }
248  ok = ok && !lua_pcall(L, 0, 0, error_handler);
249  if (!ok) {
250  const char *error_msg = lua_tostring(L, -1);
251  if (!error_msg)
252  error_msg = "(error object is not a string)";
253  lua_pop(L, 2); // Pop error message and error handler
254  throw ModError("Failed to load and run script from " +
255  script_path + ":\n" + error_msg);
256  }
257  lua_pop(L, 1); // Pop error handler
258 }

295 // Push the list of callbacks (a lua table).
296 // Then push nargs arguments.
297 // Then call this function, which
298 // - runs the callbacks
299 // - replaces the table and arguments with the return value,
300 // computed depending on mode
301 // This function must only be called with scriptlock held (i.e. inside of a
302 // code block with SCRIPTAPI_PRECHECKHEADER declared)
303 void ScriptApiBase::runCallbacksRaw(int nargs,
304  RunCallbacksMode mode, const char *fxn)
305 {
306 #ifndef SERVER
307  // Hard fail for bad guarded callbacks
308  // Only run callbacks when the scripting enviroment is loaded
309  FATAL_ERROR_IF(m_type == ScriptingType::Client &&
310  !getClient()->modsLoaded(), fxn);
311 #endif
312 
313 #ifdef SCRIPTAPI_LOCK_DEBUG
314  assert(m_lock_recursion_count > 0);
315 #endif
316  lua_State *L = getStack();
317  FATAL_ERROR_IF(lua_gettop(L) < nargs + 1, "Not enough arguments");
318 
319  // Insert error handler
320  PUSH_ERROR_HANDLER(L);
321  int error_handler = lua_gettop(L) - nargs - 1;
322  lua_insert(L, error_handler);
323 
324  // Insert run_callbacks between error handler and table
325  lua_getglobal(L, "core");
326  lua_getfield(L, -1, "run_callbacks");
327  lua_remove(L, -2);
328  lua_insert(L, error_handler + 1);
329 
330  // Insert mode after table
331  lua_pushnumber(L, (int)mode);
332  lua_insert(L, error_handler + 3);
333 
334  // Stack now looks like this:
335  // ... <error handler> <run_callbacks> <table> <mode> <arg#1> <arg#2> ... <arg#n>
336 
337  int result = lua_pcall(L, nargs + 2, 1, error_handler);
338  if (result != 0)
339  scriptError(result, fxn);
340 
341  lua_remove(L, error_handler);
342 }

360 void ScriptApiBase::stackDump(std::ostream &o)
361 {
362  int top = lua_gettop(m_luastack);
363  for (int i = 1; i <= top; i++) { /* repeat for each level */
364  int t = lua_type(m_luastack, i);
365  switch (t) {
366  case LUA_TSTRING: /* strings */
367  o << "\"" << readParam<std::string>(m_luastack, i) << "\"";
368  break;
369  case LUA_TBOOLEAN: /* booleans */
370  o << (readParam<bool>(m_luastack, i) ? "true" : "false");
371  break;
372  case LUA_TNUMBER: /* numbers */ {
373  char buf[10];
374  porting::mt_snprintf(buf, sizeof(buf), "%lf", lua_tonumber(m_luastack, i));
375  o << buf;
376  break;
377  }
378  default: /* other values */
379  o << lua_typename(m_luastack, t);
380  break;
381  }
382  o << " ";
383  }
384  o << std::endl;
385 }

399 /*
400  * How ObjectRefs are handled in Lua:
401  * When an active object is created, an ObjectRef is created on the Lua side
402  * and stored in core.object_refs[id].
403  * Methods that require an ObjectRef to a certain object retrieve it from that
404  * table instead of creating their own.(*)
405  * When an active object is removed, the existing ObjectRef is invalidated
406  * using ::set_null() and removed from the core.object_refs table.
407  * (*) An exception to this are NULL ObjectRefs and anonymous ObjectRefs
408  * for objects without ID.
409  * It's unclear what the latter are needed for and their use is problematic
410  * since we lose control over the ref and the contained pointer.
411  */
412 
413 void ScriptApiBase::addObjectReference(ServerActiveObject *cobj)
414 {
415  SCRIPTAPI_PRECHECKHEADER
416  //infostream<<"scriptapi_add_object_reference: id="<<cobj->getId()<<std::endl;
417 
418  // Create object on stack
419  ObjectRef::create(L, cobj); // Puts ObjectRef (as userdata) on stack
420  int object = lua_gettop(L);
421 
422  // Get core.object_refs table
423  lua_getglobal(L, "core");
424  lua_getfield(L, -1, "object_refs");
425  luaL_checktype(L, -1, LUA_TTABLE);
426  int objectstable = lua_gettop(L);
427 
428  // object_refs[id] = object
429  lua_pushnumber(L, cobj->getId()); // Push id
430  lua_pushvalue(L, object); // Copy object to top of stack
431  lua_settable(L, objectstable);
432 }

434 void ScriptApiBase::removeObjectReference(ServerActiveObject *cobj)
435 {
436  SCRIPTAPI_PRECHECKHEADER
437  //infostream<<"scriptapi_rm_object_reference: id="<<cobj->getId()<<std::endl;
438 
439  // Get core.object_refs table
440  lua_getglobal(L, "core");
441  lua_getfield(L, -1, "object_refs");
442  luaL_checktype(L, -1, LUA_TTABLE);
443  int objectstable = lua_gettop(L);
444 
445  // Get object_refs[id]
446  lua_pushnumber(L, cobj->getId()); // Push id
447  lua_gettable(L, objectstable);
448  // Set object reference to NULL
449  ObjectRef::set_null(L);
450  lua_pop(L, 1); // pop object
451 
452  // Set object_refs[id] = nil
453  lua_pushnumber(L, cobj->getId()); // Push id
454  lua_pushnil(L);
455  lua_settable(L, objectstable);
456 }
