109 void TestNodeResolver::testNodeResolving(NodeDefManager *ndef)
110 {
111  Foobar foobar;
112  size_t i;
113 
114  foobar.m_nodenames.emplace_back("default:torch");
115 
116  foobar.m_nodenames.emplace_back("default:dirt_with_grass");
117  foobar.m_nodenames.emplace_back("default:water");
118  foobar.m_nodenames.emplace_back("default:abloobloobloo");
119  foobar.m_nodenames.emplace_back("default:stone");
120  foobar.m_nodenames.emplace_back("default:shmegoldorf");
121  foobar.m_nnlistsizes.push_back(5);
122 
123  foobar.m_nodenames.emplace_back("group:liquids");
124  foobar.m_nnlistsizes.push_back(1);
125 
126  foobar.m_nodenames.emplace_back("default:warf");
127  foobar.m_nodenames.emplace_back("default:stone");
128  foobar.m_nodenames.emplace_back("default:bloop");
129  foobar.m_nnlistsizes.push_back(3);
130 
131  foobar.m_nnlistsizes.push_back(0);
132 
133  foobar.m_nodenames.emplace_back("default:brick");
134  foobar.m_nodenames.emplace_back("default:desert_stone");
135  foobar.m_nodenames.emplace_back("default:shnitzle");
136 
137  ndef->pendNodeResolve(&foobar);
138  UASSERT(foobar.m_ndef == ndef);
139 
140  ndef->setNodeRegistrationStatus(true);
141  ndef->runNodeResolveCallbacks();
142 
143  // Check that we read single nodes successfully
144  UASSERTEQ(content_t, foobar.test_nr_node1, t_CONTENT_TORCH);
145  UASSERTEQ(content_t, foobar.test_nr_node2, t_CONTENT_BRICK);
146  UASSERTEQ(content_t, foobar.test_nr_node3, t_CONTENT_BRICK);
147  UASSERTEQ(content_t, foobar.test_nr_node4, CONTENT_AIR);
148  UASSERTEQ(content_t, foobar.test_nr_node5, CONTENT_IGNORE);
149 
150  // Check that we read all the regular list items
151  static const content_t expected_test_nr_list[] = {
152  t_CONTENT_GRASS,
153  t_CONTENT_WATER,
154  t_CONTENT_STONE,
155  };
156  UASSERTEQ(size_t, foobar.test_nr_list.size(), 3);
157  for (i = 0; i != foobar.test_nr_list.size(); i++)
158  UASSERTEQ(content_t, foobar.test_nr_list[i], expected_test_nr_list[i]);
159 
160  // Check that we read all the list items that were from a group entry
161  static const content_t expected_test_nr_list_group[] = {
162  t_CONTENT_WATER,
163  t_CONTENT_LAVA,
164  };
165  UASSERTEQ(size_t, foobar.test_nr_list_group.size(), 2);
166  for (i = 0; i != foobar.test_nr_list_group.size(); i++) {
167  UASSERT(CONTAINS(foobar.test_nr_list_group,
168  expected_test_nr_list_group[i]));
169  }
170 
171  // Check that we read all the items we're able to in a required list
172  static const content_t expected_test_nr_list_required[] = {
173  CONTENT_AIR,
174  t_CONTENT_STONE,
175  CONTENT_AIR,
176  };
177  UASSERTEQ(size_t, foobar.test_nr_list_required.size(), 3);
178  for (i = 0; i != foobar.test_nr_list_required.size(); i++)
179  UASSERTEQ(content_t, foobar.test_nr_list_required[i],
180  expected_test_nr_list_required[i]);
181 
182  // Check that the edge case of 0 is successful
183  UASSERTEQ(size_t, foobar.test_nr_list_empty.size(), 0);
184 }

187 void TestNodeResolver::testPendingResolveCancellation(NodeDefManager *ndef)
188 {
189  Foobaz foobaz1;
190  foobaz1.test_content1 = 1234;
191  foobaz1.test_content2 = 5678;
192  foobaz1.m_nodenames.emplace_back("default:dirt_with_grass");
193  foobaz1.m_nodenames.emplace_back("default:abloobloobloo");

195 
196  Foobaz foobaz2;
197  foobaz2.test_content1 = 1234;
198  foobaz2.test_content2 = 5678;
199  foobaz2.m_nodenames.emplace_back("default:dirt_with_grass");
200  foobaz2.m_nodenames.emplace_back("default:abloobloobloo");
201  ndef->pendNodeResolve(&foobaz2);
202 
203  ndef->cancelNodeResolveCallback(&foobaz1);
204 
205  ndef->setNodeRegistrationStatus(true);
206  ndef->runNodeResolveCallbacks();
207 
208  UASSERT(foobaz1.test_content1 == 1234);
209  UASSERT(foobaz1.test_content2 == 5678);
210  UASSERT(foobaz2.test_content1 == t_CONTENT_GRASS);
211  UASSERT(foobaz2.test_content2 == CONTENT_IGNORE);
212 }
