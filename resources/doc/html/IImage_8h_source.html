215  static core::dimension2du getMipMapsSize(const core::dimension2du& sizeLevel0, u32 mipmapLevel)
216  {
217  core::dimension2du result(sizeLevel0);
218  u32 i=0;
219  while (i != mipmapLevel)
220  {
221  if (result.Width>1)
222  result.Width >>= 1;
223  if (result.Height>1)
224  result.Height>>=1;
225  ++i;
226 
227  if ( result.Width == 1 && result.Height == 1 && i < mipmapLevel )
228  return core::dimension2du(0,0);
229  }
230  return result;
231  }

239  void* getMipMapsData(irr::u32 mipLevel=1) const
240  {
241  if ( MipMapsData && mipLevel > 0)
242  {
243  size_t dataSize = 0;
244  core::dimension2du mipSize(Size);
245  u32 i = 1; // We want the start of data for this level, not end.
246 
247  while (i != mipLevel)
248  {
249  if (mipSize.Width > 1)
250  mipSize.Width >>= 1;
251 
252  if (mipSize.Height > 1)
253  mipSize.Height >>= 1;
254 
255  dataSize += getDataSizeFromFormat(Format, mipSize.Width, mipSize.Height);
256 
257  ++i;
258  if ( mipSize.Width == 1 && mipSize.Height == 1 && i < mipLevel)
259  return 0;
260  }
261 
262  return MipMapsData + dataSize;
263  }
264 
265  return 0;
266  }

277  void setMipMapsData(void* data, bool ownForeignMemory)
278  {
279  if (data != MipMapsData)
280  {
281  if (DeleteMipMapsMemory)
282  {
283  delete[] MipMapsData;
284 
285  DeleteMipMapsMemory = false;
286  }
287 
288  if (data)
289  {
290  if (ownForeignMemory)
291  {
292  MipMapsData = static_cast<u8*>(data);
293 
294  DeleteMipMapsMemory = false;
295  }
296  else
297  {
298  u32 dataSize = 0;
299  u32 width = Size.Width;
300  u32 height = Size.Height;
301 
302  do
303  {
304  if (width > 1)
305  width >>= 1;
306 
307  if (height > 1)
308  height >>= 1;
309 
310  dataSize += getDataSizeFromFormat(Format, width, height);
311  } while (width != 1 || height != 1);
312 
313  MipMapsData = new u8[dataSize];
314  memcpy(MipMapsData, data, dataSize);
315 
316  DeleteMipMapsMemory = true;
317  }
318  }
319  else
320  {
321  MipMapsData = 0;
322  }
323  }

450  static u32 getDataSizeFromFormat(ECOLOR_FORMAT format, u32 width, u32 height)
451  {
452  u32 imageSize = 0;
453 
454  switch (format)
455  {
456  case ECF_DXT1:
457  imageSize = ((width + 3) / 4) * ((height + 3) / 4) * 8;
458  break;
459  case ECF_DXT2:
460  case ECF_DXT3:
461  case ECF_DXT4:
462  case ECF_DXT5:
463  imageSize = ((width + 3) / 4) * ((height + 3) / 4) * 16;
464  break;
465  case ECF_PVRTC_RGB2:
466  case ECF_PVRTC_ARGB2:
467  imageSize = (core::max_<u32>(width, 16) * core::max_<u32>(height, 8) * 2 + 7) / 8;
468  break;
469  case ECF_PVRTC_RGB4:
470  case ECF_PVRTC_ARGB4:
471  imageSize = (core::max_<u32>(width, 8) * core::max_<u32>(height, 8) * 4 + 7) / 8;
472  break;
473  case ECF_PVRTC2_ARGB2:
474  imageSize = core::ceil32(width / 8.0f) * core::ceil32(height / 4.0f) * 8;
475  break;
476  case ECF_PVRTC2_ARGB4:
477  case ECF_ETC1:
478  case ECF_ETC2_RGB:
479  imageSize = core::ceil32(width / 4.0f) * core::ceil32(height / 4.0f) * 8;
480  break;
481  case ECF_ETC2_ARGB:
482  imageSize = core::ceil32(width / 4.0f) * core::ceil32(height / 4.0f) * 16;
483  break;
484  default: // uncompressed formats
485  imageSize = getBitsPerPixelFromFormat(format) / 8 * width;
486  imageSize *= height;
487  break;
488  }
489 
490  return imageSize;
491  }
