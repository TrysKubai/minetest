166 bool CFileSystem::moveFileArchive(u32 sourceIndex, s32 relative)
167 {
168  bool r = false;
169  const s32 dest = (s32) sourceIndex + relative;
170  const s32 dir = relative < 0 ? -1 : 1;
171  const s32 sourceEnd = ((s32) FileArchives.size() ) - 1;
172  IFileArchive *t;
173 
174  for (s32 s = (s32) sourceIndex;s != dest; s += dir)
175  {
176  if (s < 0 || s > sourceEnd || s + dir < 0 || s + dir > sourceEnd)
177  continue;
178 
179  t = FileArchives[s + dir];
180  FileArchives[s + dir] = FileArchives[s];
181  FileArchives[s] = t;
182  r = true;
183  }
184  return r;
185 }

189 bool CFileSystem::addFileArchive(const io::path& filename, bool ignoreCase,
190  bool ignorePaths, E_FILE_ARCHIVE_TYPE archiveType,
191  const core::stringc& password,
192  IFileArchive** retArchive)
193 {
194  IFileArchive* archive = 0;
195  bool ret = false;
196 
197  // see if archive is already added
198 
199  s32 i;
200 
201  // do we know what type it should be?
202  if (archiveType == EFAT_UNKNOWN || archiveType == EFAT_FOLDER)
203  {
204  // try to load archive based on file name
205  for (i = ArchiveLoader.size()-1; i >=0 ; --i)
206  {
207  if (ArchiveLoader[i]->isALoadableFileFormat(filename))
208  {
209  archive = ArchiveLoader[i]->createArchive(filename, ignoreCase, ignorePaths);
210  if (archive)
211  break;
212  }
213  }
214 
215  // try to load archive based on content
216  if (!archive)
217  {
218  io::IReadFile* file = createAndOpenFile(filename);
219  if (file)
220  {
221  for (i = ArchiveLoader.size()-1; i >= 0; --i)
222  {
223  file->seek(0);
224  if (ArchiveLoader[i]->isALoadableFileFormat(file))
225  {
226  file->seek(0);
227  archive = ArchiveLoader[i]->createArchive(file, ignoreCase, ignorePaths);
228  if (archive)
229  break;
230  }
231  }
232  file->drop();
233  }
234  }
235  }
236  else
237  {
238  // try to open archive based on archive loader type
239 
240  io::IReadFile* file = 0;
241 
242  for (i = ArchiveLoader.size()-1; i >= 0; --i)
243  {
244  if (ArchiveLoader[i]->isALoadableFileFormat(archiveType))
245  {
246  // attempt to open file
247  if (!file)
248  file = createAndOpenFile(filename);
249 
250  // is the file open?
251  if (file)
252  {
253  // attempt to open archive
254  file->seek(0);
255  if (ArchiveLoader[i]->isALoadableFileFormat(file))
256  {
257  file->seek(0);
258  archive = ArchiveLoader[i]->createArchive(file, ignoreCase, ignorePaths);
259  if (archive)
260  break;
261  }
262  }
263  else
264  {
265  // couldn't open file
266  break;
267  }
268  }
269  }
270 
271  // if open, close the file
272  if (file)
273  file->drop();
274  }
275 
276  if (archive)
277  {
278  FileArchives.push_back(archive);
279  if (password.size())
280  archive->Password=password;
281  if (retArchive)
282  *retArchive = archive;
283  ret = true;
284  }
285  else
286  {
287  os::Printer::log("Could not create archive for", filename, ELL_ERROR);
288  }
289 
290  return ret;
291 }

451 const io::path& CFileSystem::getWorkingDirectory()
452 {
453  EFileSystemType type = FileSystemType;
454 
455  if (type != FILESYSTEM_NATIVE)
456  {
457  type = FILESYSTEM_VIRTUAL;
458  }
459  else
460  {
461  #if defined(_IRR_WINDOWS_API_)
462  fschar_t tmp[_MAX_PATH];
463  _getcwd(tmp, _MAX_PATH);
464  WorkingDirectory[FILESYSTEM_NATIVE] = tmp;
465  WorkingDirectory[FILESYSTEM_NATIVE].replace('\\', '/');
466  #endif
467 
468  #if (defined(_IRR_POSIX_API_) || defined(_IRR_OSX_PLATFORM_))
469 
470  // getting the CWD is rather complex as we do not know the size
471  // so try it until the call was successful
472  // Note that neither the first nor the second parameter may be 0 according to POSIX
473 
474  u32 pathSize=256;
475  char *tmpPath = new char[pathSize];
476  while ((pathSize < (1<<16)) && !(getcwd(tmpPath,pathSize)))
477  {
478  delete [] tmpPath;
479  pathSize *= 2;
480  tmpPath = new char[pathSize];
481  }
482  if (tmpPath)
483  {
484  WorkingDirectory[FILESYSTEM_NATIVE] = tmpPath;
485  delete [] tmpPath;
486  }
487  #endif
488 
489  WorkingDirectory[type].validate();
490  }
491 
492  return WorkingDirectory[type];
493 }

523 io::path CFileSystem::getAbsolutePath(const io::path& filename) const
524 {
525  if ( filename.empty() )
526  return filename;
527 #if defined(_IRR_WINDOWS_API_)
528  fschar_t *p=0;
529  fschar_t fpath[_MAX_PATH];
530  p = _fullpath(fpath, filename.c_str(), _MAX_PATH);
531  core::stringc tmp(p);
532  tmp.replace('\\', '/');
533  return tmp;
534 #elif (defined(_IRR_POSIX_API_) || defined(_IRR_OSX_PLATFORM_))
535  c8* p=0;
536  c8 fpath[4096];
537  fpath[0]=0;
538  p = realpath(filename.c_str(), fpath);
539  if (!p)
540  {
541  // content in fpath is unclear at this point
542  if (!fpath[0]) // seems like fpath wasn't altered, use our best guess
543  {
544  io::path tmp(filename);
545  return flattenFilename(tmp);
546  }
547  else
548  return io::path(fpath);
549  }
550  if (filename[filename.size()-1]=='/')
551  return io::path(p)+_IRR_TEXT("/");
552  else
553  return io::path(p);
554 #else
555  return io::path(filename);
556 #endif

579 io::path CFileSystem::getFileBasename(const io::path& filename, bool keepExtension) const
580 {
581  // find last forward or backslash
582  s32 lastSlash = filename.findLast('/');
583  const s32 lastBackSlash = filename.findLast('\\');
584  lastSlash = core::max_(lastSlash, lastBackSlash);
585 
586  // get number of chars after last dot
587  s32 end = 0;
588  if (!keepExtension)
589  {
590  // take care to search only after last slash to check only for
591  // dots in the filename
592  end = filename.findLast('.');
593  if (end == -1 || end < lastSlash)
594  end=0;
595  else
596  end = filename.size()-end;
597  }
598 
599  if ((u32)lastSlash < filename.size())
600  return filename.subString(lastSlash+1, filename.size()-lastSlash-1-end);
601  else if (end != 0)
602  return filename.subString(0, filename.size()-end);
603  else
604  return filename;
605 }

609 io::path& CFileSystem::flattenFilename(io::path& directory, const io::path& root) const
610 {
611  directory.replace('\\', '/');
612  if (directory.lastChar() != '/')
613  directory.append('/');
614 
615  io::path dir;
616  io::path subdir;
617 
618  s32 lastpos = 0;
619  s32 pos = 0;
620  bool lastWasRealDir=false;
621 
622  while ((pos = directory.findNext('/', lastpos)) >= 0)
623  {
624  subdir = directory.subString(lastpos, pos - lastpos + 1);
625 
626  if (subdir == _IRR_TEXT("../"))
627  {
628  if (lastWasRealDir)
629  {
630  deletePathFromPath(dir, 2);
631  lastWasRealDir=(dir.size()!=0);
632  }
633  else
634  {
635  dir.append(subdir);
636  lastWasRealDir=false;
637  }
638  }
639  else if (subdir == _IRR_TEXT("/"))
640  {
641  dir = root;
642  }
643  else if (subdir != _IRR_TEXT("./"))
644  {
645  dir.append(subdir);
646  lastWasRealDir=true;
647  }
648 
649  lastpos = pos + 1;
650  }
651  directory = dir;
652  return directory;
653 }

657 path CFileSystem::getRelativeFilename(const path& filename, const path& directory) const
658 {
659  if ( filename.empty() || directory.empty() )
660  return filename;
661 
662  io::path path1, file, ext;
663  core::splitFilename(getAbsolutePath(filename), &path1, &file, &ext);
664  io::path path2(getAbsolutePath(directory));
665  std::list<io::path> list1, list2;
666  path1.split(list1, _IRR_TEXT("/\\"), 2);
667  path2.split(list2, _IRR_TEXT("/\\"), 2);
668  std::list<io::path>::const_iterator it1,it2;
669  it1=list1.begin();
670  it2=list2.begin();
671 
672  #if defined (_IRR_WINDOWS_API_)
673  fschar_t partition1 = 0, partition2 = 0;
674  io::path prefix1, prefix2;
675  if ( it1 != list1.end() )
676  prefix1 = *it1;
677  if ( it2 != list2.end() )
678  prefix2 = *it2;
679  if ( prefix1.size() > 1 && prefix1[1] == _IRR_TEXT(':') )
680  partition1 = core::locale_lower(prefix1[0]);
681  if ( prefix2.size() > 1 && prefix2[1] == _IRR_TEXT(':') )
682  partition2 = core::locale_lower(prefix2[0]);
683 
684  // must have the same prefix or we can't resolve it to a relative filename
685  if ( partition1 != partition2 )
686  {
687  return filename;
688  }
689  #endif
690 
691 
692  for (; it1 != list1.end() && it2 != list2.end()
693 #if defined (_IRR_WINDOWS_API_)
694  && (io::path(*it1).make_lower()==io::path(*it2).make_lower())
695 #else
696  && (*it1==*it2)
697 #endif
698  ;)
699  {
700  ++it1;
701  ++it2;
702  }
703  path1=_IRR_TEXT("");
704  for (; it2 != list2.end(); ++it2)
705  path1 += _IRR_TEXT("../");
706  while (it1 != list1.end())
707  {
708  path1 += *it1++;
709  path1 += _IRR_TEXT('/');
710  }
711  path1 += file;
712  if (ext.size())
713  {
714  path1 += _IRR_TEXT('.');
715  path1 += ext;
716  }
717  return path1;
718 }

731 IFileList* CFileSystem::createFileList()
732 {
733  CFileList* r = 0;
734  io::path Path = getWorkingDirectory();
735  Path.replace('\\', '/');
736  if (!Path.empty() && Path.lastChar() != '/')
737  Path.append('/');
738 
740  if (FileSystemType == FILESYSTEM_NATIVE)
741  {
742  // --------------------------------------------
744  #ifdef _IRR_WINDOWS_API_
745 
746  r = new CFileList(Path, true, false);
747 
748  // TODO: Should be unified once mingw adapts the proper types
749 #if defined(__GNUC__)
750  long hFile; //mingw return type declaration
751 #else
752  intptr_t hFile;
753 #endif
754 
755  struct _tfinddata_t c_file;
756  if( (hFile = _tfindfirst( _T("*"), &c_file )) != -1L )
757  {
758  do
759  {
760  r->addItem(Path + c_file.name, 0, c_file.size, (_A_SUBDIR & c_file.attrib) != 0, 0);
761  }
762  while( _tfindnext( hFile, &c_file ) == 0 );
763 
764  _findclose( hFile );
765  }
766 
767  #endif
768 
769  // --------------------------------------------
771  #if (defined(_IRR_POSIX_API_) || defined(_IRR_OSX_PLATFORM_))
772 
773 
774  r = new CFileList(Path, false, false);
775 
776  r->addItem(Path + _IRR_TEXT(".."), 0, 0, true, 0);
777 
779  DIR* dirHandle=opendir(Path.c_str());
780  if (dirHandle)
781  {
782  struct dirent *dirEntry;
783  while ((dirEntry=readdir(dirHandle)))
784  {
785  u32 size = 0;
786  bool isDirectory = false;
787 
788  if((strcmp(dirEntry->d_name, ".")==0) ||
789  (strcmp(dirEntry->d_name, "..")==0))
790  {
791  continue;
792  }
793  struct stat buf;
794  if (stat(dirEntry->d_name, &buf)==0)
795  {
796  size = buf.st_size;
797  isDirectory = S_ISDIR(buf.st_mode);
798  }
799  #if !defined(_IRR_SOLARIS_PLATFORM_) && !defined(__CYGWIN__) && !defined(__HAIKU__)
800  // only available on some systems
801  else
802  {
803  isDirectory = dirEntry->d_type == DT_DIR;
804  }
805  #endif
806 
807  r->addItem(Path + dirEntry->d_name, 0, size, isDirectory, 0);
808  }
809  closedir(dirHandle);
810  }
811  #endif
812  }
813  else
814  {
816  r = new CFileList(Path, false, false);
817 
819  SFileListEntry e2;
820  SFileListEntry e3;
821 
823  r->addItem(Path + _IRR_TEXT("."), 0, 0, true, 0);
824 
826  r->addItem(Path + _IRR_TEXT(".."), 0, 0, true, 0);
827 
829  for (u32 i=0; i < FileArchives.size(); ++i)
830  {
831  const IFileList *merge = FileArchives[i]->getFileList();
832 
833  for (u32 j=0; j < merge->getFileCount(); ++j)
834  {
