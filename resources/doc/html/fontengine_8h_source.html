41 struct FontSpec {
42  FontSpec(unsigned int font_size, FontMode mode, bool bold, bool italic) :
43  size(font_size),
44  mode(mode),
45  bold(bold),
46  italic(italic) {}
47 
48  u16 getHash() const
49  {
50  return (mode << 2) | (static_cast<u8>(bold) << 1) | static_cast<u8>(italic);
51  }

54  FontMode mode;

57 };
