152  // Orders blocks by distance to the camera
153  class MapBlockComparer
154  {
155  public:
156  MapBlockComparer(const v3s16 &camera_block) : m_camera_block(camera_block) {}
157 
158  bool operator() (const v3s16 &left, const v3s16 &right) const
159  {
160  auto distance_left = left.getDistanceFromSQ(m_camera_block);
161  auto distance_right = right.getDistanceFromSQ(m_camera_block);
162  return distance_left > distance_right || (distance_left == distance_right && left > right);
163  }
164 
165  private:
166  v3s16 m_camera_block;
