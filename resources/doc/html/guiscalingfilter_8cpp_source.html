27 /* Maintain a static cache to store the images that correspond to textures
28  * in a format that's manipulable by code. Some platforms exhibit issues
29  * converting textures back into images repeatedly, and some don't even
30  * allow it at all.
31  */

34 /* Maintain a static cache of all pre-scaled textures. These need to be
35  * cleared as well when the cached images.

37 std::map<io::path, video::ITexture *> g_txrCache;

39 /* Manually insert an image into the cache, useful to avoid texture-to-image
40  * conversion whenever we can intercept it.
41  */
42 void guiScalingCache(const io::path &key, video::IVideoDriver *driver, video::IImage *value)
43 {
44  if (!g_settings->getBool("gui_scaling_filter"))
45  return;
46 
47  if (g_imgCache.find(key) != g_imgCache.end())
48  return; // Already cached.
49 
50  video::IImage *copied = driver->createImage(value->getColorFormat(),
51  value->getDimension());
52  value->copyTo(copied);
53  g_imgCache[key] = copied;
54 }

56 // Manually clear the cache, e.g. when switching to different worlds.
57 void guiScalingCacheClear()
58 {
59  for (auto &it : g_imgCache) {
60  if (it.second)
61  it.second->drop();
62  }
63  g_imgCache.clear();
64  for (auto &it : g_txrCache) {
65  if (it.second)
66  RenderingEngine::get_video_driver()->removeTexture(it.second);
67  }
68  g_txrCache.clear();
69 }

71 /* Get a cached, high-quality pre-scaled texture for display purposes. If the
72  * texture is not already cached, attempt to create it. Returns a pre-scaled texture,
73  * or the original texture if unable to pre-scale it.
74  */
75 video::ITexture *guiScalingResizeCached(video::IVideoDriver *driver,
76  video::ITexture *src, const core::rect<s32> &srcrect,
77  const core::rect<s32> &destrect)
78 {
79  if (src == NULL)
80  return src;
81  if (!g_settings->getBool("gui_scaling_filter"))
82  return src;
83 
84  // Calculate scaled texture name.
85  char rectstr[200];
86  porting::mt_snprintf(rectstr, sizeof(rectstr), "%d:%d:%d:%d:%d:%d",
87  srcrect.UpperLeftCorner.X,
88  srcrect.UpperLeftCorner.Y,
89  srcrect.getWidth(),
90  srcrect.getHeight(),
91  destrect.getWidth(),
92  destrect.getHeight());
93  io::path origname = src->getName().getPath();
94  io::path scalename = origname + "@guiScalingFilter:" + rectstr;
95 
96  // Search for existing scaled texture.
97  auto it_txr = g_txrCache.find(scalename);
98  video::ITexture *scaled = (it_txr != g_txrCache.end()) ? it_txr->second : nullptr;
99  if (scaled)
100  return scaled;
101 
102  // Try to find the texture converted to an image in the cache.
103  // If the image was not found, try to extract it from the texture.
104  auto it_img = g_imgCache.find(origname);
105  video::IImage *srcimg = (it_img != g_imgCache.end()) ? it_img->second : nullptr;
106  if (!srcimg) {
107  if (!g_settings->getBool("gui_scaling_filter_txr2img"))
108  return src;
109  srcimg = driver->createImageFromData(src->getColorFormat(),
110  src->getSize(), src->lock(video::ETLM_READ_ONLY), false);
111  src->unlock();
112  g_imgCache[origname] = srcimg;
113  }
114 
115  // Create a new destination image and scale the source into it.
116  imageCleanTransparent(srcimg, 0);
117  video::IImage *destimg = driver->createImage(src->getColorFormat(),
118  core::dimension2d<u32>((u32)destrect.getWidth(),
119  (u32)destrect.getHeight()));
120  imageScaleNNAA(srcimg, srcrect, destimg);
121 
122  // Some platforms are picky about textures being powers of 2, so expand
123  // the image dimensions to the next power of 2, if necessary.
124  if (!driver->queryFeature(video::EVDF_TEXTURE_NPOT)) {
125  video::IImage *po2img = driver->createImage(src->getColorFormat(),
126  core::dimension2d<u32>(npot2((u32)destrect.getWidth()),
127  npot2((u32)destrect.getHeight())));
128  po2img->fill(video::SColor(0, 0, 0, 0));
129  destimg->copyTo(po2img);
130  destimg->drop();
131  destimg = po2img;
132  }
133 
134  // Convert the scaled image back into a texture.
135  scaled = driver->addTexture(scalename, destimg);
136  destimg->drop();
137  g_txrCache[scalename] = scaled;
138 
139  return scaled;
140 }

155 /* Replacement for driver->draw2DImage() that uses the high-quality pre-scaled
156  * texture, if configured.
157  */
158 void draw2DImageFilterScaled(video::IVideoDriver *driver, video::ITexture *txr,
159  const core::rect<s32> &destrect, const core::rect<s32> &srcrect,
160  const core::rect<s32> *cliprect, const video::SColor *const colors,
161  bool usealpha)
162 {
163  // Attempt to pre-scale image in software in high quality.
164  video::ITexture *scaled = guiScalingResizeCached(driver, txr, srcrect, destrect);
165  if (scaled == NULL)
166  return;
167 
168  // Correct source rect based on scaled image.
169  const core::rect<s32> mysrcrect = (scaled != txr)
170  ? core::rect<s32>(0, 0, destrect.getWidth(), destrect.getHeight())
171  : srcrect;
172 
173  driver->draw2DImage(scaled, destrect, mysrcrect, cliprect, colors, usealpha);
174 }

176 void draw2DImage9Slice(video::IVideoDriver *driver, video::ITexture *texture,
177  const core::rect<s32> &destrect, const core::rect<s32> &srcrect,
178  const core::rect<s32> &middlerect, const core::rect<s32> *cliprect,
179  const video::SColor *const colors)
180 {
181  // `-x` is interpreted as `w - x`
182  core::rect<s32> middle = middlerect;
183 
184  if (middlerect.LowerRightCorner.X < 0)
185  middle.LowerRightCorner.X += srcrect.getWidth();
186  if (middlerect.LowerRightCorner.Y < 0)
187  middle.LowerRightCorner.Y += srcrect.getHeight();
188 
189  core::vector2di lower_right_offset = core::vector2di(srcrect.getWidth(),
190  srcrect.getHeight()) - middle.LowerRightCorner;
191 
192  for (int y = 0; y < 3; ++y) {
193  for (int x = 0; x < 3; ++x) {
194  core::rect<s32> src = srcrect;
195  core::rect<s32> dest = destrect;
196 
197  switch (x) {
198  case 0:
199  dest.LowerRightCorner.X = destrect.UpperLeftCorner.X + middle.UpperLeftCorner.X;
200  src.LowerRightCorner.X = srcrect.UpperLeftCorner.X + middle.UpperLeftCorner.X;
201  break;
202 
203  case 1:
204  dest.UpperLeftCorner.X += middle.UpperLeftCorner.X;
205  dest.LowerRightCorner.X -= lower_right_offset.X;
206  src.UpperLeftCorner.X += middle.UpperLeftCorner.X;
207  src.LowerRightCorner.X -= lower_right_offset.X;
208  break;
209 
210  case 2:
211  dest.UpperLeftCorner.X = destrect.LowerRightCorner.X - lower_right_offset.X;
212  src.UpperLeftCorner.X = srcrect.LowerRightCorner.X - lower_right_offset.X;
213  break;
214  }
215 
216  switch (y) {
217  case 0:
218  dest.LowerRightCorner.Y = destrect.UpperLeftCorner.Y + middle.UpperLeftCorner.Y;
219  src.LowerRightCorner.Y = srcrect.UpperLeftCorner.Y + middle.UpperLeftCorner.Y;
220  break;
221 
222  case 1:
223  dest.UpperLeftCorner.Y += middle.UpperLeftCorner.Y;
224  dest.LowerRightCorner.Y -= lower_right_offset.Y;
225  src.UpperLeftCorner.Y += middle.UpperLeftCorner.Y;
226  src.LowerRightCorner.Y -= lower_right_offset.Y;
227  break;
228 
229  case 2:
230  dest.UpperLeftCorner.Y = destrect.LowerRightCorner.Y - lower_right_offset.Y;
231  src.UpperLeftCorner.Y = srcrect.LowerRightCorner.Y - lower_right_offset.Y;
232  break;
233  }
234 
235  draw2DImageFilterScaled(driver, texture, dest, src, cliprect, colors, true);
236  }
237  }
