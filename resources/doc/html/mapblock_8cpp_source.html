88 {
89  // Ignore if no stored objects (to not set changed flag)
90  if (m_static_objects.getAllStored().empty())
91  return false;
92 
93  verbosestream << "MapBlock::onObjectsActivation(): "
94  << "activating objects of block " << PP(getPos()) << " ("
95  << m_static_objects.getStoredSize() << " objects)" << std::endl;
96 
97  if (m_static_objects.getStoredSize() > g_settings->getU16("max_objects_per_block")) {
98  errorstream << "suspiciously large amount of objects detected: "
99  << m_static_objects.getStoredSize() << " in "
100  << PP(getPos()) << "; removing all of them." << std::endl;
101  // Clear stored list
102  m_static_objects.clearStored();
103  raiseModified(MOD_STATE_WRITE_NEEDED, MOD_REASON_TOO_MANY_OBJECTS);
104  return false;
105  }
106 
107  return true;

129 {
130  // Run script callbacks for elapsed node_timers
131  std::vector<NodeTimer> elapsed_timers = m_node_timers.step(dtime);
132  if (!elapsed_timers.empty()) {
133  MapNode n;
134  v3s16 p;
135  for (const NodeTimer &elapsed_timer : elapsed_timers) {
136  n = getNodeNoEx(elapsed_timer.position);
137  p = elapsed_timer.position + getPosRelative();
138  if (on_timer_cb(p, n, elapsed_timer.elapsed))
139  setNodeTimer(NodeTimer(elapsed_timer.timeout, 0, elapsed_timer.position));
140  }
141  }

163 std::string MapBlock::getModifiedReasonString()
164 {
165  std::string reason;
166 
167  const u32 ubound = MYMIN(sizeof(m_modified_reason) * CHAR_BIT,
168  ARRLEN(modified_reason_strings));
169 
170  for (u32 i = 0; i != ubound; i++) {
171  if ((m_modified_reason & (1 << i)) == 0)
172  continue;
173 
174  reason += modified_reason_strings[i];
175  reason += ", ";
176  }
177 
178  if (reason.length() > 2)
179  reason.resize(reason.length() - 2);
180 
181  return reason;
182 }

205 void MapBlock::actuallyUpdateDayNightDiff()
206 {
207  const NodeDefManager *nodemgr = m_gamedef->ndef();
208 
209  // Running this function un-expires m_day_night_differs
210  m_day_night_differs_expired = false;
211 
212  bool differs = false;
213 
214  /*
215  Check if any lighting value differs
216  */
217 
218  MapNode previous_n(CONTENT_IGNORE);
219  for (u32 i = 0; i < nodecount; i++) {
220  MapNode n = data[i];
221 
222  // If node is identical to previous node, don't verify if it differs
223  if (n == previous_n)
224  continue;
225 
226  differs = !n.isLightDayNightEq(nodemgr->getLightingFlags(n));
227  if (differs)
228  break;
229  previous_n = n;
230  }
231 
232  /*
233  If some lighting values differ, check if the whole thing is
234  just air. If it is just air, differs = false
235  */
236  if (differs) {
237  bool only_air = true;
238  for (u32 i = 0; i < nodecount; i++) {
239  MapNode &n = data[i];
240  if (n.getContent() != CONTENT_AIR) {
241  only_air = false;
242  break;
243  }
244  }
245  if (only_air)
246  differs = false;
247  }
248 
249  // Set member variable
250  m_day_night_differs = differs;
251 }

262 // List relevant id-name pairs for ids in the block using nodedef
263 // Renumbers the content IDs (starting at 0 and incrementing)
264 static void getBlockNodeIdMapping(NameIdMapping *nimap, MapNode *nodes,
265  const NodeDefManager *nodedef)
266 {
267  // The static memory requires about 65535 * sizeof(int) RAM in order to be
268  // sure we can handle all content ids. But it's absolutely worth it as it's
269  // a speedup of 4 for one of the major time consuming functions on storing
270  // mapblocks.
271  thread_local std::unique_ptr<content_t[]> mapping;
272  static_assert(sizeof(content_t) == 2, "content_t must be 16-bit");
273  if (!mapping)
274  mapping = std::make_unique<content_t[]>(USHRT_MAX + 1);
275 
276  memset(mapping.get(), 0xFF, (USHRT_MAX + 1) * sizeof(content_t));
277 
278  std::unordered_set<content_t> unknown_contents;
279  content_t id_counter = 0;
280  for (u32 i = 0; i < MapBlock::nodecount; i++) {
281  content_t global_id = nodes[i].getContent();
282  content_t id = CONTENT_IGNORE;
283 
284  // Try to find an existing mapping
285  if (mapping[global_id] != 0xFFFF) {
286  id = mapping[global_id];
287  } else {
288  // We have to assign a new mapping
289  id = id_counter++;
290  mapping[global_id] = id;
291 
292  const ContentFeatures &f = nodedef->get(global_id);
293  const std::string &name = f.name;
294  if (name.empty())
295  unknown_contents.insert(global_id);
296  else
297  nimap->set(id, name);
298  }
299 
300  // Update the MapNode
301  nodes[i].setContent(id);
302  }
303  for (u16 unknown_content : unknown_contents) {
304  errorstream << "getBlockNodeIdMapping(): IGNORING ERROR: "
305  << "Name for node id " << unknown_content << " not known" << std::endl;
306  }
307 }

309 // Correct ids in the block to match nodedef based on names.
310 // Unknown ones are added to nodedef.
311 // Will not update itself to match id-name pairs in nodedef.
312 static void correctBlockNodeIds(const NameIdMapping *nimap, MapNode *nodes,
313  IGameDef *gamedef)
314 {
315  const NodeDefManager *nodedef = gamedef->ndef();
316  // This means the block contains incorrect ids, and we contain
317  // the information to convert those to names.
318  // nodedef contains information to convert our names to globally
319  // correct ids.
320  std::unordered_set<content_t> unnamed_contents;
321  std::unordered_set<std::string> unallocatable_contents;
322 
323  bool previous_exists = false;
324  content_t previous_local_id = CONTENT_IGNORE;
325  content_t previous_global_id = CONTENT_IGNORE;
326 
327  for (u32 i = 0; i < MapBlock::nodecount; i++) {
328  content_t local_id = nodes[i].getContent();
329  // If previous node local_id was found and same than before, don't lookup maps
330  // apply directly previous resolved id
331  // This permits to massively improve loading performance when nodes are similar
332  // example: default:air, default:stone are massively present
333  if (previous_exists && local_id == previous_local_id) {
334  nodes[i].setContent(previous_global_id);
335  continue;
336  }
337 
338  std::string name;
339  if (!nimap->getName(local_id, name)) {
340  unnamed_contents.insert(local_id);
341  previous_exists = false;
342  continue;
343  }
344 
345  content_t global_id;
346  if (!nodedef->getId(name, global_id)) {
347  global_id = gamedef->allocateUnknownNodeId(name);
348  if (global_id == CONTENT_IGNORE) {
349  unallocatable_contents.insert(name);
350  previous_exists = false;
351  continue;
352  }
353  }
354  nodes[i].setContent(global_id);
355 
356  // Save previous node local_id & global_id result
357  previous_local_id = local_id;
358  previous_global_id = global_id;
359  previous_exists = true;
360  }
361 
362  for (const content_t c: unnamed_contents) {
363  errorstream << "correctBlockNodeIds(): IGNORING ERROR: "
364  << "Block contains id " << c
365  << " with no name mapping" << std::endl;
366  }
367  for (const std::string &node_name: unallocatable_contents) {
368  errorstream << "correctBlockNodeIds(): IGNORING ERROR: "
369  << "Could not allocate global id for node name \""
370  << node_name << "\"" << std::endl;
371  }
372 }

374 void MapBlock::serialize(std::ostream &os_compressed, u8 version, bool disk, int compression_level)
375 {
376  if(!ser_ver_supported(version))
377  throw VersionMismatchException("ERROR: MapBlock format not supported");
378 
379  FATAL_ERROR_IF(version < SER_FMT_VER_LOWEST_WRITE, "Serialization version error");
380 
381  std::ostringstream os_raw(std::ios_base::binary);
382  std::ostream &os = version >= 29 ? os_raw : os_compressed;
383 
384  // First byte
385  u8 flags = 0;
386  if(is_underground)
387  flags |= 0x01;
388  if(getDayNightDiff())
389  flags |= 0x02;
390  if (!m_generated)
391  flags |= 0x08;
392  writeU8(os, flags);
393  if (version >= 27) {
394  writeU16(os, m_lighting_complete);
395  }
396 
397  /*
398  Bulk node data
399  */
400  NameIdMapping nimap;
401  SharedBuffer<u8> buf;
402  const u8 content_width = 2;
403  const u8 params_width = 2;
404  if(disk)
405  {
406  MapNode *tmp_nodes = new MapNode[nodecount];
407  memcpy(tmp_nodes, data, nodecount * sizeof(MapNode));
408  getBlockNodeIdMapping(&nimap, tmp_nodes, m_gamedef->ndef());
409 
410  buf = MapNode::serializeBulk(version, tmp_nodes, nodecount,
411  content_width, params_width);
412  delete[] tmp_nodes;
413 
414  // write timestamp and node/id mapping first
415  if (version >= 29) {
416  writeU32(os, getTimestamp());
417 
418  nimap.serialize(os);
419  }
420  }
421  else
422  {
423  buf = MapNode::serializeBulk(version, data, nodecount,
424  content_width, params_width);
425  }
426 
427  writeU8(os, content_width);
428  writeU8(os, params_width);
429  if (version >= 29) {
430  os.write(reinterpret_cast<char*>(*buf), buf.getSize());
431  } else {
432  // prior to 29 node data was compressed individually
433  compress(buf, os, version, compression_level);
434  }
435 
436  /*
437  Node metadata
438  */
439  if (version >= 29) {
440  m_node_metadata.serialize(os, version, disk);
441  } else {
442  // use os_raw from above to avoid allocating another stream object
443  m_node_metadata.serialize(os_raw, version, disk);
444  // prior to 29 node data was compressed individually
445  compress(os_raw.str(), os, version, compression_level);
446  }
447 
448  /*
449  Data that goes to disk, but not the network
450  */
451  if (disk) {
452  if (version <= 24) {
453  // Node timers
454  m_node_timers.serialize(os, version);
455  }
456 
457  // Static objects
458  m_static_objects.serialize(os);
459 
460  if (version < 29) {
461  // Timestamp
462  writeU32(os, getTimestamp());
463 
464  // Write block-specific node definition id mapping
465  nimap.serialize(os);
466  }
467 
468  if (version >= 25) {
469  // Node timers
470  m_node_timers.serialize(os, version);
471  }
472  }
473 
474  if (version >= 29) {
475  // now compress the whole thing
476  compress(os_raw.str(), os_compressed, version, compression_level);
477  }
478 }

485 void MapBlock::deSerialize(std::istream &in_compressed, u8 version, bool disk)
486 {
487  if(!ser_ver_supported(version))
488  throw VersionMismatchException("ERROR: MapBlock format not supported");
489 
490  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())<<std::endl);
491 
492  m_day_night_differs_expired = false;
493 
494  if(version <= 21)
495  {
496  deSerialize_pre22(in_compressed, version, disk);
497  return;
498  }
499 
500  // Decompress the whole block (version >= 29)
501  std::stringstream in_raw(std::ios_base::binary | std::ios_base::in | std::ios_base::out);
502  if (version >= 29)
503  decompress(in_compressed, in_raw, version);
504  std::istream &is = version >= 29 ? in_raw : in_compressed;
505 
506  u8 flags = readU8(is);
507  is_underground = (flags & 0x01) != 0;
508  m_day_night_differs = (flags & 0x02) != 0;
509  if (version < 27)
510  m_lighting_complete = 0xFFFF;
511  else
512  m_lighting_complete = readU16(is);
513  m_generated = (flags & 0x08) == 0;
514 
515  NameIdMapping nimap;
516  if (disk && version >= 29) {
517  // Timestamp
518  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
519  <<": Timestamp"<<std::endl);
520  setTimestampNoChangedFlag(readU32(is));
521  m_disk_timestamp = m_timestamp;
522 
523  // Node/id mapping
524  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
525  <<": NameIdMapping"<<std::endl);
526  nimap.deSerialize(is);
527  }
528 
529  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
530  <<": Bulk node data"<<std::endl);
531  u8 content_width = readU8(is);
532  u8 params_width = readU8(is);
533  if(content_width != 1 && content_width != 2)
534  throw SerializationError("MapBlock::deSerialize(): invalid content_width");
535  if(params_width != 2)
536  throw SerializationError("MapBlock::deSerialize(): invalid params_width");
537 
538  /*
539  Bulk node data
540  */
541  if (version >= 29) {
542  MapNode::deSerializeBulk(is, version, data, nodecount,
543  content_width, params_width);
544  } else {
545  // use in_raw from above to avoid allocating another stream object
546  decompress(is, in_raw, version);
547  MapNode::deSerializeBulk(in_raw, version, data, nodecount,
548  content_width, params_width);
549  }
550 
551  /*
552  NodeMetadata
553  */
554  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
555  <<": Node metadata"<<std::endl);
556  if (version >= 29) {
557  m_node_metadata.deSerialize(is, m_gamedef->idef());
558  } else {
559  try {
560  // reuse in_raw
561  in_raw.str("");
562  in_raw.clear();
563  decompress(is, in_raw, version);
564  if (version >= 23)
565  m_node_metadata.deSerialize(in_raw, m_gamedef->idef());
566  else
567  content_nodemeta_deserialize_legacy(in_raw,
568  &m_node_metadata, &m_node_timers,
569  m_gamedef->idef());
570  } catch(SerializationError &e) {
571  warningstream<<"MapBlock::deSerialize(): Ignoring an error"
572  <<" while deserializing node metadata at ("
573  <<PP(getPos())<<": "<<e.what()<<std::endl;
574  }
575  }
576 
577  /*
578  Data that is only on disk
579  */
580  if (disk) {
581  // Node timers
582  if (version == 23) {
583  // Read unused zero
584  readU8(is);
585  }
586  if (version == 24) {
587  TRACESTREAM(<< "MapBlock::deSerialize " << PP(getPos())
588  << ": Node timers (ver==24)" << std::endl);
589  m_node_timers.deSerialize(is, version);
590  }
591 
592  // Static objects
593  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
594  <<": Static objects"<<std::endl);
595  m_static_objects.deSerialize(is);
596 
597  if (version < 29) {
598  // Timestamp
599  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
600  <<": Timestamp"<<std::endl);
601  setTimestampNoChangedFlag(readU32(is));
602  m_disk_timestamp = m_timestamp;
603 
604  // Node/id mapping
605  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
606  <<": NameIdMapping"<<std::endl);
607  nimap.deSerialize(is);
608  }
609 
610  // Dynamically re-set ids based on node names
611  correctBlockNodeIds(&nimap, data, m_gamedef);
612 
613  if(version >= 25){
614  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
615  <<": Node timers (ver>=25)"<<std::endl);
616  m_node_timers.deSerialize(is, version);
617  }
618  }
619 
620  TRACESTREAM(<<"MapBlock::deSerialize "<<PP(getPos())
621  <<": Done."<<std::endl);
622 }

662 void MapBlock::deSerialize_pre22(std::istream &is, u8 version, bool disk)
663 {
664  // Initialize default flags
665  is_underground = false;
666  m_day_night_differs = false;
667  m_lighting_complete = 0xFFFF;
668  m_generated = true;
669 
670  // Make a temporary buffer
671  u32 ser_length = MapNode::serializedLength(version);
672  SharedBuffer<u8> databuf_nodelist(nodecount * ser_length);
673 
674  // These have no compression
675  if (version <= 3 || version == 5 || version == 6) {
676  char tmp;
677  is.read(&tmp, 1);
678  if (is.gcount() != 1)
679  throw SerializationError(std::string(FUNCTION_NAME)
680  + ": not enough input data");
681  is_underground = tmp;
682  is.read((char *)*databuf_nodelist, nodecount * ser_length);
683  if ((u32)is.gcount() != nodecount * ser_length)
684  throw SerializationError(std::string(FUNCTION_NAME)
685  + ": not enough input data");
686  } else if (version <= 10) {
687  u8 t8;
688  is.read((char *)&t8, 1);
689  is_underground = t8;
690 
691  {
692  // Uncompress and set material data
693  std::ostringstream os(std::ios_base::binary);
694  decompress(is, os, version);
695  std::string s = os.str();
696  if (s.size() != nodecount)
697  throw SerializationError(std::string(FUNCTION_NAME)
