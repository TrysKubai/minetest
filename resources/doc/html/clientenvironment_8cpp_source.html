142 void ClientEnvironment::step(float dtime)
143 {
144  /* Step time of day */
145  stepTimeOfDay(dtime);
146 
147  // Get some settings
148  bool fly_allowed = m_client->checkLocalPrivilege("fly");
149  bool free_move = fly_allowed && g_settings->getBool("free_move");
150 
151  // Get local player
152  LocalPlayer *lplayer = getLocalPlayer();
153  assert(lplayer);
154  // collision info queue
155  std::vector<CollisionInfo> player_collisions;
156 
157  /*
158  Get the speed the player is going
159  */
160  bool is_climbing = lplayer->is_climbing;
161 
162  f32 player_speed = lplayer->getSpeed().getLength();
163 
164  /*
165  Maximum position increment
166  */
167  //f32 position_max_increment = 0.05*BS;
168  f32 position_max_increment = 0.1*BS;
169 
170  // Maximum time increment (for collision detection etc)
171  // time = distance / speed
172  f32 dtime_max_increment = 1;
173  if(player_speed > 0.001)
174  dtime_max_increment = position_max_increment / player_speed;
175 
176  // Maximum time increment is 10ms or lower
177  if(dtime_max_increment > 0.01)
178  dtime_max_increment = 0.01;
179 
180  // Don't allow overly huge dtime
181  if(dtime > 0.5)
182  dtime = 0.5;
183 
184  /*
185  Stuff that has a maximum time increment
186  */
187 
188  u32 steps = ceil(dtime / dtime_max_increment);
189  f32 dtime_part = dtime / steps;
190  for (; steps > 0; --steps) {
191  /*
192  Local player handling
193  */
194 
195  // Control local player
196  lplayer->applyControl(dtime_part, this);
197 
198  // Apply physics
199  lplayer->gravity = 0;
200  if (!free_move) {
201  // Gravity
202  if (!is_climbing && !lplayer->in_liquid)
203  // HACK the factor 2 for gravity is arbitrary and should be removed eventually
204  lplayer->gravity = 2 * lplayer->movement_gravity * lplayer->physics_override.gravity;
205 
206  // Liquid floating / sinking
207  if (!is_climbing && lplayer->in_liquid &&
208  !lplayer->swimming_vertical &&
209  !lplayer->swimming_pitch)
210  // HACK the factor 2 for gravity is arbitrary and should be removed eventually
211  lplayer->gravity = 2 * lplayer->movement_liquid_sink;
212 
213  // Movement resistance
214  if (lplayer->move_resistance > 0) {
215  v3f speed = lplayer->getSpeed();
216 
217  // How much the node's move_resistance blocks movement, ranges
218  // between 0 and 1. Should match the scale at which liquid_viscosity
219  // increase affects other liquid attributes.
220  static const f32 resistance_factor = 0.3f;
221 
222  v3f d_wanted;
223  bool in_liquid_stable = lplayer->in_liquid_stable || lplayer->in_liquid;
224  if (in_liquid_stable) {
225  d_wanted = -speed / lplayer->movement_liquid_fluidity;
226  } else {
227  d_wanted = -speed / BS;
228  }
229  f32 dl = d_wanted.getLength();
230  if (in_liquid_stable) {
231  if (dl > lplayer->movement_liquid_fluidity_smooth)
232  dl = lplayer->movement_liquid_fluidity_smooth;
233  }
234 
235  dl *= (lplayer->move_resistance * resistance_factor) +
236  (1 - resistance_factor);
237  v3f d = d_wanted.normalize() * (dl * dtime_part * 100.0f);
238  speed += d;
239 
240  lplayer->setSpeed(speed);
241  }
242  }
243 
244  /*
245  Move the lplayer.
246  This also does collision detection.
247  */
248 
249  lplayer->move(dtime_part, this, position_max_increment,
250  &player_collisions);
251  }
252 
253  bool player_immortal = false;
254  f32 player_fall_factor = 1.0f;
255  GenericCAO *playercao = lplayer->getCAO();
256  if (playercao) {
257  player_immortal = playercao->isImmortal();
258  int addp_p = itemgroup_get(playercao->getGroups(),
259  "fall_damage_add_percent");
260  // convert armor group into an usable fall damage factor
261  player_fall_factor = 1.0f + (float)addp_p / 100.0f;
262  }
263 
264  for (const CollisionInfo &info : player_collisions) {
265  v3f speed_diff = info.new_speed - info.old_speed;;
266  // Handle only fall damage
267  // (because otherwise walking against something in fast_move kills you)
268  if (speed_diff.Y < 0 || info.old_speed.Y >= 0)
269  continue;
270  // Get rid of other components
271  speed_diff.X = 0;
272  speed_diff.Z = 0;
273  f32 pre_factor = 1; // 1 hp per node/s
274  f32 tolerance = BS*14; // 5 without damage
275  if (info.type == COLLISION_NODE) {
276  const ContentFeatures &f = m_client->ndef()->
277  get(m_map->getNode(info.node_p));
278  // Determine fall damage modifier
279  int addp_n = itemgroup_get(f.groups, "fall_damage_add_percent");
280  // convert node group to an usable fall damage factor
281  f32 node_fall_factor = 1.0f + (float)addp_n / 100.0f;
282  // combine both player fall damage modifiers
283  pre_factor = node_fall_factor * player_fall_factor;
284  }
285  float speed = pre_factor * speed_diff.getLength();
286 
287  if (speed > tolerance && !player_immortal && pre_factor > 0.0f) {
288  f32 damage_f = (speed - tolerance) / BS;
289  u16 damage = (u16)MYMIN(damage_f + 0.5, U16_MAX);
290  if (damage != 0) {
291  damageLocalPlayer(damage, true);
292  m_client->getEventManager()->put(
293  new SimpleTriggerEvent(MtEvent::PLAYER_FALLING_DAMAGE));
294  }
295  }
296  }
297 
298  if (m_client->modsLoaded())
299  m_script->environment_step(dtime);
300 
301  // Update lighting on local player (used for wield item)
302  u32 day_night_ratio = getDayNightRatio();
303  {
304  // Get node at head
305 
306  // On InvalidPositionException, use this as default
307  // (day: LIGHT_SUN, night: 0)
308  MapNode node_at_lplayer(CONTENT_AIR, 0x0f, 0);
309 
310  v3s16 p = lplayer->getLightPosition();
311  node_at_lplayer = m_map->getNode(p);
312 
313  u16 light = getInteriorLight(node_at_lplayer, 0, m_client->ndef());
314  lplayer->light_color = encode_light(light, 0); // this transfers light.alpha
315  final_color_blend(&lplayer->light_color, light, day_night_ratio);
316  }
317 
318  /*
319  Step active objects and update lighting of them
320  */
321 
322  bool update_lighting = m_active_object_light_update_interval.step(dtime, 0.21);
323  auto cb_state = [this, dtime, update_lighting, day_night_ratio] (ClientActiveObject *cao) {
324  // Step object
325  cao->step(dtime, this);
326 
327  if (update_lighting)
328  cao->updateLight(day_night_ratio);
329  };
330 
331  m_ao_manager.step(dtime, cb_state);
332 
333  /*
334  Step and handle simple objects
335  */
336  g_profiler->avg("ClientEnv: CSO count [#]", m_simple_objects.size());
337  for (auto i = m_simple_objects.begin(); i != m_simple_objects.end();) {
338  ClientSimpleObject *simple = *i;
339 
340  simple->step(dtime);
341  if(simple->m_to_be_removed) {
342  delete simple;
343  i = m_simple_objects.erase(i);
344  }
345  else {
346  ++i;
347  }
348  }
349 }

379  const std::string &init_data)
380 {
381  ClientActiveObject* obj =
382  ClientActiveObject::create((ActiveObjectType) type, m_client, this);
383  if(obj == NULL)
384  {
385  infostream<<"ClientEnvironment::addActiveObject(): "
386  <<"id="<<id<<" type="<<type<<": Couldn't create object"

388  return;
389  }
390 
391  obj->setId(id);
392 
393  try
394  {
395  obj->initialize(init_data);
396  }
397  catch(SerializationError &e)
398  {
399  errorstream<<"ClientEnvironment::addActiveObject():"
400  <<" id="<<id<<" type="<<type

402  <<e.what()
403  <<": init_data="<<serializeJsonString(init_data)
404  <<std::endl;
405  }
406 
407  u16 new_id = addActiveObject(obj);
408  // Object initialized:
409  if ((obj = getActiveObject(new_id))) {
410  // Final step is to update all children which are already known
411  // Data provided by AO_CMD_SPAWN_INFANT
412  const auto &children = obj->getAttachmentChildIds();

494 void ClientEnvironment::getSelectedActiveObjects(
495  const core::line3d<f32> &shootline_on_map,
496  std::vector<PointedThing> &objects)
497 {
498  std::vector<DistanceSortedActiveObject> allObjects;
499  m_ao_manager.getActiveSelectableObjects(shootline_on_map, allObjects);
500  const v3f line_vector = shootline_on_map.getVector();
501 
502  for (const auto &allObject : allObjects) {
503  ClientActiveObject *obj = allObject.obj;
504  aabb3f selection_box;
505  if (!obj->getSelectionBox(&selection_box))
506  continue;
507 
508  v3f current_intersection;
509  v3f current_normal, current_raw_normal;
510  const v3f rel_pos = shootline_on_map.start - obj->getPosition();
511  bool collision;
512  GenericCAO* gcao = dynamic_cast<GenericCAO*>(obj);
513  if (gcao != nullptr && gcao->getProperties().rotate_selectionbox) {
514  gcao->getSceneNode()->updateAbsolutePosition();
515  const v3f deg = obj->getSceneNode()->getAbsoluteTransformation().getRotationDegrees();
516  collision = boxLineCollision(selection_box, deg,
517  rel_pos, line_vector, &current_intersection, &current_normal, &current_raw_normal);
518  } else {
519  collision = boxLineCollision(selection_box, rel_pos, line_vector,
520  &current_intersection, &current_normal);
521  current_raw_normal = current_normal;
522  }
523  if (collision) {
524  current_intersection += obj->getPosition();
525  objects.emplace_back(obj->getId(), current_intersection, current_normal, current_raw_normal,
526  (current_intersection - shootline_on_map.start).getLengthSQ());
527  }
528  }
529 }
