121 static void load_button_texture(button_info *btn, const char *path,

191 void AutoHideButtonBar::addButton(touch_gui_button_id button_id,
192  const wchar_t *caption, const char *btn_image)
193 {
194 
195  if (!m_initialized) {
196  errorstream << "AutoHideButtonBar::addButton not yet initialized!"
197  << std::endl;
198  return;
199  }
200  int button_size = 0;
201 
202  if ((m_dir == AHBB_Dir_Top_Bottom) || (m_dir == AHBB_Dir_Bottom_Top))
203  button_size = m_lower_right.X - m_upper_left.X;
204  else
205  button_size = m_lower_right.Y - m_upper_left.Y;
206 
207  irr::core::rect<int> current_button;
208 
209  if ((m_dir == AHBB_Dir_Right_Left) || (m_dir == AHBB_Dir_Left_Right)) {
210  int x_start = 0;
211  int x_end = 0;
212 
213  if (m_dir == AHBB_Dir_Left_Right) {
214  x_start = m_lower_right.X + (button_size * 1.25 * m_buttons.size())
215  + (button_size * 0.25);
216  x_end = x_start + button_size;
217  } else {
218  x_end = m_upper_left.X - (button_size * 1.25 * m_buttons.size())
219  - (button_size * 0.25);
220  x_start = x_end - button_size;
221  }
222 
223  current_button = rect<s32>(x_start, m_upper_left.Y, x_end,
224  m_lower_right.Y);
225  } else {
226  double y_start = 0;
227  double y_end = 0;
228 
229  if (m_dir == AHBB_Dir_Top_Bottom) {
230  y_start = m_lower_right.X + (button_size * 1.25 * m_buttons.size())
231  + (button_size * 0.25);
232  y_end = y_start + button_size;
233  } else {
234  y_end = m_upper_left.X - (button_size * 1.25 * m_buttons.size())
235  - (button_size * 0.25);
236  y_start = y_end - button_size;
237  }
238 
239  current_button = rect<s32>(m_upper_left.X, y_start,
240  m_lower_right.Y, y_end);
241  }
242 
243  auto *btn = new button_info();
244  btn->guibutton = m_guienv->addButton(current_button,
245  nullptr, button_id, caption, nullptr);
246  btn->guibutton->grab();
247  btn->guibutton->setVisible(false);
248  btn->guibutton->setEnabled(false);
249  btn->repeatcounter = -1;
250  btn->keycode = id2keycode(button_id);
251  btn->immediate_release = true;
252  btn->ids.clear();
253 
254  load_button_texture(btn, btn_image, current_button, m_texturesource,
255  m_driver);
256 
257  m_buttons.push_back(btn);
258 }

271 bool AutoHideButtonBar::isButton(const SEvent &event)
272 {
273  IGUIElement *rootguielement = m_guienv->getRootGUIElement();
274 
275  if (rootguielement == nullptr)
276  return false;
277 
278  gui::IGUIElement *element = rootguielement->getElementFromPoint(
279  core::position2d<s32>(event.TouchInput.X, event.TouchInput.Y));
280 
281  if (element == nullptr)
282  return false;
283 
284  if (m_active) {
285  // check for all buttons in vector
286  auto iter = m_buttons.begin();
287 
288  while (iter != m_buttons.end()) {
289  if ((*iter)->guibutton == element) {
290 
291  auto *translated = new SEvent();
292  memset(translated, 0, sizeof(SEvent));
293  translated->EventType = irr::EET_KEY_INPUT_EVENT;
294  translated->KeyInput.Key = (*iter)->keycode;
295  translated->KeyInput.Control = false;
296  translated->KeyInput.Shift = false;
297  translated->KeyInput.Char = 0;
298 
299  // add this event
300  translated->KeyInput.PressedDown = true;
301  m_receiver->OnEvent(*translated);
302 
303  // remove this event
304  translated->KeyInput.PressedDown = false;
305  m_receiver->OnEvent(*translated);
306 
307  delete translated;
308 
309  (*iter)->ids.push_back(event.TouchInput.ID);
310 
311  m_timeout = 0;
312 
313  if ((*iter)->togglable == 1) {
314  (*iter)->togglable = 2;
315  load_button_texture(*iter, (*iter)->textures[1],
316  (*iter)->guibutton->getRelativePosition(),
317  m_texturesource, m_driver);
318  } else if ((*iter)->togglable == 2) {
319  (*iter)->togglable = 1;
320  load_button_texture(*iter, (*iter)->textures[0],
321  (*iter)->guibutton->getRelativePosition(),
322  m_texturesource, m_driver);
323  }
324 
325  return true;
326  }
327  ++iter;
328  }
329  } else {
330  // check for starter button only
331  if (element == m_starter.guibutton) {
332  m_starter.ids.push_back(event.TouchInput.ID);
333  m_starter.guibutton->setVisible(false);
334  m_starter.guibutton->setEnabled(false);
335  m_active = true;
336  m_timeout = 0;
337 
338  auto iter = m_buttons.begin();
339 
340  while (iter != m_buttons.end()) {
341  (*iter)->guibutton->setVisible(true);
342  (*iter)->guibutton->setEnabled(true);
343  ++iter;
344  }
345 
346  return true;
347  }
348  }
349  return false;
350 }

623 void TouchScreenGUI::handleButtonEvent(touch_gui_button_id button,
624  size_t eventID, bool action)
625 {
626  button_info *btn = &m_buttons[button];
627  auto *translated = new SEvent();
628  memset(translated, 0, sizeof(SEvent));
629  translated->EventType = irr::EET_KEY_INPUT_EVENT;
630  translated->KeyInput.Key = btn->keycode;
631  translated->KeyInput.Control = false;
632  translated->KeyInput.Shift = false;
633  translated->KeyInput.Char = 0;
634 
635  // add this event
636  if (action) {
637  assert(std::find(btn->ids.begin(), btn->ids.end(), eventID) == btn->ids.end());
638 
639  btn->ids.push_back(eventID);
640 
641  if (btn->ids.size() > 1) return;
642 
643  btn->repeatcounter = 0;
644  translated->KeyInput.PressedDown = true;
645  translated->KeyInput.Key = btn->keycode;
646  m_receiver->OnEvent(*translated);
647  }
648 
649  // remove event
650  if ((!action) || (btn->immediate_release)) {
651  auto pos = std::find(btn->ids.begin(), btn->ids.end(), eventID);
652  // has to be in touch list
653  assert(pos != btn->ids.end());

727 void TouchScreenGUI::translateEvent(const SEvent &event)
728 {
729  if (!m_initialized)
730  return;
731  if (!m_visible) {
732  infostream
733  << "TouchScreenGUI::translateEvent got event but not visible!"
734  << std::endl;
735  return;
736  }
737 
738  if (event.EventType != EET_TOUCH_INPUT_EVENT)
739  return;
740 
741  if (event.TouchInput.Event == ETIE_PRESSED_DOWN) {
742  /*
743  * Add to own copy of event list...
744  * android would provide this information but Irrlicht guys don't
745  * wanna design an efficient interface
746  */
747  id_status toadd{};
748  toadd.id = event.TouchInput.ID;
749  toadd.X = event.TouchInput.X;
750  toadd.Y = event.TouchInput.Y;
751  m_known_ids.push_back(toadd);
752 
753  size_t eventID = event.TouchInput.ID;
754 
755  touch_gui_button_id button =
756  getButtonID(event.TouchInput.X, event.TouchInput.Y);
757 
758  // handle button events
759  if (button != after_last_element_id) {
760  handleButtonEvent(button, eventID, true);
761  m_settingsbar.deactivate();
762  m_rarecontrolsbar.deactivate();
763  } else if (isHUDButton(event)) {
764  m_settingsbar.deactivate();
765  m_rarecontrolsbar.deactivate();
766  // already handled in isHUDButton()
767  } else if (m_settingsbar.isButton(event)) {
768  m_rarecontrolsbar.deactivate();
769  // already handled in isSettingsBarButton()
770  } else if (m_rarecontrolsbar.isButton(event)) {
771  m_settingsbar.deactivate();
772  // already handled in isSettingsBarButton()
773  } else {
774  // handle non button events
775  if (m_settingsbar.active() || m_rarecontrolsbar.active()) {
776  m_settingsbar.deactivate();
777  m_rarecontrolsbar.deactivate();
778  return;
779  }
780 
781  s32 dxj = event.TouchInput.X - button_size * 5.0f / 2.0f;
782  s32 dyj = event.TouchInput.Y - (s32)m_screensize.Y + button_size * 5.0f / 2.0f;
783 
784  /* Select joystick when left 1/3 of screen dragged or
785  * when joystick tapped (fixed joystick position)
786  */
787  if ((m_fixed_joystick && dxj * dxj + dyj * dyj <= button_size * button_size * 1.5 * 1.5) ||
788  (!m_fixed_joystick && event.TouchInput.X < m_screensize.X / 3.0f)) {
789  // If we don't already have a starting point for joystick make this the one.
790  if (!m_has_joystick_id) {
791  m_has_joystick_id = true;
792  m_joystick_id = event.TouchInput.ID;
793  m_joystick_has_really_moved = false;
794 
795  m_joystick_btn_off->guibutton->setVisible(false);
796  m_joystick_btn_bg->guibutton->setVisible(true);
797  m_joystick_btn_center->guibutton->setVisible(true);
798 
799  // If it's a fixed joystick, don't move the joystick "button".
800  if (!m_fixed_joystick)
801  m_joystick_btn_bg->guibutton->setRelativePosition(v2s32(
802  event.TouchInput.X - button_size * 3.0f / 2.0f,
803  event.TouchInput.Y - button_size * 3.0f / 2.0f));
804 
805  m_joystick_btn_center->guibutton->setRelativePosition(v2s32(
806  event.TouchInput.X - button_size / 2.0f,
807  event.TouchInput.Y - button_size / 2.0f));
808  }
809  } else {
810  // If we don't already have a moving point make this the moving one.
811  if (!m_has_move_id) {
812  m_has_move_id = true;
813  m_move_id = event.TouchInput.ID;
814  m_move_has_really_moved = false;
815  m_move_downtime = porting::getTimeMs();
816  m_move_downlocation = v2s32(event.TouchInput.X, event.TouchInput.Y);
817  m_move_sent_as_mouse_event = false;
818  if (m_draw_crosshair)
819  m_move_downlocation = v2s32(m_screensize.X / 2, m_screensize.Y / 2);
820  }
821  }
822  }
823 
824  m_pointerpos[event.TouchInput.ID] = v2s32(event.TouchInput.X, event.TouchInput.Y);
825  }
826  else if (event.TouchInput.Event == ETIE_LEFT_UP) {
827  verbosestream
828  << "Up event for pointerid: " << event.TouchInput.ID << std::endl;
829  handleReleaseEvent(event.TouchInput.ID);
830  } else {
831  assert(event.TouchInput.Event == ETIE_MOVED);
832 
833  if (!(m_has_joystick_id && m_fixed_joystick) &&
834  m_pointerpos[event.TouchInput.ID] ==
835  v2s32(event.TouchInput.X, event.TouchInput.Y))
836  return;
837 
838  if (m_has_move_id) {
839  if (event.TouchInput.ID == m_move_id &&
840  (!m_move_sent_as_mouse_event || m_draw_crosshair)) {
841  double distance = sqrt(
842  (m_pointerpos[event.TouchInput.ID].X - event.TouchInput.X) *
843  (m_pointerpos[event.TouchInput.ID].X - event.TouchInput.X) +
844  (m_pointerpos[event.TouchInput.ID].Y - event.TouchInput.Y) *
845  (m_pointerpos[event.TouchInput.ID].Y - event.TouchInput.Y));
846 
847  if ((distance > m_touchscreen_threshold) ||
848  (m_move_has_really_moved)) {
849  m_move_has_really_moved = true;
850  s32 X = event.TouchInput.X;
851  s32 Y = event.TouchInput.Y;
852 
853  // update camera_yaw and camera_pitch
854  s32 dx = X - m_pointerpos[event.TouchInput.ID].X;
855  s32 dy = Y - m_pointerpos[event.TouchInput.ID].Y;
856  m_pointerpos[event.TouchInput.ID] = v2s32(X, Y);
857 
858  // adapt to similar behavior as pc screen
859  const double d = g_settings->getFloat("mouse_sensitivity", 0.001f, 10.0f) * 3.0f;
860 
861  m_camera_yaw_change -= dx * d;
862  m_camera_pitch = MYMIN(MYMAX(m_camera_pitch + (dy * d), -180), 180);
863 
864  // update shootline
865  // no need to update (X, Y) when using crosshair since the shootline is not used
866  m_shootline = m_device
867  ->getSceneManager()
868  ->getSceneCollisionManager()
869  ->getRayFromScreenCoordinates(v2s32(X, Y));
870  }
871  } else if ((event.TouchInput.ID == m_move_id) &&
872  (m_move_sent_as_mouse_event)) {
873  m_shootline = m_device
874  ->getSceneManager()
875  ->getSceneCollisionManager()
876  ->getRayFromScreenCoordinates(
877  v2s32(event.TouchInput.X, event.TouchInput.Y));
878  }
879  }
880 
881  if (m_has_joystick_id && event.TouchInput.ID == m_joystick_id) {
882  s32 X = event.TouchInput.X;
883  s32 Y = event.TouchInput.Y;
884 
885  s32 dx = X - m_pointerpos[event.TouchInput.ID].X;
886  s32 dy = Y - m_pointerpos[event.TouchInput.ID].Y;
887  if (m_fixed_joystick) {
888  dx = X - button_size * 5.0f / 2.0f;
889  dy = Y - (s32)m_screensize.Y + button_size * 5.0f / 2.0f;
890  }
891 
892  double distance_sq = dx * dx + dy * dy;
893 
894  s32 dxj = event.TouchInput.X - button_size * 5.0f / 2.0f;
895  s32 dyj = event.TouchInput.Y - (s32)m_screensize.Y + button_size * 5.0f / 2.0f;
896  bool inside_joystick = (dxj * dxj + dyj * dyj <= button_size * button_size * 1.5 * 1.5);
897 
898  if (m_joystick_has_really_moved || inside_joystick ||
899  (!m_fixed_joystick &&
900  distance_sq > m_touchscreen_threshold * m_touchscreen_threshold)) {
901  m_joystick_has_really_moved = true;
902  double distance = sqrt(distance_sq);
903 
904  // angle in degrees
905  double angle = acos(dx / distance) * 180 / M_PI;
906  if (dy < 0)
907  angle *= -1;
908  // rotate to make comparing easier
909  angle = fmod(angle + 180 + 22.5, 360);
910 
911  // reset state before applying
912  for (bool & joystick_status : m_joystick_status)
913  joystick_status = false;
914 
915  if (distance <= m_touchscreen_threshold) {
916  // do nothing
917  } else if (angle < 45)
918  m_joystick_status[j_left] = true;
919  else if (angle < 90) {
920  m_joystick_status[j_forward] = true;
921  m_joystick_status[j_left] = true;
922  } else if (angle < 135)
923  m_joystick_status[j_forward] = true;
924  else if (angle < 180) {
925  m_joystick_status[j_forward] = true;
926  m_joystick_status[j_right] = true;
927  } else if (angle < 225)
928  m_joystick_status[j_right] = true;
929  else if (angle < 270) {
930  m_joystick_status[j_backward] = true;
931  m_joystick_status[j_right] = true;
932  } else if (angle < 315)
933  m_joystick_status[j_backward] = true;
934  else if (angle <= 360) {
935  m_joystick_status[j_backward] = true;
936  m_joystick_status[j_left] = true;
937  }
938 
939  if (distance > button_size) {
940  m_joystick_status[j_aux1] = true;
941  // move joystick "button"
942  s32 ndx = button_size * dx / distance - button_size / 2.0f;
943  s32 ndy = button_size * dy / distance - button_size / 2.0f;
944  if (m_fixed_joystick) {
945  m_joystick_btn_center->guibutton->setRelativePosition(v2s32(
946  button_size * 5.0f / 2.0f + ndx,
947  m_screensize.Y - button_size * 5.0f / 2.0f + ndy));
948  } else {
949  m_joystick_btn_center->guibutton->setRelativePosition(v2s32(
950  m_pointerpos[event.TouchInput.ID].X + ndx,
951  m_pointerpos[event.TouchInput.ID].Y + ndy));
952  }
953  } else {
954  m_joystick_btn_center->guibutton->setRelativePosition(
955  v2s32(X - button_size / 2, Y - button_size / 2));
956  }
957  }
958  }
959 
960  if (!m_has_move_id && !m_has_joystick_id)
961  handleChangedButton(event);
962  }

965 void TouchScreenGUI::handleChangedButton(const SEvent &event)
966 {
967  for (unsigned int i = 0; i < after_last_element_id; i++) {
968  if (m_buttons[i].ids.empty())
969  continue;
970 
971  for (auto iter = m_buttons[i].ids.begin();
972  iter != m_buttons[i].ids.end(); ++iter) {
973  if (event.TouchInput.ID == *iter) {
974  int current_button_id =
975  getButtonID(event.TouchInput.X, event.TouchInput.Y);
976 
977  if (current_button_id == i)
978  continue;
979 
980  // remove old button
981  handleButtonEvent((touch_gui_button_id) i, *iter, false);
982 
983  if (current_button_id == after_last_element_id)
984  return;
985 
986  handleButtonEvent((touch_gui_button_id) current_button_id, *iter, true);
987  return;
988  }
989  }
990  }
991 
992  int current_button_id = getButtonID(event.TouchInput.X, event.TouchInput.Y);
993 
994  if (current_button_id == after_last_element_id)
995  return;
996 
997  button_info *btn = &m_buttons[current_button_id];
998  if (std::find(btn->ids.begin(), btn->ids.end(), event.TouchInput.ID)
999  == btn->ids.end())
1000  handleButtonEvent((touch_gui_button_id) current_button_id,
1001  event.TouchInput.ID, true);
