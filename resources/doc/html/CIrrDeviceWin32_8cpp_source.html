140  void SJoystickWin32Control::directInputAddJoystick(LPCDIDEVICEINSTANCE lpddi)
141  {
142  //Get the GUID of the joystuck
143  const GUID guid = lpddi->guidInstance;
144 
145  JoystickInfo activeJoystick;
146  activeJoystick.Index=ActiveJoysticks.size();
147  activeJoystick.guid=guid;
148  activeJoystick.Name=lpddi->tszProductName;
149  if (FAILED(DirectInputDevice->CreateDevice(guid, &activeJoystick.lpdijoy, NULL)))
150  {
151  os::Printer::log("Could not create DirectInput device", ELL_WARNING);
152  return;
153  }
154 
155  activeJoystick.devcaps.dwSize=sizeof(activeJoystick.devcaps);
156  if (FAILED(activeJoystick.lpdijoy->GetCapabilities(&activeJoystick.devcaps)))
157  {
158  os::Printer::log("Could not create DirectInput device", ELL_WARNING);
159  return;
160  }
161 
162  if (FAILED(activeJoystick.lpdijoy->SetCooperativeLevel(Device->HWnd, DISCL_BACKGROUND | DISCL_EXCLUSIVE)))
163  {
164  os::Printer::log("Could not set DirectInput device cooperative level", ELL_WARNING);
165  return;
166  }
167 
168  if (FAILED(activeJoystick.lpdijoy->SetDataFormat(&c_dfDIJoystick2)))
169  {
170  os::Printer::log("Could not set DirectInput device data format", ELL_WARNING);
171  return;
172  }
173 
174  if (FAILED(activeJoystick.lpdijoy->Acquire()))
175  {
176  os::Printer::log("Could not set DirectInput cooperative level", ELL_WARNING);
177  return;
178  }
179 
180  DIJOYSTATE2 info;
181  if (FAILED(activeJoystick.lpdijoy->GetDeviceState(sizeof(info),&info)))
182  {
183  os::Printer::log("Could not read DirectInput device state", ELL_WARNING);
184  return;
185  }
186 
187  ZeroMemory(activeJoystick.axisValid,sizeof(activeJoystick.axisValid));
188  activeJoystick.axisValid[0]= (info.lX!=0) ? 1 : 0;
189  activeJoystick.axisValid[1]= (info.lY!=0) ? 1 : 0;
190  activeJoystick.axisValid[2]= (info.lZ!=0) ? 1 : 0;
191  activeJoystick.axisValid[3]= (info.lRx!=0) ? 1 : 0;
192  activeJoystick.axisValid[4]= (info.lRy!=0) ? 1 : 0;
193  activeJoystick.axisValid[5]= (info.lRz!=0) ? 1 : 0;
194 
195  int caxis=0;
196  for (u8 i=0; i<6; i++)
197  {
198  if (activeJoystick.axisValid[i])
199  caxis++;
200  }
201 
202  for (u8 i=0; i<(activeJoystick.devcaps.dwAxes)-caxis; i++)
203  {
204  if (i+caxis < 8)
205  activeJoystick.axisValid[i+caxis]=1;
206  }
207 
208  ActiveJoysticks.push_back(activeJoystick);
209  }

212 void SJoystickWin32Control::pollJoysticks()
213 {
214 #if defined _IRR_COMPILE_WITH_JOYSTICK_EVENTS_
215 #ifdef _IRR_COMPILE_WITH_DIRECTINPUT_JOYSTICK_
216  if(0 == ActiveJoysticks.size())
217  return;
218 
219  u32 joystick;
220  DIJOYSTATE2 info;
221 
222  for(joystick = 0; joystick < ActiveJoysticks.size(); ++joystick)
223  {
224  // needs to be reset for each joystick
225  // request ALL values and POV as continuous if possible
226 
227  const DIDEVCAPS & caps = ActiveJoysticks[joystick].devcaps;
228  // if no POV is available don't ask for POV values
229 
230  if (!FAILED(ActiveJoysticks[joystick].lpdijoy->GetDeviceState(sizeof(info),&info)))
231  {
232  SEvent event;
233 
234  event.EventType = irr::EET_JOYSTICK_INPUT_EVENT;
235  event.JoystickEvent.Joystick = (u8)joystick;
236 
237  event.JoystickEvent.POV = (u16)info.rgdwPOV[0];
238  // set to undefined if no POV value was returned or the value
239  // is out of range
240  if ((caps.dwPOVs==0) || (event.JoystickEvent.POV > 35900))
241  event.JoystickEvent.POV = 65535;
242 
243  for(int axis = 0; axis < SEvent::SJoystickEvent::NUMBER_OF_AXES; ++axis)
244  event.JoystickEvent.Axis[axis] = 0;
245 
246  u16 dxAxis=0;
247  u16 irrAxis=0;
248 
249  while (dxAxis < 6 && irrAxis <caps.dwAxes)
250  {
251  bool axisFound=0;
252  s32 axisValue=0;
253 
254  switch (dxAxis)
255  {
256  case 0:
257  axisValue=info.lX;
258  break;
259  case 1:
260  axisValue=info.lY;
261  break;
262  case 2:
263  axisValue=info.lZ;
264  break;
265  case 3:
266  axisValue=info.lRx;
267  break;
268  case 4:
269  axisValue=info.lRy;
270  break;
271  case 5:
272  axisValue=info.lRz;
273  break;
274  case 6:
275  axisValue=info.rglSlider[0];
276  break;
277  case 7:
278  axisValue=info.rglSlider[1];
279  break;
280  default:
281  break;
282  }
283 
284  if (ActiveJoysticks[joystick].axisValid[dxAxis]>0)
285  axisFound=1;
286 
287  if (axisFound)
288  {
289  s32 val=axisValue - 32768;
290 
291  if (val <-32767) val=-32767;
292  if (val > 32767) val=32767;
293  event.JoystickEvent.Axis[irrAxis]=(s16)(val);
294  irrAxis++;
295  }
296 
297  dxAxis++;
298  }
299 
300  u32 buttons=0;
301  BYTE* bytebuttons=info.rgbButtons;
302  for (u16 i=0; i<32; i++)
303  {
304  if (bytebuttons[i] >0)
305  {
306  buttons |= (1 << i);
307  }
308  }
309  event.JoystickEvent.ButtonStates = buttons;
310 
311  (void)Device->postEventFromUser(event);
312  }
313  }
314 #else
315  if (0 == ActiveJoysticks.size())
316  return;
317 
318  u32 joystick;
319  JOYINFOEX info;
320 
321  for(joystick = 0; joystick < ActiveJoysticks.size(); ++joystick)
322  {
323  // needs to be reset for each joystick
324  // request ALL values and POV as continuous if possible
325  info.dwSize = sizeof(info);
326  info.dwFlags = JOY_RETURNALL|JOY_RETURNPOVCTS;
327  const JOYCAPS & caps = ActiveJoysticks[joystick].Caps;
328  // if no POV is available don't ask for POV values
329  if (!(caps.wCaps & JOYCAPS_HASPOV))
330  info.dwFlags &= ~(JOY_RETURNPOV|JOY_RETURNPOVCTS);
331  if(JOYERR_NOERROR == joyGetPosEx(ActiveJoysticks[joystick].Index, &info))
332  {
333  SEvent event;
334 
335  event.EventType = irr::EET_JOYSTICK_INPUT_EVENT;
336  event.JoystickEvent.Joystick = (u8)joystick;
337 
338  event.JoystickEvent.POV = (u16)info.dwPOV;
339  // set to undefined if no POV value was returned or the value
340  // is out of range
341  if (!(info.dwFlags & JOY_RETURNPOV) || (event.JoystickEvent.POV > 35900))
342  event.JoystickEvent.POV = 65535;
343 
344  for(int axis = 0; axis < SEvent::SJoystickEvent::NUMBER_OF_AXES; ++axis)
345  event.JoystickEvent.Axis[axis] = 0;
346 
347  event.JoystickEvent.ButtonStates = info.dwButtons;
348 
349  switch(caps.wNumAxes)
350  {
351  default:
352  case 6:
353  event.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_V] =
354  (s16)((65535 * (info.dwVpos - caps.wVmin)) / (caps.wVmax - caps.wVmin) - 32768);
355 
356  case 5:
357  event.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_U] =
358  (s16)((65535 * (info.dwUpos - caps.wUmin)) / (caps.wUmax - caps.wUmin) - 32768);
359 
360  case 4:
361  event.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_R] =
362  (s16)((65535 * (info.dwRpos - caps.wRmin)) / (caps.wRmax - caps.wRmin) - 32768);
363 
364  case 3:
365  event.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_Z] =
366  (s16)((65535 * (info.dwZpos - caps.wZmin)) / (caps.wZmax - caps.wZmin) - 32768);
367 
368  case 2:
369  event.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_Y] =
370  (s16)((65535 * (info.dwYpos - caps.wYmin)) / (caps.wYmax - caps.wYmin) - 32768);
371 
372  case 1:
373  event.JoystickEvent.Axis[SEvent::SJoystickEvent::AXIS_X] =
374  (s16)((65535 * (info.dwXpos - caps.wXmin)) / (caps.wXmax - caps.wXmin) - 32768);
375  }
376 
377  (void)Device->postEventFromUser(event);
378  }
379  }
380 #endif
381 #endif // _IRR_COMPILE_WITH_JOYSTICK_EVENTS_

386 irr::core::stringc SJoystickWin32Control::findJoystickName(int index, const JOYCAPS &caps) const
387 {
388 #if defined _IRR_COMPILE_WITH_JOYSTICK_EVENTS_
389 
390  // As a default use the name given in the joystick structure.
391  // It is always the same name, independent of joystick.
392  irr::core::stringc result(caps.szPname);
393 
394  core::stringc key = core::stringc(REGSTR_PATH_JOYCONFIG)+ "\\" + caps.szRegKey + "\\" + REGSTR_KEY_JOYCURR;
395  HKEY hTopKey = HKEY_LOCAL_MACHINE;
396  HKEY hKey;
397  long regresult = RegOpenKeyExA(hTopKey, key.c_str(), 0, KEY_READ, &hKey);
398  if (regresult != ERROR_SUCCESS)
399  {
400  hTopKey = HKEY_CURRENT_USER;
401  regresult = RegOpenKeyExA(hTopKey, key.c_str(), 0, KEY_READ, &hKey);
402  }
403  if (regresult != ERROR_SUCCESS)
404  return result;
405 
406  /* find the registry key name for the joystick's properties */
407  char regname[256];
408  DWORD regsize = sizeof(regname);
409  core::stringc regvalue = core::stringc("Joystick")+core::stringc(index+1) + REGSTR_VAL_JOYOEMNAME;
410  regresult = RegQueryValueExA(hKey, regvalue.c_str(), 0, 0, (LPBYTE)regname, &regsize);
411  RegCloseKey(hKey);
412  if (regresult != ERROR_SUCCESS)
413  return result;
414 
415  /* open that registry key */
416  core::stringc regkey = core::stringc(REGSTR_PATH_JOYOEM) + "\\" + regname;
417  regresult = RegOpenKeyExA(hTopKey, regkey.c_str(), 0, KEY_READ, &hKey);
418  if (regresult != ERROR_SUCCESS)
419  return result;
420 
421  /* find the size for the OEM name text */
422  regsize = sizeof(regvalue);
423  regresult = RegQueryValueEx(hKey, REGSTR_VAL_JOYOEMNAME, 0, 0,
424  NULL, &regsize);
425  if (regresult == ERROR_SUCCESS)
426  {
427  char *name;
428  /* allocate enough memory for the OEM name text ... */
429  name = new char[regsize];
430  if (name)
431  {
432  /* ... and read it from the registry */
433  regresult = RegQueryValueEx(hKey, REGSTR_VAL_JOYOEMNAME, 0, 0,
434  (LPBYTE)name, &regsize );
435  result = name;
436  }
437  delete[] name;
438  }
439  RegCloseKey(hKey);
440 
441  return result;
442 #endif
443  return "";
444 }

446 bool SJoystickWin32Control::activateJoysticks(core::array<SJoystickInfo> & joystickInfo)
447 {
448 #if defined _IRR_COMPILE_WITH_JOYSTICK_EVENTS_
449  joystickInfo.clear();
450  ActiveJoysticks.clear();
451 #ifdef _IRR_COMPILE_WITH_DIRECTINPUT_JOYSTICK_
452  if (!DirectInputDevice || (DirectInputDevice->EnumDevices(DI8DEVCLASS_GAMECTRL, SJoystickWin32Control::EnumJoysticks, this, DIEDFL_ATTACHEDONLY )))
453  {
454  os::Printer::log("Could not enum DirectInput8 controllers", ELL_WARNING);
455  return false;
456  }
457 
458  for(u32 joystick = 0; joystick < ActiveJoysticks.size(); ++joystick)
459  {
460  JoystickInfo& activeJoystick = ActiveJoysticks[joystick];
461  SJoystickInfo info;
462  info.Axes=activeJoystick.devcaps.dwAxes;
463  info.Buttons=activeJoystick.devcaps.dwButtons;
464  info.Name=activeJoystick.Name;
465  info.PovHat = (activeJoystick.devcaps.dwPOVs != 0)
466  ? SJoystickInfo::POV_HAT_PRESENT : SJoystickInfo::POV_HAT_ABSENT;
467  joystickInfo.push_back(info);
468  }
469  return true;
470 #else
471  const u32 numberOfJoysticks = ::joyGetNumDevs();
472  JOYINFOEX info;
473  info.dwSize = sizeof(info);
474  info.dwFlags = JOY_RETURNALL;
475 
476  JoystickInfo activeJoystick;
477  SJoystickInfo returnInfo;
478 
479  joystickInfo.reallocate(numberOfJoysticks);
480  ActiveJoysticks.reallocate(numberOfJoysticks);
481 
482  u32 joystick = 0;
483  for(; joystick < numberOfJoysticks; ++joystick)
484  {
485  if(JOYERR_NOERROR == joyGetPosEx(joystick, &info)
486  &&
487  JOYERR_NOERROR == joyGetDevCaps(joystick,
488  &activeJoystick.Caps,
489  sizeof(activeJoystick.Caps)))
490  {
491  activeJoystick.Index = joystick;
492  ActiveJoysticks.push_back(activeJoystick);
493 
494  returnInfo.Joystick = (u8)joystick;
495  returnInfo.Axes = activeJoystick.Caps.wNumAxes;
496  returnInfo.Buttons = activeJoystick.Caps.wNumButtons;
497  returnInfo.Name = findJoystickName(joystick, activeJoystick.Caps);
498  returnInfo.PovHat = ((activeJoystick.Caps.wCaps & JOYCAPS_HASPOV) == JOYCAPS_HASPOV)
499  ? SJoystickInfo::POV_HAT_PRESENT : SJoystickInfo::POV_HAT_ABSENT;
500 
501  joystickInfo.push_back(returnInfo);
502  }
503  }
504 
505  for(joystick = 0; joystick < joystickInfo.size(); ++joystick)
506  {
507  char logString[256];
508  snprintf_irr(logString, sizeof(logString), "Found joystick %d, %d axes, %d buttons '%s'",
509  joystick, joystickInfo[joystick].Axes,
510  joystickInfo[joystick].Buttons, joystickInfo[joystick].Name.c_str());
511  os::Printer::log(logString, ELL_INFORMATION);
512  }
513 
514  return true;
515 #endif
516 #else
517  return false;
518 #endif // _IRR_COMPILE_WITH_JOYSTICK_EVENTS_
519 }

549 LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
550 {
551  #ifndef WHEEL_DELTA
552  #define WHEEL_DELTA 120
553  #endif
554 
555  irr::CIrrDeviceWin32* dev = 0;
556  irr::SEvent event;
557 
558  static irr::s32 ClickCount=0;
559  if (GetCapture() != hWnd && ClickCount > 0)
560  ClickCount = 0;
561 
562 
563  struct messageMap
564  {
565  irr::s32 group;
566  UINT winMessage;
567  irr::s32 irrMessage;
568  };
569 
570  static messageMap mouseMap[] =
571  {
572  {0, WM_LBUTTONDOWN, irr::EMIE_LMOUSE_PRESSED_DOWN},
573  {1, WM_LBUTTONUP, irr::EMIE_LMOUSE_LEFT_UP},
574  {0, WM_RBUTTONDOWN, irr::EMIE_RMOUSE_PRESSED_DOWN},
575  {1, WM_RBUTTONUP, irr::EMIE_RMOUSE_LEFT_UP},
576  {0, WM_MBUTTONDOWN, irr::EMIE_MMOUSE_PRESSED_DOWN},
577  {1, WM_MBUTTONUP, irr::EMIE_MMOUSE_LEFT_UP},
578  {2, WM_MOUSEMOVE, irr::EMIE_MOUSE_MOVED},
579  {3, WM_MOUSEWHEEL, irr::EMIE_MOUSE_WHEEL},
580  {-1, 0, 0}
581  };
582 
583  // handle grouped events
584  messageMap * m = mouseMap;
585  while ( m->group >=0 && m->winMessage != message )
586  m += 1;
587 
588  if ( m->group >= 0 )
589  {
590  if ( m->group == 0 ) // down
591  {
592  ClickCount++;
593  SetCapture(hWnd);
594  }
595  else
596  if ( m->group == 1 ) // up
597  {
598  ClickCount--;
599  if (ClickCount<1)
600  {
601  ClickCount=0;
602  ReleaseCapture();
603  }
604  }
605 
606  event.EventType = irr::EET_MOUSE_INPUT_EVENT;
607  event.MouseInput.Event = (irr::EMOUSE_INPUT_EVENT) m->irrMessage;
608  event.MouseInput.X = (short)LOWORD(lParam);
609  event.MouseInput.Y = (short)HIWORD(lParam);
610  event.MouseInput.Shift = ((LOWORD(wParam) & MK_SHIFT) != 0);
611  event.MouseInput.Control = ((LOWORD(wParam) & MK_CONTROL) != 0);
612  // left and right mouse buttons
613  event.MouseInput.ButtonStates = wParam & ( MK_LBUTTON | MK_RBUTTON);
614  // middle and extra buttons
615  if (wParam & MK_MBUTTON)
616  event.MouseInput.ButtonStates |= irr::EMBSM_MIDDLE;
617  if (wParam & MK_XBUTTON1)
618  event.MouseInput.ButtonStates |= irr::EMBSM_EXTRA1;
619  if (wParam & MK_XBUTTON2)
620  event.MouseInput.ButtonStates |= irr::EMBSM_EXTRA2;
621  event.MouseInput.Wheel = 0.f;
622 
623  // wheel
624  if ( m->group == 3 )

635  if (dev)

637  dev->postEventFromUser(event);
638 
639  if ( event.MouseInput.Event >= irr::EMIE_LMOUSE_PRESSED_DOWN && event.MouseInput.Event <= irr::EMIE_MMOUSE_PRESSED_DOWN )
640  {
641  irr::u32 clicks = dev->checkSuccessiveClicks(event.MouseInput.X, event.MouseInput.Y, event.MouseInput.Event);
642  if ( clicks == 2 )

644  event.MouseInput.Event = (irr::EMOUSE_INPUT_EVENT)(irr::EMIE_LMOUSE_DOUBLE_CLICK + event.MouseInput.Event-irr::EMIE_LMOUSE_PRESSED_DOWN);
645  dev->postEventFromUser(event);
646  }

648  {
649  event.MouseInput.Event = (irr::EMOUSE_INPUT_EVENT)(irr::EMIE_LMOUSE_TRIPLE_CLICK + event.MouseInput.Event-irr::EMIE_LMOUSE_PRESSED_DOWN);

651  }
652  }

655  }

657  switch (message)
658  {
659  case WM_PAINT:
660  {
661  PAINTSTRUCT ps;
662  BeginPaint(hWnd, &ps);
663  EndPaint(hWnd, &ps);
664  }

666 
667  case WM_ERASEBKGND:
668  return 0;
669 
670  case WM_SYSKEYDOWN:
671  case WM_SYSKEYUP:
672  case WM_KEYDOWN:
673  case WM_KEYUP:

675  BYTE allKeys[256];
676 
677  event.EventType = irr::EET_KEY_INPUT_EVENT;
678  event.KeyInput.Key = (irr::EKEY_CODE)wParam;
679  event.KeyInput.PressedDown = (message==WM_KEYDOWN || message == WM_SYSKEYDOWN);

682  {
683  event.KeyInput.Key = (irr::EKEY_CODE)MapVirtualKey( ((lParam>>16) & 255), MAPVK_VSC_TO_VK_EX );
684  }
685  if ( event.KeyInput.Key == irr::KEY_CONTROL )
686  {
687  event.KeyInput.Key = (irr::EKEY_CODE)MapVirtualKey( ((lParam>>16) & 255), MAPVK_VSC_TO_VK_EX );
688  // some keyboards will just return LEFT for both - left and right keys. So also check extend bit.
689  if (lParam & 0x1000000)
690  event.KeyInput.Key = irr::KEY_RCONTROL;
691  }
692  if ( event.KeyInput.Key == irr::KEY_MENU )
693  {
694  event.KeyInput.Key = (irr::EKEY_CODE)MapVirtualKey( ((lParam>>16) & 255), MAPVK_VSC_TO_VK_EX );
695  if (lParam & 0x1000000)
696  event.KeyInput.Key = irr::KEY_RMENU;
697  }

700 
701  event.KeyInput.Shift = ((allKeys[VK_SHIFT] & 0x80)!=0);
702  event.KeyInput.Control = ((allKeys[VK_CONTROL] & 0x80)!=0);

705  WCHAR keyChars[2];

707  int conversionResult = ToUnicodeEx(static_cast<UINT>(wParam),scanCode,allKeys,keyChars,2,0,KEYBOARD_INPUT_HKL);
708  if (conversionResult == 1)
709  event.KeyInput.Char = keyChars[0];
710  else
711  event.KeyInput.Char = 0;
712 
713  // allow composing characters like '@' with Alt Gr on non-US keyboards
714  if ((allKeys[VK_MENU] & 0x80) != 0)
715  event.KeyInput.Control = 0;
716 
717  dev = getDeviceFromHWnd(hWnd);
718  if (dev)
719  dev->postEventFromUser(event);
720 
721  if (message == WM_SYSKEYDOWN || message == WM_SYSKEYUP)
722  return DefWindowProcW(hWnd, message, wParam, lParam);
723  else
724  return 0;
725  }
726 
727  case WM_SIZE:
728  {
729  // resize
730  dev = getDeviceFromHWnd(hWnd);
731  if (dev)
732  dev->OnResized();
733  }
734  return 0;
735 
736  case WM_DESTROY:
737  PostQuitMessage(0);
738  return 0;
739 
740  case WM_SYSCOMMAND:
741  // prevent screensaver or monitor powersave mode from starting
742  if ((wParam & 0xFFF0) == SC_SCREENSAVE ||
743  (wParam & 0xFFF0) == SC_MONITORPOWER ||
744  (wParam & 0xFFF0) == SC_KEYMENU
745  )
746  return 0;
747 
748  break;
749 
750  case WM_USER:
751  event.EventType = irr::EET_USER_EVENT;
752  event.UserEvent.UserData1 = static_cast<size_t>(wParam);
753  event.UserEvent.UserData2 = static_cast<size_t>(lParam);
754  dev = getDeviceFromHWnd(hWnd);
755 
756  if (dev)
757  dev->postEventFromUser(event);
758 
759  return 0;
760 
761  case WM_SETCURSOR:
762  // because Windows forgot about that in the meantime
763  dev = getDeviceFromHWnd(hWnd);
764  if (dev)
765  {
766  dev->getCursorControl()->setActiveIcon( dev->getCursorControl()->getActiveIcon() );
767  dev->getCursorControl()->setVisible( dev->getCursorControl()->isVisible() );
768  }
769  break;
770 
771  case WM_INPUTLANGCHANGE:
772  // get the new codepage used for keyboard input
773  KEYBOARD_INPUT_HKL = GetKeyboardLayout(0);
774  return 0;
775  }
776  return DefWindowProcW(hWnd, message, wParam, lParam);

1408 // Based on http://www.codeguru.com/cpp/w-p/win32/cursors/article.php/c4529/
1409 HCURSOR CIrrDeviceWin32::TextureToCursor(HWND hwnd, irr::video::ITexture * tex, const core::rect<s32>& sourceRect, const core::position2d<s32> &hotspot)
1410 {
1411  //
1412  // create the bitmaps needed for cursors from the texture
1413 
1414  HDC dc = GetDC(hwnd);
1415  HDC andDc = CreateCompatibleDC(dc);
1416  HDC xorDc = CreateCompatibleDC(dc);
1417  HBITMAP andBitmap = CreateCompatibleBitmap(dc, sourceRect.getWidth(), sourceRect.getHeight());
1418  HBITMAP xorBitmap = CreateCompatibleBitmap(dc, sourceRect.getWidth(), sourceRect.getHeight());
1419 
1420  HBITMAP oldAndBitmap = (HBITMAP)SelectObject(andDc, andBitmap);
1421  HBITMAP oldXorBitmap = (HBITMAP)SelectObject(xorDc, xorBitmap);
1422 
1423 
1424  video::ECOLOR_FORMAT format = tex->getColorFormat();
1425  u32 bytesPerPixel = video::IImage::getBitsPerPixelFromFormat(format) / 8;
1426  u32 bytesLeftGap = sourceRect.UpperLeftCorner.X * bytesPerPixel;
1427  u32 bytesRightGap = tex->getPitch() - sourceRect.LowerRightCorner.X * bytesPerPixel;
1428  const u8* data = (const u8*)tex->lock(video::ETLM_READ_ONLY, 0);
1429  data += sourceRect.UpperLeftCorner.Y*tex->getPitch();
1430  for ( s32 y = 0; y < sourceRect.getHeight(); ++y )
1431  {
1432  data += bytesLeftGap;
1433  for ( s32 x = 0; x < sourceRect.getWidth(); ++x )
1434  {
1435  video::SColor pixelCol;
1436  pixelCol.setData((const void*)data, format);
1437  data += bytesPerPixel;
1438 
1439  if ( pixelCol.getAlpha() == 0 ) // transparent
1440  {
1441  SetPixel(andDc, x, y, RGB(255,255,255));
1442  SetPixel(xorDc, x, y, RGB(0,0,0));
1443  }
1444  else // color
1445  {
1446  SetPixel(andDc, x, y, RGB(0,0,0));
1447  SetPixel(xorDc, x, y, RGB(pixelCol.getRed(), pixelCol.getGreen(), pixelCol.getBlue()));
1448  }
1449  }
1450  data += bytesRightGap;
1451  }
1452  tex->unlock();
1453 
1454  SelectObject(andDc, oldAndBitmap);
1455  SelectObject(xorDc, oldXorBitmap);
1456 
1457  DeleteDC(xorDc);
1458  DeleteDC(andDc);
1459 
1460  ReleaseDC(hwnd, dc);
1461 
1462  // create the cursor
1463 
1464  ICONINFO iconinfo;
1465  iconinfo.fIcon = false; // type is cursor not icon
1466  iconinfo.xHotspot = hotspot.X;
1467  iconinfo.yHotspot = hotspot.Y;
1468  iconinfo.hbmMask = andBitmap;
1469  iconinfo.hbmColor = xorBitmap;
1470 
1471  HCURSOR cursor = CreateIconIndirect(&iconinfo);
1472 
1473  DeleteObject(andBitmap);
1474  DeleteObject(xorBitmap);
1475 
1476  return cursor;
1477 }
