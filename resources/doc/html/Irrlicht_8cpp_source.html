58  extern "C" IRRLICHT_API IrrlichtDevice* IRRCALLCONV createDeviceEx(const SIrrlichtCreationParameters& params)
59  {
60 
61  IrrlichtDevice* dev = 0;
62 
63 #ifdef _IRR_COMPILE_WITH_WINDOWS_DEVICE_
64  if (params.DeviceType == EIDT_WIN32 || (!dev && params.DeviceType == EIDT_BEST))
65  dev = new CIrrDeviceWin32(params);
66 #endif
67 
68 #ifdef _IRR_COMPILE_WITH_OSX_DEVICE_
69  if (params.DeviceType == EIDT_OSX || (!dev && params.DeviceType == EIDT_BEST))
70  dev = new CIrrDeviceMacOSX(params);
71 #endif
72 
73 #ifdef _IRR_COMPILE_WITH_X11_DEVICE_
74  if (params.DeviceType == EIDT_X11 || (!dev && params.DeviceType == EIDT_BEST))
75  dev = new CIrrDeviceLinux(params);
76 #endif
77 
78 #ifdef _IRR_COMPILE_WITH_ANDROID_DEVICE_
79  if (params.DeviceType == EIDT_ANDROID || (!dev && params.DeviceType == EIDT_BEST))
80  dev = new CIrrDeviceAndroid(params);
81 #endif
82 
83 #ifdef _IRR_COMPILE_WITH_SDL_DEVICE_
84  if (params.DeviceType == EIDT_SDL || (!dev && params.DeviceType == EIDT_BEST))
85  dev = new CIrrDeviceSDL(params);
86 #endif
87 
88  if (dev && !dev->getVideoDriver() && params.DriverType != video::EDT_NULL)
89  {
90  dev->closeDevice(); // destroy window
91  dev->run(); // consume quit message
92  dev->drop();
93  dev = 0;
94  }
95 
96  return dev;
97  }
