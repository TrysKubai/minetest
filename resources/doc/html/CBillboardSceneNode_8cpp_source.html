68 void CBillboardSceneNode::render()
69 {
70  video::IVideoDriver* driver = SceneManager->getVideoDriver();
71  ICameraSceneNode* camera = SceneManager->getActiveCamera();
72 
73  if (!camera || !driver)
74  return;
75 
76  // make billboard look to camera
77  updateMesh(camera);
78 
79  driver->setTransform(video::ETS_WORLD, core::IdentityMatrix);
80  driver->setMaterial(Buffer->Material);
81  driver->drawMeshBuffer(Buffer);
82 
83  if (DebugDataVisible & scene::EDS_BBOX)
84  {
85  driver->setTransform(video::ETS_WORLD, AbsoluteTransformation);
86  video::SMaterial m;
87  m.Lighting = false;
88  driver->setMaterial(m);
89  driver->draw3DBox(BBoxSafe, video::SColor(0,208,195,152));
90  }
91 }

93 void CBillboardSceneNode::updateMesh(const irr::scene::ICameraSceneNode* camera)
94 {
95  // billboard looks toward camera
96  core::vector3df pos = getAbsolutePosition();
97 
98  core::vector3df campos = camera->getAbsolutePosition();
99  core::vector3df target = camera->getTarget();
100  core::vector3df up = camera->getUpVector();
101  core::vector3df view = target - campos;
102  view.normalize();
103 
104  core::vector3df horizontal = up.crossProduct(view);
105  if ( horizontal.getLength() == 0 )
106  {
107  horizontal.set(up.Y,up.X,up.Z);
108  }
109  horizontal.normalize();
110  core::vector3df topHorizontal = horizontal * 0.5f * TopEdgeWidth;
111  horizontal *= 0.5f * Size.Width;
112 
113  // pointing down!
114  core::vector3df vertical = horizontal.crossProduct(view);
115  vertical.normalize();
116  vertical *= 0.5f * Size.Height;
117 
118  view *= -1.0f;
119 
120  core::array<video::S3DVertex>& vertices = Buffer->Vertices;
121 
122  for (s32 i=0; i<4; ++i)
123  vertices[i].Normal = view;
124 
125  /* Vertices are:
126  2--1
127  |\ |
128  | \|
129  3--0
130  */
131  vertices[0].Pos = pos + horizontal + vertical;
132  vertices[1].Pos = pos + topHorizontal - vertical;
133  vertices[2].Pos = pos - topHorizontal - vertical;
134  vertices[3].Pos = pos - horizontal + vertical;
135 
136  Buffer->setDirty(EBT_VERTEX);
137  Buffer->recalculateBoundingBox();
138 }

154 void CBillboardSceneNode::setSize(const core::dimension2d<f32>& size)
155 {
156  Size = size;
157 
158  if (core::equals(Size.Width, 0.0f))
159  Size.Width = 1.0f;
160  TopEdgeWidth = Size.Width;
161 
162  if (core::equals(Size.Height, 0.0f))
163  Size.Height = 1.0f;
164 
165  const f32 extent = 0.5f*sqrtf(Size.Width*Size.Width + Size.Height*Size.Height);
166  BBoxSafe.MinEdge.set(-extent,-extent,-extent);
167  BBoxSafe.MaxEdge.set(extent,extent,extent);
168 }

171 void CBillboardSceneNode::setSize(f32 height, f32 bottomEdgeWidth, f32 topEdgeWidth)
172 {
173  Size.set(bottomEdgeWidth, height);
174  TopEdgeWidth = topEdgeWidth;
175 
176  if (core::equals(Size.Height, 0.0f))
177  Size.Height = 1.0f;
178 
179  if (core::equals(Size.Width, 0.f) && core::equals(TopEdgeWidth, 0.f))
180  {
181  Size.Width = 1.0f;
182  TopEdgeWidth = 1.0f;
183  }
184 
185  const f32 extent = 0.5f*sqrtf(Size.Width*Size.Width + Size.Height*Size.Height);
186  BBoxSafe.MinEdge.set(-extent,-extent,-extent);
187  BBoxSafe.MaxEdge.set(extent,extent,extent);
188 }
