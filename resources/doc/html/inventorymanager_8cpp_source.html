46 void InventoryLocation::serialize(std::ostream &os) const
47 {
48  switch (type) {
49  case InventoryLocation::UNDEFINED:
50  os<<"undefined";
51  break;
52  case InventoryLocation::CURRENT_PLAYER:
53  os<<"current_player";
54  break;
55  case InventoryLocation::PLAYER:
56  os<<"player:"<<name;
57  break;
58  case InventoryLocation::NODEMETA:
59  os<<"nodemeta:"<<p.X<<","<<p.Y<<","<<p.Z;
60  break;
61  case InventoryLocation::DETACHED:
62  os<<"detached:"<<name;
63  break;
64  default:
65  FATAL_ERROR("Unhandled inventory location type");
66  }
67 }

69 void InventoryLocation::deSerialize(std::istream &is)
70 {
71  std::string tname;
72  std::getline(is, tname, ':');
73  if (tname == "undefined") {
74  type = InventoryLocation::UNDEFINED;
75  } else if (tname == "current_player") {
76  type = InventoryLocation::CURRENT_PLAYER;
77  } else if (tname == "player") {
78  type = InventoryLocation::PLAYER;
79  std::getline(is, name, '\n');
80  } else if (tname == "nodemeta") {
81  type = InventoryLocation::NODEMETA;
82  std::string pos;
83  std::getline(is, pos, '\n');
84  Strfnd fn(pos);
85  p.X = stoi(fn.next(","));
86  p.Y = stoi(fn.next(","));
87  p.Z = stoi(fn.next(","));
88  } else if (tname == "detached") {
89  type = InventoryLocation::DETACHED;
90  std::getline(is, name, '\n');
91  } else {
92  infostream<<"Unknown InventoryLocation type=\""<<tname<<"\""<<std::endl;
93  throw SerializationError("Unknown InventoryLocation type");
94  }
95 }

245 void IMoveAction::apply(InventoryManager *mgr, ServerActiveObject *player, IGameDef *gamedef)
246 {
247  Inventory *inv_from = mgr->getInventory(from_inv);
248  Inventory *inv_to = mgr->getInventory(to_inv);
249 
250  if (!inv_from) {
251  infostream << "IMoveAction::apply(): FAIL: source inventory not found: "
252  << "from_inv=\""<<from_inv.dump() << "\""
253  << ", to_inv=\"" << to_inv.dump() << "\"" << std::endl;
254  return;
255  }
256  if (!inv_to) {
257  infostream << "IMoveAction::apply(): FAIL: destination inventory not found: "
258  << "from_inv=\"" << from_inv.dump() << "\""
259  << ", to_inv=\"" << to_inv.dump() << "\"" << std::endl;
260  return;
261  }
262 
263  auto get_borrow_checked_invlist = [](Inventory *inv, const std::string &listname)
264  -> InventoryList::ResizeLocked
265  {
266  InventoryList *list = inv->getList(listname);
267  if (!list)
268  return nullptr;
269  return list->resizeLock();
270  };
271 
272  auto list_from = get_borrow_checked_invlist(inv_from, from_list);
273  auto list_to = get_borrow_checked_invlist(inv_to, to_list);
274 
275  if (!list_from) {
276  infostream << "IMoveAction::apply(): FAIL: source list not found: "
277  << "from_inv=\"" << from_inv.dump() << "\""
278  << ", from_list=\"" << from_list << "\"" << std::endl;
279  return;
280  }
281  if (!list_to) {
282  infostream << "IMoveAction::apply(): FAIL: destination list not found: "
283  << "to_inv=\"" << to_inv.dump() << "\""
284  << ", to_list=\"" << to_list << "\"" << std::endl;
285  return;
286  }
287 
288  if (move_somewhere) {
289  list_from.reset();
290 
291  s16 old_to_i = to_i;
292  u16 old_count = count;
293  caused_by_move_somewhere = true;
294  move_somewhere = false;
295 
296  infostream << "IMoveAction::apply(): moving item somewhere"
297  << " msom=" << move_somewhere
298  << " count=" << count
299  << " from inv=\"" << from_inv.dump() << "\""
300  << " list=\"" << from_list << "\""
301  << " i=" << from_i
302  << " to inv=\"" << to_inv.dump() << "\""
303  << " list=\"" << to_list << "\""
304  << std::endl;
305 
306  // Try to add the item to destination list
307  s16 dest_size = list_to->getSize();
308  auto add_to = [&](s16 dest_i) {
309  // release resize lock while the callbacks are happening
310  list_to.reset();
311 
312  to_i = dest_i;
313  apply(mgr, player, gamedef);
314  assert(move_count <= count);
315  count -= move_count;
316 
317  list_to = get_borrow_checked_invlist(inv_to, to_list);
318  if (!list_to) {
319  // list_to was removed. simulate an empty list
320  dest_size = 0;
321  return;
322  }
323  dest_size = list_to->getSize();
324  };
325  // First try all the non-empty slots
326  for (s16 dest_i = 0; dest_i < dest_size && count > 0; dest_i++) {
327  if (!list_to->getItem(dest_i).empty())
328  add_to(dest_i);
329  }
330  // Then try all the empty ones
331  for (s16 dest_i = 0; dest_i < dest_size && count > 0; dest_i++) {
332  if (list_to->getItem(dest_i).empty())
333  add_to(dest_i);
334  }
335 
336  to_i = old_to_i;
337  count = old_count;
338  caused_by_move_somewhere = false;
339  move_somewhere = true;
340  return;
341  }
342 
343  if (from_i < 0 || list_from->getSize() <= (u32) from_i) {
344  infostream << "IMoveAction::apply(): FAIL: source index out of bounds: "
345  << "size of from_list=\"" << list_from->getSize() << "\""
346  << ", from_index=\"" << from_i << "\"" << std::endl;
347  return;
348  }
349 
350  if (to_i < 0 || list_to->getSize() <= (u32) to_i) {
351  infostream << "IMoveAction::apply(): FAIL: destination index out of bounds: "
352  << "size of to_list=\"" << list_to->getSize() << "\""
353  << ", to_index=\"" << to_i << "\"" << std::endl;
354  return;
355  }
356 
357  /*
358  Do not handle rollback if both inventories are that of the same player
359  */
360  bool ignore_rollback = (
361  from_inv.type == InventoryLocation::PLAYER &&
362  from_inv == to_inv);
363 
364  /*
365  Collect information of endpoints
366  */
367 
368  ItemStack src_item = list_from->getItem(from_i);
369  if (count > 0 && count < src_item.count)
370  src_item.count = count;
371  if (src_item.empty())
372  return;
373 
374  int src_can_take_count = 0xffff;
375  int dst_can_put_count = 0xffff;
376 
377  // this is needed for swapping items inside one inventory to work
378  ItemStack restitem;
379  bool allow_swap = !list_to->itemFits(to_i, src_item, &restitem)
380  && restitem.count == src_item.count
381  && !caused_by_move_somewhere;
382  move_count = src_item.count - restitem.count;
383 
384  // Shift-click: Cannot fill this stack, proceed with next slot
385  if (caused_by_move_somewhere && move_count == 0) {
386  return;
387  }
388 
389  if (allow_swap) {
390  // Swap will affect the entire stack if it can performed.
391  src_item = list_from->getItem(from_i);
392  count = src_item.count;
393  }
394 
395  if (from_inv == to_inv) {
396  // Move action within the same inventory
397  src_can_take_count = allowMove(src_item.count, player);
398 
399  bool swap_expected = allow_swap;
400  allow_swap = allow_swap
401  && (src_can_take_count == -1 || src_can_take_count >= src_item.count);
402  if (allow_swap) {
403  int try_put_count = list_to->getItem(to_i).count;
404  swapDirections();
405  dst_can_put_count = allowMove(try_put_count, player);
406  allow_swap = allow_swap
407  && (dst_can_put_count == -1 || dst_can_put_count >= try_put_count);
408  swapDirections();
409  } else {
410  dst_can_put_count = src_can_take_count;
411  }
412  if (swap_expected != allow_swap)
413  src_can_take_count = dst_can_put_count = 0;
414  } else {
415  // Take from one inventory, put into another
416  int src_item_count = src_item.count;
417  if (caused_by_move_somewhere)
418  // When moving somewhere: temporarily use the actual movable stack
419  // size to ensure correct callback execution.
420  src_item.count = move_count;
421  dst_can_put_count = allowPut(src_item, player);
422  src_can_take_count = allowTake(src_item, player);
423  if (caused_by_move_somewhere)
424  // Reset source item count
425  src_item.count = src_item_count;
426  bool swap_expected = allow_swap;
427  allow_swap = allow_swap
428  && (src_can_take_count == -1 || src_can_take_count >= src_item.count)
429  && (dst_can_put_count == -1 || dst_can_put_count >= src_item.count);
430  // A swap is expected, which means that we have to
431  // run the "allow" callbacks a second time with swapped inventories
432  if (allow_swap) {
433  ItemStack dst_item = list_to->getItem(to_i);
434  swapDirections();
435 
436  int src_can_take = allowPut(dst_item, player);
437  int dst_can_put = allowTake(dst_item, player);
438  allow_swap = allow_swap
439  && (src_can_take == -1 || src_can_take >= dst_item.count)
440  && (dst_can_put == -1 || dst_can_put >= dst_item.count);
441  swapDirections();
442  }
443  if (swap_expected != allow_swap)
444  src_can_take_count = dst_can_put_count = 0;
445  }
446 
447  int old_count = count;
448 
449  /* Modify count according to collected data */
450  count = src_item.count;
451  if (src_can_take_count != -1 && count > src_can_take_count)
452  count = src_can_take_count;
453  if (dst_can_put_count != -1 && count > dst_can_put_count)
454  count = dst_can_put_count;
455 
456  /* Limit according to source item count */
457  if (count > list_from->getItem(from_i).count)
458  count = list_from->getItem(from_i).count;
459 
460  /* If no items will be moved, don't go further */
461  if (count == 0) {
462  if (caused_by_move_somewhere)
463  // Set move count to zero, as no items have been moved
464  move_count = 0;
465 
466  // Undo client prediction. See 'clientApply'
467  if (from_inv.type == InventoryLocation::PLAYER)
468  list_from->setModified();
469 
470  if (to_inv.type == InventoryLocation::PLAYER)
471  list_to->setModified();
472 
473  infostream<<"IMoveAction::apply(): move was completely disallowed:"
474  <<" count="<<old_count
475  <<" from inv=\""<<from_inv.dump()<<"\""
476  <<" list=\""<<from_list<<"\""
477  <<" i="<<from_i
478  <<" to inv=\""<<to_inv.dump()<<"\""
479  <<" list=\""<<to_list<<"\""
480  <<" i="<<to_i
481  <<std::endl;
482 
483  return;
484  }
485 
486  src_item = list_from->getItem(from_i);
487  src_item.count = count;
488  ItemStack from_stack_was = list_from->getItem(from_i);
489  ItemStack to_stack_was = list_to->getItem(to_i);
490 
491  /*
492  Perform actual move
493 
494  If something is wrong (source item is empty, destination is the
495  same as source), nothing happens
496  */
497  bool did_swap = false;
498  move_count = list_from->moveItem(from_i,
499  list_to.get(), to_i, count, allow_swap, &did_swap);
500  if (caused_by_move_somewhere)
501  count = old_count;
502  assert(allow_swap == did_swap);
503 
504  // If source is infinite, reset it's stack
505  if (src_can_take_count == -1) {
506  // For the caused_by_move_somewhere == true case we didn't force-put the item,
507  // which guarantees there is no leftover, and code below would duplicate the
508  // (not replaced) to_stack_was item.
509  if (!caused_by_move_somewhere) {
510  // If destination stack is of different type and there are leftover
511  // items, attempt to put the leftover items to a different place in the
512  // destination inventory.
513  // The client-side GUI will try to guess if this happens.
514  if (from_stack_was.name != to_stack_was.name) {
515  for (u32 i = 0; i < list_to->getSize(); i++) {
516  if (list_to->getItem(i).empty()) {
517  list_to->changeItem(i, to_stack_was);
518  break;
519  }
520  }
521  }
522  }
523  if (move_count > 0 || did_swap) {
524  list_from->deleteItem(from_i);
525  list_from->addItem(from_i, from_stack_was);
526  }
527  }
528  // If destination is infinite, reset it's stack and take count from source
529  if (dst_can_put_count == -1) {
530  list_to->deleteItem(to_i);
531  list_to->addItem(to_i, to_stack_was);
532  list_from->deleteItem(from_i);
533  list_from->addItem(from_i, from_stack_was);
534  list_from->takeItem(from_i, count);
535  }
536 
537  infostream << "IMoveAction::apply(): moved"
538  << " msom=" << move_somewhere
539  << " caused=" << caused_by_move_somewhere
540  << " count=" << count
541  << " from inv=\"" << from_inv.dump() << "\""
542  << " list=\"" << from_list << "\""
543  << " i=" << from_i
544  << " to inv=\"" << to_inv.dump() << "\""
545  << " list=\"" << to_list << "\""
546  << " i=" << to_i
547  << std::endl;
548 
549  // If we are inside the move somewhere loop, we don't need to report
550  // anything if nothing happened
551  if (caused_by_move_somewhere && move_count == 0)
552  return;
553 
554  /*
555  Record rollback information
556  */
557  if (!ignore_rollback && gamedef->rollback()) {
558  IRollbackManager *rollback = gamedef->rollback();
559 
560  // If source is not infinite, record item take
561  if (src_can_take_count != -1) {
562  RollbackAction action;
563  std::string loc;
564  {
565  std::ostringstream os(std::ios::binary);
566  from_inv.serialize(os);
567  loc = os.str();
568  }
569  action.setModifyInventoryStack(loc, from_list, from_i, false,
570  src_item);
571  rollback->reportAction(action);
572  }
573  // If destination is not infinite, record item put
574  if (dst_can_put_count != -1) {
575  RollbackAction action;
576  std::string loc;
577  {
578  std::ostringstream os(std::ios::binary);
579  to_inv.serialize(os);
580  loc = os.str();
581  }
582  action.setModifyInventoryStack(loc, to_list, to_i, true,
583  src_item);
584  rollback->reportAction(action);
585  }
586  }
587 
588  /*
589  Report move to endpoints
590  */
591  list_to.reset();
592 
593  // Source = destination => move
594  if (from_inv == to_inv) {
595  onMove(count, player);
596  if (did_swap) {
597  // Item is now placed in source list
598  src_item = list_from->getItem(from_i);
599  swapDirections();
600  onMove(src_item.count, player);
601  swapDirections();
602  }
603  mgr->setInventoryModified(from_inv);
604  } else {
605  int src_item_count = src_item.count;
606  if (caused_by_move_somewhere)
607  // When moving somewhere: temporarily use the actual movable stack
608  // size to ensure correct callback execution.
609  src_item.count = move_count;
610  onPutAndOnTake(src_item, player);
611  if (caused_by_move_somewhere)
612  // Reset source item count
613  src_item.count = src_item_count;
614  if (did_swap) {
615  // Item is now placed in source list
616  src_item = list_from->getItem(from_i);
617  swapDirections();
618  onPutAndOnTake(src_item, player);
619  swapDirections();
620  }
621  mgr->setInventoryModified(to_inv);
622  mgr->setInventoryModified(from_inv);
623  }
624 }

626 void IMoveAction::clientApply(InventoryManager *mgr, IGameDef *gamedef)
627 {
628  // Optional InventoryAction operation that is run on the client
629  // to make lag less apparent.
630 
631  Inventory *inv_from = mgr->getInventory(from_inv);
632  Inventory *inv_to = mgr->getInventory(to_inv);
633  if (!inv_from || !inv_to)
634  return;
635 
636  InventoryLocation current_player;
637  current_player.setCurrentPlayer();
638  Inventory *inv_player = mgr->getInventory(current_player);

743  take_count = src_can_take_count;

746  src_item = list_from->getItem(from_i);
747

749  ItemStack item1 = list_from->getItem(from_i);
750  item1.count = take_count;
751  if(PLAYER_TO_SA(player)->item_OnDrop(item1, player,
752  player->getBasePosition())) {
753  int actually_dropped_count = take_count - item1.count;
754

759  if (from_inv.type == InventoryLocation::PLAYER)

761  return;
762  }

765  if (src_can_take_count != -1) {
766  // Take item from source list

768

786  */
787  list_from_lock.reset();

789  switch (from_inv.type) {
790  case InventoryLocation::DETACHED:
791  PLAYER_TO_SA(player)->detached_inventory_OnTake(
792  *this, src_item, player);
793  break;
794  case InventoryLocation::NODEMETA:
795  PLAYER_TO_SA(player)->nodemeta_inventory_OnTake(
796  *this, src_item, player);
797  break;
798  case InventoryLocation::PLAYER:
799  PLAYER_TO_SA(player)->player_inventory_OnTake(
800  *this, src_item, player);
801  break;
802  default:
803  break;

805 
806  /*
807  Record rollback information
808  */
809  if (!ignore_src_rollback && gamedef->rollback()) {
810  IRollbackManager *rollback = gamedef->rollback();
811 
812  // If source is not infinite, record item take
813  if (src_can_take_count != -1) {
814  RollbackAction action;

819  loc = os.str();

822  false, src_item);
