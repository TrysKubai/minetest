16  OpenGLVersion COpenGLES2Driver::getVersionFromOpenGL() const {
17  auto version_string = reinterpret_cast<const char *>(glGetString(GL_VERSION));
18  int major, minor;
19  if (sscanf(version_string, "OpenGL ES %d.%d", &major, &minor) != 2) {
20  os::Printer::log("Failed to parse OpenGL ES version string", version_string, ELL_ERROR);
21  return {OpenGLSpec::ES, 0, 0, 0};
22  }
23  return {OpenGLSpec::ES, (u8)major, (u8)minor, 0};
24  }

26  void COpenGLES2Driver::initFeatures() {
27  assert (Version.Spec == OpenGLSpec::ES);
28  assert (Version.Major >= 2);
29  if (Version.Major >= 3)
30  initExtensionsNew();
31  else
32  initExtensionsOld();
33 
34  if (Version.Major >= 3) {
35  // NOTE floating-point formats may not be suitable for render targets.
36  TextureFormats[ECF_A1R5G5B5] = {GL_RGB5_A1, GL_RGBA, GL_UNSIGNED_SHORT_5_5_5_1, CColorConverter::convert_A1R5G5B5toR5G5B5A1};
37  TextureFormats[ECF_R5G6B5] = {GL_RGB565, GL_RGB, GL_UNSIGNED_SHORT_5_6_5};
38  TextureFormats[ECF_R8G8B8] = {GL_RGB8, GL_RGB, GL_UNSIGNED_BYTE};
39  TextureFormats[ECF_A8R8G8B8] = {GL_RGBA8, GL_RGBA, GL_UNSIGNED_BYTE, CColorConverter::convert_A8R8G8B8toA8B8G8R8};
40  TextureFormats[ECF_R16F] = {GL_R16F, GL_RED, GL_HALF_FLOAT};
41  TextureFormats[ECF_G16R16F] = {GL_RG16F, GL_RG, GL_HALF_FLOAT};
42  TextureFormats[ECF_A16B16G16R16F] = {GL_RGBA16F, GL_RGBA, GL_HALF_FLOAT};
43  TextureFormats[ECF_R32F] = {GL_R32F, GL_RED, GL_FLOAT};
44  TextureFormats[ECF_G32R32F] = {GL_RG32F, GL_RG, GL_FLOAT};
45  TextureFormats[ECF_A32B32G32R32F] = {GL_RGBA32F, GL_RGBA, GL_FLOAT};
46  TextureFormats[ECF_R8] = {GL_R8, GL_RED, GL_UNSIGNED_BYTE};
47  TextureFormats[ECF_R8G8] = {GL_RG8, GL_RG, GL_UNSIGNED_BYTE};
48  TextureFormats[ECF_D16] = {GL_DEPTH_COMPONENT16, GL_DEPTH_COMPONENT, GL_UNSIGNED_SHORT};
49  TextureFormats[ECF_D24S8] = {GL_DEPTH24_STENCIL8, GL_DEPTH_STENCIL, GL_UNSIGNED_INT_24_8};
50 
51  if (FeatureAvailable[IRR_GL_EXT_texture_format_BGRA8888])
52  TextureFormats[ECF_A8R8G8B8] = {GL_RGBA, GL_BGRA, GL_UNSIGNED_BYTE};
53  else if (FeatureAvailable[IRR_GL_APPLE_texture_format_BGRA8888])
54  TextureFormats[ECF_A8R8G8B8] = {GL_BGRA, GL_BGRA, GL_UNSIGNED_BYTE};
55 
56  if (FeatureAvailable[IRR_GL_OES_depth32])
57  TextureFormats[ECF_D32] = {GL_DEPTH_COMPONENT32, GL_DEPTH_COMPONENT, GL_UNSIGNED_INT};
58  } else {
59  // NOTE These are *texture* formats. They may or may not be suitable
60  // for render targets. The specs only talks on *sized* formats for the
61  // latter but forbids creating textures with sized internal formats,
62  // reserving them for renderbuffers.
63 
64  static const GLenum HALF_FLOAT_OES = 0x8D61; // not equal to GL_HALF_FLOAT
65  TextureFormats[ECF_A1R5G5B5] = {GL_RGBA, GL_RGBA, GL_UNSIGNED_SHORT_5_5_5_1, CColorConverter::convert_A1R5G5B5toR5G5B5A1};
66  TextureFormats[ECF_R5G6B5] = {GL_RGB, GL_RGB, GL_UNSIGNED_SHORT_5_6_5};
67  TextureFormats[ECF_R8G8B8] = {GL_RGB, GL_RGB, GL_UNSIGNED_BYTE};
68  TextureFormats[ECF_A8R8G8B8] = {GL_RGBA, GL_RGBA, GL_UNSIGNED_BYTE, CColorConverter::convert_A8R8G8B8toA8B8G8R8};
69 
70  if (FeatureAvailable[IRR_GL_EXT_texture_format_BGRA8888])
71  TextureFormats[ECF_A8R8G8B8] = {GL_RGBA, GL_BGRA, GL_UNSIGNED_BYTE};
72  else if (FeatureAvailable[IRR_GL_APPLE_texture_format_BGRA8888])
73  TextureFormats[ECF_A8R8G8B8] = {GL_BGRA, GL_BGRA, GL_UNSIGNED_BYTE};
74 
75  if (FeatureAvailable[IRR_GL_OES_texture_half_float]) {
76  TextureFormats[ECF_A16B16G16R16F] = {GL_RGBA, GL_RGBA, HALF_FLOAT_OES};
77  }
78  if (FeatureAvailable[IRR_GL_OES_texture_float]) {
79  TextureFormats[ECF_A32B32G32R32F] = {GL_RGBA, GL_RGBA, GL_FLOAT};
80  }
81  if (FeatureAvailable[IRR_GL_EXT_texture_rg]) {
82  TextureFormats[ECF_R8] = {GL_RED, GL_RED, GL_UNSIGNED_BYTE};
83  TextureFormats[ECF_R8G8] = {GL_RG, GL_RG, GL_UNSIGNED_BYTE};
84 
85  if (FeatureAvailable[IRR_GL_OES_texture_half_float]) {
86  TextureFormats[ECF_R16F] = {GL_RED, GL_RED, HALF_FLOAT_OES};
87  TextureFormats[ECF_G16R16F] = {GL_RG, GL_RG, HALF_FLOAT_OES};
88  }
89  if (FeatureAvailable[IRR_GL_OES_texture_float]) {
90  TextureFormats[ECF_R32F] = {GL_RED, GL_RED, GL_FLOAT};
91  TextureFormats[ECF_G32R32F] = {GL_RG, GL_RG, GL_FLOAT};
92  }
93  }
94 
95  if (FeatureAvailable[IRR_GL_OES_depth_texture]) {
96  TextureFormats[ECF_D16] = {GL_DEPTH_COMPONENT, GL_DEPTH_COMPONENT, GL_UNSIGNED_SHORT};
97  if (FeatureAvailable[IRR_GL_OES_depth32])
98  TextureFormats[ECF_D32] = {GL_DEPTH_COMPONENT, GL_DEPTH_COMPONENT, GL_UNSIGNED_INT};
99  if (FeatureAvailable[IRR_GL_OES_packed_depth_stencil])
100  TextureFormats[ECF_D24S8] = {GL_DEPTH_STENCIL, GL_DEPTH_STENCIL, GL_UNSIGNED_INT_24_8};
101  }
102  }
103 
104  const bool MRTSupported = Version.Major >= 3 || queryExtension("GL_EXT_draw_buffers");
105  AnisotropicFilterSupported = queryExtension("GL_EXT_texture_filter_anisotropic");
106  BlendMinMaxSupported = (Version.Major >= 3) || FeatureAvailable[IRR_GL_EXT_blend_minmax];
107  const bool TextureLODBiasSupported = queryExtension("GL_EXT_texture_lod_bias");
108 
109  // COGLESCoreExtensionHandler::Feature
110  static_assert(MATERIAL_MAX_TEXTURES <= 8, "Only up to 8 textures are guaranteed");
111  Feature.BlendOperation = true;
112  Feature.ColorAttachment = 1;
113  if (MRTSupported)
114  Feature.ColorAttachment = GetInteger(GL_MAX_COLOR_ATTACHMENTS);
115  Feature.MaxTextureUnits = MATERIAL_MAX_TEXTURES;
116  if (MRTSupported)
117  Feature.MultipleRenderTarget = GetInteger(GL_MAX_DRAW_BUFFERS);
118 
119  // COGLESCoreExtensionHandler
120  if (AnisotropicFilterSupported)
121  MaxAnisotropy = GetInteger(GL_MAX_TEXTURE_MAX_ANISOTROPY_EXT);
122  if (Version.Major >= 3 || queryExtension("GL_EXT_draw_range_elements"))
123  MaxIndices = GetInteger(GL_MAX_ELEMENTS_INDICES);
124  MaxTextureSize = GetInteger(GL_MAX_TEXTURE_SIZE);
125  if (TextureLODBiasSupported)
126  glGetFloatv(GL_MAX_TEXTURE_LOD_BIAS, &MaxTextureLODBias);
127  glGetFloatv(GL_ALIASED_LINE_WIDTH_RANGE, DimAliasedLine); // NOTE: this is not in the OpenGL ES 2.0 spec...
128  glGetFloatv(GL_ALIASED_POINT_SIZE_RANGE, DimAliasedPoint);
129  }
