49 bool ActiveObjectMgr::registerObject(ClientActiveObject *obj)
50 {
51  assert(obj); // Pre-condition
52  if (obj->getId() == 0) {
53  u16 new_id = getFreeId();
54  if (new_id == 0) {
55  infostream << "Client::ActiveObjectMgr::registerObject(): "
56  << "no free id available" << std::endl;
57 
58  delete obj;
59  return false;
60  }
61  obj->setId(new_id);
62  }
63 
64  if (!isFreeId(obj->getId())) {
65  infostream << "Client::ActiveObjectMgr::registerObject(): "
66  << "id is not free (" << obj->getId() << ")" << std::endl;
67  delete obj;
68  return false;
69  }
70  infostream << "Client::ActiveObjectMgr::registerObject(): "
71  << "added (id=" << obj->getId() << ")" << std::endl;
72  m_active_objects[obj->getId()] = obj;
73  return true;
74 }

94 void ActiveObjectMgr::getActiveObjects(const v3f &origin, f32 max_d,
95  std::vector<DistanceSortedActiveObject> &dest)
96 {
97  f32 max_d2 = max_d * max_d;
98  for (auto &ao_it : m_active_objects) {
99  ClientActiveObject *obj = ao_it.second;
100 
101  f32 d2 = (obj->getPosition() - origin).getLengthSQ();
102 
103  if (d2 > max_d2)
104  continue;
105 
106  dest.emplace_back(obj, d2);
107  }

110 void ActiveObjectMgr::getActiveSelectableObjects(const core::line3d<f32> &shootline,
111  std::vector<DistanceSortedActiveObject> &dest)
112 {
113  // Imagine a not-axis-aligned cuboid oriented into the direction of the shootline,
114  // with the width of the object's selection box radius * 2 and with length of the
115  // shootline (+selection box radius forwards and backwards). We check whether
116  // the selection box center is inside this cuboid.
117 
118  f32 max_d = shootline.getLength();
119  v3f dir = shootline.getVector().normalize();
120  // arbitrary linearly independent vector and orthogonal dirs
121  v3f li2dir = dir + (std::fabs(dir.X) < 0.5f ? v3f(1,0,0) : v3f(0,1,0));
122  v3f dir_ortho1 = dir.crossProduct(li2dir).normalize();
123  v3f dir_ortho2 = dir.crossProduct(dir_ortho1);
124 
125  for (auto &ao_it : m_active_objects) {
126  ClientActiveObject *obj = ao_it.second;
127 
128  aabb3f selection_box;
129  if (!obj->getSelectionBox(&selection_box))
130  continue;
131 
132  // possible optimization: get rid of the sqrt here
133  f32 selection_box_radius = selection_box.getRadius();
134 
135  v3f pos_diff = obj->getPosition() + selection_box.getCenter() - shootline.start;
136 
137  f32 d = dir.dotProduct(pos_diff);
138 
139  // backward- and far-plane
140  if (d + selection_box_radius < 0.0f || d - selection_box_radius > max_d)
141  continue;
142 
143  // side-planes
144  if (std::fabs(dir_ortho1.dotProduct(pos_diff)) > selection_box_radius
145  || std::fabs(dir_ortho2.dotProduct(pos_diff)) > selection_box_radius)
146  continue;
147 
148  dest.emplace_back(obj, d);
149  }
150 }
