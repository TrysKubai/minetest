83 template <typename T, std::size_t count, bool cache>
84 class CachedShaderSetting {
85  const char *m_name;
86  T m_sent[count];
87  bool has_been_set = false;
88  bool is_pixel;
89 protected:
90  CachedShaderSetting(const char *name, bool is_pixel) :
91  m_name(name), is_pixel(is_pixel)
92  {}
93 public:
94  void set(const T value[count], video::IMaterialRendererServices *services)
95  {
96  if (cache && has_been_set && std::equal(m_sent, m_sent + count, value))
97  return;
98  if (is_pixel)
99  services->setPixelShaderConstant(services->getPixelShaderConstantID(m_name), value, count);
100  else
101  services->setVertexShaderConstant(services->getVertexShaderConstantID(m_name), value, count);
102 
103  if (cache) {
104  std::copy(value, value + count, m_sent);
105  has_been_set = true;
106  }
107  }
108 };

124 template <typename T, std::size_t count, bool cache, bool is_pixel>
125 class CachedStructShaderSetting {
126  const char *m_name;
127  T m_sent[count];
128  bool has_been_set = false;
129  std::array<const char*, count> m_fields;
130 public:
131  CachedStructShaderSetting(const char *name, std::array<const char*, count> &&fields) :
132  m_name(name), m_fields(std::move(fields))
133  {}
134 
135  void set(const T value[count], video::IMaterialRendererServices *services)
136  {
137  if (cache && has_been_set && std::equal(m_sent, m_sent + count, value))
138  return;
139 
140  for (std::size_t i = 0; i < count; i++) {
141  std::string uniform_name = std::string(m_name) + "." + m_fields[i];
142 
143  if (is_pixel)
144  services->setPixelShaderConstant(services->getPixelShaderConstantID(uniform_name.c_str()), value + i, 1);
145  else
146  services->setVertexShaderConstant(services->getVertexShaderConstantID(uniform_name.c_str()), value + i, 1);
147  }
148 
149  if (cache) {
150  std::copy(value, value + count, m_sent);
151  has_been_set = true;
152  }
153  }
154 };
