171 void GUIFormSpecMenu::setInitialFocus()
172 {
173  // Set initial focus according to following order of precedence:
174  // 1. first empty editbox
175  // 2. first editbox
176  // 3. first table
177  // 4. last button
178  // 5. first focusable (not statictext, not tabheader)
179  // 6. first child element
180 
181  const auto& children = getChildren();
182 
183  // 1. first empty editbox
184  for (gui::IGUIElement *it : children) {
185  if (it->getType() == gui::EGUIET_EDIT_BOX
186  && it->getText()[0] == 0) {
187  Environment->setFocus(it);
188  return;
189  }
190  }
191 
192  // 2. first editbox
193  for (gui::IGUIElement *it : children) {
194  if (it->getType() == gui::EGUIET_EDIT_BOX) {
195  Environment->setFocus(it);
196  return;
197  }
198  }
199 
200  // 3. first table
201  for (gui::IGUIElement *it : children) {
202  if (it->getTypeName() == std::string("GUITable")) {
203  Environment->setFocus(it);
204  return;
205  }
206  }
207 
208  // 4. last button
209  for (auto it = children.rbegin(); it != children.rend(); ++it) {
210  if ((*it)->getType() == gui::EGUIET_BUTTON) {
211  Environment->setFocus(*it);
212  return;
213  }
214  }
215 
216  // 5. first focusable (not statictext, not tabheader)
217  for (gui::IGUIElement *it : children) {
218  if (it->getType() != gui::EGUIET_STATIC_TEXT &&
219  it->getType() != gui::EGUIET_TAB_CONTROL) {
220  Environment->setFocus(it);
221  return;
222  }
223  }
224 
225  // 6. first child element
226  if (children.empty())
227  Environment->setFocus(this);
228  else
229  Environment->setFocus(children.front());
230 }

336 void GUIFormSpecMenu::parseScrollContainer(parserData *data, const std::string &element)
337 {
338  std::vector<std::string> parts;
339  if (!precheckElement("scroll_container start", element, 4, 5, parts))
340  return;
341 
342  std::vector<std::string> v_pos = split(parts[0], ',');
343  std::vector<std::string> v_geom = split(parts[1], ',');
344  std::string scrollbar_name = parts[2];
345  std::string orientation = parts[3];
346  f32 scroll_factor = 0.1f;
347  if (parts.size() >= 5 && !parts[4].empty())
348  scroll_factor = stof(parts[4]);
349 
350  MY_CHECKPOS("scroll_container", 0);
351  MY_CHECKGEOM("scroll_container", 1);
352 
353  v2s32 pos = getRealCoordinateBasePos(v_pos);
354  v2s32 geom = getRealCoordinateGeometry(v_geom);
355 
356  if (orientation == "vertical")
357  scroll_factor *= -imgsize.Y;
358  else if (orientation == "horizontal")
359  scroll_factor *= -imgsize.X;
360  else
361  warningstream << "GUIFormSpecMenu::parseScrollContainer(): "
362  << "Invalid scroll_container orientation: " << orientation
363  << std::endl;
364 
365  // old parent (at first: this)
366  // ^ is parent of clipper
367  // ^ is parent of mover
368  // ^ is parent of other elements
369 
370  // make clipper
371  core::rect<s32> rect_clipper = core::rect<s32>(pos, pos + geom);
372 
373  gui::IGUIElement *clipper = new gui::IGUIElement(EGUIET_ELEMENT, Environment,
374  data->current_parent, 0, rect_clipper);
375 
376  // make mover
377  FieldSpec spec_mover(
378  "",
379  L"",
380  L"",
381  258 + m_fields.size()
382  );
383 
384  core::rect<s32> rect_mover = core::rect<s32>(0, 0, geom.X, geom.Y);
385 
386  GUIScrollContainer *mover = new GUIScrollContainer(Environment,
387  clipper, spec_mover.fid, rect_mover, orientation, scroll_factor);
388 
389  data->current_parent = mover;
390 
391  m_scroll_containers.emplace_back(scrollbar_name, mover);
392 
393  m_fields.push_back(spec_mover);
394 
395  clipper->drop();
396 
397  // remove interferring offset of normal containers
398  container_stack.push(pos_offset);
399  pos_offset.X = 0.0f;
400  pos_offset.Y = 0.0f;
401 }

403 void GUIFormSpecMenu::parseScrollContainerEnd(parserData *data)
404 {
405  if (data->current_parent == this || data->current_parent->getParent() == this ||
406  container_stack.empty()) {
407  errorstream << "Invalid scroll_container end element, "
408  << "no matching scroll_container start element" << std::endl;
409  return;
410  }
411

413  // pos_offset is only set by containers and scroll_containers.
414  // scroll_containers always set it to 0,0 which means that if it is
415  // not 0,0, it is a normal container that was opened last, not a
416  // scroll_container
417  errorstream << "Invalid scroll_container end element, "
418  << "an inner container was left open" << std::endl;
419  return;
420  }
421 
422  data->current_parent = data->current_parent->getParent()->getParent();
423  pos_offset = container_stack.top();
424  container_stack.pop();
425 }

427 void GUIFormSpecMenu::parseList(parserData *data, const std::string &element)
428 {
429  MY_CHECKCLIENT("list");
430 
431  std::vector<std::string> parts;
432  if (!precheckElement("list", element, 4, 5, parts))
433  return;
434 
435  std::string location = parts[0];
436  std::string listname = parts[1];
437  std::vector<std::string> v_pos = split(parts[2],',');
438  std::vector<std::string> v_geom = split(parts[3],',');
439  std::string startindex;
440  if (parts.size() == 5)
441  startindex = parts[4];
442 
443  MY_CHECKPOS("list",2);
444  MY_CHECKGEOM("list",3);
445 
446  InventoryLocation loc;
447 
448  if (location == "context" || location == "current_name")
449  loc = m_current_inventory_location;
450  else
451  loc.deSerialize(location);
452 
453  v2s32 geom;
454  geom.X = stoi(v_geom[0]);
455  geom.Y = stoi(v_geom[1]);
456 
457  s32 start_i = 0;
458  if (!startindex.empty())
459  start_i = stoi(startindex);
460 
461  if (geom.X < 0 || geom.Y < 0 || start_i < 0) {
462  errorstream << "Invalid list element: '" << element << "'" << std::endl;
463  return;
464  }
465 
466  if (!data->explicit_size)
467  warningstream << "invalid use of list without a size[] element" << std::endl;
468 
469  FieldSpec spec(
470  "",
471  L"",
472  L"",
473  258 + m_fields.size(),
474  3
475  );
476 
477  auto style = getDefaultStyleForElement("list", spec.fname);
478 
479  v2f32 slot_scale = style.getVector2f(StyleSpec::SIZE, v2f32(0, 0));
480  v2f32 slot_size(
481  slot_scale.X <= 0 ? imgsize.X : std::max<f32>(slot_scale.X * imgsize.X, 1),
482  slot_scale.Y <= 0 ? imgsize.Y : std::max<f32>(slot_scale.Y * imgsize.Y, 1)
483  );
484 
485  v2f32 slot_spacing = style.getVector2f(StyleSpec::SPACING, v2f32(-1, -1));
486  v2f32 default_spacing = data->real_coordinates ?
487  v2f32(imgsize.X * 0.25f, imgsize.Y * 0.25f) :
488  v2f32(spacing.X - imgsize.X, spacing.Y - imgsize.Y);
489 
490  slot_spacing.X = slot_spacing.X < 0 ? default_spacing.X :
491  imgsize.X * slot_spacing.X;
492  slot_spacing.Y = slot_spacing.Y < 0 ? default_spacing.Y :
493  imgsize.Y * slot_spacing.Y;
494 
495  slot_spacing += slot_size;
496 
497  v2s32 pos = data->real_coordinates ? getRealCoordinateBasePos(v_pos) :
498  getElementBasePos(&v_pos);
499 
500  core::rect<s32> rect = core::rect<s32>(pos.X, pos.Y,
501  pos.X + (geom.X - 1) * slot_spacing.X + slot_size.X,
502  pos.Y + (geom.Y - 1) * slot_spacing.Y + slot_size.Y);
503 
504  GUIInventoryList *e = new GUIInventoryList(Environment, data->current_parent,
505  spec.fid, rect, m_invmgr, loc, listname, geom, start_i,
506  v2s32(slot_size.X, slot_size.Y), slot_spacing, this,
507  data->inventorylist_options, m_font);
508 
509  e->setNotClipped(style.getBool(StyleSpec::NOCLIP, false));
510 
511  m_inventorylists.push_back(e);
512  m_fields.push_back(spec);
513 }
