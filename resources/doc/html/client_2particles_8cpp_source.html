40 Particle::Particle(
41  IGameDef *gamedef,
42  LocalPlayer *player,
43  ClientEnvironment *env,
44  const ParticleParameters &p,
45  const ClientTexRef& texture,
46  v2f texpos,
47  v2f texsize,
48  video::SColor color
49 ):
50  scene::ISceneNode(((Client *)gamedef)->getSceneManager()->getRootSceneNode(),
51  ((Client *)gamedef)->getSceneManager()),
52  m_texture(texture)
53 {
54  // Misc
55  m_gamedef = gamedef;
56  m_env = env;
57 
58  // translate blend modes to GL blend functions
59  video::E_BLEND_FACTOR bfsrc, bfdst;
60  video::E_BLEND_OPERATION blendop;
61  const auto blendmode = texture.tex != nullptr
62  ? texture.tex -> blendmode
63  : ParticleParamTypes::BlendMode::alpha;
64 
65  switch (blendmode) {
66  case ParticleParamTypes::BlendMode::add:
67  bfsrc = video::EBF_SRC_ALPHA;
68  bfdst = video::EBF_DST_ALPHA;
69  blendop = video::EBO_ADD;
70  break;
71 
72  case ParticleParamTypes::BlendMode::sub:
73  bfsrc = video::EBF_SRC_ALPHA;
74  bfdst = video::EBF_DST_ALPHA;
75  blendop = video::EBO_REVSUBTRACT;
76  break;
77 
78  case ParticleParamTypes::BlendMode::screen:
79  bfsrc = video::EBF_ONE;
80  bfdst = video::EBF_ONE_MINUS_SRC_COLOR;
81  blendop = video::EBO_ADD;
82  break;
83 
84  default: // includes ParticleParamTypes::BlendMode::alpha
85  bfsrc = video::EBF_SRC_ALPHA;
86  bfdst = video::EBF_ONE_MINUS_SRC_ALPHA;
87  blendop = video::EBO_ADD;
88  break;
89  }
90 
91  // Texture
92  m_material.setFlag(video::EMF_LIGHTING, false);
93  m_material.setFlag(video::EMF_BACK_FACE_CULLING, false);
94  m_material.setFlag(video::EMF_BILINEAR_FILTER, false);
95  m_material.setFlag(video::EMF_FOG_ENABLE, true);
96 
97  // correctly render layered transparent particles -- see #10398
98  m_material.setFlag(video::EMF_ZWRITE_ENABLE, true);
99 
100  // enable alpha blending and set blend mode
101  m_material.MaterialType = video::EMT_ONETEXTURE_BLEND;
102  m_material.MaterialTypeParam = video::pack_textureBlendFunc(
103  bfsrc, bfdst,
104  video::EMFN_MODULATE_1X,
105  video::EAS_TEXTURE | video::EAS_VERTEX_COLOR);
106  m_material.BlendOperation = blendop;
107  m_material.setTexture(0, m_texture.ref);
108  m_texpos = texpos;
109  m_texsize = texsize;
110  m_animation = p.animation;
111 
112  // Color
113  m_base_color = color;
114  m_color = color;
115 
116  // Particle related
117  m_pos = p.pos;
118  m_velocity = p.vel;
119  m_acceleration = p.acc;
120  m_drag = p.drag;
121  m_jitter = p.jitter;
122  m_bounce = p.bounce;
123  m_expiration = p.expirationtime;
124  m_player = player;
125  m_size = p.size;
126  m_collisiondetection = p.collisiondetection;
127  m_collision_removal = p.collision_removal;
128  m_object_collision = p.object_collision;
129  m_vertical = p.vertical;
130  m_glow = p.glow;
131  m_alpha = 0;
132  m_parent = nullptr;
133 
134  // Irrlicht stuff
135  const float c = p.size / 2;
136  m_collisionbox = aabb3f(-c, -c, -c, c, c, c);
137  this->setAutomaticCulling(scene::EAC_OFF);
138 
139  // Init lighting
140  updateLight();
141 
142  // Init model
143  updateVertices();
144 }

174 void Particle::step(float dtime)
175 {
176  m_time += dtime;
177 
178  // apply drag (not handled by collisionMoveSimple) and brownian motion
179  v3f av = vecAbsolute(m_velocity);
180  av -= av * (m_drag * dtime);
181  m_velocity = av*vecSign(m_velocity) + v3f(m_jitter.pickWithin())*dtime;
182 
183  if (m_collisiondetection) {
184  aabb3f box = m_collisionbox;
185  v3f p_pos = m_pos * BS;
186  v3f p_velocity = m_velocity * BS;
187  collisionMoveResult r = collisionMoveSimple(m_env, m_gamedef, BS * 0.5f,
188  box, 0.0f, dtime, &p_pos, &p_velocity, m_acceleration * BS, nullptr,
189  m_object_collision);
190 
191  f32 bounciness = m_bounce.pickWithin();
192  if (r.collides && (m_collision_removal || bounciness > 0)) {
193  if (m_collision_removal) {
194  // force expiration of the particle
195  m_expiration = -1.0f;
196  } else if (bounciness > 0) {
197  /* cheap way to get a decent bounce effect is to only invert the
198  * largest component of the velocity vector, so e.g. you don't
199  * have a rock immediately bounce back in your face when you try
200  * to skip it across the water (as would happen if we simply
201  * downscaled and negated the velocity vector). this means
202  * bounciness will work properly for cubic objects, but meshes
203  * with diagonal angles and entities will not yield the correct
204  * visual. this is probably unavoidable */
205  if (av.Y > av.X && av.Y > av.Z) {
206  m_velocity.Y = -(m_velocity.Y * bounciness);
207  } else if (av.X > av.Y && av.X > av.Z) {
208  m_velocity.X = -(m_velocity.X * bounciness);
209  } else if (av.Z > av.Y && av.Z > av.X) {
210  m_velocity.Z = -(m_velocity.Z * bounciness);
211  } else { // well now we're in a bit of a pickle
212  m_velocity = -(m_velocity * bounciness);
213  }
214  }
215  } else {
216  m_velocity = p_velocity / BS;
217  }
218  m_pos = p_pos / BS;
219  } else {
220  // apply velocity and acceleration to position
221  m_pos += (m_velocity + m_acceleration * 0.5f * dtime) * dtime;
222  // apply acceleration to velocity
223  m_velocity += m_acceleration * dtime;
224  }
225 
226  if (m_animation.type != TAT_NONE) {
227  m_animation_time += dtime;
228  int frame_length_i, frame_count;
229  m_animation.determineParams(
230  m_material.getTexture(0)->getSize(),
231  &frame_count, &frame_length_i, NULL);
232  float frame_length = frame_length_i / 1000.0;
233  while (m_animation_time > frame_length) {
234  m_animation_frame++;
235  m_animation_time -= frame_length;
236  }
237  }
238 
239  // animate particle alpha in accordance with settings
240  if (m_texture.tex != nullptr)
241  m_alpha = m_texture.tex -> alpha.blend(m_time / (m_expiration+0.1f));
242  else
243  m_alpha = 1.f;
244 
245  // Update lighting
246  updateLight();
247 
248  // Update model
249  updateVertices();
250 
251  // Update position -- see #10398
252  v3s16 camera_offset = m_env->getCameraOffset();
253  setPosition(m_pos*BS - intToFloat(camera_offset, BS));
254 }

256 void Particle::updateLight()
257 {
258  u8 light = 0;
259  bool pos_ok;
260 
261  v3s16 p = v3s16(
262  floor(m_pos.X+0.5),
263  floor(m_pos.Y+0.5),
264  floor(m_pos.Z+0.5)
265  );
266  MapNode n = m_env->getClientMap().getNode(p, &pos_ok);
267  if (pos_ok)
268  light = n.getLightBlend(m_env->getDayNightRatio(),
269  m_gamedef->ndef()->getLightingFlags(n));
270  else
271  light = blend_light(m_env->getDayNightRatio(), LIGHT_SUN, 0);
272 
273  u8 m_light = decode_light(light + m_glow);
274  m_color.set(m_alpha*255,
275  m_light * m_base_color.getRed() / 255,
276  m_light * m_base_color.getGreen() / 255,
277  m_light * m_base_color.getBlue() / 255);
278 }

280 void Particle::updateVertices()
281 {
282  f32 tx0, tx1, ty0, ty1;
283  v2f scale;
284 
285  if (m_texture.tex != nullptr)
286  scale = m_texture.tex -> scale.blend(m_time / (m_expiration+0.1));
287  else
288  scale = v2f(1.f, 1.f);
289 
290  if (m_animation.type != TAT_NONE) {
291  const v2u32 texsize = m_material.getTexture(0)->getSize();
292  v2f texcoord, framesize_f;
293  v2u32 framesize;
294  texcoord = m_animation.getTextureCoords(texsize, m_animation_frame);
295  m_animation.determineParams(texsize, NULL, NULL, &framesize);
296  framesize_f = v2f(framesize.X / (float) texsize.X, framesize.Y / (float) texsize.Y);
297 
298  tx0 = m_texpos.X + texcoord.X;
299  tx1 = m_texpos.X + texcoord.X + framesize_f.X * m_texsize.X;
300  ty0 = m_texpos.Y + texcoord.Y;
301  ty1 = m_texpos.Y + texcoord.Y + framesize_f.Y * m_texsize.Y;
302  } else {
303  tx0 = m_texpos.X;
304  tx1 = m_texpos.X + m_texsize.X;
305  ty0 = m_texpos.Y;
306  ty1 = m_texpos.Y + m_texsize.Y;
307  }
308 
309  auto half = m_size * .5f,
310  hx = half * scale.X,
311  hy = half * scale.Y;
312  m_vertices[0] = video::S3DVertex(-hx, -hy,
313  0, 0, 0, 0, m_color, tx0, ty1);
314  m_vertices[1] = video::S3DVertex(hx, -hy,
315  0, 0, 0, 0, m_color, tx1, ty1);
316  m_vertices[2] = video::S3DVertex(hx, hy,
317  0, 0, 0, 0, m_color, tx1, ty0);
318  m_vertices[3] = video::S3DVertex(-hx, hy,
319  0, 0, 0, 0, m_color, tx0, ty0);
320 
321 
322  // see #10398
323  // v3s16 camera_offset = m_env->getCameraOffset();
324  // particle position is now handled by step()
325  m_box.reset(v3f());
326 
327  for (video::S3DVertex &vertex : m_vertices) {
328  if (m_vertical) {
329  v3f ppos = m_player->getPosition()/BS;
330  vertex.Pos.rotateXZBy(std::atan2(ppos.Z - m_pos.Z, ppos.X - m_pos.X) /
331  core::DEGTORAD + 90);
332  } else {
333  vertex.Pos.rotateYZBy(m_player->getPitch());
334  vertex.Pos.rotateXZBy(m_player->getYaw());
335  }
336  m_box.addInternalPoint(vertex.Pos);
337  }
338 }

344 ParticleSpawner::ParticleSpawner(
345  IGameDef *gamedef,
346  LocalPlayer *player,
347  const ParticleSpawnerParameters &p,
348  u16 attached_id,
349  std::unique_ptr<ClientTexture[]>& texpool,
350  size_t texcount,
351  ParticleManager *p_manager
352 ):
353  m_particlemanager(p_manager), p(p)
354 {
355  m_gamedef = gamedef;
356  m_player = player;
357  m_attached_id = attached_id;
358  m_texpool = std::move(texpool);
359  m_texcount = texcount;
360  m_time = 0;
361  m_active = 0;
362  m_dying = false;
363 
364  m_spawntimes.reserve(p.amount + 1);
365  for (u16 i = 0; i <= p.amount; i++) {
366  float spawntime = myrand_float() * p.time;
367  m_spawntimes.push_back(spawntime);
368  }
369 
370  size_t max_particles = 0; // maximum number of particles likely to be visible at any given time
371  if (p.time != 0) {
372  auto maxGenerations = p.time / std::min(p.exptime.start.min, p.exptime.end.min);
373  max_particles = p.amount / maxGenerations;
374  } else {
375  auto longestLife = std::max(p.exptime.start.max, p.exptime.end.max);
376  max_particles = p.amount * longestLife;
377  }
378 
379  p_manager->reserveParticleSpace(max_particles * 1.2);
380 }

390 void ParticleSpawner::spawnParticle(ClientEnvironment *env, float radius,
391  const core::matrix4 *attached_absolute_pos_rot_matrix)
392 {
393  float fac = 0;
394  if (p.time != 0) { // ensure safety from divide-by-zeroes
395  fac = m_time / (p.time+0.1f);
396  }
397 
398  auto r_pos = p.pos.blend(fac);
399  auto r_vel = p.vel.blend(fac);
400  auto r_acc = p.acc.blend(fac);
401  auto r_drag = p.drag.blend(fac);
402  auto r_radius = p.radius.blend(fac);
403  auto r_jitter = p.jitter.blend(fac);
404  auto r_bounce = p.bounce.blend(fac);
405  v3f attractor_origin = p.attractor_origin.blend(fac);
406  v3f attractor_direction = p.attractor_direction.blend(fac);
407  auto attractor_obj = findObjectByID(env, p.attractor_attachment);
408  auto attractor_direction_obj = findObjectByID(env, p.attractor_direction_attachment);
409 
410  auto r_exp = p.exptime.blend(fac);
411  auto r_size = p.size.blend(fac);
412  auto r_attract = p.attract.blend(fac);
413  auto attract = r_attract.pickWithin();
414 
415  v3f ppos = m_player->getPosition() / BS;
416  v3f pos = r_pos.pickWithin();
417  v3f sphere_radius = r_radius.pickWithin();
418 
419  // Need to apply this first or the following check
420  // will be wrong for attached spawners
421  if (attached_absolute_pos_rot_matrix) {
422  pos *= BS;
423  attached_absolute_pos_rot_matrix->transformVect(pos);
424  pos /= BS;
425  v3s16 camera_offset = m_particlemanager->m_env->getCameraOffset();
426  pos.X += camera_offset.X;
427  pos.Y += camera_offset.Y;
428  pos.Z += camera_offset.Z;
429  }
430 
431  if (pos.getDistanceFromSQ(ppos) > radius*radius)
432  return;
433 
434  // Parameters for the single particle we're about to spawn
435  ParticleParameters pp;
436  pp.pos = pos;
437 
438  pp.vel = r_vel.pickWithin();
439  pp.acc = r_acc.pickWithin();
440  pp.drag = r_drag.pickWithin();
441  pp.jitter = r_jitter;
442  pp.bounce = r_bounce;
443 
444  if (attached_absolute_pos_rot_matrix) {
445  // Apply attachment rotation
446  attached_absolute_pos_rot_matrix->rotateVect(pp.vel);
447  attached_absolute_pos_rot_matrix->rotateVect(pp.acc);
448  }
449 
450  if (attractor_obj)
451  attractor_origin += attractor_obj->getPosition() / BS;
452  if (attractor_direction_obj) {
453  auto *attractor_absolute_pos_rot_matrix = attractor_direction_obj->getAbsolutePosRotMatrix();
454  if (attractor_absolute_pos_rot_matrix)
455  attractor_absolute_pos_rot_matrix->rotateVect(attractor_direction);
456  }
457 
458  pp.expirationtime = r_exp.pickWithin();
459 
460  if (sphere_radius != v3f()) {
461  f32 l = sphere_radius.getLength();
462  v3f mag = sphere_radius;
463  mag.normalize();
464 
465  v3f ofs = v3f(l,0,0);
466  ofs.rotateXZBy(myrand_range(0.f,360.f));
467  ofs.rotateYZBy(myrand_range(0.f,360.f));
468  ofs.rotateXYBy(myrand_range(0.f,360.f));
469 
470  pp.pos += ofs * mag;
471  }
472 
473  if (p.attractor_kind != ParticleParamTypes::AttractorKind::none && attract != 0) {
474  v3f dir;
475  f32 dist = 0; /* =0 necessary to silence warning */
476  switch (p.attractor_kind) {
477  case ParticleParamTypes::AttractorKind::none:
478  break;
479 
480  case ParticleParamTypes::AttractorKind::point: {
481  dist = pp.pos.getDistanceFrom(attractor_origin);
482  dir = pp.pos - attractor_origin;
483  dir.normalize();
484  break;
485  }
486 
487  case ParticleParamTypes::AttractorKind::line: {
488  // https://github.com/minetest/minetest/issues/11505#issuecomment-915612700
489  const auto& lorigin = attractor_origin;
490  v3f ldir = attractor_direction;
491  ldir.normalize();
492  auto origin_to_point = pp.pos - lorigin;
493  auto scalar_projection = origin_to_point.dotProduct(ldir);
494  auto point_on_line = lorigin + (ldir * scalar_projection);
495 
496  dist = pp.pos.getDistanceFrom(point_on_line);
497  dir = (point_on_line - pp.pos);
498  dir.normalize();
499  dir *= -1; // flip it around so strength=1 attracts, not repulses
500  break;
501  }
502 
503  case ParticleParamTypes::AttractorKind::plane: {
504  // https://github.com/minetest/minetest/issues/11505#issuecomment-915612700
505  const v3f& porigin = attractor_origin;
506  v3f normal = attractor_direction;
507  normal.normalize();
508  v3f point_to_origin = porigin - pp.pos;
509  f32 factor = normal.dotProduct(point_to_origin);
510  if (numericAbsolute(factor) == 0.0f) {
511  dir = normal;
512  } else {
513  factor = numericSign(factor);
514  dir = normal * factor;
515  }
516  dist = numericAbsolute(normal.dotProduct(pp.pos - porigin));
517  dir *= -1; // flip it around so strength=1 attracts, not repulses
518  break;
519  }
520  }
521 
522  f32 speedTowards = numericAbsolute(attract) * dist;
523  v3f avel = dir * speedTowards;
524  if (attract > 0 && speedTowards > 0) {
525  avel *= -1;
526  if (p.attractor_kill) {
527  // make sure the particle dies after crossing the attractor threshold
528  f32 timeToCenter = dist / speedTowards;
529  if (timeToCenter < pp.expirationtime)
530  pp.expirationtime = timeToCenter;
531  }
532  }
533  pp.vel += avel;
534  }
535 
536  p.copyCommon(pp);
537 
538  ClientTexRef texture;
539  v2f texpos, texsize;
540  video::SColor color(0xFFFFFFFF);
541 
542  if (p.node.getContent() != CONTENT_IGNORE) {
543  const ContentFeatures &f =
544  m_particlemanager->m_env->getGameDef()->ndef()->get(p.node);
545  if (!ParticleManager::getNodeParticleParams(p.node, f, pp, &texture.ref,
546  texpos, texsize, &color, p.node_tile))
547  return;
548  } else {
549  if (m_texcount == 0)
550  return;
551  texture = decltype(texture)(m_texpool[m_texcount == 1 ? 0 : myrand_range(0,m_texcount-1)]);
552  texpos = v2f(0.0f, 0.0f);
553  texsize = v2f(1.0f, 1.0f);
554  if (texture.tex->animated)
555  pp.animation = texture.tex->animation;
556  }
557 
558  // synchronize animation length with particle life if desired
559  if (pp.animation.type != TAT_NONE) {
560  // FIXME: this should be moved into a TileAnimationParams class method
561  if (pp.animation.type == TAT_VERTICAL_FRAMES &&
562  pp.animation.vertical_frames.length < 0) {
563  auto& a = pp.animation.vertical_frames;
564  // we add a tiny extra value to prevent the first frame
565  // from flickering back on just before the particle dies
566  a.length = (pp.expirationtime / -a.length) + 0.1;
567  } else if (pp.animation.type == TAT_SHEET_2D &&
568  pp.animation.sheet_2d.frame_length < 0) {
569  auto& a = pp.animation.sheet_2d;
570  auto frames = a.frames_w * a.frames_h;
571  auto runtime = (pp.expirationtime / -a.frame_length) + 0.1;
572  pp.animation.sheet_2d.frame_length = frames / runtime;
573  }
574  }
575 
576  // Allow keeping default random size
577  if (p.size.start.max > 0.0f || p.size.end.max > 0.0f)
578  pp.size = r_size.pickWithin();
579 
580  ++m_active;
581  auto pa = new Particle(
582  m_gamedef,
583  m_player,
584  env,
585  pp,
586  texture,
587  texpos,
588  texsize,
589  color
590  );
591  pa->m_parent = this;
592  m_particlemanager->addParticle(pa);
593 }

595 void ParticleSpawner::step(float dtime, ClientEnvironment *env)
596 {
597  m_time += dtime;
598 
599  static thread_local const float radius =
600  g_settings->getS16("max_block_send_distance") * MAP_BLOCKSIZE;
601 
602  bool unloaded = false;
603  const core::matrix4 *attached_absolute_pos_rot_matrix = nullptr;
604  if (m_attached_id) {
605  if (GenericCAO *attached = env->getGenericCAO(m_attached_id)) {
606  attached_absolute_pos_rot_matrix = attached->getAbsolutePosRotMatrix();
607  } else {
608  unloaded = true;
609  }
610  }
611 
612  if (p.time != 0) {
613  // Spawner exists for a predefined timespan
614  for (auto i = m_spawntimes.begin(); i != m_spawntimes.end(); ) {
615  if ((*i) <= m_time && p.amount > 0) {
616  --p.amount;
617

619  // attached to an unloaded object or distant from player.
620  if (!unloaded)
621  spawnParticle(env, radius, attached_absolute_pos_rot_matrix);
622 
623  i = m_spawntimes.erase(i);
624  } else {
625  ++i;
626  }
627  }
628  } else {
629  // Spawner exists for an infinity timespan, spawn on a per-second base
630 
631  // Skip this step if attached to an unloaded object
632  if (unloaded)
633  return;
634 
635  for (int i = 0; i <= p.amount; i++) {
636  if (myrand_float() < dtime)
637  spawnParticle(env, radius, attached_absolute_pos_rot_matrix);
638  }

717 void ParticleManager::handleParticleEvent(ClientEvent *event, Client *client,
718  LocalPlayer *player)
719 {
720  switch (event->type) {
721  case CE_DELETE_PARTICLESPAWNER: {
722  deleteParticleSpawner(event->delete_particlespawner.id);
723  // no allocated memory in delete event
724  break;
725  }
726  case CE_ADD_PARTICLESPAWNER: {
727  deleteParticleSpawner(event->add_particlespawner.id);
728 
729  const ParticleSpawnerParameters &p = *event->add_particlespawner.p;
730 
731  // texture pool
732  std::unique_ptr<ClientTexture[]> texpool = nullptr;
733  size_t txpsz = 0;
734  if (!p.texpool.empty()) {
735  txpsz = p.texpool.size();
736  texpool = decltype(texpool)(new ClientTexture [txpsz]);
737 
738  for (size_t i = 0; i < txpsz; ++i) {
739  texpool[i] = ClientTexture(p.texpool[i], client->tsrc());
740  }
741  } else {
742  // no texpool in use, use fallback texture
743  txpsz = 1;
744  texpool = decltype(texpool)(new ClientTexture[1] {
745  ClientTexture(p.texture, client->tsrc())
746  });
747  }
748 
749  auto toadd = new ParticleSpawner(client, player,
750  p,
751  event->add_particlespawner.attached_id,
752  texpool,
753  txpsz,
754  this);
755 
756  addParticleSpawner(event->add_particlespawner.id, toadd);
757 
758  delete event->add_particlespawner.p;
759  break;
760  }
761  case CE_SPAWN_PARTICLE: {
762  ParticleParameters &p = *event->spawn_particle;
763 
764  ClientTexRef texture;
765  v2f texpos, texsize;
766  video::SColor color(0xFFFFFFFF);
767 
768  f32 oldsize = p.size;
769 
770  if (p.node.getContent() != CONTENT_IGNORE) {
771  const ContentFeatures &f = m_env->getGameDef()->ndef()->get(p.node);
772  getNodeParticleParams(p.node, f, p, &texture.ref, texpos,
773  texsize, &color, p.node_tile);
774  } else {
775  /* with no particlespawner to own the texture, we need
776  * to save it on the heap. it will be freed when the
777  * particle is destroyed */
778  auto texstore = new ClientTexture(p.texture, client->tsrc());
779 
780  texture = ClientTexRef(*texstore);
781  texpos = v2f(0.0f, 0.0f);
782  texsize = v2f(1.0f, 1.0f);
783  }
784 
785  // Allow keeping default random size
786  if (oldsize > 0.0f)
787  p.size = oldsize;
788 
789  if (texture.ref) {
790  Particle *toadd = new Particle(client, player, m_env,
791  p, texture, texpos, texsize, color);
792 
793  addParticle(toadd);
794  }
795 
796  delete event->spawn_particle;
797  break;
798  }
799  default: break;
800  }
801 }

803 bool ParticleManager::getNodeParticleParams(const MapNode &n,
804  const ContentFeatures &f, ParticleParameters &p, video::ITexture **texture,
805  v2f &texpos, v2f &texsize, video::SColor *color, u8 tilenum)
806 {
807  // No particles for "airlike" nodes
808  if (f.drawtype == NDT_AIRLIKE)
809  return false;
810 
811  // Texture
812  u8 texid;
813  if (tilenum > 0 && tilenum <= 6)
814  texid = tilenum - 1;
815  else
816  texid = myrand_range(0,5);
817  const TileLayer &tile = f.tiles[texid].layers[0];
818  p.animation.type = TAT_NONE;
819 
820  // Only use first frame of animated texture
821  if (tile.material_flags & MATERIAL_FLAG_ANIMATION)
822  *texture = (*tile.frames)[0].texture;
823  else
824  *texture = tile.texture;
825 
826  float size = (myrand_range(0,8)) / 64.0f;
827  p.size = BS * size;
828  if (tile.scale)
829  size /= tile.scale;
830  texsize = v2f(size * 2.0f, size * 2.0f);
831  texpos.X = (myrand_range(0,64)) / 64.0f - texsize.X;
832  texpos.Y = (myrand_range(0,64)) / 64.0f - texsize.Y;
833 
834  if (tile.has_color)
835  *color = tile.color;
836  else
837  n.getColor(f, color);
838 
839  return true;
840 }
