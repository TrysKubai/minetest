26 // Simple 2D bitmap class with just the functionality needed here
27 class Bitmap {
28  u32 linesize, lines;
29  std::vector<u8> data;
30 
31  static inline u32 bytepos(u32 index) { return index >> 3; }
32  static inline u8 bitpos(u32 index) { return index & 7; }
33 
34 public:
35  Bitmap(u32 width, u32 height) : linesize(width), lines(height),
36  data(bytepos(width * height) + 1) {}
37 
38  inline bool get(u32 x, u32 y) const {
39  u32 index = y * linesize + x;
40  return data[bytepos(index)] & (1 << bitpos(index));
41  }
42 
43  inline void set(u32 x, u32 y) {
44  u32 index = y * linesize + x;
45  data[bytepos(index)] |= 1 << bitpos(index);
46  }
47 
48  inline bool all() const {
49  for (u32 i = 0; i < data.size() - 1; i++) {
50  if (data[i] != 0xff)
51  return false;
52  }
53  // last byte not entirely filled
54  for (u8 i = 0; i < bitpos(linesize * lines); i++) {
55  bool value_of_bit = data.back() & (1 << i);
56  if (!value_of_bit)
57  return false;
58  }
59  return true;
60  }
61 
62  inline void copy(Bitmap &to) const {
63  assert(to.linesize == linesize && to.lines == lines);
64  to.data = data;
65  }
66 };

68 /* Fill in RGB values for transparent pixels, to correct for odd colors
69  * appearing at borders when blending. This is because many PNG optimizers
70  * like to discard RGB values of transparent pixels, but when blending then
71  * with non-transparent neighbors, their RGB values will show up nonetheless.
72  *
73  * This function modifies the original image in-place.
74  *
75  * Parameter "threshold" is the alpha level below which pixels are considered
76  * transparent. Should be 127 when the texture is used with ALPHA_CHANNEL_REF,
77  * 0 when alpha blending is used.
78  */
79 void imageCleanTransparent(video::IImage *src, u32 threshold)
80 {
81  core::dimension2d<u32> dim = src->getDimension();
82 
83  Bitmap bitmap(dim.Width, dim.Height);
84 
85  // First pass: Mark all opaque pixels
86  // Note: loop y around x for better cache locality.
87  for (u32 ctry = 0; ctry < dim.Height; ctry++)
88  for (u32 ctrx = 0; ctrx < dim.Width; ctrx++) {
89  if (src->getPixel(ctrx, ctry).getAlpha() > threshold)
90  bitmap.set(ctrx, ctry);
91  }
92 
93  // Exit early if all pixels opaque
94  if (bitmap.all())
95  return;
96 
97  Bitmap newmap = bitmap;
98 
99  // Cap iterations to keep runtime reasonable, for higher-res textures we can
100  // get away with filling less pixels.
101  int iter_max = 11 - std::max(dim.Width, dim.Height) / 16;
102  iter_max = std::max(iter_max, 2);
103 
104  // Then repeatedly look for transparent pixels, filling them in until
105  // we're finished.
106  for (int iter = 0; iter < iter_max; iter++) {
107 
108  for (u32 ctry = 0; ctry < dim.Height; ctry++)
109  for (u32 ctrx = 0; ctrx < dim.Width; ctrx++) {
110  // Skip pixels we have already processed
111  if (bitmap.get(ctrx, ctry))
112  continue;
113 
114  // Sample size and total weighted r, g, b values
115  u32 ss = 0, sr = 0, sg = 0, sb = 0;
116 
117  // Walk each neighbor pixel (clipped to image bounds)
118  for (u32 sy = (ctry < 1) ? 0 : (ctry - 1);
119  sy <= (ctry + 1) && sy < dim.Height; sy++)
120  for (u32 sx = (ctrx < 1) ? 0 : (ctrx - 1);
121  sx <= (ctrx + 1) && sx < dim.Width; sx++) {
122  // Ignore pixels we haven't processed
123  if (!bitmap.get(sx, sy))
124  continue;
125 
126  // Add RGB values weighted by alpha IF the pixel is opaque, otherwise
127  // use full weight since we want to propagate colors.
128  video::SColor d = src->getPixel(sx, sy);
129  u32 a = d.getAlpha() <= threshold ? 255 : d.getAlpha();
130  ss += a;
131  sr += a * d.getRed();
132  sg += a * d.getGreen();
133  sb += a * d.getBlue();
134  }
135 
136  // Set pixel to average weighted by alpha
137  if (ss > 0) {
138  video::SColor c = src->getPixel(ctrx, ctry);
139  c.setRed(sr / ss);
140  c.setGreen(sg / ss);
141  c.setBlue(sb / ss);
142  src->setPixel(ctrx, ctry, c);
143  newmap.set(ctrx, ctry);
144  }
145  }
146 
147  if (newmap.all())
148  return;
149 
150  // Apply changes to bitmap for next run. This is done so we don't introduce
151  // a bias in color propagation in the direction pixels are processed.
152  newmap.copy(bitmap);
153 
154  }
155 }

157 /* Scale a region of an image into another image, using nearest-neighbor with
158  * anti-aliasing; treat pixels as crisp rectangles, but blend them at boundaries
159  * to prevent non-integer scaling ratio artifacts. Note that this may cause
160  * some blending at the edges where pixels don't line up perfectly, but this
161  * filter is designed to produce the most accurate results for both upscaling
162  * and downscaling.
163  */
164 void imageScaleNNAA(video::IImage *src, const core::rect<s32> &srcrect, video::IImage *dest)
165 {
166  double sx, sy, minsx, maxsx, minsy, maxsy, area, ra, ga, ba, aa, pw, ph, pa;
167  u32 dy, dx;
168  video::SColor pxl;
169 
170  // Cache rectangle boundaries.
171  double sox = srcrect.UpperLeftCorner.X * 1.0;
172  double soy = srcrect.UpperLeftCorner.Y * 1.0;
173  double sw = srcrect.getWidth() * 1.0;
174  double sh = srcrect.getHeight() * 1.0;
175 
176  // Walk each destination image pixel.
177  // Note: loop y around x for better cache locality.
178  core::dimension2d<u32> dim = dest->getDimension();
179  for (dy = 0; dy < dim.Height; dy++)
180  for (dx = 0; dx < dim.Width; dx++) {
181 
182  // Calculate floating-point source rectangle bounds.
183  // Do some basic clipping, and for mirrored/flipped rects,
184  // make sure min/max are in the right order.
185  minsx = sox + (dx * sw / dim.Width);
186  minsx = rangelim(minsx, 0, sox + sw);
187  maxsx = minsx + sw / dim.Width;
188  maxsx = rangelim(maxsx, 0, sox + sw);
189  if (minsx > maxsx)
190  SWAP(double, minsx, maxsx);
191  minsy = soy + (dy * sh / dim.Height);
192  minsy = rangelim(minsy, 0, soy + sh);
193  maxsy = minsy + sh / dim.Height;
194  maxsy = rangelim(maxsy, 0, soy + sh);
195  if (minsy > maxsy)
196  SWAP(double, minsy, maxsy);
197 
198  // Total area, and integral of r, g, b values over that area,
199  // initialized to zero, to be summed up in next loops.
200  area = 0;
201  ra = 0;
202  ga = 0;
203  ba = 0;
204  aa = 0;
205 
206  // Loop over the integral pixel positions described by those bounds.
207  for (sy = floor(minsy); sy < maxsy; sy++)
208  for (sx = floor(minsx); sx < maxsx; sx++) {
209 
210  // Calculate width, height, then area of dest pixel
211  // that's covered by this source pixel.
212  pw = 1;
213  if (minsx > sx)
214  pw += sx - minsx;
215  if (maxsx < (sx + 1))
216  pw += maxsx - sx - 1;
217  ph = 1;
218  if (minsy > sy)
219  ph += sy - minsy;
220  if (maxsy < (sy + 1))
221  ph += maxsy - sy - 1;
222  pa = pw * ph;
223 
224  // Get source pixel and add it to totals, weighted
225  // by covered area and alpha.
226  pxl = src->getPixel((u32)sx, (u32)sy);
227  area += pa;
228  ra += pa * pxl.getRed();
229  ga += pa * pxl.getGreen();
230  ba += pa * pxl.getBlue();
231  aa += pa * pxl.getAlpha();
232  }
233 
234  // Set the destination image pixel to the average color.
235  if (area > 0) {
236  pxl.setRed(ra / area + 0.5);
237  pxl.setGreen(ga / area + 0.5);
238  pxl.setBlue(ba / area + 0.5);
239  pxl.setAlpha(aa / area + 0.5);
240  } else {
241  pxl.setRed(0);
242  pxl.setGreen(0);
243  pxl.setBlue(0);
244  pxl.setAlpha(0);
245  }
246  dest->setPixel(dx, dy, pxl);
247  }
248 }
