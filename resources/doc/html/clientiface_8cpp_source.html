75 LuaEntitySAO *getAttachedObject(PlayerSAO *sao, ServerEnvironment *env)
76 {
77  if (!sao->isAttached())
78  return nullptr;
79 
80  int id;
81  std::string bone;
82  v3f dummy;
83  bool force_visible;
84  sao->getAttachment(&id, &bone, &dummy, &dummy, &force_visible);
85  ServerActiveObject *ao = env->getActiveObject(id);
86  while (id && ao) {
87  ao->getAttachment(&id, &bone, &dummy, &dummy, &force_visible);
88  if (id)
89  ao = env->getActiveObject(id);
90  }
91  return dynamic_cast<LuaEntitySAO *>(ao);
92 }

94 void RemoteClient::GetNextBlocks (
95  ServerEnvironment *env,
96  EmergeManager * emerge,
97  float dtime,
98  std::vector<PrioritySortedBlockTransfer> &dest)
99 {
100  // Increment timers
101  m_nothing_to_send_pause_timer -= dtime;
102  m_map_send_completion_timer += dtime;
103 
104  if (m_map_send_completion_timer > g_settings->getFloat("server_unload_unused_data_timeout") * 0.8f) {
105  infostream << "Server: Player " << m_name << ", peer_id=" << peer_id
106  << ": full map send is taking too long ("
107  << m_map_send_completion_timer
108  << "s), restarting to avoid visible blocks being unloaded."
109  << std::endl;
110  m_map_send_completion_timer = 0.0f;
111  m_nearest_unsent_d = 0;
112  }
113 
114  if (m_nothing_to_send_pause_timer >= 0)
115  return;
116 
117  RemotePlayer *player = env->getPlayer(peer_id);
118  // This can happen sometimes; clients and players are not in perfect sync.
119  if (!player)
120  return;
121 
122  PlayerSAO *sao = player->getPlayerSAO();
123  if (!sao)
124  return;
125 
126  // Won't send anything if already sending
127  if (m_blocks_sending.size() >= m_max_simul_sends) {
128  //infostream<<"Not sending any blocks, Queue full."<<std::endl;
129  return;
130  }
131 
132  v3f playerpos = sao->getBasePosition();
133  // if the player is attached, get the velocity from the attached object
134  LuaEntitySAO *lsao = getAttachedObject(sao, env);
135  const v3f &playerspeed = lsao? lsao->getVelocity() : player->getSpeed();
136  v3f playerspeeddir(0,0,0);
137  if (playerspeed.getLength() > 1.0f * BS)
138  playerspeeddir = playerspeed / playerspeed.getLength();
139  // Predict to next block
140  v3f playerpos_predicted = playerpos + playerspeeddir * (MAP_BLOCKSIZE * BS);
141 
142  v3s16 center_nodepos = floatToInt(playerpos_predicted, BS);
143 
144  v3s16 center = getNodeBlockPos(center_nodepos);
145 
146  // Camera position and direction
147  v3f camera_pos = sao->getEyePosition();
148  v3f camera_dir = v3f(0,0,1);
149  camera_dir.rotateYZBy(sao->getLookPitch());
150  camera_dir.rotateXZBy(sao->getRotation().Y);
151 
152  if (sao->getCameraInverted())
153  camera_dir = -camera_dir;
154 
155  u16 max_simul_sends_usually = m_max_simul_sends;
156 
157  /*
158  Check the time from last addNode/removeNode.
159 
160  Decrease send rate if player is building stuff.
161  */
162  m_time_from_building += dtime;
163  if (m_time_from_building < m_min_time_from_building) {
164  max_simul_sends_usually
165  = LIMITED_MAX_SIMULTANEOUS_BLOCK_SENDS;
166  }
167 
168  /*
169  Number of blocks sending + number of blocks selected for sending
170  */
171  u32 num_blocks_selected = m_blocks_sending.size();
172 
173  /*
174  next time d will be continued from the d from which the nearest
175  unsent block was found this time.
176 
177  This is because not necessarily any of the blocks found this
178  time are actually sent.
179  */
180  s32 new_nearest_unsent_d = -1;
181 
182  // Get view range and camera fov (radians) from the client
183  s16 wanted_range = sao->getWantedRange() + 1;
184  float camera_fov = sao->getFov();
185 
186  /*
187  Get the starting value of the block finder radius.
188  */
189  if (m_last_center != center) {
190  m_nearest_unsent_d = 0;
191  m_last_center = center;
192  m_map_send_completion_timer = 0.0f;
193  }
194  // reset the unsent distance if the view angle has changed more that 10% of the fov
195  // (this matches isBlockInSight which allows for an extra 10%)
196  if (camera_dir.dotProduct(m_last_camera_dir) < std::cos(camera_fov * 0.1f)) {
197  m_nearest_unsent_d = 0;
198  m_last_camera_dir = camera_dir;
199  m_map_send_completion_timer = 0.0f;
200  }
201  if (m_nearest_unsent_d > 0) {
202  // make sure any blocks modified since the last time we sent blocks are resent
203  for (const v3s16 &p : m_blocks_modified) {
204  m_nearest_unsent_d = std::min(m_nearest_unsent_d, center.getDistanceFrom(p));
205  }
206  }
207  m_blocks_modified.clear();
208 
209  s16 d_start = m_nearest_unsent_d;
210 
211  // Distrust client-sent FOV and get server-set player object property
212  // zoom FOV (degrees) as a check to avoid hacked clients using FOV to load
213  // distant world.
214  // (zoom is disabled by value 0)
215  float prop_zoom_fov = sao->getZoomFOV() < 0.001f ?
216  0.0f :
217  std::max(camera_fov, sao->getZoomFOV() * core::DEGTORAD);
218 
219  const s16 full_d_max = std::min(adjustDist(m_max_send_distance, prop_zoom_fov),
220  wanted_range);
221  const s16 d_opt = std::min(adjustDist(m_block_optimize_distance, prop_zoom_fov),
222  wanted_range);
223  const s16 d_blocks_in_sight = full_d_max * BS * MAP_BLOCKSIZE;
224 
225  s16 d_max_gen = std::min(adjustDist(m_max_gen_distance, prop_zoom_fov),
226  wanted_range);
227 
228  s16 d_max = full_d_max;
229 
230  // Don't loop very much at a time
231  s16 max_d_increment_at_time = 2;
232  if (d_max > d_start + max_d_increment_at_time)
233  d_max = d_start + max_d_increment_at_time;
234 
235  // cos(angle between velocity and camera) * |velocity|
236  // Limit to 0.0f in case player moves backwards.
237  f32 dot = rangelim(camera_dir.dotProduct(playerspeed), 0.0f, 300.0f);
238 
239  // Reduce the field of view when a player moves and looks forward.
240  // limit max fov effect to 50%, 60% at 20n/s fly speed
241  camera_fov = camera_fov / (1 + dot / 300.0f);
242 
243  s32 nearest_emerged_d = -1;
244  s32 nearest_emergefull_d = -1;
245  s32 nearest_sent_d = -1;
246  //bool queue_is_full = false;
247 
248  const v3s16 cam_pos_nodes = floatToInt(camera_pos, BS);
249 
250  s16 d;
251  for (d = d_start; d <= d_max; d++) {
252  /*
253  Get the border/face dot coordinates of a "d-radiused"
254  box
255  */
256  std::vector<v3s16> list = FacePositionCache::getFacePositions(d);
257 
258  std::vector<v3s16>::iterator li;
259  for (li = list.begin(); li != list.end(); ++li) {
260  v3s16 p = *li + center;
261 
262  /*
263  Send throttling
264  - Don't allow too many simultaneous transfers
265  - EXCEPT when the blocks are very close
266 
267  Also, don't send blocks that are already flying.
268  */
269 
270  // Start with the usual maximum
271  u16 max_simul_dynamic = max_simul_sends_usually;
272 
273  // If block is very close, allow full maximum
274  if (d <= BLOCK_SEND_DISABLE_LIMITS_MAX_D)
275  max_simul_dynamic = m_max_simul_sends;
276 
277  /*
278  Do not go over max mapgen limit
279  */
280  if (blockpos_over_max_limit(p))
281  continue;
282 
283  // If this is true, inexistent block will be made from scratch
284  bool generate = d <= d_max_gen;
285 
286  /*
287  Don't generate or send if not in sight
288  FIXME This only works if the client uses a small enough
289  FOV setting. The default of 72 degrees is fine.
290  Also retrieve a smaller view cone in the direction of the player's
291  movement.
292  (0.1 is about 4 degrees)
293  */
294  f32 dist;
295  if (!(isBlockInSight(p, camera_pos, camera_dir, camera_fov,
296  d_blocks_in_sight, &dist) ||
297  (playerspeed.getLength() > 1.0f * BS &&
298  isBlockInSight(p, camera_pos, playerspeeddir, 0.1f,
299  d_blocks_in_sight)))) {
300  continue;
301  }
302 
303  /*
304  Check if map has this block
305  */
306  MapBlock *block = env->getMap().getBlockNoCreateNoEx(p);
307  if (block) {
308  // First: Reset usage timer, this block will be of use in the future.
309  block->resetUsageTimer();
310  }
311 
312  // Don't select too many blocks for sending
313  if (num_blocks_selected >= max_simul_dynamic) {
314  //queue_is_full = true;
315  goto queue_full_break;
316  }
317 
318  // Don't send blocks that are currently being transferred
319  if (m_blocks_sending.find(p) != m_blocks_sending.end())
320  continue;
321 
322  /*
323  Don't send already sent blocks
324  */
325  if (m_blocks_sent.find(p) != m_blocks_sent.end())
326  continue;
327 
328  bool block_not_found = false;
329  if (block) {
330  // Check whether the block exists (with data)
331  if (!block->isGenerated())
332  block_not_found = true;
333 
334  /*
335  If block is not close, don't send it unless it is near
336  ground level.
337 
338  Block is near ground level if night-time mesh
339  differs from day-time mesh.
340  */
341  if (d >= d_opt) {
342  if (!block->getIsUnderground() && !block->getDayNightDiff())
343  continue;
344  }
345 
346  /*
347  Check occlusion cache first.
348  */
349  if (m_blocks_occ.find(p) != m_blocks_occ.end())
350  continue;
351 
352  if (m_occ_cull && !block_not_found &&
353  env->getMap().isBlockOccluded(block, cam_pos_nodes)) {
354  m_blocks_occ.insert(p);
355  continue;
356  }
357  }
358 
359  /*
360  If block has been marked to not exist on disk (dummy) or is
361  not generated and generating new ones is not wanted, skip block.
362  */
363  if (!generate && block_not_found) {
364  // get next one.
365  continue;
366  }
367 
368  /*
369  Add inexistent block to emerge queue.
370  */
371  if (block == NULL || block_not_found) {
372  if (emerge->enqueueBlockEmerge(peer_id, p, generate)) {
373  if (nearest_emerged_d == -1)
374  nearest_emerged_d = d;
375  } else {
376  if (nearest_emergefull_d == -1)
377  nearest_emergefull_d = d;
378  goto queue_full_break;
379  }
380 
381  // get next one.
382  continue;
383  }
384 
385  if (nearest_sent_d == -1)
386  nearest_sent_d = d;
387 
388  /*
389  Add block to send queue
390  */
391  PrioritySortedBlockTransfer q((float)dist, p, peer_id);
392 
393  dest.push_back(q);
394 
395  num_blocks_selected += 1;
396  }
397  }
398 queue_full_break:
399 
400  // If nothing was found for sending and nothing was queued for
401  // emerging, continue next time browsing from here
402  if (nearest_emerged_d != -1) {
403  new_nearest_unsent_d = nearest_emerged_d;
404  } else if (nearest_emergefull_d != -1) {
405  new_nearest_unsent_d = nearest_emergefull_d;
406  } else {
407  if (d > full_d_max) {
408  new_nearest_unsent_d = 0;
409  m_nothing_to_send_pause_timer = 2.0f;
410  infostream << "Server: Player " << m_name << ", RemoteClient " << peer_id << ": full map send completed after " << m_map_send_completion_timer << "s, restarting" << std::endl;
411  m_map_send_completion_timer = 0.0f;
412  } else {
413  if (nearest_sent_d != -1)
414  new_nearest_unsent_d = nearest_sent_d;
415  else
416  new_nearest_unsent_d = d;
417  }
418  }
419 
420  if (new_nearest_unsent_d != -1 && m_nearest_unsent_d != new_nearest_unsent_d) {
421  m_nearest_unsent_d = new_nearest_unsent_d;
422  // if the distance has changed, clear the occlusion cache
423  m_blocks_occ.clear();
424  }
425 }

471 void RemoteClient::notifyEvent(ClientStateEvent event)
472 {
473  std::ostringstream myerror;
474  switch (m_state)
475  {
476  case CS_Invalid:
477  //intentionally do nothing
478  break;
479  case CS_Created:
480  switch (event) {
481  case CSE_Hello:
482  m_state = CS_HelloSent;
483  break;
484  case CSE_Disconnect:
485  m_state = CS_Disconnecting;
486  break;
487  case CSE_SetDenied:
488  m_state = CS_Denied;
489  break;
490  /* GotInit2 SetDefinitionsSent SetMediaSent */
491  default:
492  myerror << "Created: Invalid client state transition! " << event;
493  throw ClientStateError(myerror.str());
494  }
495  break;
496  case CS_Denied:
497  /* don't do anything if in denied state */
498  break;
499  case CS_HelloSent:
500  switch(event)
501  {
502  case CSE_AuthAccept:
503  m_state = CS_AwaitingInit2;
504  resetChosenMech();
505  break;
506  case CSE_Disconnect:
507  m_state = CS_Disconnecting;
508  break;
509  case CSE_SetDenied:
510  m_state = CS_Denied;
511  resetChosenMech();
512  break;
513  default:
514  myerror << "HelloSent: Invalid client state transition! " << event;
515  throw ClientStateError(myerror.str());
516  }
517  break;
518  case CS_AwaitingInit2:
519  switch(event)
520  {
521  case CSE_GotInit2:
522  confirmSerializationVersion();
523  m_state = CS_InitDone;
524  break;
525  case CSE_Disconnect:
526  m_state = CS_Disconnecting;
527  break;
528  case CSE_SetDenied:
529  m_state = CS_Denied;
530  break;
531 
532  /* Init SetDefinitionsSent SetMediaSent */
533  default:
534  myerror << "InitSent: Invalid client state transition! " << event;
535  throw ClientStateError(myerror.str());
536  }
537  break;
538 
539  case CS_InitDone:
540  switch(event)
541  {
542  case CSE_SetDefinitionsSent:
543  m_state = CS_DefinitionsSent;
544  break;
545  case CSE_Disconnect:
546  m_state = CS_Disconnecting;
547  break;
548  case CSE_SetDenied:
549  m_state = CS_Denied;
550  break;
551 
552  /* Init GotInit2 SetMediaSent */
553  default:
554  myerror << "InitDone: Invalid client state transition! " << event;
555  throw ClientStateError(myerror.str());
556  }
557  break;
558  case CS_DefinitionsSent:
559  switch(event)
560  {
561  case CSE_SetClientReady:
562  m_state = CS_Active;
563  break;
564  case CSE_Disconnect:
565  m_state = CS_Disconnecting;
566  break;
567  case CSE_SetDenied:
568  m_state = CS_Denied;
569  break;
570  /* Init GotInit2 SetDefinitionsSent */
571  default:
572  myerror << "DefinitionsSent: Invalid client state transition! " << event;
573  throw ClientStateError(myerror.str());
574  }
575  break;
576  case CS_Active:
577  switch(event)
578  {
579  case CSE_SetDenied:
580  m_state = CS_Denied;
581  break;
582  case CSE_Disconnect:
583  m_state = CS_Disconnecting;
584  break;
585  case CSE_SudoSuccess:
586  m_state = CS_SudoMode;
587  resetChosenMech();
588  break;
589  /* Init GotInit2 SetDefinitionsSent SetMediaSent SetDenied */
590  default:
591  myerror << "Active: Invalid client state transition! " << event;
592  throw ClientStateError(myerror.str());
593  break;
594  }
595  break;
596  case CS_SudoMode:
597  switch(event)
598  {
599  case CSE_SetDenied:
600  m_state = CS_Denied;
601  break;
602  case CSE_Disconnect:
603  m_state = CS_Disconnecting;
604  break;
605  case CSE_SudoLeave:
606  m_state = CS_Active;
607  break;
608  default:
609  myerror << "Active: Invalid client state transition! " << event;
610  throw ClientStateError(myerror.str());
611  break;
612  }
613  break;
614  case CS_Disconnecting:
615  /* we are already disconnecting */
616  break;
617  }
618 }

698 void ClientInterface::UpdatePlayerList()
699 {
700  if (m_env) {
701  std::vector<session_t> clients = getClientIDs();
702  m_clients_names.clear();
703 
704  if (!clients.empty())
705  infostream<<"Players:"<<std::endl;
706
