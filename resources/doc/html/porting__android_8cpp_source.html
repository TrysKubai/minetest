72 jclass findClass(const std::string &classname)
73 {
74  if (jnienv == nullptr)
75  return nullptr;
76 
77  jclass nativeactivity = jnienv->FindClass("android/app/NativeActivity");
78  jmethodID getClassLoader = jnienv->GetMethodID(
79  nativeactivity, "getClassLoader", "()Ljava/lang/ClassLoader;");
80  jobject cls = jnienv->CallObjectMethod(
81  app_global->activity->clazz, getClassLoader);
82  jclass classLoader = jnienv->FindClass("java/lang/ClassLoader");
83  jmethodID findClass = jnienv->GetMethodID(classLoader, "loadClass",
84  "(Ljava/lang/String;)Ljava/lang/Class;");
85  jstring strClassName = jnienv->NewStringUTF(classname.c_str());
86  return (jclass) jnienv->CallObjectMethod(cls, findClass, strClassName);
87 }

89 void initAndroid()
90 {
91  porting::jnienv = nullptr;
92  JavaVM *jvm = app_global->activity->vm;
93  JavaVMAttachArgs lJavaVMAttachArgs;
94  lJavaVMAttachArgs.version = JNI_VERSION_1_6;
95  lJavaVMAttachArgs.name = PROJECT_NAME_C "NativeThread";
96  lJavaVMAttachArgs.group = nullptr;
97 
98  if (jvm->AttachCurrentThread(&porting::jnienv, &lJavaVMAttachArgs) == JNI_ERR) {
99  errorstream << "Failed to attach native thread to jvm" << std::endl;
100  exit(-1);
101  }
102 
103  nativeActivity = findClass("net/minetest/minetest/GameActivity");
104  if (nativeActivity == nullptr)
105  errorstream <<
106  "porting::initAndroid unable to find Java native activity class" <<
107  std::endl;
108 
109 #ifdef GPROF
110  // in the start-up code
111  __android_log_print(ANDROID_LOG_ERROR, PROJECT_NAME_C,
112  "Initializing GPROF profiler");
113  monstartup("libMinetest.so");
114 #endif
115 }

129 static std::string readJavaString(jstring j_str)
130 {
131  // Get string as a UTF-8 C string
132  const char *c_str = jnienv->GetStringUTFChars(j_str, nullptr);
133  // Save it
134  std::string str(c_str);
135  // And free the C string
136  jnienv->ReleaseStringUTFChars(j_str, c_str);
137  return str;
138 }

140 void initializePathsAndroid()

142  // Set user and share paths
143  {
144  jmethodID getUserDataPath = jnienv->GetMethodID(nativeActivity,
145  "getUserDataPath", "()Ljava/lang/String;");
146  FATAL_ERROR_IF(getUserDataPath==nullptr,
147  "porting::initializePathsAndroid unable to find Java getUserDataPath method");
148  jobject result = jnienv->CallObjectMethod(app_global->activity->clazz, getUserDataPath);
149  std::string str = readJavaString((jstring) result);
150  path_user = str;
151  path_share = str;
152  path_locale = str + DIR_DELIM + "locale";

161  jobject result = jnienv->CallObjectMethod(app_global->activity->clazz, getCachePath);
162  path_cache = readJavaString((jstring) result);
163 
164  migrateCachePath();
