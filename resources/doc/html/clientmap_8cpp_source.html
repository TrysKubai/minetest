43 void MeshBufListList::add(scene::IMeshBuffer *buf, v3s16 position, u8 layer)
44 {
45  // Append to the correct layer
46  std::vector<MeshBufList> &list = lists[layer];
47  const video::SMaterial &m = buf->getMaterial();
48  for (MeshBufList &l : list) {
49  // comparing a full material is quite expensive so we don't do it if
50  // not even first texture is equal
51  if (l.m.TextureLayer[0].Texture != m.TextureLayer[0].Texture)
52  continue;
53 
54  if (l.m == m) {
55  l.bufs.emplace_back(position, buf);
56  return;
57  }
58  }
59  MeshBufList l;
60  l.m = m;
61  l.bufs.emplace_back(position, buf);
62  list.emplace_back(l);
63 }

176 void ClientMap::getBlocksInViewRange(v3s16 cam_pos_nodes,
177  v3s16 *p_blocks_min, v3s16 *p_blocks_max, float range)
178 {
179  if (range <= 0.0f)
180  range = m_control.wanted_range;
181 
182  v3s16 box_nodes_d = range * v3s16(1, 1, 1);
183  // Define p_nodes_min/max as v3s32 because 'cam_pos_nodes -/+ box_nodes_d'
184  // can exceed the range of v3s16 when a large view range is used near the
185  // world edges.
186  v3s32 p_nodes_min(
187  cam_pos_nodes.X - box_nodes_d.X,
188  cam_pos_nodes.Y - box_nodes_d.Y,
189  cam_pos_nodes.Z - box_nodes_d.Z);
190  v3s32 p_nodes_max(
191  cam_pos_nodes.X + box_nodes_d.X,
192  cam_pos_nodes.Y + box_nodes_d.Y,
193  cam_pos_nodes.Z + box_nodes_d.Z);
194  // Take a fair amount as we will be dropping more out later
195  // Umm... these additions are a bit strange but they are needed.
196  *p_blocks_min = v3s16(
197  p_nodes_min.X / MAP_BLOCKSIZE - 3,
198  p_nodes_min.Y / MAP_BLOCKSIZE - 3,
199  p_nodes_min.Z / MAP_BLOCKSIZE - 3);
200  *p_blocks_max = v3s16(
201  p_nodes_max.X / MAP_BLOCKSIZE + 1,
202  p_nodes_max.Y / MAP_BLOCKSIZE + 1,
203  p_nodes_max.Z / MAP_BLOCKSIZE + 1);
204 }

206 class MapBlockFlags
207 {
208 public:
209  static constexpr u16 CHUNK_EDGE = 8;
210  static constexpr u16 CHUNK_MASK = CHUNK_EDGE - 1;
211  static constexpr std::size_t CHUNK_VOLUME = CHUNK_EDGE * CHUNK_EDGE * CHUNK_EDGE; // volume of a chunk
212 
213  MapBlockFlags(v3s16 min_pos, v3s16 max_pos)
214  : min_pos(min_pos), volume((max_pos - min_pos) / CHUNK_EDGE + 1)
215  {
216  chunks.resize(volume.X * volume.Y * volume.Z);
217  }
218 
219  class Chunk
220  {
221  public:
222  inline u8 &getBits(v3s16 pos)
223  {
224  std::size_t address = getAddress(pos);
225  return bits[address];
226  }
227 
228  private:
229  inline std::size_t getAddress(v3s16 pos) {
230  std::size_t address = (pos.X & CHUNK_MASK) + (pos.Y & CHUNK_MASK) * CHUNK_EDGE + (pos.Z & CHUNK_MASK) * (CHUNK_EDGE * CHUNK_EDGE);
231  return address;
232  }
233 
234  std::array<u8, CHUNK_VOLUME> bits;
235  };
236 
237  Chunk &getChunk(v3s16 pos)
238  {
239  v3s16 delta = (pos - min_pos) / CHUNK_EDGE;
240  std::size_t address = delta.X + delta.Y * volume.X + delta.Z * volume.X * volume.Y;
241  Chunk *chunk = chunks[address].get();
242  if (!chunk) {
243  chunk = new Chunk();
244  chunks[address].reset(chunk);
245  }
246  return *chunk;
247  }
248 private:
249  std::vector<std::unique_ptr<Chunk>> chunks;
250  v3s16 min_pos;
251  v3s16 volume;
252 };

254 void ClientMap::updateDrawList()
255 {
256  ScopeProfiler sp(g_profiler, "CM::updateDrawList()", SPT_AVG);
257 
258  m_needs_update_drawlist = false;
259 
260  for (auto &i : m_drawlist) {
261  MapBlock *block = i.second;
262  block->refDrop();
263  }
264  m_drawlist.clear();
265 
266  for (auto &block : m_keeplist) {
267  block->refDrop();
268  }
269  m_keeplist.clear();
270 
271  const v3s16 cam_pos_nodes = floatToInt(m_camera_position, BS);
272 
273  v3s16 p_blocks_min;
274  v3s16 p_blocks_max;
275  getBlocksInViewRange(cam_pos_nodes, &p_blocks_min, &p_blocks_max);
276 
277  // Number of blocks occlusion culled
278  u32 blocks_occlusion_culled = 0;
279  // Number of blocks frustum culled
280  u32 blocks_frustum_culled = 0;
281 
282  MeshGrid mesh_grid = m_client->getMeshGrid();
283 
284  // No occlusion culling when free_move is on and camera is inside ground
285  // No occlusion culling for chunk sizes of 4 and above
286  // because the current occlusion culling test is highly inefficient at these sizes
287  bool occlusion_culling_enabled = mesh_grid.cell_size < 4;
288  if (m_control.allow_noclip) {
289  MapNode n = getNode(cam_pos_nodes);
290  if (n.getContent() == CONTENT_IGNORE || m_nodedef->get(n).solidness == 2)
291  occlusion_culling_enabled = false;
292  }
293 
294  const v3s16 camera_block = getContainerPos(cam_pos_nodes, MAP_BLOCKSIZE);
295  m_drawlist = std::map<v3s16, MapBlock*, MapBlockComparer>(MapBlockComparer(camera_block));
296 
297  auto is_frustum_culled = m_client->getCamera()->getFrustumCuller();
298 
299  // Uncomment to debug occluded blocks in the wireframe mode
300  // TODO: Include this as a flag for an extended debugging setting
301  // if (occlusion_culling_enabled && m_control.show_wireframe)
302  // occlusion_culling_enabled = porting::getTimeS() & 1;
303 
304  // Set of mesh holding blocks
305  std::set<v3s16> shortlist;
306 
307  /*
308  When range_all is enabled, enumerate all blocks visible in the
309  frustum and display them.
310  */
311  if (m_control.range_all || m_loops_occlusion_culler) {
312  // Number of blocks currently loaded by the client
313  u32 blocks_loaded = 0;
314  // Number of blocks with mesh in rendering range
315  u32 blocks_in_range_with_mesh = 0;
316 
317  MapBlockVect sectorblocks;
318 
319  for (auto &sector_it : m_sectors) {
320  MapSector *sector = sector_it.second;
321  v2s16 sp = sector->getPos();
322 
323  blocks_loaded += sector->size();
324  if (!m_control.range_all) {
325  if (sp.X < p_blocks_min.X || sp.X > p_blocks_max.X ||
326  sp.Y < p_blocks_min.Z || sp.Y > p_blocks_max.Z)
327  continue;
328  }
329 
330  sectorblocks.clear();
331  sector->getBlocks(sectorblocks);
332 
333  // Loop through blocks in sector
334  for (MapBlock *block : sectorblocks) {
335  MapBlockMesh *mesh = block->mesh;
336 
337  // Calculate the coordinates for range and frustum culling
338  v3f mesh_sphere_center;
339  f32 mesh_sphere_radius;
340 
341  v3s16 block_pos_nodes = block->getPos() * MAP_BLOCKSIZE;
342 
343  if (mesh) {
344  mesh_sphere_center = intToFloat(block_pos_nodes, BS)
345  + mesh->getBoundingSphereCenter();
346  mesh_sphere_radius = mesh->getBoundingRadius();
347  } else {
348  mesh_sphere_center = intToFloat(block_pos_nodes, BS)
349  + v3f((MAP_BLOCKSIZE * 0.5f - 0.5f) * BS);
350  mesh_sphere_radius = 0.0f;
351  }
352 
353  // First, perform a simple distance check.
354  if (!m_control.range_all &&
355  mesh_sphere_center.getDistanceFrom(m_camera_position) >
356  m_control.wanted_range * BS + mesh_sphere_radius)
357  continue; // Out of range, skip.
358 
359  // Keep the block alive as long as it is in range.
360  block->resetUsageTimer();
361  blocks_in_range_with_mesh++;
362 
363  // Frustum culling
364  // Only do coarse culling here, to account for fast camera movement.
365  // This is needed because this function is not called every frame.
366  float frustum_cull_extra_radius = 300.0f;
367  if (is_frustum_culled(mesh_sphere_center,
368  mesh_sphere_radius + frustum_cull_extra_radius)) {
369  blocks_frustum_culled++;
370  continue;
371  }
372 
373  // Raytraced occlusion culling - send rays from the camera to the block's corners
374  if (!m_control.range_all && occlusion_culling_enabled && m_enable_raytraced_culling &&
375  mesh &&
376  isMeshOccluded(block, mesh_grid.cell_size, cam_pos_nodes)) {
377  blocks_occlusion_culled++;
378  continue;
379  }
380 
381  if (mesh_grid.cell_size > 1) {
382  // Block meshes are stored in the corner block of a chunk
383  // (where all coordinate are divisible by the chunk size)
384  // Add them to the de-dup set.
385  shortlist.emplace(mesh_grid.getMeshPos(block->getPos()));
386  // All other blocks we can grab and add to the keeplist right away.
387  m_keeplist.push_back(block);
388  block->refGrab();
389  } else if (mesh) {
390  // without mesh chunking we can add the block to the drawlist
391  block->refGrab();
392  m_drawlist.emplace(block->getPos(), block);
393  }
394  }
395  }
396 
397  g_profiler->avg("MapBlock meshes in range [#]", blocks_in_range_with_mesh);
398  g_profiler->avg("MapBlocks loaded [#]", blocks_loaded);
399  } else {
400  // Blocks visited by the algorithm
401  u32 blocks_visited = 0;
402  // Block sides that were not traversed
403  u32 sides_skipped = 0;
404 
405  std::queue<v3s16> blocks_to_consider;
406 
407  v3s16 camera_mesh = mesh_grid.getMeshPos(camera_block);
408  v3s16 camera_cell = mesh_grid.getCellPos(camera_block);
409 
410  // Bits per block:
411  // [ visited | 0 | 0 | 0 | 0 | Z visible | Y visible | X visible ]
412  MapBlockFlags meshes_seen(mesh_grid.getCellPos(p_blocks_min), mesh_grid.getCellPos(p_blocks_max) + 1);
413 
414  // Start breadth-first search with the block the camera is in
415  blocks_to_consider.push(camera_mesh);
416  meshes_seen.getChunk(camera_cell).getBits(camera_cell) = 0x07; // mark all sides as visible
417 
418  // Recursively walk the space and pick mapblocks for drawing
419  while (!blocks_to_consider.empty()) {
420 
421  v3s16 block_coord = blocks_to_consider.front();
422  blocks_to_consider.pop();
423 
424  v3s16 cell_coord = mesh_grid.getCellPos(block_coord);
425  auto &flags = meshes_seen.getChunk(cell_coord).getBits(cell_coord);
426 
427  // Only visit each block once (it may have been queued up to three times)
428  if ((flags & 0x80) == 0x80)
429  continue;
430  flags |= 0x80;
431 
432  blocks_visited++;
433 
434  // Get the sector, block and mesh
435  MapSector *sector = this->getSectorNoGenerate(v2s16(block_coord.X, block_coord.Z));
436 
437  MapBlock *block = sector ? sector->getBlockNoCreateNoEx(block_coord.Y) : nullptr;
438 
439  MapBlockMesh *mesh = block ? block->mesh : nullptr;
440 
441  // Calculate the coordinates for range and frustum culling
442  v3f mesh_sphere_center;
443  f32 mesh_sphere_radius;
444 
445  v3s16 block_pos_nodes = block_coord * MAP_BLOCKSIZE;
446 
447  if (mesh) {
448  mesh_sphere_center = intToFloat(block_pos_nodes, BS)
449  + mesh->getBoundingSphereCenter();
450  mesh_sphere_radius = mesh->getBoundingRadius();
451  } else {
452  mesh_sphere_center = intToFloat(block_pos_nodes, BS) + v3f((mesh_grid.cell_size * MAP_BLOCKSIZE * 0.5f - 0.5f) * BS);
453  mesh_sphere_radius = 0.87f * mesh_grid.cell_size * MAP_BLOCKSIZE * BS;
454  }
455 
456  // First, perform a simple distance check.
457  if (!m_control.range_all &&
458  mesh_sphere_center.getDistanceFrom(intToFloat(cam_pos_nodes, BS)) >
459  m_control.wanted_range * BS + mesh_sphere_radius)
460  continue; // Out of range, skip.
461 
462  // Frustum culling
463  // Only do coarse culling here, to account for fast camera movement.
464  // This is needed because this function is not called every frame.
465  float frustum_cull_extra_radius = 300.0f;
466  if (is_frustum_culled(mesh_sphere_center,
467  mesh_sphere_radius + frustum_cull_extra_radius)) {
468  blocks_frustum_culled++;
469  continue;
470  }
471 
472  // Calculate the vector from the camera block to the current block
473  // We use it to determine through which sides of the current block we can continue the search
474  v3s16 look = block_coord - camera_mesh;
475 
476  // Occluded near sides will further occlude the far sides
477  u8 visible_outer_sides = flags & 0x07;
478 
479  // Raytraced occlusion culling - send rays from the camera to the block's corners
480  if (occlusion_culling_enabled && m_enable_raytraced_culling &&
481  block && mesh &&
482  visible_outer_sides != 0x07 && isMeshOccluded(block, mesh_grid.cell_size, cam_pos_nodes)) {
483  blocks_occlusion_culled++;
484  continue;
485  }
486 
487  if (mesh_grid.cell_size > 1) {
488  // Block meshes are stored in the corner block of a chunk
489  // (where all coordinate are divisible by the chunk size)
490  // Add them to the de-dup set.
491  shortlist.emplace(block_coord.X, block_coord.Y, block_coord.Z);
492  // All other blocks we can grab and add to the keeplist right away.
493  if (block) {
494  m_keeplist.push_back(block);
495  block->refGrab();
496  }
497  } else if (mesh) {
498  // without mesh chunking we can add the block to the drawlist
499  block->refGrab();
500  m_drawlist.emplace(block_coord, block);
501  }
502 
503  // Decide which sides to traverse next or to block away
504 
505  // First, find the near sides that would occlude the far sides
506  // * A near side can itself be occluded by a nearby block (the test above ^^)
507  // * A near side can be visible but fully opaque by itself (e.g. ground at the 0 level)
508 
509  // mesh solid sides are +Z-Z+Y-Y+X-X
510  // if we are inside the block's coordinates on an axis,
511  // treat these sides as opaque, as they should not allow to reach the far sides
512  u8 block_inner_sides = (look.X == 0 ? 3 : 0) |
513  (look.Y == 0 ? 12 : 0) |
514  (look.Z == 0 ? 48 : 0);
515 
516  // get the mask for the sides that are relevant based on the direction
517  u8 near_inner_sides = (look.X > 0 ? 1 : 2) |
518  (look.Y > 0 ? 4 : 8) |
519  (look.Z > 0 ? 16 : 32);
520 
521  // This bitset is +Z-Z+Y-Y+X-X (See MapBlockMesh), and axis is XYZ.
522  // Get he block's transparent sides
523  u8 transparent_sides = (occlusion_culling_enabled && block) ? ~block->solid_sides : 0x3F;
524 
525  // compress block transparent sides to ZYX mask of see-through axes
526  u8 near_transparency = (block_inner_sides == 0x3F) ? near_inner_sides : (transparent_sides & near_inner_sides);
527 
528  // when we are inside the camera block, do not block any sides
529  if (block_inner_sides == 0x3F)
530  block_inner_sides = 0;
531 
532  near_transparency &= ~block_inner_sides & 0x3F;
533 
534  near_transparency |= (near_transparency >> 1);
535  near_transparency = (near_transparency & 1) |
536  ((near_transparency >> 1) & 2) |
537  ((near_transparency >> 2) & 4);
538 
539  // combine with known visible sides that matter
540  near_transparency &= visible_outer_sides;
541 
542  // The rule for any far side to be visible:
543  // * Any of the adjacent near sides is transparent (different axes)
544  // * The opposite near side (same axis) is transparent, if it is the dominant axis of the look vector
545 
546  // Calculate vector from camera to mapblock center. Because we only need relation between
547  // coordinates we scale by 2 to avoid precision loss.
548  v3s16 precise_look = 2 * (block_pos_nodes - cam_pos_nodes) + mesh_grid.cell_size * MAP_BLOCKSIZE - 1;
549 
550  // dominant axis flag
551  u8 dominant_axis = (abs(precise_look.X) > abs(precise_look.Y) && abs(precise_look.X) > abs(precise_look.Z)) |
552  ((abs(precise_look.Y) > abs(precise_look.Z) && abs(precise_look.Y) > abs(precise_look.X)) << 1) |
553  ((abs(precise_look.Z) > abs(precise_look.X) && abs(precise_look.Z) > abs(precise_look.Y)) << 2);
554 
555  // Queue next blocks for processing:
556  // - Examine "far" sides of the current blocks, i.e. never move towards the camera
557  // - Only traverse the sides that are not occluded
558  // - Only traverse the sides that are not opaque
559  // When queueing, mark the relevant side on the next block as 'visible'
560  for (s16 axis = 0; axis < 3; axis++) {
561 
562  // Select a bit from transparent_sides for the side
563  u8 far_side_mask = 1 << (2 * axis);
564 
565  // axis flag
566  u8 my_side = 1 << axis;
567  u8 adjacent_sides = my_side ^ 0x07;
568 
569  auto traverse_far_side = [&](s8 next_pos_offset) {
570  // far side is visible if adjacent near sides are transparent, or if opposite side on dominant axis is transparent
571  bool side_visible = ((near_transparency & adjacent_sides) | (near_transparency & my_side & dominant_axis)) != 0;
572  side_visible = side_visible && ((far_side_mask & transparent_sides) != 0);
573 
574  v3s16 next_pos = block_coord;
575  next_pos[axis] += next_pos_offset;
576 
577  v3s16 next_cell = mesh_grid.getCellPos(next_pos);
578 
579  // If a side is a see-through, mark the next block's side as visible, and queue
580  if (side_visible) {

895 static bool getVisibleBrightness(Map *map, const v3f &p0, v3f dir, float step,
896  float step_multiplier, float start_distance, float end_distance,
897  const NodeDefManager *ndef, u32 daylight_factor, float sunlight_min_d,
898  int *result, bool *sunlight_seen)
899 {
900  int brightness_sum = 0;
901  int brightness_count = 0;
902  float distance = start_distance;
903  dir.normalize();
904  v3f pf = p0;
905  pf += dir * distance;
906  int noncount = 0;
907  bool nonlight_seen = false;
908  bool allow_allowing_non_sunlight_propagates = false;
909  bool allow_non_sunlight_propagates = false;
910  // Check content nearly at camera position
911  {
912  v3s16 p = floatToInt(p0 /*+ dir * 3*BS*/, BS);
913  MapNode n = map->getNode(p);
914  if(ndef->getLightingFlags(n).has_light &&
915  !ndef->getLightingFlags(n).sunlight_propagates)
916  allow_allowing_non_sunlight_propagates = true;
917  }
918  // If would start at CONTENT_IGNORE, start closer
919  {
920  v3s16 p = floatToInt(pf, BS);
921  MapNode n = map->getNode(p);
922  if(n.getContent() == CONTENT_IGNORE){
923  float newd = 2*BS;
924  pf = p0 + dir * 2*newd;
925  distance = newd;
926  sunlight_min_d = 0;
927  }
928  }
929  for (int i=0; distance < end_distance; i++) {
930  pf += dir * step;
931  distance += step;
932  step *= step_multiplier;
933 
934  v3s16 p = floatToInt(pf, BS);
935  MapNode n = map->getNode(p);
936  ContentLightingFlags f = ndef->getLightingFlags(n);
937  if (allow_allowing_non_sunlight_propagates && i == 0 &&
938  f.has_light && !f.sunlight_propagates) {
939  allow_non_sunlight_propagates = true;
940  }
941 
942  if (!f.has_light || (!f.sunlight_propagates && !allow_non_sunlight_propagates)){
943  nonlight_seen = true;
944  noncount++;
945  if(noncount >= 4)
946  break;
947  continue;
948  }
949 
950  if (distance >= sunlight_min_d && !*sunlight_seen && !nonlight_seen)
951  if (n.getLight(LIGHTBANK_DAY, f) == LIGHT_SUN)
952  *sunlight_seen = true;
953  noncount = 0;
954  brightness_sum += decode_light(n.getLightBlend(daylight_factor, f));
955  brightness_count++;
956  }
957  *result = 0;
958  if(brightness_count == 0)
959  return false;
960  *result = brightness_sum / brightness_count;
961  /*std::cerr<<"Sampled "<<brightness_count<<" points; result="
962  <<(*result)<<std::endl;*/
963  return true;
964 }
