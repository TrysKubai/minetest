121  // How much sun & moon transition should affect horizon color
122  float m_horizon_blend()
123  {
124  if (!m_sunlight_seen)
125  return 0;
126  float x = m_time_of_day >= 0.5 ? (1 - m_time_of_day) * 2
127  : m_time_of_day * 2;
128 
129  if (x <= 0.3)
130  return 0;
131  if (x <= 0.4) // when the sun and moon are aligned
132  return (x - 0.3) * 10;
133  if (x <= 0.5)
134  return (0.5 - x) * 10;
135  return 0;
136  }

138  // Mix two colors by a given amount
139  static video::SColor m_mix_scolor(video::SColor col1, video::SColor col2, f32 factor)
140  {
141  video::SColor result = video::SColor(
142  col1.getAlpha() * (1 - factor) + col2.getAlpha() * factor,
143  col1.getRed() * (1 - factor) + col2.getRed() * factor,
144  col1.getGreen() * (1 - factor) + col2.getGreen() * factor,
145  col1.getBlue() * (1 - factor) + col2.getBlue() * factor);
146  return result;
147  }
148  static video::SColorf m_mix_scolorf(video::SColorf col1, video::SColorf col2, f32 factor)
149  {
150  video::SColorf result =
151  video::SColorf(col1.r * (1 - factor) + col2.r * factor,
152  col1.g * (1 - factor) + col2.g * factor,
153  col1.b * (1 - factor) + col2.b * factor,
154  col1.a * (1 - factor) + col2.a * factor);
155  return result;
156  }
